8//
9// This pass performs various vector pseudo peephole optimisations after
10// instruction selection.
11//
12// Currently it converts vmerge.vvm to vmv.v.v
13// PseudoVMERGE_VVM %false, %false, %true, %allonesmask, %vl, %sew
14// ->
15// PseudoVMV_V_V %false, %true, %vl, %sew
16//
17// And masked pseudos to unmasked pseudos
18// PseudoVADD_V_V_MASK %passthru, %a, %b, %allonesmask, %vl, sew, policy
19// ->
20// PseudoVADD_V_V %passthru %a, %b, %vl, sew, policy
21//
22// It also converts AVLs to VLMAX where possible
23// %vl = VLENB * something
24// PseudoVADD_V_V %passthru, %a, %b, %vl, sew, policy
25// ->
26// PseudoVADD_V_V %passthru, %a, %b, -1, sew, policy
27//
28//===----------------------------------------------------------------------===//

89/// Given two VL operands, do we know that LHS <= RHS?
90static bool isVLKnownLE(const MachineOperand &LHS, const MachineOperand &RHS) {
91 if (LHS.isReg() && RHS.isReg() && LHS.getReg().isVirtual() &&
92 LHS.getReg() == RHS.getReg())
93 return true;
94 if (RHS.isImm() && RHS.getImm() == RISCV::VLMaxSentinel)
95 return true;
96 if (LHS.isImm() && LHS.getImm() == RISCV::VLMaxSentinel)
97 return false;
98 if (!LHS.isImm() || !RHS.isImm())
99 return false;
100 return LHS.getImm() <= RHS.getImm();
101}

103/// Given \p User that has an input operand with EEW=SEW, which uses the dest
104/// operand of \p Src with an unknown EEW, return true if their EEWs match.
105bool RISCVVectorPeephole::hasSameEEW(const MachineInstr &User,
106 const MachineInstr &Src) const {
107 unsigned UserLog2SEW =
108 User.getOperand(RISCVII::getSEWOpNum(User.getDesc())).getImm();
109 unsigned SrcLog2SEW =
110 Src.getOperand(RISCVII::getSEWOpNum(Src.getDesc())).getImm();
111 unsigned SrcLog2EEW = RISCV::getDestLog2EEW(
112 TII->get(RISCV::getRVVMCOpcode(Src.getOpcode())), SrcLog2SEW);
113 return SrcLog2EEW == UserLog2SEW;
114}

116// Attempt to reduce the VL of an instruction whose sole use is feeding a
117// instruction with a narrower VL. This currently works backwards from the
118// user instruction (which might have a smaller VL).
119bool RISCVVectorPeephole::tryToReduceVL(MachineInstr &MI) const {
120 // Note that the goal here is a bit multifaceted.
121 // 1) For store's reducing the VL of the value being stored may help to
122 // reduce VL toggles. This is somewhat of an artifact of the fact we
123 // promote arithmetic instructions but VL predicate stores.
124 // 2) For vmv.v.v reducing VL eagerly on the source instruction allows us
125 // to share code with the foldVMV_V_V transform below.
126 //
127 // Note that to the best of our knowledge, reducing VL is generally not
128 // a significant win on real hardware unless we can also reduce LMUL which
129 // this code doesn't try to do.
130 //
131 // TODO: We can handle a bunch more instructions here, and probably
132 // recurse backwards through operands too.
133 unsigned SrcIdx = 0;
134 switch (RISCV::getRVVMCOpcode(MI.getOpcode())) {
135 default:
136 return false;
137 case RISCV::VSE8_V:
138 case RISCV::VSE16_V:
139 case RISCV::VSE32_V:
140 case RISCV::VSE64_V:
141 break;
142 case RISCV::VMV_V_V:
143 SrcIdx = 2;
144 break;
145 case RISCV::VMERGE_VVM:
146 SrcIdx = 3; // TODO: We can also handle the false operand.
147 break;
148 case RISCV::VREDSUM_VS:
149 case RISCV::VREDMAXU_VS:
150 case RISCV::VREDMAX_VS:
151 case RISCV::VREDMINU_VS:
152 case RISCV::VREDMIN_VS:
153 case RISCV::VREDAND_VS:
154 case RISCV::VREDOR_VS:
155 case RISCV::VREDXOR_VS:
156 case RISCV::VWREDSUM_VS:
157 case RISCV::VWREDSUMU_VS:
158 case RISCV::VFREDUSUM_VS:
159 case RISCV::VFREDOSUM_VS:
160 case RISCV::VFREDMAX_VS:
161 case RISCV::VFREDMIN_VS:
162 case RISCV::VFWREDUSUM_VS:
163 case RISCV::VFWREDOSUM_VS:
164 SrcIdx = 2;
165 break;
166 }
167
168 MachineOperand &VL = MI.getOperand(RISCVII::getVLOpNum(MI.getDesc()));
169 if (VL.isImm() && VL.getImm() == RISCV::VLMaxSentinel)
170 return false;
171
172 Register SrcReg = MI.getOperand(SrcIdx).getReg();
173 // Note: one *use*, not one *user*.
174 if (!MRI->hasOneUse(SrcReg))
175 return false;
176
177 MachineInstr *Src = MRI->getVRegDef(SrcReg);
178 if (!Src || Src->hasUnmodeledSideEffects() ||
179 Src->getParent() != MI.getParent() || Src->getNumDefs() != 1 ||
180 !RISCVII::hasVLOp(Src->getDesc().TSFlags) ||
181 !RISCVII::hasSEWOp(Src->getDesc().TSFlags))
182 return false;
183
184 // Src's dest needs to have the same EEW as MI's input.
185 if (!hasSameEEW(MI, *Src))
186 return false;
187
188 bool ElementsDependOnVL = RISCVII::elementsDependOnVL(
189 TII->get(RISCV::getRVVMCOpcode(Src->getOpcode())).TSFlags);
190 if (ElementsDependOnVL || Src->mayRaiseFPException())
191 return false;
192
193 MachineOperand &SrcVL = Src->getOperand(RISCVII::getVLOpNum(Src->getDesc()));
194 if (VL.isIdenticalTo(SrcVL) || !isVLKnownLE(VL, SrcVL))
195 return false;
196
197 if (!ensureDominates(VL, *Src))
198 return false;
199
200 if (VL.isImm())
201 SrcVL.ChangeToImmediate(VL.getImm());
202 else if (VL.isReg())
203 SrcVL.ChangeToRegister(VL.getReg(), false);
204
205 // TODO: For instructions with a passthru, we could clear the passthru
206 // and tail policy since we've just proven the tail is not demanded.
207 return true;
208}

210/// Check if an operand is an immediate or a materialized ADDI $x0, imm.
211std::optional<unsigned>
212RISCVVectorPeephole::getConstant(const MachineOperand &VL) const {
213 if (VL.isImm())
214 return VL.getImm();
215
216 MachineInstr *Def = MRI->getVRegDef(VL.getReg());
217 if (!Def || Def->getOpcode() != RISCV::ADDI ||
218 Def->getOperand(1).getReg() != RISCV::X0)
219 return std::nullopt;
220 return Def->getOperand(2).getImm();
221}

223/// Convert AVLs that are known to be VLMAX to the VLMAX sentinel.
224bool RISCVVectorPeephole::convertToVLMAX(MachineInstr &MI) const {
225 if (!RISCVII::hasVLOp(MI.getDesc().TSFlags) ||
226 !RISCVII::hasSEWOp(MI.getDesc().TSFlags))
227 return false;
228
229 auto LMUL = RISCVVType::decodeVLMUL(RISCVII::getLMul(MI.getDesc().TSFlags));
230 // Fixed-point value, denominator=8
231 unsigned LMULFixed = LMUL.second ? (8 / LMUL.first) : 8 * LMUL.first;
232 unsigned Log2SEW = MI.getOperand(RISCVII::getSEWOpNum(MI.getDesc())).getImm();
233 // A Log2SEW of 0 is an operation on mask registers only
234 unsigned SEW = Log2SEW ? 1 << Log2SEW : 8;
235 assert(RISCVVType::isValidSEW(SEW) && "Unexpected SEW");
236 assert(8 * LMULFixed / SEW > 0);
237
238 // If the exact VLEN is known then we know VLMAX, check if the AVL == VLMAX.
239 MachineOperand &VL = MI.getOperand(RISCVII::getVLOpNum(MI.getDesc()));
240 if (auto VLen = ST->getRealVLen(), AVL = getConstant(VL);
241 VLen && AVL && (*VLen * LMULFixed) / SEW == *AVL * 8) {
242 VL.ChangeToImmediate(RISCV::VLMaxSentinel);
243 return true;
244 }
245
246 // If an AVL is a VLENB that's possibly scaled to be equal to VLMAX, convert
247 // it to the VLMAX sentinel value.
248 if (!VL.isReg())
249 return false;
250 MachineInstr *Def = MRI->getVRegDef(VL.getReg());
251 if (!Def)
252 return false;
253
254 // Fixed-point value, denominator=8
255 uint64_t ScaleFixed = 8;
256 // Check if the VLENB was potentially scaled with slli/srli
257 if (Def->getOpcode() == RISCV::SLLI) {
258 assert(Def->getOperand(2).getImm() < 64);
259 ScaleFixed <<= Def->getOperand(2).getImm();
260 Def = MRI->getVRegDef(Def->getOperand(1).getReg());
261 } else if (Def->getOpcode() == RISCV::SRLI) {
262 assert(Def->getOperand(2).getImm() < 64);
263 ScaleFixed >>= Def->getOperand(2).getImm();
264 Def = MRI->getVRegDef(Def->getOperand(1).getReg());
265 }
266
267 if (!Def || Def->getOpcode() != RISCV::PseudoReadVLENB)
268 return false;
269
270 // AVL = (VLENB * Scale)
271 //
272 // VLMAX = (VLENB * 8 * LMUL) / SEW
273 //
274 // AVL == VLMAX
275 // -> VLENB * Scale == (VLENB * 8 * LMUL) / SEW
276 // -> Scale == (8 * LMUL) / SEW
277 if (ScaleFixed != 8 * LMULFixed / SEW)
278 return false;
279
280 VL.ChangeToImmediate(RISCV::VLMaxSentinel);
281
282 return true;
283}

285bool RISCVVectorPeephole::isAllOnesMask(const MachineInstr *MaskDef) const {
286 assert(MaskDef && MaskDef->isCopy() &&
287 MaskDef->getOperand(0).getReg() == RISCV::V0);
288 Register SrcReg = TRI->lookThruCopyLike(MaskDef->getOperand(1).getReg(), MRI);
289 if (!SrcReg.isVirtual())
290 return false;
291 MaskDef = MRI->getVRegDef(SrcReg);
292 if (!MaskDef)
293 return false;
294
295 // TODO: Check that the VMSET is the expected bitwidth? The pseudo has
296 // undefined behaviour if it's the wrong bitwidth, so we could choose to
297 // assume that it's all-ones? Same applies to its VL.
298 switch (MaskDef->getOpcode()) {
299 case RISCV::PseudoVMSET_M_B1:
300 case RISCV::PseudoVMSET_M_B2:
301 case RISCV::PseudoVMSET_M_B4:
302 case RISCV::PseudoVMSET_M_B8:
303 case RISCV::PseudoVMSET_M_B16:
304 case RISCV::PseudoVMSET_M_B32:
305 case RISCV::PseudoVMSET_M_B64:
306 return true;
307 default:
308 return false;
309 }
310}

312/// Convert unit strided unmasked loads and stores to whole-register equivalents
313/// to avoid the dependency on $vl and $vtype.
314///
315/// %x = PseudoVLE8_V_M1 %passthru, %ptr, %vlmax, policy
316/// PseudoVSE8_V_M1 %v, %ptr, %vlmax
317///
318/// ->
319///
320/// %x = VL1RE8_V %ptr
321/// VS1R_V %v, %ptr
322bool RISCVVectorPeephole::convertToWholeRegister(MachineInstr &MI) const {
323#define CASE_WHOLE_REGISTER_LMUL_SEW(lmul, sew) \
324 case RISCV::PseudoVLE##sew##_V_M##lmul: \
325 NewOpc = RISCV::VL##lmul##RE##sew##_V; \
326 break; \
327 case RISCV::PseudoVSE##sew##_V_M##lmul: \
328 NewOpc = RISCV::VS##lmul##R_V; \
329 break;
330#define CASE_WHOLE_REGISTER_LMUL(lmul) \
331 CASE_WHOLE_REGISTER_LMUL_SEW(lmul, 8) \
332 CASE_WHOLE_REGISTER_LMUL_SEW(lmul, 16) \
333 CASE_WHOLE_REGISTER_LMUL_SEW(lmul, 32) \
334 CASE_WHOLE_REGISTER_LMUL_SEW(lmul, 64)
335
336 unsigned NewOpc;
337 switch (MI.getOpcode()) {
338 CASE_WHOLE_REGISTER_LMUL(1)
339 CASE_WHOLE_REGISTER_LMUL(2)
340 CASE_WHOLE_REGISTER_LMUL(4)
341 CASE_WHOLE_REGISTER_LMUL(8)
342 default:
343 return false;
344 }
345
346 MachineOperand &VLOp = MI.getOperand(RISCVII::getVLOpNum(MI.getDesc()));
347 if (!VLOp.isImm() || VLOp.getImm() != RISCV::VLMaxSentinel)
348 return false;
349
350 // Whole register instructions aren't pseudos so they don't have
351 // policy/SEW/AVL ops, and they don't have passthrus.
352 if (RISCVII::hasVecPolicyOp(MI.getDesc().TSFlags))
353 MI.removeOperand(RISCVII::getVecPolicyOpNum(MI.getDesc()));
354 MI.removeOperand(RISCVII::getSEWOpNum(MI.getDesc()));
355 MI.removeOperand(RISCVII::getVLOpNum(MI.getDesc()));
356 if (RISCVII::isFirstDefTiedToFirstUse(MI.getDesc()))
357 MI.removeOperand(1);
358
359 MI.setDesc(TII->get(NewOpc));
360
361 return true;
362}

364static unsigned getVMV_V_VOpcodeForVMERGE_VVM(const MachineInstr &MI) {
365#define CASE_VMERGE_TO_VMV(lmul) \
366 case RISCV::PseudoVMERGE_VVM_##lmul: \
367 return RISCV::PseudoVMV_V_V_##lmul;
368 switch (MI.getOpcode()) {
369 default:
370 return 0;
371 CASE_VMERGE_TO_VMV(MF8)
372 CASE_VMERGE_TO_VMV(MF4)
373 CASE_VMERGE_TO_VMV(MF2)
374 CASE_VMERGE_TO_VMV(M1)
375 CASE_VMERGE_TO_VMV(M2)
376 CASE_VMERGE_TO_VMV(M4)
377 CASE_VMERGE_TO_VMV(M8)
378 }

381/// Convert a PseudoVMERGE_VVM with an all ones mask to a PseudoVMV_V_V.
382///
383/// %x = PseudoVMERGE_VVM %passthru, %false, %true, %allones, sew, vl
384/// ->
385/// %x = PseudoVMV_V_V %passthru, %true, vl, sew, tu_mu
386bool RISCVVectorPeephole::convertAllOnesVMergeToVMv(MachineInstr &MI) const {
387 unsigned NewOpc = getVMV_V_VOpcodeForVMERGE_VVM(MI);
388 if (!NewOpc)
389 return false;
390 assert(MI.getOperand(4).isReg() && MI.getOperand(4).getReg() == RISCV::V0);
391 if (!isAllOnesMask(V0Defs.lookup(&MI)))
392 return false;
393
394 MI.setDesc(TII->get(NewOpc));
395 MI.removeOperand(2); // False operand
396 MI.removeOperand(3); // Mask operand
397 MI.addOperand(
398 MachineOperand::CreateImm(RISCVII::TAIL_UNDISTURBED_MASK_UNDISTURBED));
399
400 // vmv.v.v doesn't have a mask operand, so we may be able to inflate the
401 // register class for the destination and passthru operands e.g. VRNoV0 -> VR
402 MRI->recomputeRegClass(MI.getOperand(0).getReg());
403 if (MI.getOperand(1).getReg() != RISCV::NoRegister)
404 MRI->recomputeRegClass(MI.getOperand(1).getReg());
405 return true;
406}

408/// If a PseudoVMERGE_VVM's true operand is a masked pseudo and both have the
409/// same mask, and the masked pseudo's passthru is the same as the false
410/// operand, we can convert the PseudoVMERGE_VVM to a PseudoVMV_V_V.
411///
412/// %true = PseudoVADD_VV_M1_MASK %false, %x, %y, %mask, vl1, sew, policy
413/// %x = PseudoVMERGE_VVM %passthru, %false, %true, %mask, vl2, sew
414/// ->
415/// %true = PseudoVADD_VV_M1_MASK %false, %x, %y, %mask, vl1, sew, policy
416/// %x = PseudoVMV_V_V %passthru, %true, vl2, sew, tu_mu
417bool RISCVVectorPeephole::convertSameMaskVMergeToVMv(MachineInstr &MI) {
418 unsigned NewOpc = getVMV_V_VOpcodeForVMERGE_VVM(MI);
419 if (!NewOpc)
420 return false;
421 MachineInstr *True = MRI->getVRegDef(MI.getOperand(3).getReg());
422 if (!True || !RISCV::getMaskedPseudoInfo(True->getOpcode()) ||
423 !hasSameEEW(MI, *True))
424 return false;
425
426 const MachineInstr *TrueV0Def = V0Defs.lookup(True);
427 const MachineInstr *MIV0Def = V0Defs.lookup(&MI);
428 assert(TrueV0Def && TrueV0Def->isCopy() && MIV0Def && MIV0Def->isCopy());
429 if (TrueV0Def->getOperand(1).getReg() != MIV0Def->getOperand(1).getReg())
430 return false;
431
432 // True's passthru needs to be equivalent to False
433 Register TruePassthruReg = True->getOperand(1).getReg();
434 Register FalseReg = MI.getOperand(2).getReg();
435 if (TruePassthruReg != FalseReg) {
436 // If True's passthru is undef see if we can change it to False
437 if (TruePassthruReg != RISCV::NoRegister ||
438 !MRI->hasOneUse(MI.getOperand(3).getReg()) ||
439 !ensureDominates(MI.getOperand(2), *True))
440 return false;
441 True->getOperand(1).setReg(MI.getOperand(2).getReg());
442 // If True is masked then its passthru needs to be in VRNoV0.
443 MRI->constrainRegClass(True->getOperand(1).getReg(),
444 TII->getRegClass(True->getDesc(), 1, TRI,
445 *True->getParent()->getParent()));
446 }
447
448 MI.setDesc(TII->get(NewOpc));
449 MI.removeOperand(2); // False operand
450 MI.removeOperand(3); // Mask operand
451 MI.addOperand(
452 MachineOperand::CreateImm(RISCVII::TAIL_UNDISTURBED_MASK_UNDISTURBED));
453
454 // vmv.v.v doesn't have a mask operand, so we may be able to inflate the
455 // register class for the destination and passthru operands e.g. VRNoV0 -> VR
456 MRI->recomputeRegClass(MI.getOperand(0).getReg());
457 if (MI.getOperand(1).getReg() != RISCV::NoRegister)
458 MRI->recomputeRegClass(MI.getOperand(1).getReg());
459 return true;
460}

462bool RISCVVectorPeephole::convertToUnmasked(MachineInstr &MI) const {
463 const RISCV::RISCVMaskedPseudoInfo *I =
464 RISCV::getMaskedPseudoInfo(MI.getOpcode());
465 if (!I)
466 return false;
467
468 if (!isAllOnesMask(V0Defs.lookup(&MI)))
469 return false;
470
471 // There are two classes of pseudos in the table - compares and
472 // everything else. See the comment on RISCVMaskedPseudo for details.
473 const unsigned Opc = I->UnmaskedPseudo;
474 const MCInstrDesc &MCID = TII->get(Opc);
475 [[maybe_unused]] const bool HasPolicyOp =
476 RISCVII::hasVecPolicyOp(MCID.TSFlags);
477 const bool HasPassthru = RISCVII::isFirstDefTiedToFirstUse(MCID);
478#ifndef NDEBUG
479 const MCInstrDesc &MaskedMCID = TII->get(MI.getOpcode());
480 assert(RISCVII::hasVecPolicyOp(MaskedMCID.TSFlags) ==
481 RISCVII::hasVecPolicyOp(MCID.TSFlags) &&
482 "Masked and unmasked pseudos are inconsistent");
483 assert(HasPolicyOp == HasPassthru && "Unexpected pseudo structure");
484#endif
485 (void)HasPolicyOp;
486
487 MI.setDesc(MCID);
488
489 // TODO: Increment all MaskOpIdxs in tablegen by num of explicit defs?
490 unsigned MaskOpIdx = I->MaskOpIdx + MI.getNumExplicitDefs();
491 MI.removeOperand(MaskOpIdx);
492
493 // The unmasked pseudo will no longer be constrained to the vrnov0 reg class,
494 // so try and relax it to vr.
495 MRI->recomputeRegClass(MI.getOperand(0).getReg());
496 unsigned PassthruOpIdx = MI.getNumExplicitDefs();
497 if (HasPassthru) {
498 if (MI.getOperand(PassthruOpIdx).getReg() != RISCV::NoRegister)
499 MRI->recomputeRegClass(MI.getOperand(PassthruOpIdx).getReg());
500 } else
501 MI.removeOperand(PassthruOpIdx);
502
503 return true;
504}

506/// Check if it's safe to move From down to To, checking that no physical
507/// registers are clobbered.
508static bool isSafeToMove(const MachineInstr &From, const MachineInstr &To) {
509 assert(From.getParent() == To.getParent() && !From.hasImplicitDef());
510 SmallVector<Register> PhysUses;
511 for (const MachineOperand &MO : From.all_uses())
512 if (MO.getReg().isPhysical())
513 PhysUses.push_back(MO.getReg());
514 bool SawStore = false;
515 for (auto II = From.getIterator(); II != To.getIterator(); II++) {
516 for (Register PhysReg : PhysUses)
517 if (II->definesRegister(PhysReg, nullptr))
518 return false;
519 if (II->mayStore()) {
520 SawStore = true;
521 break;
522 }
523 }
524 return From.isSafeToMove(SawStore);
525}

527/// Given A and B are in the same MBB, returns true if A comes before B.
528static bool dominates(MachineBasicBlock::const_iterator A,
529 MachineBasicBlock::const_iterator B) {
530 assert(A->getParent() == B->getParent());
531 const MachineBasicBlock *MBB = A->getParent();
532 auto MBBEnd = MBB->end();
533 if (B == MBBEnd)
534 return true;
535
536 MachineBasicBlock::const_iterator I = MBB->begin();
537 for (; &*I != A && &*I != B; ++I)
538 ;
539
540 return &*I == A;
541}

543/// If the register in \p MO doesn't dominate \p Src, try to move \p Src so it
544/// does. Returns false if doesn't dominate and we can't move. \p MO must be in
545/// the same basic block as \Src.
546bool RISCVVectorPeephole::ensureDominates(const MachineOperand &MO,
547 MachineInstr &Src) const {
548 assert(MO.getParent()->getParent() == Src.getParent());
549 if (!MO.isReg() || MO.getReg() == RISCV::NoRegister)
550 return true;
551
552 MachineInstr *Def = MRI->getVRegDef(MO.getReg());
553 if (Def->getParent() == Src.getParent() && !dominates(Def, Src)) {
554 if (!isSafeToMove(Src, *Def->getNextNode()))
555 return false;
556 Src.moveBefore(Def->getNextNode());
557 }
558
559 return true;
560}

562/// If a PseudoVMV_V_V's passthru is undef then we can replace it with its input
563bool RISCVVectorPeephole::foldUndefPassthruVMV_V_V(MachineInstr &MI) {
564 if (RISCV::getRVVMCOpcode(MI.getOpcode()) != RISCV::VMV_V_V)
565 return false;
566 if (MI.getOperand(1).getReg() != RISCV::NoRegister)
567 return false;
568
569 // If the input was a pseudo with a policy operand, we can give it a tail
570 // agnostic policy if MI's undef tail subsumes the input's.
571 MachineInstr *Src = MRI->getVRegDef(MI.getOperand(2).getReg());
572 if (Src && !Src->hasUnmodeledSideEffects() &&
573 MRI->hasOneUse(MI.getOperand(2).getReg()) &&
574 RISCVII::hasVLOp(Src->getDesc().TSFlags) &&
575 RISCVII::hasVecPolicyOp(Src->getDesc().TSFlags) && hasSameEEW(MI, *Src)) {
576 const MachineOperand &MIVL = MI.getOperand(3);
577 const MachineOperand &SrcVL =
578 Src->getOperand(RISCVII::getVLOpNum(Src->getDesc()));
579
580 MachineOperand &SrcPolicy =
581 Src->getOperand(RISCVII::getVecPolicyOpNum(Src->getDesc()));
582
583 if (isVLKnownLE(MIVL, SrcVL))
584 SrcPolicy.setImm(SrcPolicy.getImm() | RISCVII::TAIL_AGNOSTIC);
585 }
586
587 MRI->replaceRegWith(MI.getOperand(0).getReg(), MI.getOperand(2).getReg());
588 MI.eraseFromParent();
589 V0Defs.erase(&MI);
590 return true;
591}

593/// If a PseudoVMV_V_V is the only user of its input, fold its passthru and VL
594/// into it.
595///
596/// %x = PseudoVADD_V_V_M1 %passthru, %a, %b, %vl1, sew, policy
597/// %y = PseudoVMV_V_V_M1 %passthru, %x, %vl2, sew, policy
598/// (where %vl1 <= %vl2, see related tryToReduceVL)
599///
600/// ->
601///
602/// %y = PseudoVADD_V_V_M1 %passthru, %a, %b, vl1, sew, policy
603bool RISCVVectorPeephole::foldVMV_V_V(MachineInstr &MI) {
604 if (RISCV::getRVVMCOpcode(MI.getOpcode()) != RISCV::VMV_V_V)
605 return false;
606
607 MachineOperand &Passthru = MI.getOperand(1);
608
609 if (!MRI->hasOneUse(MI.getOperand(2).getReg()))
610 return false;
611
612 MachineInstr *Src = MRI->getVRegDef(MI.getOperand(2).getReg());
613 if (!Src || Src->hasUnmodeledSideEffects() ||
614 Src->getParent() != MI.getParent() || Src->getNumDefs() != 1 ||
615 !RISCVII::isFirstDefTiedToFirstUse(Src->getDesc()) ||
616 !RISCVII::hasVLOp(Src->getDesc().TSFlags) ||
617 !RISCVII::hasVecPolicyOp(Src->getDesc().TSFlags))
618 return false;
619
620 // Src's dest needs to have the same EEW as MI's input.
621 if (!hasSameEEW(MI, *Src))
622 return false;
623
624 // Src needs to have the same passthru as VMV_V_V
625 MachineOperand &SrcPassthru = Src->getOperand(1);
626 if (SrcPassthru.getReg() != RISCV::NoRegister &&
627 SrcPassthru.getReg() != Passthru.getReg())
628 return false;
629
630 // Src VL will have already been reduced if legal (see tryToReduceVL),
631 // so we don't need to handle a smaller source VL here. However, the
632 // user's VL may be larger
633 MachineOperand &SrcVL = Src->getOperand(RISCVII::getVLOpNum(Src->getDesc()));
634 if (!isVLKnownLE(SrcVL, MI.getOperand(3)))
635 return false;
636
637 // If the new passthru doesn't dominate Src, try to move Src so it does.
638 if (!ensureDominates(Passthru, *Src))
639 return false;
640
641 if (SrcPassthru.getReg() != Passthru.getReg()) {
642 SrcPassthru.setReg(Passthru.getReg());
643 // If Src is masked then its passthru needs to be in VRNoV0.
644 if (Passthru.getReg() != RISCV::NoRegister)
645 MRI->constrainRegClass(Passthru.getReg(),
646 TII->getRegClass(Src->getDesc(), 1, TRI,
647 *Src->getParent()->getParent()));
648 }
649
650 // If MI was tail agnostic and the VL didn't increase, preserve it.
651 int64_t Policy = RISCVII::TAIL_UNDISTURBED_MASK_UNDISTURBED;
652 if ((MI.getOperand(5).getImm() & RISCVII::TAIL_AGNOSTIC) &&
653 isVLKnownLE(MI.getOperand(3), SrcVL))
654 Policy |= RISCVII::TAIL_AGNOSTIC;
655 Src->getOperand(RISCVII::getVecPolicyOpNum(Src->getDesc())).setImm(Policy);
656
657 MRI->replaceRegWith(MI.getOperand(0).getReg(), Src->getOperand(0).getReg());
658 MI.eraseFromParent();
659 V0Defs.erase(&MI);
660
661 return true;
662}

664bool RISCVVectorPeephole::runOnMachineFunction(MachineFunction &MF) {
665 if (skipFunction(MF.getFunction()))
666 return false;
667
668 // Skip if the vector extension is not enabled.
669 ST = &MF.getSubtarget<RISCVSubtarget>();
670 if (!ST->hasVInstructions())
671 return false;
672
673 TII = ST->getInstrInfo();
674 MRI = &MF.getRegInfo();
675 TRI = MRI->getTargetRegisterInfo();
676
677 bool Changed = false;
678
679 // Masked pseudos coming out of isel will have their mask operand in the form:
680 //
681 // $v0:vr = COPY %mask:vr
682 // %x:vr = Pseudo_MASK %a:vr, %b:br, $v0:vr
683 //
684 // Because $v0 isn't in SSA, keep track of its definition at each use so we
685 // can check mask operands.
686 for (const MachineBasicBlock &MBB : MF) {
687 const MachineInstr *CurrentV0Def = nullptr;
688 for (const MachineInstr &MI : MBB) {
689 if (MI.readsRegister(RISCV::V0, TRI))
690 V0Defs[&MI] = CurrentV0Def;
691
692 if (MI.definesRegister(RISCV::V0, TRI))
693 CurrentV0Def = &MI;
694 }
695 }
696
697 for (MachineBasicBlock &MBB : MF) {
698 for (MachineInstr &MI : make_early_inc_range(MBB)) {
699 Changed |= convertToVLMAX(MI);
700 Changed |= tryToReduceVL(MI);
701 Changed |= convertToUnmasked(MI);
702 Changed |= convertToWholeRegister(MI);
703 Changed |= convertAllOnesVMergeToVMv(MI);
704 Changed |= convertSameMaskVMergeToVMv(MI);
705 if (foldUndefPassthruVMV_V_V(MI)) {
706 Changed |= true;
707 continue; // MI is erased
708 }
709 Changed |= foldVMV_V_V(MI);
710 }
711 }
712
713 return Changed;
714}
