465static inline bool addrspacesMayAlias(unsigned AS1, unsigned AS2) {
466 static_assert(AMDGPUAS::MAX_AMDGPU_ADDRESS <= 9, "Addr space out of range");
467
468 if (AS1 > AMDGPUAS::MAX_AMDGPU_ADDRESS || AS2 > AMDGPUAS::MAX_AMDGPU_ADDRESS)
469 return true;
470
471 // This array is indexed by address space value enum elements 0 ... to 9
472 // clang-format off
473 static const bool ASAliasRules[10][10] = {
474 /* Flat Global Region Group Constant Private Const32 BufFatPtr BufRsrc BufStrdPtr */
475 /* Flat */ {true, true, false, true, true, true, true, true, true, true},
476 /* Global */ {true, true, false, false, true, false, true, true, true, true},
477 /* Region */ {false, false, true, false, false, false, false, false, false, false},
478 /* Group */ {true, false, false, true, false, false, false, false, false, false},
479 /* Constant */ {true, true, false, false, false, false, true, true, true, true},
480 /* Private */ {true, false, false, false, false, true, false, false, false, false},
481 /* Constant 32-bit */ {true, true, false, false, true, false, false, true, true, true},
482 /* Buffer Fat Ptr */ {true, true, false, false, true, false, true, true, true, true},
483 /* Buffer Resource */ {true, true, false, false, true, false, true, true, true, true},
484 /* Buffer Strided Ptr */ {true, true, false, false, true, false, true, true, true, true},
485 };
486 // clang-format on
487
488 return ASAliasRules[AS1][AS2];
489}
