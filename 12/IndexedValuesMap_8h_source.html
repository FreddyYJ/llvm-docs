21template <typename T> class IndexedValuesMap {

23 uint64_t getValueIndex(T Value) {
24 auto [It, Inserted] = ValueToIndexMap.try_emplace(Value, Values.size());

26 Values.push_back(Value);
27 return It->second;
28 }
29
