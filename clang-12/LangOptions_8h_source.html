136 /// Attempt to be ABI-compatible with code generated by Clang 3.8.x
137 /// (SVN r257626). This causes <1 x long long> to be passed in an
138 /// integer register instead of an SSE register on x64_64.
139 Ver3_8,
140
141 /// Attempt to be ABI-compatible with code generated by Clang 4.0.x
142 /// (SVN r291814). This causes move operations to be ignored when
143 /// determining whether a class type can be passed or returned directly.
144 Ver4,
145
146 /// Attempt to be ABI-compatible with code generated by Clang 6.0.x
147 /// (SVN r321711). This causes determination of whether a type is
148 /// standard-layout to ignore collisions between empty base classes
149 /// and between base classes and member subobjects, which affects
150 /// whether we reuse base class tail padding in some ABIs.
151 Ver6,
152
153 /// Attempt to be ABI-compatible with code generated by Clang 7.0.x
154 /// (SVN r338536). This causes alignof (C++) and _Alignof (C11) to be
155 /// compatible with __alignof (i.e., return the preferred alignment)
156 /// rather than returning the required alignment.
157 Ver7,
158
159 /// Attempt to be ABI-compatible with code generated by Clang 9.0.x
160 /// (SVN r351319). This causes vectors of __int128 to be passed in memory

162 /// NetBSD.
163 Ver9,
164

422/// Floating point control options
423class FPOptionsOverride;
424class FPOptions {
425public:
426 // We start by defining the layout.
427 using storage_type = uint16_t;
428
429 using RoundingMode = llvm::RoundingMode;
430
431 static constexpr unsigned StorageBitSize = 8 * sizeof(storage_type);
432
433 // Define a fake option named "First" so that we have a PREVIOUS even for the
434 // real first option.
435 static constexpr storage_type FirstShift = 0, FirstWidth = 0;
436#define OPTION(NAME, TYPE, WIDTH, PREVIOUS) \
437 static constexpr storage_type NAME##Shift = \
438 PREVIOUS##Shift + PREVIOUS##Width; \
439 static constexpr storage_type NAME##Width = WIDTH; \
440 static constexpr storage_type NAME##Mask = ((1 << NAME##Width) - 1) \
441 << NAME##Shift;
442#include "clang/Basic/FPOptions.def"
443
444 static constexpr storage_type TotalWidth = 0
445#define OPTION(NAME, TYPE, WIDTH, PREVIOUS) +WIDTH
446#include "clang/Basic/FPOptions.def"
447 ;
448 static_assert(TotalWidth <= StorageBitSize, "Too short type for FPOptions");
449
450private:
451 storage_type Value;
452
453public:
454 FPOptions() : Value(0) {
455 setFPContractMode(LangOptions::FPM_Off);
456 setRoundingMode(static_cast<RoundingMode>(LangOptions::FPR_ToNearest));
457 setFPExceptionMode(LangOptions::FPE_Ignore);
458 }
459 explicit FPOptions(const LangOptions &LO) {
460 Value = 0;
461 // The language fp contract option FPM_FastHonorPragmas has the same effect
462 // as FPM_Fast in frontend. For simplicity, use FPM_Fast uniformly in
463 // frontend.
464 auto LangOptContractMode = LO.getDefaultFPContractMode();
465 if (LangOptContractMode == LangOptions::FPM_FastHonorPragmas)
466 LangOptContractMode = LangOptions::FPM_Fast;
467 setFPContractMode(LangOptContractMode);
468 setRoundingMode(LO.getFPRoundingMode());
469 setFPExceptionMode(LO.getFPExceptionMode());
470 setAllowFPReassociate(LO.AllowFPReassoc);
471 setNoHonorNaNs(LO.NoHonorNaNs);
472 setNoHonorInfs(LO.NoHonorInfs);
473 setNoSignedZero(LO.NoSignedZero);
474 setAllowReciprocal(LO.AllowRecip);
475 setAllowApproxFunc(LO.ApproxFunc);
476 if (getFPContractMode() == LangOptions::FPM_On &&
477 getRoundingMode() == llvm::RoundingMode::Dynamic &&
478 getFPExceptionMode() == LangOptions::FPE_Strict)
479 // If the FP settings are set to the "strict" model, then
480 // FENV access is set to true. (ffp-model=strict)
481 setAllowFEnvAccess(true);
482 else
483 setAllowFEnvAccess(LangOptions::FPM_Off);
484 }
485
486 bool allowFPContractWithinStatement() const {
487 return getFPContractMode() == LangOptions::FPM_On;
488 }
489 void setAllowFPContractWithinStatement() {
490 setFPContractMode(LangOptions::FPM_On);
491 }
492
493 bool allowFPContractAcrossStatement() const {
494 return getFPContractMode() == LangOptions::FPM_Fast;
495 }
496 void setAllowFPContractAcrossStatement() {
497 setFPContractMode(LangOptions::FPM_Fast);
498 }
499
500 bool isFPConstrained() const {
501 return getRoundingMode() != llvm::RoundingMode::NearestTiesToEven ||
502 getFPExceptionMode() != LangOptions::FPE_Ignore ||
503 getAllowFEnvAccess();
504 }
505
506 bool operator==(FPOptions other) const { return Value == other.Value; }
507
508 /// Return the default value of FPOptions that's used when trailing
509 /// storage isn't required.
510 static FPOptions defaultWithoutTrailingStorage(const LangOptions &LO);
511
512 storage_type getAsOpaqueInt() const { return Value; }
513 static FPOptions getFromOpaqueInt(storage_type Value) {
514 FPOptions Opts;
515 Opts.Value = Value;
516 return Opts;
517 }
518
519 // We can define most of the accessors automatically:
520#define OPTION(NAME, TYPE, WIDTH, PREVIOUS) \
521 TYPE get##NAME() const { \
522 return static_cast<TYPE>((Value & NAME##Mask) >> NAME##Shift); \
523 } \
524 void set##NAME(TYPE value) { \
525 Value = (Value & ~NAME##Mask) | (storage_type(value) << NAME##Shift); \
526 }
527#include "clang/Basic/FPOptions.def"
528 LLVM_DUMP_METHOD void dump();

531/// Represents difference between two FPOptions values.
532///
533/// The effect of language constructs changing the set of floating point options
534/// is usually a change of some FP properties while leaving others intact. This
535/// class describes such changes by keeping information about what FP options
536/// are overridden.
537///
538/// The integral set of FP options, described by the class FPOptions, may be
539/// represented as a default FP option set, defined by language standard and
540/// command line options, with the overrides introduced by pragmas.
541///
542/// The is implemented as a value of the new FPOptions plus a mask showing which
543/// fields are actually set in it.
544class FPOptionsOverride {
545 FPOptions Options = FPOptions::getFromOpaqueInt(0);
546 FPOptions::storage_type OverrideMask = 0;
547
548public:
549 using RoundingMode = llvm::RoundingMode;
550
551 /// The type suitable for storing values of FPOptionsOverride. Must be twice
552 /// as wide as bit size of FPOption.
553 using storage_type = uint32_t;
554 static_assert(sizeof(storage_type) >= 2 * sizeof(FPOptions::storage_type),
555 "Too short type for FPOptionsOverride");
556
557 /// Bit mask selecting bits of OverrideMask in serialized representation of
558 /// FPOptionsOverride.
559 static constexpr storage_type OverrideMaskBits =
560 (static_cast<storage_type>(1) << FPOptions::StorageBitSize) - 1;
561
562 FPOptionsOverride() {}
563 FPOptionsOverride(const LangOptions &LO)
564 : Options(LO), OverrideMask(OverrideMaskBits) {}
565 FPOptionsOverride(FPOptions FPO)
566 : Options(FPO), OverrideMask(OverrideMaskBits) {}
567
568 bool requiresTrailingStorage() const { return OverrideMask != 0; }
569
570 void setAllowFPContractWithinStatement() {
571 setFPContractModeOverride(LangOptions::FPM_On);
572 }
573
574 void setAllowFPContractAcrossStatement() {
575 setFPContractModeOverride(LangOptions::FPM_Fast);
576 }
577
578 void setDisallowFPContract() {
579 setFPContractModeOverride(LangOptions::FPM_Off);
580 }
581
582 void setFPPreciseEnabled(bool Value) {
583 setAllowFPReassociateOverride(!Value);
584 setNoHonorNaNsOverride(!Value);
585 setNoHonorInfsOverride(!Value);
586 setNoSignedZeroOverride(!Value);
587 setAllowReciprocalOverride(!Value);
588 setAllowApproxFuncOverride(!Value);
589 if (Value)
590 /* Precise mode implies fp_contract=on and disables ffast-math */
591 setAllowFPContractWithinStatement();
592 else
593 /* Precise mode disabled sets fp_contract=fast and enables ffast-math */
594 setAllowFPContractAcrossStatement();
595 }
596
597 storage_type getAsOpaqueInt() const {
598 return (static_cast<storage_type>(Options.getAsOpaqueInt())
599 << FPOptions::StorageBitSize) |
600 OverrideMask;
601 }
602 static FPOptionsOverride getFromOpaqueInt(storage_type I) {
603 FPOptionsOverride Opts;
604 Opts.OverrideMask = I & OverrideMaskBits;
605 Opts.Options = FPOptions::getFromOpaqueInt(I >> FPOptions::StorageBitSize);
606 return Opts;
607 }
608
609 FPOptions applyOverrides(FPOptions Base) {
610 FPOptions Result =
611 FPOptions::getFromOpaqueInt((Base.getAsOpaqueInt() & ~OverrideMask) |
612 (Options.getAsOpaqueInt() & OverrideMask));
613 return Result;
614 }
615
616 FPOptions applyOverrides(const LangOptions &LO) {
617 return applyOverrides(FPOptions(LO));
618 }
619
620 bool operator==(FPOptionsOverride other) const {
621 return Options == other.Options && OverrideMask == other.OverrideMask;
622 }
623 bool operator!=(FPOptionsOverride other) const { return !(*this == other); }
624
625#define OPTION(NAME, TYPE, WIDTH, PREVIOUS) \
626 bool has##NAME##Override() const { \
627 return OverrideMask & FPOptions::NAME##Mask; \
628 } \
629 TYPE get##NAME##Override() const { \
630 assert(has##NAME##Override()); \
631 return Options.get##NAME(); \
632 } \
633 void clear##NAME##Override() { \
634 /* Clear the actual value so that we don't have spurious differences when \
635 * testing equality. */ \
636 Options.set##NAME(TYPE(0)); \
637 OverrideMask &= ~FPOptions::NAME##Mask; \
638 } \
639 void set##NAME##Override(TYPE value) { \
640 Options.set##NAME(value); \
641 OverrideMask |= FPOptions::NAME##Mask; \
642 }
643#include "clang/Basic/FPOptions.def"
644 LLVM_DUMP_METHOD void dump();
645};
