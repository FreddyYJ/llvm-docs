72// PTX ABI requires all scalar argument/return values to have
73// bit-size as a power of two of at least 32 bits.
74inline unsigned promoteScalarArgumentSize(unsigned size) {
75 if (size <= 32)
76 return 32;
77 if (size <= 64)
78 return 64;
79 return size;
80}

88inline std::string OrderingToString(Ordering Order) {
89 switch (Order) {
90 case Ordering::NotAtomic:
91 return "NotAtomic";
92 case Ordering::Relaxed:
93 return "Relaxed";
94 case Ordering::Acquire:
95 return "Acquire";
96 case Ordering::Release:
97 return "Release";
98 case Ordering::AcquireRelease:
99 return "AcquireRelease";
100 case Ordering::SequentiallyConsistent:
101 return "SequentiallyConsistent";
102 case Ordering::Volatile:
103 return "Volatile";
104 case Ordering::RelaxedMMIO:
105 return "RelaxedMMIO";
106 }
107 report_fatal_error(formatv("Unknown NVPTX::Ordering \"{}\".",
108 static_cast<OrderingUnderlyingType>(Order)));
109}
110
111inline raw_ostream &operator<<(raw_ostream &O, Ordering Order) {

113 return O;
114}
