17void ConstantFPRange::makeEmpty() {
18 auto &Sem = Lower.getSemantics();
19 Lower = APFloat::getInf(Sem, /*Negative=*/false);
20 Upper = APFloat::getInf(Sem, /*Negative=*/true);
21 MayBeQNaN = false;
22 MayBeSNaN = false;
23}

25void ConstantFPRange::makeFull() {

37ConstantFPRange::ConstantFPRange(const fltSemantics &Sem, bool IsFullSet)
38 : Lower(Sem, APFloat::uninitialized), Upper(Sem, APFloat::uninitialized) {
39 Lower = APFloat::getInf(Sem, /*Negative=*/IsFullSet);
40 Upper = APFloat::getInf(Sem, /*Negative=*/!IsFullSet);
41 MayBeQNaN = IsFullSet;
42 MayBeSNaN = IsFullSet;

45ConstantFPRange::ConstantFPRange(const APFloat &Value)
46 : Lower(Value.getSemantics(), APFloat::uninitialized),
47 Upper(Value.getSemantics(), APFloat::uninitialized) {
48 if (Value.isNaN()) {
49 makeEmpty();
50 bool IsSNaN = Value.isSignaling();
51 MayBeQNaN = !IsSNaN;
52 MayBeSNaN = IsSNaN;
53 } else {
54 Lower = Upper = Value;
55 MayBeQNaN = MayBeSNaN = false;
56 }
57}

59// We treat that -0 is less than 0 here.
60static APFloat::cmpResult strictCompare(const APFloat &LHS,
61 const APFloat &RHS) {
62 assert(!LHS.isNaN() && !RHS.isNaN() && "Unordered compare");
63 if (LHS.isZero() && RHS.isZero()) {
64 if (LHS.isNegative() == RHS.isNegative())
65 return APFloat::cmpEqual;
66 return LHS.isNegative() ? APFloat::cmpLessThan : APFloat::cmpGreaterThan;
67 }
68 return LHS.compare(RHS);
69}

71static bool isNonCanonicalEmptySet(const APFloat &Lower, const APFloat &Upper) {
72 return strictCompare(Lower, Upper) == APFloat::cmpGreaterThan &&
73 !(Lower.isInfinity() && Upper.isInfinity());
74}

76static void canonicalizeRange(APFloat &Lower, APFloat &Upper) {
77 if (isNonCanonicalEmptySet(Lower, Upper)) {
78 Lower = APFloat::getInf(Lower.getSemantics(), /*Negative=*/false);
79 Upper = APFloat::getInf(Upper.getSemantics(), /*Negative=*/true);
80 }
81}

83ConstantFPRange::ConstantFPRange(APFloat LowerVal, APFloat UpperVal,
84 bool MayBeQNaN, bool MayBeSNaN)
85 : Lower(std::move(LowerVal)), Upper(std::move(UpperVal)) {
86 assert(&Lower.getSemantics() == &Upper.getSemantics() &&
87 "Should only use the same semantics");
88 assert(!isNonCanonicalEmptySet(Lower, Upper) && "Non-canonical form");
89 this->MayBeQNaN = MayBeQNaN;
90 this->MayBeSNaN = MayBeSNaN;

140bool ConstantFPRange::contains(const APFloat &Val) const {
141 assert(&getSemantics() == &Val.getSemantics() &&
142 "Should only use the same semantics");
143
144 if (Val.isNaN())
145 return Val.isSignaling() ? MayBeSNaN : MayBeQNaN;
146 return strictCompare(Lower, Val) != APFloat::cmpGreaterThan &&
147 strictCompare(Val, Upper) != APFloat::cmpGreaterThan;
148}

150bool ConstantFPRange::contains(const ConstantFPRange &CR) const {
151 assert(&getSemantics() == &CR.getSemantics() &&
152 "Should only use the same semantics");
153
154 if (CR.MayBeQNaN && !MayBeQNaN)
155 return false;
156
157 if (CR.MayBeSNaN && !MayBeSNaN)
158 return false;
159
160 return strictCompare(Lower, CR.Lower) != APFloat::cmpGreaterThan &&
161 strictCompare(CR.Upper, Upper) != APFloat::cmpGreaterThan;
162}

176bool ConstantFPRange::operator==(const ConstantFPRange &CR) const {
177 if (MayBeSNaN != CR.MayBeSNaN || MayBeQNaN != CR.MayBeQNaN)
178 return false;
179 return Lower.bitwiseIsEqual(CR.Lower) && Upper.bitwiseIsEqual(CR.Upper);

182FPClassTest ConstantFPRange::classify() const {
183 uint32_t Mask = fcNone;
184 if (MayBeSNaN)
185 Mask |= fcSNan;
186 if (MayBeQNaN)
187 Mask |= fcQNan;
188 if (!isNaNOnly()) {
189 FPClassTest LowerMask = Lower.classify();
190 FPClassTest UpperMask = Upper.classify();
191 assert(LowerMask <= UpperMask && "Range is nan-only.");
192 for (uint32_t I = LowerMask; I <= UpperMask; I <<= 1)
193 Mask |= I;
194 }
195 return static_cast<FPClassTest>(Mask);
196}

198void ConstantFPRange::print(raw_ostream &OS) const {
199 if (isFullSet())
200 OS << "full-set";
201 else if (isEmptySet())
202 OS << "empty-set";
203 else {
204 bool NaNOnly = isNaNOnly();
205 if (!NaNOnly)
206 OS << '[' << Lower << ", " << Upper << ']';
207
208 if (MayBeSNaN || MayBeQNaN) {
209 if (!NaNOnly)
210 OS << " with ";
211 if (MayBeSNaN && MayBeQNaN)
212 OS << "NaN";
213 else if (MayBeSNaN)
214 OS << "SNaN";
215 else if (MayBeQNaN)
216 OS << "QNaN";
217 }
218 }

225ConstantFPRange
226ConstantFPRange::intersectWith(const ConstantFPRange &CR) const {
227 assert(&getSemantics() == &CR.getSemantics() &&
228 "Should only use the same semantics");
229 APFloat NewLower = maxnum(Lower, CR.Lower);
230 APFloat NewUpper = minnum(Upper, CR.Upper);
231 canonicalizeRange(NewLower, NewUpper);
232 return ConstantFPRange(std::move(NewLower), std::move(NewUpper),
233 MayBeQNaN & CR.MayBeQNaN, MayBeSNaN & CR.MayBeSNaN);
234}

236ConstantFPRange ConstantFPRange::unionWith(const ConstantFPRange &CR) const {
237 assert(&getSemantics() == &CR.getSemantics() &&
238 "Should only use the same semantics");
239 return ConstantFPRange(minnum(Lower, CR.Lower), maxnum(Upper, CR.Upper),
240 MayBeQNaN | CR.MayBeQNaN, MayBeSNaN | CR.MayBeSNaN);
