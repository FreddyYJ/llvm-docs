53SmallVector<std::unique_ptr<Region>> Region::createRegionsFromMD(Function &F) {
54 SmallVector<std::unique_ptr<Region>> Regions;
55 DenseMap<MDNode *, Region *> MDNToRegion;
56 auto &Ctx = F.getContext();
57 for (BasicBlock &BB : F) {
58 for (Instruction &Inst : BB) {
59 if (auto *MDN = cast<llvm::Instruction>(Inst.Val)->getMetadata(MDKind)) {
60 Region *R = nullptr;
61 auto It = MDNToRegion.find(MDN);
62 if (It == MDNToRegion.end()) {
63 Regions.push_back(std::make_unique<Region>(Ctx));
64 R = Regions.back().get();
65 MDNToRegion[MDN] = R;
66 } else {
67 R = It->second;
68 }
69 R->add(&Inst);
70 }
71 }
72 }
73 return Regions;
74}
