9// whether given two BasicBlocks A and B there is a path from A to B that

63bool SuspendCrossingInfo::hasPathCrossingSuspendPoint(BasicBlock *From,
64 BasicBlock *To) const {
65 size_t const FromIndex = Mapping.blockToIndex(From);
66 size_t const ToIndex = Mapping.blockToIndex(To);
67 bool const Result = Block[ToIndex].Kills[FromIndex];
68 LLVM_DEBUG(if (Result) dbgs() << From->getName() << " => " << To->getName()
69 << " crosses suspend point\n");
70 return Result;
71}

73bool SuspendCrossingInfo::hasPathOrLoopCrossingSuspendPoint(
74 BasicBlock *From, BasicBlock *To) const {
75 size_t const FromIndex = Mapping.blockToIndex(From);
76 size_t const ToIndex = Mapping.blockToIndex(To);
77 bool Result = Block[ToIndex].Kills[FromIndex] ||
78 (From == To && Block[ToIndex].KillLoop);
79 LLVM_DEBUG(if (Result) dbgs() << From->getName() << " => " << To->getName()
80 << " crosses suspend point (path or loop)\n");
81 return Result;
82}

84template <bool Initialize>
85bool SuspendCrossingInfo::computeBlockData(
86 const ReversePostOrderTraversal<Function *> &RPOT) {
87 bool Changed = false;
88
89 for (const BasicBlock *BB : RPOT) {
90 auto BBNo = Mapping.blockToIndex(BB);
91 auto &B = Block[BBNo];
92
93 // We don't need to count the predecessors when initialization.
94 if constexpr (!Initialize)
95 // If all the predecessors of the current Block don't change,
96 // the BlockData for the current block must not change too.
97 if (all_of(predecessors(B), [this](BasicBlock *BB) {
98 return !Block[Mapping.blockToIndex(BB)].Changed;
99 })) {
100 B.Changed = false;
101 continue;
102 }
103
104 // Saved Consumes and Kills bitsets so that it is easy to see
105 // if anything changed after propagation.
106 auto SavedConsumes = B.Consumes;
107 auto SavedKills = B.Kills;
108
109 for (BasicBlock *PI : predecessors(B)) {
110 auto PrevNo = Mapping.blockToIndex(PI);
111 auto &P = Block[PrevNo];
112
113 // Propagate Kills and Consumes from predecessors into B.
114 B.Consumes |= P.Consumes;
115 B.Kills |= P.Kills;
116
117 // If block P is a suspend block, it should propagate kills into block
118 // B for every block P consumes.
119 if (P.Suspend)
120 B.Kills |= P.Consumes;
121 }
122
123 if (B.Suspend) {
124 // If block B is a suspend block, it should kill all of the blocks it
125 // consumes.
126 B.Kills |= B.Consumes;
127 } else if (B.End) {
128 // If block B is an end block, it should not propagate kills as the
129 // blocks following coro.end() are reached during initial invocation
130 // of the coroutine while all the data are still available on the
131 // stack or in the registers.
132 B.Kills.reset();
133 } else {
134 // This is reached when B block it not Suspend nor coro.end and it
135 // need to make sure that it is not in the kill set.
136 B.KillLoop |= B.Kills[BBNo];
137 B.Kills.reset(BBNo);
138 }
139
140 if constexpr (!Initialize) {
141 B.Changed = (B.Kills != SavedKills) || (B.Consumes != SavedConsumes);
142 Changed |= B.Changed;
143 }
144 }
145
146 return Changed;
147}

149SuspendCrossingInfo::SuspendCrossingInfo(
150 Function &F, const SmallVectorImpl<AnyCoroSuspendInst *> &CoroSuspends,
151 const SmallVectorImpl<AnyCoroEndInst *> &CoroEnds)
152 : Mapping(F) {
153 const size_t N = Mapping.size();
154 Block.resize(N);
155
156 // Initialize every block so that it consumes itself
157 for (size_t I = 0; I < N; ++I) {
158 auto &B = Block[I];
159 B.Consumes.resize(N);
160 B.Kills.resize(N);
161 B.Consumes.set(I);
162 B.Changed = true;
163 }
164
165 // Mark all CoroEnd Blocks. We do not propagate Kills beyond coro.ends as
166 // the code beyond coro.end is reachable during initial invocation of the
167 // coroutine.
168 for (auto *CE : CoroEnds) {
169 // Verify CoroEnd was normalized
170 assert(CE->getParent()->getFirstInsertionPt() == CE->getIterator() &&
171 CE->getParent()->size() <= 2 && "CoroEnd must be in its own BB");
172
173 getBlockData(CE->getParent()).End = true;
174 }
175
176 // Mark all suspend blocks and indicate that they kill everything they
177 // consume. Note, that crossing coro.save also requires a spill, as any code
178 // between coro.save and coro.suspend may resume the coroutine and all of the
179 // state needs to be saved by that time.
180 auto markSuspendBlock = [&](IntrinsicInst *BarrierInst) {
181 BasicBlock *SuspendBlock = BarrierInst->getParent();
182 auto &B = getBlockData(SuspendBlock);
183 B.Suspend = true;
184 B.Kills |= B.Consumes;
185 };
186 for (auto *CSI : CoroSuspends) {
187 // Verify CoroSuspend was normalized
188 assert(CSI->getParent()->getFirstInsertionPt() == CSI->getIterator() &&
189 CSI->getParent()->size() <= 2 &&
190 "CoroSuspend must be in its own BB");
191
192 markSuspendBlock(CSI);
193 if (auto *Save = CSI->getCoroSave())
194 markSuspendBlock(Save);
195 }
196
197 // It is considered to be faster to use RPO traversal for forward-edges
198 // dataflow analysis.
199 ReversePostOrderTraversal<Function *> RPOT(&F);
200 computeBlockData</*Initialize=*/true>(RPOT);
201 while (computeBlockData</*Initialize*/ false>(RPOT))
202 ;
203
204 LLVM_DEBUG(dump());
205}
