27/// Function object that provides a total ordering on QualType values.
28struct QualTypeOrdering {
29 bool operator()(QualType T1, QualType T2) const {
30 return std::less<void*>()(T1.getAsOpaquePtr(), T2.getAsOpaquePtr());
31 }
32};
