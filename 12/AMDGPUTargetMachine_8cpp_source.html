117static bool onlyAllocateSGPRs(const TargetRegisterInfo &TRI,
118 const MachineRegisterInfo &MRI,
119 const Register Reg) {
120 const TargetRegisterClass *RC = MRI.getRegClass(Reg);
121 return static_cast<const SIRegisterInfo &>(TRI).isSGPRClass(RC);
122}
123
124static bool onlyAllocateVGPRs(const TargetRegisterInfo &TRI,
125 const MachineRegisterInfo &MRI,
126 const Register Reg) {
127 const TargetRegisterClass *RC = MRI.getRegClass(Reg);
128 return !static_cast<const SIRegisterInfo &>(TRI).isSGPRClass(RC);
129}
130
131static bool onlyAllocateWWMRegs(const TargetRegisterInfo &TRI,
132 const MachineRegisterInfo &MRI,
133 const Register Reg) {
134 const SIMachineFunctionInfo *MFI =
135 MRI.getMF().getInfo<SIMachineFunctionInfo>();
136 const TargetRegisterClass *RC = MRI.getRegClass(Reg);
137 return !static_cast<const SIRegisterInfo &>(TRI).isSGPRClass(RC) &&
138 MFI->checkFlag(Reg, AMDGPU::VirtRegFlag::WWM_REG);
139}

534static ScheduleDAGInstrs *
535createGCNMaxOccupancyMachineScheduler(MachineSchedContext *C) {
536 const GCNSubtarget &ST = C->MF->getSubtarget<GCNSubtarget>();
537 ScheduleDAGMILive *DAG =
538 new GCNScheduleDAGMILive(C, std::make_unique<GCNMaxOccupancySchedStrategy>(C));
539 DAG->addMutation(createLoadClusterDAGMutation(DAG->TII, DAG->TRI));
540 if (ST.shouldClusterStores())
541 DAG->addMutation(createStoreClusterDAGMutation(DAG->TII, DAG->TRI));
542 DAG->addMutation(createIGroupLPDAGMutation(AMDGPU::SchedulingPhase::Initial));
543 DAG->addMutation(createAMDGPUMacroFusionDAGMutation());
544 DAG->addMutation(createAMDGPUExportClusteringDAGMutation());
545 return DAG;
546}

572static ScheduleDAGInstrs *
573createIterativeILPMachineScheduler(MachineSchedContext *C) {
574 const GCNSubtarget &ST = C->MF->getSubtarget<GCNSubtarget>();
575 auto DAG = new GCNIterativeScheduler(C,
576 GCNIterativeScheduler::SCHEDULE_ILP);
577 DAG->addMutation(createLoadClusterDAGMutation(DAG->TII, DAG->TRI));
578 if (ST.shouldClusterStores())
579 DAG->addMutation(createStoreClusterDAGMutation(DAG->TII, DAG->TRI));
580 DAG->addMutation(createAMDGPUMacroFusionDAGMutation());
581 return DAG;

771 if (EnableLibCallSimplify)
772 FPM.addPass(AMDGPUSimplifyLibCallsPass());

776 [this](CGSCCPassManager &PM, OptimizationLevel Level) {

780 FunctionPassManager FPM;
781
782 // Add promote kernel arguments pass to the opt pipeline right before

791 FPM.addPass(InferAddressSpacesPass());
792

794 // anything, and before other cleanup optimizations.
795 FPM.addPass(AMDGPULowerKernelAttributesPass());

798 // Promote alloca to vector before SROA and loop unroll. If we
799 // manage to eliminate allocas before unroll we may choose to unroll
800 // less.

802 }
803
804 PM.addPass(createCGSCCToFunctionPassAdaptor(std::move(FPM)));
805 });
806
807 // FIXME: Why is AMDGPUAttributor not in CGSCC?
808 PB.registerOptimizerLastEPCallback(
809 [this](ModulePassManager &MPM, OptimizationLevel Level) {
810 if (Level != OptimizationLevel::O0) {
811 MPM.addPass(AMDGPUAttributorPass(*this));
812 }
813 });
814

816 [this](ModulePassManager &PM, OptimizationLevel Level) {

819 // module is partitioned for codegen.

822 if (EnableLowerModuleLDS)

826 });

832 if (FilterName == "vgpr")
833 return onlyAllocateVGPRs;
834 return nullptr;
835 });

852unsigned AMDGPUTargetMachine::getAssumedAddrSpace(const Value *V) const {
853 const auto *LD = dyn_cast<LoadInst>(V);
854 if (!LD) // TODO: Handle invariant load like constant.
855 return AMDGPUAS::UNKNOWN_ADDRESS_SPACE;
856
857 // It must be a generic pointer loaded.
858 assert(V->getType()->getPointerAddressSpace() == AMDGPUAS::FLAT_ADDRESS);
859
860 const auto *Ptr = LD->getPointerOperand();
861 if (Ptr->getType()->getPointerAddressSpace() != AMDGPUAS::CONSTANT_ADDRESS)
862 return AMDGPUAS::UNKNOWN_ADDRESS_SPACE;
863 // For a generic pointer loaded from the constant memory, it could be assumed
864 // as a global pointer since the constant memory is only populated on the
865 // host side. As implied by the offload programming model, only global
866 // pointers could be referenced on the host side.
867 return AMDGPUAS::GLOBAL_ADDRESS;
868}

870std::pair<const Value *, unsigned>
871AMDGPUTargetMachine::getPredicatedAddrSpace(const Value *V) const {
872 if (auto *II = dyn_cast<IntrinsicInst>(V)) {
873 switch (II->getIntrinsicID()) {
874 case Intrinsic::amdgcn_is_shared:
875 return std::pair(II->getArgOperand(0), AMDGPUAS::LOCAL_ADDRESS);
876 case Intrinsic::amdgcn_is_private:
877 return std::pair(II->getArgOperand(0), AMDGPUAS::PRIVATE_ADDRESS);
878 default:
879 break;
880 }
881 return std::pair(nullptr, -1);
882 }
883 // Check the global pointer predication based on
884 // (!is_share(p) && !is_private(p)). Note that logic 'and' is commutative and
885 // the order of 'is_shared' and 'is_private' is not significant.
886 Value *Ptr;
887 if (match(
888 const_cast<Value *>(V),
889 m_c_And(m_Not(m_Intrinsic<Intrinsic::amdgcn_is_shared>(m_Value(Ptr))),
890 m_Not(m_Intrinsic<Intrinsic::amdgcn_is_private>(
891 m_Deferred(Ptr))))))
892 return std::pair(Ptr, AMDGPUAS::GLOBAL_ADDRESS);
893
894 return std::pair(nullptr, -1);
895}

1093void AMDGPUPassConfig::addIRPasses() {
1094 const AMDGPUTargetMachine &TM = getAMDGPUTargetMachine();
1095
1096 Triple::ArchType Arch = TM.getTargetTriple().getArch();
1097 if (RemoveIncompatibleFunctions && Arch == Triple::amdgcn)
1098 addPass(createAMDGPURemoveIncompatibleFunctionsPass(&TM));
1099
1100 // There is no reason to run these.
1101 disablePass(&StackMapLivenessID);
1102 disablePass(&FuncletLayoutID);
1103 disablePass(&PatchableFunctionID);
1104
1105 addPass(createAMDGPUPrintfRuntimeBinding());
1106 if (LowerCtorDtor)
1107 addPass(createAMDGPUCtorDtorLoweringLegacyPass());
1108
1109 if (isPassEnabled(EnableImageIntrinsicOptimizer))
1110 addPass(createAMDGPUImageIntrinsicOptimizerPass(&TM));
1111
1112 // This can be disabled by passing ::Disable here or on the command line
1113 // with --expand-variadics-override=disable.
1114 addPass(createExpandVariadicsPass(ExpandVariadicsMode::Lowering));
1115
1116 // Function calls are not supported, so make sure we inline everything.
1117 addPass(createAMDGPUAlwaysInlinePass());
1118 addPass(createAlwaysInlinerLegacyPass());
1119
1120 // Handle uses of OpenCL image2d_t, image3d_t and sampler_t arguments.
1121 if (Arch == Triple::r600)
1122 addPass(createR600OpenCLImageTypeLoweringPass());
1123
1124 // Replace OpenCL enqueued block function pointers with global variables.
1125 addPass(createAMDGPUOpenCLEnqueuedBlockLoweringPass());
1126
1127 // Lower LDS accesses to global memory pass if address sanitizer is enabled.
1128 if (EnableSwLowerLDS)
1129 addPass(createAMDGPUSwLowerLDSLegacyPass(&TM));
1130
1131 // Runs before PromoteAlloca so the latter can account for function uses
1132 if (EnableLowerModuleLDS) {
1133 addPass(createAMDGPULowerModuleLDSLegacyPass(&TM));
1134 }
1135
1136 if (TM.getOptLevel() > CodeGenOptLevel::None)
1137 addPass(createInferAddressSpacesPass());
1138
1139 // Run atomic optimizer before Atomic Expand
1140 if ((TM.getTargetTriple().getArch() == Triple::amdgcn) &&
1141 (TM.getOptLevel() >= CodeGenOptLevel::Less) &&
1142 (AMDGPUAtomicOptimizerStrategy != ScanOptions::None)) {
1143 addPass(createAMDGPUAtomicOptimizerPass(AMDGPUAtomicOptimizerStrategy));
1144 }
1145
1146 addPass(createAtomicExpandLegacyPass());
1147
1148 if (TM.getOptLevel() > CodeGenOptLevel::None) {
1149 addPass(createAMDGPUPromoteAlloca());
1150
1151 if (isPassEnabled(EnableScalarIRPasses))
1152 addStraightLineScalarOptimizationPasses();
1153
1154 if (EnableAMDGPUAliasAnalysis) {
1155 addPass(createAMDGPUAAWrapperPass());
1156 addPass(createExternalAAWrapperPass([](Pass &P, Function &,
1157 AAResults &AAR) {
1158 if (auto *WrapperPass = P.getAnalysisIfAvailable<AMDGPUAAWrapperPass>())
1159 AAR.addAAResult(WrapperPass->getResult());
1160 }));
1161 }
1162
1163 if (TM.getTargetTriple().getArch() == Triple::amdgcn) {
1164 // TODO: May want to move later or split into an early and late one.
1165 addPass(createAMDGPUCodeGenPreparePass());
1166 }
1167
1168 // Try to hoist loop invariant parts of divisions AMDGPUCodeGenPrepare may
1169 // have expanded.
1170 if (TM.getOptLevel() > CodeGenOptLevel::Less)
1171 addPass(createLICMPass());
1172 }
1173
1174 TargetPassConfig::addIRPasses();
1175
1176 // EarlyCSE is not always strong enough to clean up what LSR produces. For
1177 // example, GVN can combine
1178 //
1179 // %0 = add %a, %b
1180 // %1 = add %b, %a
1181 //
1182 // and
1183 //
1184 // %0 = shl nsw %a, 2
1185 // %1 = shl %a, 2
1186 //
1187 // but EarlyCSE can do neither of them.
1188 if (isPassEnabled(EnableScalarIRPasses))
1189 addEarlyCSEOrGVNPass();

1192void AMDGPUPassConfig::addCodeGenPrepare() {
1193 if (TM->getTargetTriple().getArch() == Triple::amdgcn) {
1194 // FIXME: This pass adds 2 hacky attributes that can be replaced with an
1195 // analysis, and should be removed.
1196 addPass(createAMDGPUAnnotateKernelFeaturesPass());
1197 }
1198
1199 if (TM->getTargetTriple().getArch() == Triple::amdgcn &&
1200 EnableLowerKernelArguments)
1201 addPass(createAMDGPULowerKernelArgumentsPass());
1202
1203 if (TM->getTargetTriple().getArch() == Triple::amdgcn) {
1204 // This lowering has been placed after codegenprepare to take advantage of
1205 // address mode matching (which is why it isn't put with the LDS lowerings).
1206 // It could be placed anywhere before uniformity annotations (an analysis
1207 // that it changes by splitting up fat pointers into their components)
1208 // but has been put before switch lowering and CFG flattening so that those
1209 // passes can run on the more optimized control flow this pass creates in
1210 // many cases.
1211 //
1212 // FIXME: This should ideally be put after the LoadStoreVectorizer.
1213 // However, due to some annoying facts about ResourceUsageAnalysis,
1214 // (especially as exercised in the resource-usage-dead-function test),
1215 // we need all the function passes codegenprepare all the way through
1216 // said resource usage analysis to run on the call graph produced

1632bool GCNTargetMachine::parseMachineFunctionInfo(
1633 const yaml::MachineFunctionInfo &MFI_, PerFunctionMIParsingState &PFS,
1634 SMDiagnostic &Error, SMRange &SourceRange) const {
1635 const yaml::SIMachineFunctionInfo &YamlMFI =
1636 static_cast<const yaml::SIMachineFunctionInfo &>(MFI_);
1637 MachineFunction &MF = PFS.MF;
1638 SIMachineFunctionInfo *MFI = MF.getInfo<SIMachineFunctionInfo>();
1639 const GCNSubtarget &ST = MF.getSubtarget<GCNSubtarget>();
1640
1641 if (MFI->initializeBaseYamlFields(YamlMFI, MF, PFS, Error, SourceRange))
1642 return true;
1643
1644 if (MFI->Occupancy == 0) {
1645 // Fixup the subtarget dependent default value.
1646 MFI->Occupancy = ST.computeOccupancy(MF.getFunction(), MFI->getLDSSize());
1647 }
1648
1649 auto parseRegister = [&](const yaml::StringValue &RegName, Register &RegVal) {
1650 Register TempReg;
1651 if (parseNamedRegisterReference(PFS, TempReg, RegName.Value, Error)) {
1652 SourceRange = RegName.SourceRange;
1653 return true;
1654 }
1655 RegVal = TempReg;
1656
1657 return false;
1658 };
1659
1660 auto parseOptionalRegister = [&](const yaml::StringValue &RegName,
1661 Register &RegVal) {
1662 return !RegName.Value.empty() && parseRegister(RegName, RegVal);
1663 };
1664
1665 if (parseOptionalRegister(YamlMFI.VGPRForAGPRCopy, MFI->VGPRForAGPRCopy))
1666 return true;
1667
1668 if (parseOptionalRegister(YamlMFI.SGPRForEXECCopy, MFI->SGPRForEXECCopy))
1669 return true;
1670
1671 if (parseOptionalRegister(YamlMFI.LongBranchReservedReg,
1672 MFI->LongBranchReservedReg))
1673 return true;
1674
1675 auto diagnoseRegisterClass = [&](const yaml::StringValue &RegName) {
1676 // Create a diagnostic for a the register string literal.
1677 const MemoryBuffer &Buffer =
1678 *PFS.SM->getMemoryBuffer(PFS.SM->getMainFileID());
1679 Error = SMDiagnostic(*PFS.SM, SMLoc(), Buffer.getBufferIdentifier(), 1,
1680 RegName.Value.size(), SourceMgr::DK_Error,
1681 "incorrect register class for field", RegName.Value,
1682 {}, {});
1683 SourceRange = RegName.SourceRange;
1684 return true;
1685 };
1686
1687 if (parseRegister(YamlMFI.ScratchRSrcReg, MFI->ScratchRSrcReg) ||
1688 parseRegister(YamlMFI.FrameOffsetReg, MFI->FrameOffsetReg) ||
1689 parseRegister(YamlMFI.StackPtrOffsetReg, MFI->StackPtrOffsetReg))
1690 return true;
1691
1692 if (MFI->ScratchRSrcReg != AMDGPU::PRIVATE_RSRC_REG &&
1693 !AMDGPU::SGPR_128RegClass.contains(MFI->ScratchRSrcReg)) {
1694 return diagnoseRegisterClass(YamlMFI.ScratchRSrcReg);
1695 }
1696
1697 if (MFI->FrameOffsetReg != AMDGPU::FP_REG &&
1698 !AMDGPU::SGPR_32RegClass.contains(MFI->FrameOffsetReg)) {
1699 return diagnoseRegisterClass(YamlMFI.FrameOffsetReg);
1700 }
1701
1702 if (MFI->StackPtrOffsetReg != AMDGPU::SP_REG &&
1703 !AMDGPU::SGPR_32RegClass.contains(MFI->StackPtrOffsetReg)) {
1704 return diagnoseRegisterClass(YamlMFI.StackPtrOffsetReg);
1705 }
1706
1707 for (const auto &YamlReg : YamlMFI.WWMReservedRegs) {
1708 Register ParsedReg;
1709 if (parseRegister(YamlReg, ParsedReg))
1710 return true;
1711
1712 MFI->reserveWWMRegister(ParsedReg);
1713 }
1714
1715 auto parseAndCheckArgument = [&](const std::optional<yaml::SIArgument> &A,
1716 const TargetRegisterClass &RC,
1717 ArgDescriptor &Arg, unsigned UserSGPRs,
1718 unsigned SystemSGPRs) {
1719 // Skip parsing if it's not present.
1720 if (!A)
1721 return false;
1722
1723 if (A->IsRegister) {
1724 Register Reg;
1725 if (parseNamedRegisterReference(PFS, Reg, A->RegisterName.Value, Error)) {
1726 SourceRange = A->RegisterName.SourceRange;
1727 return true;
1728 }
1729 if (!RC.contains(Reg))
1730 return diagnoseRegisterClass(A->RegisterName);
1731 Arg = ArgDescriptor::createRegister(Reg);
1732 } else
1733 Arg = ArgDescriptor::createStack(A->StackOffset);
1734 // Check and apply the optional mask.
1735 if (A->Mask)
1736 Arg = ArgDescriptor::createArg(Arg, *A->Mask);
1737
1738 MFI->NumUserSGPRs += UserSGPRs;
1739 MFI->NumSystemSGPRs += SystemSGPRs;
1740 return false;
1741 };
1742
1743 if (YamlMFI.ArgInfo &&
1744 (parseAndCheckArgument(YamlMFI.ArgInfo->PrivateSegmentBuffer,
1745 AMDGPU::SGPR_128RegClass,
1746 MFI->ArgInfo.PrivateSegmentBuffer, 4, 0) ||
1747 parseAndCheckArgument(YamlMFI.ArgInfo->DispatchPtr,
1748 AMDGPU::SReg_64RegClass, MFI->ArgInfo.DispatchPtr,
1749 2, 0) ||
1750 parseAndCheckArgument(YamlMFI.ArgInfo->QueuePtr, AMDGPU::SReg_64RegClass,
1751 MFI->ArgInfo.QueuePtr, 2, 0) ||
1752 parseAndCheckArgument(YamlMFI.ArgInfo->KernargSegmentPtr,
1753 AMDGPU::SReg_64RegClass,
1754 MFI->ArgInfo.KernargSegmentPtr, 2, 0) ||
1755 parseAndCheckArgument(YamlMFI.ArgInfo->DispatchID,
1756 AMDGPU::SReg_64RegClass, MFI->ArgInfo.DispatchID,
1757 2, 0) ||
1758 parseAndCheckArgument(YamlMFI.ArgInfo->FlatScratchInit,
1759 AMDGPU::SReg_64RegClass,
1760 MFI->ArgInfo.FlatScratchInit, 2, 0) ||
1761 parseAndCheckArgument(YamlMFI.ArgInfo->PrivateSegmentSize,
1762 AMDGPU::SGPR_32RegClass,
1763 MFI->ArgInfo.PrivateSegmentSize, 0, 0) ||
1764 parseAndCheckArgument(YamlMFI.ArgInfo->LDSKernelId,
1765 AMDGPU::SGPR_32RegClass,
1766 MFI->ArgInfo.LDSKernelId, 0, 1) ||
1767 parseAndCheckArgument(YamlMFI.ArgInfo->WorkGroupIDX,
1768 AMDGPU::SGPR_32RegClass, MFI->ArgInfo.WorkGroupIDX,
1769 0, 1) ||
1770 parseAndCheckArgument(YamlMFI.ArgInfo->WorkGroupIDY,
1771 AMDGPU::SGPR_32RegClass, MFI->ArgInfo.WorkGroupIDY,
1772 0, 1) ||
1773 parseAndCheckArgument(YamlMFI.ArgInfo->WorkGroupIDZ,
1774 AMDGPU::SGPR_32RegClass, MFI->ArgInfo.WorkGroupIDZ,
1775 0, 1) ||
1776 parseAndCheckArgument(YamlMFI.ArgInfo->WorkGroupInfo,
1777 AMDGPU::SGPR_32RegClass,
1778 MFI->ArgInfo.WorkGroupInfo, 0, 1) ||
1779 parseAndCheckArgument(YamlMFI.ArgInfo->PrivateSegmentWaveByteOffset,
1780 AMDGPU::SGPR_32RegClass,
1781 MFI->ArgInfo.PrivateSegmentWaveByteOffset, 0, 1) ||
1782 parseAndCheckArgument(YamlMFI.ArgInfo->ImplicitArgPtr,
1783 AMDGPU::SReg_64RegClass,
1784 MFI->ArgInfo.ImplicitArgPtr, 0, 0) ||
1785 parseAndCheckArgument(YamlMFI.ArgInfo->ImplicitBufferPtr,
1786 AMDGPU::SReg_64RegClass,
1787 MFI->ArgInfo.ImplicitBufferPtr, 2, 0) ||
1788 parseAndCheckArgument(YamlMFI.ArgInfo->WorkItemIDX,
1789 AMDGPU::VGPR_32RegClass,
1790 MFI->ArgInfo.WorkItemIDX, 0, 0) ||
1791 parseAndCheckArgument(YamlMFI.ArgInfo->WorkItemIDY,
1792 AMDGPU::VGPR_32RegClass,
1793 MFI->ArgInfo.WorkItemIDY, 0, 0) ||
1794 parseAndCheckArgument(YamlMFI.ArgInfo->WorkItemIDZ,
1795 AMDGPU::VGPR_32RegClass,
1796 MFI->ArgInfo.WorkItemIDZ, 0, 0)))
1797 return true;
1798
1799 if (ST.hasIEEEMode())
1800 MFI->Mode.IEEE = YamlMFI.Mode.IEEE;
1801 if (ST.hasDX10ClampMode())
1802 MFI->Mode.DX10Clamp = YamlMFI.Mode.DX10Clamp;
1803
1804 // FIXME: Move proper support for denormal-fp-math into base MachineFunction
1805 MFI->Mode.FP32Denormals.Input = YamlMFI.Mode.FP32InputDenormals
1806 ? DenormalMode::IEEE
1807 : DenormalMode::PreserveSign;
1808 MFI->Mode.FP32Denormals.Output = YamlMFI.Mode.FP32OutputDenormals
1809 ? DenormalMode::IEEE
1810 : DenormalMode::PreserveSign;
1811
1812 MFI->Mode.FP64FP16Denormals.Input = YamlMFI.Mode.FP64FP16InputDenormals
1813 ? DenormalMode::IEEE
1814 : DenormalMode::PreserveSign;
1815 MFI->Mode.FP64FP16Denormals.Output = YamlMFI.Mode.FP64FP16OutputDenormals
1816 ? DenormalMode::IEEE
1817 : DenormalMode::PreserveSign;
1818
1819 if (YamlMFI.HasInitWholeWave)
1820 MFI->setInitWholeWave();
1821
1822 return false;

1906void AMDGPUCodeGenPassBuilder::addCodeGenPrepare(AddIRPass &addPass) const {
1907 // AMDGPUAnnotateKernelFeaturesPass is missing here, but it will hopefully be
1908 // deleted soon.
1909
1910 if (EnableLowerKernelArguments)
1911 addPass(AMDGPULowerKernelArgumentsPass(TM));
1912
1913 // This lowering has been placed after codegenprepare to take advantage of
1914 // address mode matching (which is why it isn't put with the LDS lowerings).
1915 // It could be placed anywhere before uniformity annotations (an analysis
1916 // that it changes by splitting up fat pointers into their components)
1917 // but has been put before switch lowering and CFG flattening so that those
1918 // passes can run on the more optimized control flow this pass creates in
1919 // many cases.
1920 //
1921 // FIXME: This should ideally be put after the LoadStoreVectorizer.
1922 // However, due to some annoying facts about ResourceUsageAnalysis,
1923 // (especially as exercised in the resource-usage-dead-function test),
1924 // we need all the function passes codegenprepare all the way through
1925 // said resource usage analysis to run on the call graph produced
1926 // before codegenprepare runs (because codegenprepare will knock some
1927 // nodes out of the graph, which leads to function-level passes not
1928 // being run on them, which causes crashes in the resource usage analysis).
1929 addPass(AMDGPULowerBufferFatPointersPass(TM));
1930
1931 Base::addCodeGenPrepare(addPass);
1932
1933 if (isPassEnabled(EnableLoadStoreVectorizer))
1934 addPass(LoadStoreVectorizerPass());
1935
1936 // LowerSwitch pass may introduce unreachable blocks that can cause unexpected
1937 // behavior for subsequent passes. Placing it here seems better that these
1938 // blocks would get cleaned up by UnreachableBlockElim inserted next in the
1939 // pass flow.
1940 addPass(LowerSwitchPass());
