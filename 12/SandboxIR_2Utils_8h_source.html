26 /// \Returns the expected type of \p Value V. For most Values this is
27 /// equivalent to getType, but for stores returns the stored type, rather
28 /// than void, and for ReturnInsts returns the returned type.
29 static Type *getExpectedType(const Value *V) {
30 if (auto *I = dyn_cast<Instruction>(V)) {
31 // A Return's value operand can be null if it returns void.
32 if (auto *RI = dyn_cast<ReturnInst>(I)) {
33 if (RI->getReturnValue() == nullptr)
34 return RI->getType();
35 }
36 return getExpectedValue(I)->getType();
37 }
38 return V->getType();
39 }
40
41 /// \Returns the expected Value for this instruction. For most instructions,
42 /// this is the instruction itself, but for stores returns the stored
43 /// operand, and for ReturnInstructions returns the returned value.
44 static Value *getExpectedValue(const Instruction *I) {
45 if (auto *SI = dyn_cast<StoreInst>(I))
46 return SI->getValueOperand();
47 if (auto *RI = dyn_cast<ReturnInst>(I))
48 return RI->getReturnValue();
49 return const_cast<Instruction *>(I);
50 }
51
52 /// \Returns the number of bits required to represent the operands or return
53 /// value of \p V in \p DL.
54 static unsigned getNumBits(Value *V, const DataLayout &DL) {
55 Type *Ty = getExpectedType(V);
56 return DL.getTypeSizeInBits(Ty->LLVMTy);
57 }
58
59 /// Equivalent to MemoryLocation::getOrNone(I).
60 static std::optional<llvm::MemoryLocation>
61 memoryLocationGetOrNone(const Instruction *I) {
62 return llvm::MemoryLocation::getOrNone(cast<llvm::Instruction>(I->Val));
63 }
64
65 /// \Returns the gap between the memory locations accessed by \p I0 and
66 /// \p I1 in bytes.
67 template <typename LoadOrStoreT>
68 static std::optional<int>
69 getPointerDiffInBytes(LoadOrStoreT *I0, LoadOrStoreT *I1, ScalarEvolution &SE,
70 const DataLayout &DL) {
71 static_assert(std::is_same_v<LoadOrStoreT, LoadInst> ||
72 std::is_same_v<LoadOrStoreT, StoreInst>,
73 "Expected sandboxir::Load or sandboxir::Store!");
74 llvm::Value *Opnd0 = I0->getPointerOperand()->Val;
75 llvm::Value *Opnd1 = I1->getPointerOperand()->Val;
76 llvm::Value *Ptr0 = getUnderlyingObject(Opnd0);
77 llvm::Value *Ptr1 = getUnderlyingObject(Opnd1);
78 if (Ptr0 != Ptr1)
79 return false;
80 llvm::Type *ElemTy = llvm::Type::getInt8Ty(SE.getContext());
81 return getPointersDiff(ElemTy, Opnd0, ElemTy, Opnd1, DL, SE,
82 /*StrictCheck=*/false, /*CheckType=*/false);
83 }
84
85 /// \Returns true if \p I0 accesses a memory location lower than \p I1.
86 /// Returns false if the difference cannot be determined, if the memory
87 /// locations are equal, or if I1 accesses a memory location greater than I0.
88 template <typename LoadOrStoreT>
89 static bool atLowerAddress(LoadOrStoreT *I0, LoadOrStoreT *I1,
90 ScalarEvolution &SE, const DataLayout &DL) {
91 auto Diff = getPointerDiffInBytes(I0, I1, SE, DL);
92 if (!Diff)
93 return false;
94 return *Diff > 0;
95 }
96};
