9// Registers and executes the Sandbox IR passes.
10//
11// The pass manager contains an ordered sequence of passes that it runs in
12// order. The passes are owned by the PassRegistry, not by the PassManager.
13//
14// Note that in this design a pass manager is also a pass. So a pass manager
15// runs when it is it's turn to run in its parent pass-manager pass pipeline.
16//

30/// Base class.
31template <typename ParentPass, typename ContainedPass>
32class PassManager : public ParentPass {
33protected:
34 /// The list of passes that this pass manager will run.
35 SmallVector<ContainedPass *> Passes;
36
37 PassManager(StringRef Name) : ParentPass(Name) {}
38 PassManager(const PassManager &) = delete;
39 virtual ~PassManager() = default;
40 PassManager &operator=(const PassManager &) = delete;
41
42public:
43 /// Adds \p Pass to the pass pipeline.
44 void addPass(ContainedPass *Pass) {
45 // TODO: Check that Pass's class type works with this PassManager type.
46 Passes.push_back(Pass);
47 }
48#ifndef NDEBUG
49 void print(raw_ostream &OS) const override {
50 OS << this->getName();
51 OS << "(";
52 // TODO: This should call Pass->print(OS) because Pass may be a PM.
53 interleave(Passes, OS, [&OS](auto *Pass) { OS << Pass->getName(); }, ",");
54 OS << ")";
55 }
56 LLVM_DUMP_METHOD void dump() const override {
57 print(dbgs());
58 dbgs() << "\n";
59 }
60#endif
61 /// Similar to print() but prints one pass per line. Used for testing.
62 void printPipeline(raw_ostream &OS) const {
63 OS << this->getName() << "\n";
64 for (const auto &PassPtr : Passes)
65 PassPtr->printPipeline(OS);
66 }
67};

76/// Owns the passes and provides an API to get a pass by its name.
77class PassRegistry {
78 SmallVector<std::unique_ptr<Pass>, 8> Passes;
79 DenseMap<StringRef, Pass *> NameToPassMap;
80
81public:
82 static constexpr const char PassDelimToken = ',';
83 PassRegistry() = default;
84 /// Registers \p PassPtr and takes ownership.
85 Pass &registerPass(std::unique_ptr<Pass> &&PassPtr) {
86 auto &PassRef = *PassPtr.get();
87 NameToPassMap[PassRef.getName()] = &PassRef;
88 Passes.push_back(std::move(PassPtr));
89 return PassRef;
90 }
91 /// \Returns the pass with name \p Name, or null if not registered.
92 Pass *getPassByName(StringRef Name) const {
93 auto It = NameToPassMap.find(Name);
94 return It != NameToPassMap.end() ? It->second : nullptr;
95 }
96 /// Creates a pass pipeline and returns the first pass manager.
97 FunctionPassManager &parseAndCreatePassPipeline(StringRef Pipeline);
98
99#ifndef NDEBUG
100 void print(raw_ostream &OS) const {
101 for (const auto &PassPtr : Passes)
102 OS << PassPtr->getName() << "\n";
103 }
104 LLVM_DUMP_METHOD void dump() const;
105#endif
106};
