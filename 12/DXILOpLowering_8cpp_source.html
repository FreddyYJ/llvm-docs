33static bool isVectorArgExpansion(Function &F) {
34 switch (F.getIntrinsicID()) {
35 case Intrinsic::dx_dot2:
36 case Intrinsic::dx_dot3:
37 case Intrinsic::dx_dot4:
38 return true;
39 }
40 return false;
41}

43static SmallVector<Value *> populateOperands(Value *Arg, IRBuilder<> &Builder) {
44 SmallVector<Value *> ExtractedElements;
45 auto *VecArg = dyn_cast<FixedVectorType>(Arg->getType());
46 for (unsigned I = 0; I < VecArg->getNumElements(); ++I) {
47 Value *Index = ConstantInt::get(Type::getInt32Ty(Arg->getContext()), I);
48 Value *ExtractedElement = Builder.CreateExtractElement(Arg, Index);
49 ExtractedElements.push_back(ExtractedElement);
50 }
51 return ExtractedElements;
52}

54static SmallVector<Value *> argVectorFlatten(CallInst *Orig,
55 IRBuilder<> &Builder) {
56 // Note: arg[NumOperands-1] is a pointer and is not needed by our flattening.
57 unsigned NumOperands = Orig->getNumOperands() - 1;
58 assert(NumOperands > 0);
59 Value *Arg0 = Orig->getOperand(0);
60 [[maybe_unused]] auto *VecArg0 = dyn_cast<FixedVectorType>(Arg0->getType());
61 assert(VecArg0);
62 SmallVector<Value *> NewOperands = populateOperands(Arg0, Builder);
63 for (unsigned I = 1; I < NumOperands; ++I) {
64 Value *Arg = Orig->getOperand(I);
65 [[maybe_unused]] auto *VecArg = dyn_cast<FixedVectorType>(Arg->getType());
66 assert(VecArg);
67 assert(VecArg0->getElementType() == VecArg->getElementType());
68 assert(VecArg0->getNumElements() == VecArg->getNumElements());
69 auto NextOperandList = populateOperands(Arg, Builder);
70 NewOperands.append(NextOperandList.begin(), NextOperandList.end());
71 }
72 return NewOperands;
73}

85 /// Replace every call to \c F using \c ReplaceCall, and then erase \c F. If
86 /// there is an error replacing a call, we emit a diagnostic and return true.
87 [[nodiscard]] bool
88 replaceFunction(Function &F,
89 llvm::function_ref<Error(CallInst *CI)> ReplaceCall) {
90 for (User *U : make_early_inc_range(F.users())) {
91 CallInst *CI = dyn_cast<CallInst>(U);
92 if (!CI)
93 continue;
94
95 if (Error E = ReplaceCall(CI)) {
96 std::string Message(toString(std::move(E)));
97 DiagnosticInfoUnsupported Diag(*CI->getFunction(), Message,
98 CI->getDebugLoc());
99 M.getContext().diagnose(Diag);
100 return true;
101 }
102 }
103 if (F.user_empty())
104 F.eraseFromParent();
105 return false;
106 }

108 [[nodiscard]]
109 bool replaceFunctionWithOp(Function &F, dxil::OpCode DXILOp) {
110 bool IsVectorArgExpansion = isVectorArgExpansion(F);
111 return replaceFunction(F, [&](CallInst *CI) -> Error {
112 SmallVector<Value *> Args;
113 OpBuilder.getIRB().SetInsertPoint(CI);
114 if (IsVectorArgExpansion) {
115 SmallVector<Value *> NewArgs = argVectorFlatten(CI, OpBuilder.getIRB());
116 Args.append(NewArgs.begin(), NewArgs.end());
117 } else
118 Args.append(CI->arg_begin(), CI->arg_end());
119
120 Expected<CallInst *> OpCall =
121 OpBuilder.tryCreateOp(DXILOp, Args, CI->getName(), F.getReturnType());
122 if (Error E = OpCall.takeError())
123 return E;
124
125 CI->replaceAllUsesWith(*OpCall);
126 CI->eraseFromParent();
127 return Error::success();
128 });
129 }

131 /// Create a cast between a `target("dx")` type and `dx.types.Handle`, which
132 /// is intended to be removed by the end of lowering. This is used to allow
133 /// lowering of ops which need to change their return or argument types in a
134 /// piecemeal way - we can add the casts in to avoid updating all of the uses
135 /// or defs, and by the end all of the casts will be redundant.
136 Value *createTmpHandleCast(Value *V, Type *Ty) {
137 Function *CastFn = Intrinsic::getDeclaration(&M, Intrinsic::dx_cast_handle,
138 {Ty, V->getType()});
139 CallInst *Cast = OpBuilder.getIRB().CreateCall(CastFn, {V});
140 CleanupCasts.push_back(Cast);
141 return Cast;
142 }
143
144 void cleanupHandleCasts() {
145 SmallVector<CallInst *> ToRemove;
146 SmallVector<Function *> CastFns;
147
148 for (CallInst *Cast : CleanupCasts) {
149 // These casts were only put in to ease the move from `target("dx")` types
150 // to `dx.types.Handle in a piecemeal way. At this point, all of the
151 // non-cast uses should now be `dx.types.Handle`, and remaining casts
152 // should all form pairs to and from the now unused `target("dx")` type.
153 CastFns.push_back(Cast->getCalledFunction());
154
155 // If the cast is not to `dx.types.Handle`, it should be the first part of
156 // the pair. Keep track so we can remove it once it has no more uses.
157 if (Cast->getType() != OpBuilder.getHandleType()) {
158 ToRemove.push_back(Cast);
159 continue;
160 }
161 // Otherwise, we're the second handle in a pair. Forward the arguments and
162 // remove the (second) cast.
163 CallInst *Def = cast<CallInst>(Cast->getOperand(0));
164 assert(Def->getIntrinsicID() == Intrinsic::dx_cast_handle &&
165 "Unbalanced pair of temporary handle casts");
166 Cast->replaceAllUsesWith(Def->getOperand(0));
167 Cast->eraseFromParent();
168 }
169 for (CallInst *Cast : ToRemove) {
170 assert(Cast->user_empty() && "Temporary handle cast still has users");
171 Cast->eraseFromParent();
172 }
173
174 // Deduplicate the cast functions so that we only erase each one once.
175 llvm::sort(CastFns);
176 CastFns.erase(llvm::unique(CastFns), CastFns.end());
177 for (Function *F : CastFns)
178 F->eraseFromParent();
179
180 CleanupCasts.clear();
181 }

183 [[nodiscard]] bool lowerToCreateHandle(Function &F) {
184 IRBuilder<> &IRB = OpBuilder.getIRB();
185 Type *Int8Ty = IRB.getInt8Ty();
186 Type *Int32Ty = IRB.getInt32Ty();
187
188 return replaceFunction(F, [&](CallInst *CI) -> Error {
189 IRB.SetInsertPoint(CI);
190
191 auto *It = DRM.find(CI);
192 assert(It != DRM.end() && "Resource not in map?");
193 dxil::ResourceInfo &RI = *It;
194 const auto &Binding = RI.getBinding();
195
196 std::array<Value *, 4> Args{
197 ConstantInt::get(Int8Ty, llvm::to_underlying(RI.getResourceClass())),
198 ConstantInt::get(Int32Ty, Binding.RecordID), CI->getArgOperand(3),
199 CI->getArgOperand(4)};
200 Expected<CallInst *> OpCall =
201 OpBuilder.tryCreateOp(OpCode::CreateHandle, Args, CI->getName());
202 if (Error E = OpCall.takeError())
203 return E;
204
205 Value *Cast = createTmpHandleCast(*OpCall, CI->getType());
206
207 CI->replaceAllUsesWith(Cast);
208 CI->eraseFromParent();
209 return Error::success();
210 });
211 }

213 [[nodiscard]] bool lowerToBindAndAnnotateHandle(Function &F) {
214 IRBuilder<> &IRB = OpBuilder.getIRB();
215
216 return replaceFunction(F, [&](CallInst *CI) -> Error {
217 IRB.SetInsertPoint(CI);
218
219 auto *It = DRM.find(CI);
220 assert(It != DRM.end() && "Resource not in map?");
221 dxil::ResourceInfo &RI = *It;
222
223 const auto &Binding = RI.getBinding();
224 std::pair<uint32_t, uint32_t> Props = RI.getAnnotateProps();
225
226 // For `CreateHandleFromBinding` we need the upper bound rather than the
227 // size, so we need to be careful about the difference for "unbounded".
228 uint32_t Unbounded = std::numeric_limits<uint32_t>::max();
229 uint32_t UpperBound = Binding.Size == Unbounded
230 ? Unbounded
231 : Binding.LowerBound + Binding.Size - 1;
232 Constant *ResBind = OpBuilder.getResBind(
233 Binding.LowerBound, UpperBound, Binding.Space, RI.getResourceClass());
234 std::array<Value *, 3> BindArgs{ResBind, CI->getArgOperand(3),
235 CI->getArgOperand(4)};
236 Expected<CallInst *> OpBind = OpBuilder.tryCreateOp(
237 OpCode::CreateHandleFromBinding, BindArgs, CI->getName());
238 if (Error E = OpBind.takeError())
239 return E;
240
241 std::array<Value *, 2> AnnotateArgs{
242 *OpBind, OpBuilder.getResProps(Props.first, Props.second)};
243 Expected<CallInst *> OpAnnotate = OpBuilder.tryCreateOp(
244 OpCode::AnnotateHandle, AnnotateArgs,
245 CI->hasName() ? CI->getName() + "_annot" : Twine());
246 if (Error E = OpAnnotate.takeError())
247 return E;
248
249 Value *Cast = createTmpHandleCast(*OpAnnotate, CI->getType());
250
251 CI->replaceAllUsesWith(Cast);
252 CI->eraseFromParent();
253
254 return Error::success();
255 });
256 }

260 bool lowerHandleFromBinding(Function &F) {
261 Triple TT(Triple(M.getTargetTriple()));

267 /// Replace uses of \c Intrin with the values in the `dx.ResRet` of \c Op.
268 /// Since we expect to be post-scalarization, make an effort to avoid vectors.
269 Error replaceResRetUses(CallInst *Intrin, CallInst *Op, bool HasCheckBit) {
270 IRBuilder<> &IRB = OpBuilder.getIRB();
271
272 Instruction *OldResult = Intrin;
273 Type *OldTy = Intrin->getType();
274
275 if (HasCheckBit) {
276 auto *ST = cast<StructType>(OldTy);
277
278 Value *CheckOp = nullptr;
279 Type *Int32Ty = IRB.getInt32Ty();
280 for (Use &U : make_early_inc_range(OldResult->uses())) {
281 if (auto *EVI = dyn_cast<ExtractValueInst>(U.getUser())) {
282 ArrayRef<unsigned> Indices = EVI->getIndices();
283 assert(Indices.size() == 1);
284 // We're only interested in uses of the check bit for now.
285 if (Indices[0] != 1)
286 continue;
287 if (!CheckOp) {
288 Value *NewEVI = IRB.CreateExtractValue(Op, 4);
289 Expected<CallInst *> OpCall = OpBuilder.tryCreateOp(
290 OpCode::CheckAccessFullyMapped, {NewEVI},
291 OldResult->hasName() ? OldResult->getName() + "_check"
292 : Twine(),
293 Int32Ty);
294 if (Error E = OpCall.takeError())
295 return E;
296 CheckOp = *OpCall;
297 }
298 EVI->replaceAllUsesWith(CheckOp);
299 EVI->eraseFromParent();
300 }
301 }
302
303 OldResult = cast<Instruction>(
304 IRB.CreateExtractValue(Op, 0, OldResult->getName()));
305 OldTy = ST->getElementType(0);
306 }
307
308 // For scalars, we just extract the first element.
309 if (!isa<FixedVectorType>(OldTy)) {
310 Value *EVI = IRB.CreateExtractValue(Op, 0);
311 OldResult->replaceAllUsesWith(EVI);
312 OldResult->eraseFromParent();
313 if (OldResult != Intrin) {
314 assert(Intrin->use_empty() && "Intrinsic still has uses?");
315 Intrin->eraseFromParent();
316 }
317 return Error::success();
318 }
319
320 std::array<Value *, 4> Extracts = {};
321 SmallVector<ExtractElementInst *> DynamicAccesses;
322
323 // The users of the operation should all be scalarized, so we attempt to
324 // replace the extractelements with extractvalues directly.
325 for (Use &U : make_early_inc_range(OldResult->uses())) {
326 if (auto *EEI = dyn_cast<ExtractElementInst>(U.getUser())) {
327 if (auto *IndexOp = dyn_cast<ConstantInt>(EEI->getIndexOperand())) {
328 size_t IndexVal = IndexOp->getZExtValue();
329 assert(IndexVal < 4 && "Index into buffer load out of range");
330 if (!Extracts[IndexVal])
331 Extracts[IndexVal] = IRB.CreateExtractValue(Op, IndexVal);
332 EEI->replaceAllUsesWith(Extracts[IndexVal]);
333 EEI->eraseFromParent();
334 } else {
335 DynamicAccesses.push_back(EEI);
336 }
337 }
338 }
339
340 const auto *VecTy = cast<FixedVectorType>(OldTy);
341 const unsigned N = VecTy->getNumElements();
342
343 // If there's a dynamic access we need to round trip through stack memory so
344 // that we don't leave vectors around.
345 if (!DynamicAccesses.empty()) {
346 Type *Int32Ty = IRB.getInt32Ty();
347 Constant *Zero = ConstantInt::get(Int32Ty, 0);
348
349 Type *ElTy = VecTy->getElementType();
350 Type *ArrayTy = ArrayType::get(ElTy, N);
351 Value *Alloca = IRB.CreateAlloca(ArrayTy);
352
353 for (int I = 0, E = N; I != E; ++I) {
354 if (!Extracts[I])
355 Extracts[I] = IRB.CreateExtractValue(Op, I);
356 Value *GEP = IRB.CreateInBoundsGEP(
357 ArrayTy, Alloca, {Zero, ConstantInt::get(Int32Ty, I)});
358 IRB.CreateStore(Extracts[I], GEP);
359 }
360
361 for (ExtractElementInst *EEI : DynamicAccesses) {
362 Value *GEP = IRB.CreateInBoundsGEP(ArrayTy, Alloca,
363 {Zero, EEI->getIndexOperand()});
364 Value *Load = IRB.CreateLoad(ElTy, GEP);
365 EEI->replaceAllUsesWith(Load);
366 EEI->eraseFromParent();
367 }
368 }
369
370 // If we still have uses, then we're not fully scalarized and need to
371 // recreate the vector. This should only happen for things like exported
372 // functions from libraries.
373 if (!OldResult->use_empty()) {
374 for (int I = 0, E = N; I != E; ++I)
375 if (!Extracts[I])
376 Extracts[I] = IRB.CreateExtractValue(Op, I);
377
378 Value *Vec = UndefValue::get(OldTy);
379 for (int I = 0, E = N; I != E; ++I)
380 Vec = IRB.CreateInsertElement(Vec, Extracts[I], I);
381 OldResult->replaceAllUsesWith(Vec);
382 }
383
384 OldResult->eraseFromParent();
385 if (OldResult != Intrin) {
386 assert(Intrin->use_empty() && "Intrinsic still has uses?");
387 Intrin->eraseFromParent();
388 }
389
390 return Error::success();
391 }

393 [[nodiscard]] bool lowerTypedBufferLoad(Function &F, bool HasCheckBit) {
394 IRBuilder<> &IRB = OpBuilder.getIRB();
395 Type *Int32Ty = IRB.getInt32Ty();
396
397 return replaceFunction(F, [&](CallInst *CI) -> Error {
398 IRB.SetInsertPoint(CI);
399
400 Value *Handle =
401 createTmpHandleCast(CI->getArgOperand(0), OpBuilder.getHandleType());
402 Value *Index0 = CI->getArgOperand(1);
403 Value *Index1 = UndefValue::get(Int32Ty);
404
405 Type *OldTy = CI->getType();
406 if (HasCheckBit)
407 OldTy = cast<StructType>(OldTy)->getElementType(0);
408 Type *NewRetTy = OpBuilder.getResRetType(OldTy->getScalarType());
409
410 std::array<Value *, 3> Args{Handle, Index0, Index1};
411 Expected<CallInst *> OpCall = OpBuilder.tryCreateOp(
412 OpCode::BufferLoad, Args, CI->getName(), NewRetTy);
413 if (Error E = OpCall.takeError())
414 return E;
415 if (Error E = replaceResRetUses(CI, *OpCall, HasCheckBit))
416 return E;
417
418 return Error::success();
419 });
420 }

422 [[nodiscard]] bool lowerTypedBufferStore(Function &F) {
423 IRBuilder<> &IRB = OpBuilder.getIRB();
424 Type *Int8Ty = IRB.getInt8Ty();
425 Type *Int32Ty = IRB.getInt32Ty();
426
427 return replaceFunction(F, [&](CallInst *CI) -> Error {
428 IRB.SetInsertPoint(CI);
429
430 Value *Handle =
431 createTmpHandleCast(CI->getArgOperand(0), OpBuilder.getHandleType());
432 Value *Index0 = CI->getArgOperand(1);
433 Value *Index1 = UndefValue::get(Int32Ty);
434 // For typed stores, the mask must always cover all four elements.
435 Constant *Mask = ConstantInt::get(Int8Ty, 0xF);
436
437 Value *Data = CI->getArgOperand(2);
438 auto *DataTy = dyn_cast<FixedVectorType>(Data->getType());
439 if (!DataTy || DataTy->getNumElements() != 4)
440 return make_error<StringError>(
441 "typedBufferStore data must be a vector of 4 elements",
442 inconvertibleErrorCode());
443 Value *Data0 =
444 IRB.CreateExtractElement(Data, ConstantInt::get(Int32Ty, 0));
445 Value *Data1 =
446 IRB.CreateExtractElement(Data, ConstantInt::get(Int32Ty, 1));
447 Value *Data2 =
448 IRB.CreateExtractElement(Data, ConstantInt::get(Int32Ty, 2));
449 Value *Data3 =
450 IRB.CreateExtractElement(Data, ConstantInt::get(Int32Ty, 3));
451
452 std::array<Value *, 8> Args{Handle, Index0, Index1, Data0,
453 Data1, Data2, Data3, Mask};
454 Expected<CallInst *> OpCall =
455 OpBuilder.tryCreateOp(OpCode::BufferStore, Args, CI->getName());
456 if (Error E = OpCall.takeError())
457 return E;
458
459 CI->eraseFromParent();
460 return Error::success();
461 });
462 }

464 bool lowerIntrinsics() {
465 bool Updated = false;
466 bool HasErrors = false;
467
468 for (Function &F : make_early_inc_range(M.functions())) {
469 if (!F.isDeclaration())
470 continue;
471 Intrinsic::ID ID = F.getIntrinsicID();
472 switch (ID) {
473 default:
474 continue;
475#define DXIL_OP_INTRINSIC(OpCode, Intrin) \
476 case Intrin: \
477 HasErrors |= replaceFunctionWithOp(F, OpCode); \
478 break;
479#include "DXILOperation.inc"
480 case Intrinsic::dx_handle_fromBinding:
481 HasErrors |= lowerHandleFromBinding(F);
482 break;
483 case Intrinsic::dx_typedBufferLoad:
484 HasErrors |= lowerTypedBufferLoad(F, /*HasCheckBit=*/false);
485 break;
486 case Intrinsic::dx_typedBufferLoad_checkbit:
487 HasErrors |= lowerTypedBufferLoad(F, /*HasCheckBit=*/true);
488 break;
489 case Intrinsic::dx_typedBufferStore:
490 HasErrors |= lowerTypedBufferStore(F);
491 break;
492 }
493 Updated = true;
494 }
495 if (Updated && !HasErrors)
496 cleanupHandleCasts();
497
498 return Updated;
499 }
