20const LangStandard &LangStandard::getLangStandardForKind(Kind K) {
21 switch (K) {
22 case lang_unspecified:
23 llvm::report_fatal_error("getLangStandardForKind() on unspecified kind");
24#define LANGSTANDARD(id, name, lang, desc, features) \
25 case lang_##id: return Lang_##id;
26#include "clang/Basic/LangStandards.def"
27 }
28 llvm_unreachable("Invalid language kind!");
29}

31LangStandard::Kind LangStandard::getLangKind(StringRef Name) {

34#define LANGSTANDARD_ALIAS(id, alias) .Case(alias, lang_##id)

39const LangStandard *LangStandard::getLangStandardForName(StringRef Name) {

47LangStandard::Kind clang::getDefaultLanguageStandard(clang::Language Lang,
48 const llvm::Triple &T) {
49 switch (Lang) {
50 case Language::Unknown:
51 case Language::LLVM_IR:
52 llvm_unreachable("Invalid input kind!");
53 case Language::OpenCL:
54 return LangStandard::lang_opencl12;
55 case Language::OpenCLCXX:
56 return LangStandard::lang_openclcpp10;
57 case Language::Asm:
58 case Language::C:
59 // The PS4 uses C99 as the default C standard.
60 if (T.isPS4())
61 return LangStandard::lang_gnu99;
62 return LangStandard::lang_gnu17;
63 case Language::ObjC:
64 return LangStandard::lang_gnu11;
65 case Language::CXX:
66 case Language::ObjCXX:
67 case Language::CUDA:
68 case Language::HIP:
69 return LangStandard::lang_gnucxx17;
70 case Language::RenderScript:
71 return LangStandard::lang_c99;
72 case Language::HLSL:
73 return LangStandard::lang_hlsl2021;
74 }
75 llvm_unreachable("unhandled Language kind!");
76}
