178 Type *getType(llvm::Type *LLVMTy) {
179 if (LLVMTy == nullptr)
180 return nullptr;
181 auto Pair = LLVMTypeToTypeMap.insert({LLVMTy, nullptr});
182 auto It = Pair.first;
183 if (Pair.second)
184 It->second = std::unique_ptr<Type, TypeDeleter>(new Type(LLVMTy, *this));
185 return It->second.get();
186 }
