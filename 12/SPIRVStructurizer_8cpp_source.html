54// Returns the exact convergence region in the tree defined by `Node` for which
55// `BB` is the header, nullptr otherwise.
56const ConvergenceRegion *getRegionForHeader(const ConvergenceRegion *Node,
57 BasicBlock *BB) {
58 if (Node->Entry == BB)
59 return Node;
60
61 for (auto *Child : Node->Children) {
62 const auto *CR = getRegionForHeader(Child, BB);
63 if (CR != nullptr)
64 return CR;
65 }
66 return nullptr;
67}

69// Returns the single BasicBlock exiting the convergence region `CR`,
70// nullptr if no such exit exists.
71BasicBlock *getExitFor(const ConvergenceRegion *CR) {
72 std::unordered_set<BasicBlock *> ExitTargets;
73 for (BasicBlock *Exit : CR->Exits) {
74 for (BasicBlock *Successor : successors(Exit)) {
75 if (CR->Blocks.count(Successor) == 0)
76 ExitTargets.insert(Successor);
77 }
78 }
79
80 assert(ExitTargets.size() <= 1);
81 if (ExitTargets.size() == 0)
82 return nullptr;
83
84 return *ExitTargets.begin();
85}

87// Returns the merge block designated by I if I is a merge instruction, nullptr
88// otherwise.
89BasicBlock *getDesignatedMergeBlock(Instruction *I) {
90 IntrinsicInst *II = dyn_cast<IntrinsicInst>(I);
91 if (II == nullptr)
92 return nullptr;
93
94 if (II->getIntrinsicID() != Intrinsic::spv_loop_merge &&
95 II->getIntrinsicID() != Intrinsic::spv_selection_merge)
96 return nullptr;
97
98 BlockAddress *BA = cast<BlockAddress>(II->getOperand(0));
99 return BA->getBasicBlock();
100}

102// Returns the continue block designated by I if I is an OpLoopMerge, nullptr
103// otherwise.
104BasicBlock *getDesignatedContinueBlock(Instruction *I) {
105 IntrinsicInst *II = dyn_cast<IntrinsicInst>(I);
106 if (II == nullptr)
107 return nullptr;

112 BlockAddress *BA = cast<BlockAddress>(II->getOperand(1));

116// Returns true if Header has one merge instruction which designated Merge as
117// merge block.
118bool isDefinedAsSelectionMergeBy(BasicBlock &Header, BasicBlock &Merge) {
119 for (auto &I : Header) {
120 BasicBlock *MB = getDesignatedMergeBlock(&I);
121 if (MB == &Merge)
122 return true;
123 }
124 return false;
125}

127// Returns true if the BB has one OpLoopMerge instruction.
128bool hasLoopMergeInstruction(BasicBlock &BB) {
129 for (auto &I : BB)
130 if (getDesignatedContinueBlock(&I))
131 return true;
132 return false;

193// Do a preorder traversal of the CFG starting from the BB |Start|.
194// point. Calls |op| on each basic block encountered during the traversal.
195void visit(BasicBlock &Start, std::function<bool(BasicBlock *)> op) {
196 std::stack<BasicBlock *> ToVisit;
197 SmallPtrSet<BasicBlock *, 8> Seen;
198
199 ToVisit.push(&Start);
200 Seen.insert(ToVisit.top());
201 while (ToVisit.size() != 0) {
202 BasicBlock *BB = ToVisit.top();
203 ToVisit.pop();
204
205 if (!op(BB))
206 continue;
207
208 for (auto Succ : successors(BB)) {
209 if (Seen.contains(Succ))
210 continue;
211 ToVisit.push(Succ);
212 Seen.insert(Succ);
213 }
214 }
215}

217// Replaces the conditional and unconditional branch targets of |BB| by
218// |NewTarget| if the target was |OldTarget|. This function also makes sure the
219// associated merge instruction gets updated accordingly.
220void replaceIfBranchTargets(BasicBlock *BB, BasicBlock *OldTarget,
221 BasicBlock *NewTarget) {
222 auto *BI = cast<BranchInst>(BB->getTerminator());
223
224 // 1. Replace all matching successors.
225 for (size_t i = 0; i < BI->getNumSuccessors(); i++) {
226 if (BI->getSuccessor(i) == OldTarget)
227 BI->setSuccessor(i, NewTarget);
228 }
229
230 // Branch was unconditional, no fixup required.
231 if (BI->isUnconditional())
232 return;
233
234 // Branch had 2 successors, maybe now both are the same?
235 if (BI->getSuccessor(0) != BI->getSuccessor(1))
236 return;
237
238 // Note: we may end up here because the original IR had such branches.
239 // This means Target is not necessarily equal to NewTarget.
240 IRBuilder<> Builder(BB);
241 Builder.SetInsertPoint(BI);
242 Builder.CreateBr(BI->getSuccessor(0));
243 BI->eraseFromParent();
244
245 // The branch was the only instruction, nothing else to do.
246 if (BB->size() == 1)
247 return;
248
249 // Otherwise, we need to check: was there an OpSelectionMerge before this
250 // branch? If we removed the OpBranchConditional, we must also remove the
251 // OpSelectionMerge. This is not valid for OpLoopMerge:
252 IntrinsicInst *II =
253 dyn_cast<IntrinsicInst>(BB->getTerminator()->getPrevNode());
254 if (!II || II->getIntrinsicID() != Intrinsic::spv_selection_merge)
255 return;
256
257 Constant *C = cast<Constant>(II->getOperand(0));
258 II->eraseFromParent();
259 if (!C->isConstantUsed())
260 C->destroyConstant();
261}

263// Replaces the target of branch instruction in |BB| with |NewTarget| if it
264// was |OldTarget|. This function also fixes the associated merge instruction.
265// Note: this function does not simplify branching instructions, it only updates
266// targets. See also: simplifyBranches.
267void replaceBranchTargets(BasicBlock *BB, BasicBlock *OldTarget,
268 BasicBlock *NewTarget) {
269 auto *T = BB->getTerminator();
270 if (isa<ReturnInst>(T))
271 return;
272
273 if (isa<BranchInst>(T))
274 return replaceIfBranchTargets(BB, OldTarget, NewTarget);
275
276 if (auto *SI = dyn_cast<SwitchInst>(T)) {
277 for (size_t i = 0; i < SI->getNumSuccessors(); i++) {
278 if (SI->getSuccessor(i) == OldTarget)
279 SI->setSuccessor(i, NewTarget);
280 }
281 return;
282 }
283
284 assert(false && "Unhandled terminator type.");
285}

287// Replaces basic bloc operands |OldSrc| or OpPhi instructions in |BB| by
288// |NewSrc|. This function does not simplify the OpPhi instruction once
289// transformed.
290void replacePhiTargets(BasicBlock *BB, BasicBlock *OldSrc, BasicBlock *NewSrc) {
291 for (PHINode &Phi : BB->phis()) {
292 int index = Phi.getBasicBlockIndex(OldSrc);
293 if (index == -1)
294 continue;
295 Phi.setIncomingBlock(index, NewSrc);
296 }

341 // Returns the list of blocks that belong to a SPIR-V loop construct,
342 // including the continue construct.
343 std::vector<BasicBlock *> getLoopConstructBlocks(BasicBlock *Header,
344 BasicBlock *Merge) {
345 assert(DT.dominates(Header, Merge));
346 std::vector<BasicBlock *> Output;
347 partialOrderVisit(*Header, [&](BasicBlock *BB) {
348 if (BB == Merge)
349 return false;
350 if (DT.dominates(Merge, BB) || !DT.dominates(Header, BB))
351 return false;
352 Output.push_back(BB);
353 return true;
354 });
355 return Output;
356 }

358 // Returns the list of blocks that belong to a SPIR-V selection construct.
359 std::vector<BasicBlock *>
360 getSelectionConstructBlocks(DivergentConstruct *Node) {
361 assert(DT.dominates(Node->Header, Node->Merge));
362 BlockSet OutsideBlocks;
363 OutsideBlocks.insert(Node->Merge);
364
365 for (DivergentConstruct *It = Node->Parent; It != nullptr;
366 It = It->Parent) {
367 OutsideBlocks.insert(It->Merge);
368 if (It->Continue)
369 OutsideBlocks.insert(It->Continue);
370 }
371
372 std::vector<BasicBlock *> Output;
373 partialOrderVisit(*Node->Header, [&](BasicBlock *BB) {
374 if (OutsideBlocks.count(BB) != 0)
375 return false;
376 if (DT.dominates(Node->Merge, BB) || !DT.dominates(Node->Header, BB))
377 return false;
378 Output.push_back(BB);
379 return true;
380 });
381 return Output;
382 }

384 // Returns the list of blocks that belong to a SPIR-V switch construct.
385 std::vector<BasicBlock *> getSwitchConstructBlocks(BasicBlock *Header,
386 BasicBlock *Merge) {
387 assert(DT.dominates(Header, Merge));
388
389 std::vector<BasicBlock *> Output;
390 partialOrderVisit(*Header, [&](BasicBlock *BB) {
391 // the blocks structurally dominated by a switch header,
392 if (!DT.dominates(Header, BB))
393 return false;
394 // excluding blocks structurally dominated by the switch header’s merge
395 // block.
396 if (DT.dominates(Merge, BB) || BB == Merge)
397 return false;
398 Output.push_back(BB);
399 return true;
400 });

425 // Splits the given edges by recreating proxy nodes so that the destination
426 // OpPhi instruction can still be viable.
427 //
428 // clang-format off
429 //
430 // In SPIR-V, constructs must have a single exit/merge.
431 // Given nodes A and B in the construct, a node C outside, and the following edges.
432 // A -> C
433 // B -> C
434 //
435 // In such cases, we must create a new exit node D, that belong to the construct to make is viable:
436 // A -> D -> C
437 // B -> D -> C
438 //
439 // But if C had a phi node, adding such proxy-block breaks it. In such case, we must add 1 new block per
440 // exit, and patchup the phi node:
441 // A -> D -> D1 -> C
442 // B -> D -> D2 -> C
443 //
444 // A, B, D belongs to the construct. D is the exit. D1 and D2 are empty, just used as
445 // source operands for C's phi node.
446 //
447 // clang-format on
448 std::vector<Edge>
449 createAliasBlocksForComplexEdges(std::vector<Edge> Edges) {
450 std::unordered_map<BasicBlock *, BasicBlock *> Seen;
451 std::vector<Edge> Output;
452 Output.reserve(Edges.size());
453
454 for (auto &[Src, Dst] : Edges) {
455 auto [iterator, inserted] = Seen.insert({Src, Dst});
456 if (inserted) {
457 Output.emplace_back(Src, Dst);
458 continue;
459 }
460
461 // The exact same edge was already seen. Ignoring.
462 if (iterator->second == Dst)
463 continue;
464
465 // The same Src block branches to 2 distinct blocks. This will be an
466 // issue for the generated OpPhi. Creating alias block.
467 BasicBlock *NewSrc =
468 BasicBlock::Create(F.getContext(), "new.exit.src", &F);
469 replaceBranchTargets(Src, Dst, NewSrc);
470 replacePhiTargets(Dst, Src, NewSrc);
471
472 IRBuilder<> Builder(NewSrc);
473 Builder.CreateBr(Dst);
474
475 Seen.emplace(NewSrc, Dst);
476 Output.emplace_back(NewSrc, Dst);
477 }
478
479 return Output;
480 }

482 // Given a construct defined by |Header|, and a list of exiting edges
483 // |Edges|, creates a new single exit node, fixing up those edges.
484 BasicBlock *createSingleExitNode(BasicBlock *Header,
485 std::vector<Edge> &Edges) {
486 auto NewExit = BasicBlock::Create(F.getContext(), "new.exit", &F);
487 IRBuilder<> ExitBuilder(NewExit);
488
489 std::vector<BasicBlock *> Dsts;
490 std::unordered_map<BasicBlock *, ConstantInt *> DstToIndex;
491
492 // Given 2 edges: Src1 -> Dst, Src2 -> Dst:
493 // If Dst has an PHI node, and Src1 and Src2 are both operands, both Src1
494 // and Src2 cannot be hidden by NewExit. Create 2 new nodes: Alias1,
495 // Alias2 to which NewExit will branch before going to Dst. Then, patchup
496 // Dst PHI node to look for Alias1 and Alias2.
497 std::vector<Edge> FixedEdges = createAliasBlocksForComplexEdges(Edges);
498
499 for (auto &[Src, Dst] : FixedEdges) {
500 if (DstToIndex.count(Dst) != 0)
501 continue;
502 DstToIndex.emplace(Dst, ExitBuilder.getInt32(DstToIndex.size()));
503 Dsts.push_back(Dst);
504 }
505
506 if (Dsts.size() == 1) {
507 for (auto &[Src, Dst] : FixedEdges) {
508 replaceBranchTargets(Src, Dst, NewExit);
509 replacePhiTargets(Dst, Src, NewExit);
510 }
511 ExitBuilder.CreateBr(Dsts[0]);
512 return NewExit;
513 }
514
515 PHINode *PhiNode =
516 ExitBuilder.CreatePHI(ExitBuilder.getInt32Ty(), FixedEdges.size());
517
518 for (auto &[Src, Dst] : FixedEdges) {
519 PhiNode->addIncoming(DstToIndex[Dst], Src);
520 replaceBranchTargets(Src, Dst, NewExit);
521 replacePhiTargets(Dst, Src, NewExit);
522 }
523
524 // If we can avoid an OpSwitch, generate an OpBranch. Reason is some
525 // OpBranch are allowed to exist without a new OpSelectionMerge if one of
526 // the branch is the parent's merge node, while OpSwitches are not.
527 if (Dsts.size() == 2) {
528 Value *Condition = ExitBuilder.CreateCmp(CmpInst::ICMP_EQ,
529 DstToIndex[Dsts[0]], PhiNode);
530 ExitBuilder.CreateCondBr(Condition, Dsts[0], Dsts[1]);
531 return NewExit;
532 }
533
534 SwitchInst *Sw =
535 ExitBuilder.CreateSwitch(PhiNode, Dsts[0], Dsts.size() - 1);
536 for (auto It = Dsts.begin() + 1; It != Dsts.end(); ++It) {
537 Sw->addCase(DstToIndex[*It], *It);
538 }
539 return NewExit;
540 }
541 };

543 /// Create a value in BB set to the value associated with the branch the block
544 /// terminator will take.
545 Value *createExitVariable(
546 BasicBlock *BB,
547 const DenseMap<BasicBlock *, ConstantInt *> &TargetToValue) {
548 auto *T = BB->getTerminator();
549 if (isa<ReturnInst>(T))
550 return nullptr;
551
552 IRBuilder<> Builder(BB);
553 Builder.SetInsertPoint(T);
554
555 if (auto *BI = dyn_cast<BranchInst>(T)) {
556
557 BasicBlock *LHSTarget = BI->getSuccessor(0);
558 BasicBlock *RHSTarget =
559 BI->isConditional() ? BI->getSuccessor(1) : nullptr;
560
561 Value *LHS = TargetToValue.count(LHSTarget) != 0
562 ? TargetToValue.at(LHSTarget)
563 : nullptr;
564 Value *RHS = TargetToValue.count(RHSTarget) != 0
565 ? TargetToValue.at(RHSTarget)
566 : nullptr;
567
568 if (LHS == nullptr || RHS == nullptr)
569 return LHS == nullptr ? RHS : LHS;
570 return Builder.CreateSelect(BI->getCondition(), LHS, RHS);
571 }
572
573 // TODO: add support for switch cases.
574 llvm_unreachable("Unhandled terminator type.");
575 }

585 // Add OpLoopMerge instruction on cycles.
586 bool addMergeForLoops(Function &F) {
587 LoopInfo &LI = getAnalysis<LoopInfoWrapperPass>().getLoopInfo();
588 auto *TopLevelRegion =
589 getAnalysis<SPIRVConvergenceRegionAnalysisWrapperPass>()
590 .getRegionInfo()
591 .getTopLevelRegion();
592
593 bool Modified = false;
594 for (auto &BB : F) {
595 // Not a loop header. Ignoring for now.
596 if (!LI.isLoopHeader(&BB))
597 continue;
598 auto *L = LI.getLoopFor(&BB);
599
600 // This loop header is not the entrance of a convergence region. Ignoring
601 // this block.
602 auto *CR = getRegionForHeader(TopLevelRegion, &BB);
603 if (CR == nullptr)
604 continue;
605
606 IRBuilder<> Builder(&BB);
607
608 auto *Merge = getExitFor(CR);
609 // We are indeed in a loop, but there are no exits (infinite loop).
610 // This could be caused by a bad shader, but also could be an artifact
611 // from an earlier optimization. It is not always clear if structurally
612 // reachable means runtime reachable, so we cannot error-out. What we must
613 // do however is to make is legal on the SPIR-V point of view, hence
614 // adding an unreachable merge block.
615 if (Merge == nullptr) {
616 BranchInst *Br = cast<BranchInst>(BB.getTerminator());
617 assert(Br &&
618 "This assumes the branch is not a switch. Maybe that's wrong?");
619 assert(cast<BranchInst>(BB.getTerminator())->isUnconditional());
620
621 Merge = CreateUnreachable(F);
622 Builder.SetInsertPoint(Br);
623 Builder.CreateCondBr(Builder.getFalse(), Merge, Br->getSuccessor(0));
624 Br->eraseFromParent();
625 }
626
627 auto *Continue = L->getLoopLatch();
628
629 Builder.SetInsertPoint(BB.getTerminator());
630 auto MergeAddress = BlockAddress::get(Merge->getParent(), Merge);
631 auto ContinueAddress = BlockAddress::get(Continue->getParent(), Continue);
632 SmallVector<Value *, 2> Args = {MergeAddress, ContinueAddress};
633
634 Builder.CreateIntrinsic(Intrinsic::spv_loop_merge, {}, {Args});
635 Modified = true;
636 }
637
638 return Modified;
639 }

641 // Adds an OpSelectionMerge to the immediate dominator or each node with an
642 // in-degree of 2 or more which is not already the merge target of an
643 // OpLoopMerge/OpSelectionMerge.
644 bool addMergeForNodesWithMultiplePredecessors(Function &F) {
645 DomTreeBuilder::BBDomTree DT;
646 DT.recalculate(F);
647
648 bool Modified = false;
649 for (auto &BB : F) {
650 if (pred_size(&BB) <= 1)
651 continue;
652
653 if (hasLoopMergeInstruction(BB) && pred_size(&BB) <= 2)
654 continue;
655
656 assert(DT.getNode(&BB)->getIDom());
657 BasicBlock *Header = DT.getNode(&BB)->getIDom()->getBlock();
658
659 if (isDefinedAsSelectionMergeBy(*Header, BB))
660 continue;
661
662 IRBuilder<> Builder(Header);
663 Builder.SetInsertPoint(Header->getTerminator());
664
665 auto MergeAddress = BlockAddress::get(BB.getParent(), &BB);
666 SmallVector<Value *, 1> Args = {MergeAddress};
667 Builder.CreateIntrinsic(Intrinsic::spv_selection_merge, {}, {Args});
668
669 Modified = true;
670 }
671
672 return Modified;
673 }

675 // When a block has multiple OpSelectionMerge/OpLoopMerge instructions, sorts
676 // them to put the "largest" first. A merge instruction is defined as larger
677 // than another when its target merge block post-dominates the other target's
678 // merge block. (This ordering should match the nesting ordering of the source
679 // HLSL).
680 bool sortSelectionMerge(Function &F, BasicBlock &Block) {
681 std::vector<Instruction *> MergeInstructions;
682 for (Instruction &I : Block)
683 if (isMergeInstruction(&I))
684 MergeInstructions.push_back(&I);
685
686 if (MergeInstructions.size() <= 1)
687 return false;
688
689 Instruction *InsertionPoint = *MergeInstructions.begin();
690
691 PartialOrderingVisitor Visitor(F);
692 std::sort(MergeInstructions.begin(), MergeInstructions.end(),
693 [&Visitor](Instruction *Left, Instruction *Right) {
694 if (Left == Right)
695 return false;
696 BasicBlock *RightMerge = getDesignatedMergeBlock(Right);
697 BasicBlock *LeftMerge = getDesignatedMergeBlock(Left);
698 return !Visitor.compare(RightMerge, LeftMerge);
699 });
700
701 for (Instruction *I : MergeInstructions) {
702 I->moveBefore(InsertionPoint);
703 InsertionPoint = I;
704 }
705
706 return true;
707 }

709 // Sorts selection merge headers in |F|.
710 // A is sorted before B if the merge block designated by B is an ancestor of
711 // the one designated by A.

720 // Split basic blocks containing multiple OpLoopMerge/OpSelectionMerge
721 // instructions so each basic block contains only a single merge instruction.
722 bool splitBlocksWithMultipleHeaders(Function &F) {
723 std::stack<BasicBlock *> Work;
724 for (auto &BB : F) {
725 std::vector<Instruction *> MergeInstructions = getMergeInstructions(BB);
726 if (MergeInstructions.size() <= 1)
727 continue;
728 Work.push(&BB);
729 }

737 getMergeInstructions(*Header);

753 }

817 // Build a divergent construct tree searching from |BB|.
818 // If |Parent| is not null, this tree is attached to the parent's tree.
819 void constructDivergentConstruct(BlockSet &Visited, Splitter &S,
820 BasicBlock *BB, DivergentConstruct *Parent) {
821 if (Visited.count(BB) != 0)
822 return;
823 Visited.insert(BB);
824
825 auto MIS = getMergeInstructions(*BB);
826 if (MIS.size() == 0) {
827 for (BasicBlock *Successor : successors(BB))
828 constructDivergentConstruct(Visited, S, Successor, Parent);
829 return;
830 }
831
832 assert(MIS.size() == 1);
833 Instruction *MI = MIS[0];
834
835 BasicBlock *Merge = getDesignatedMergeBlock(MI);
836 BasicBlock *Continue = getDesignatedContinueBlock(MI);
837
838 auto Output = std::make_unique<DivergentConstruct>();
839 Output->Header = BB;
840 Output->Merge = Merge;
841 Output->Continue = Continue;
842 Output->Parent = Parent;
843
844 constructDivergentConstruct(Visited, S, Merge, Parent);
845 if (Continue)
846 constructDivergentConstruct(Visited, S, Continue, Output.get());
847
848 for (BasicBlock *Successor : successors(BB))
849 constructDivergentConstruct(Visited, S, Successor, Output.get());
850
851 if (Parent)
852 Parent->Children.emplace_back(std::move(Output));
853 }

868 // Fixup the construct |Node| to respect a set of rules defined by the SPIR-V
869 // spec.
870 bool fixupConstruct(Splitter &S, DivergentConstruct *Node) {
871 bool Modified = false;
872 for (auto &Child : Node->Children)
873 Modified |= fixupConstruct(S, Child.get());
874
875 // This construct is the root construct. Does not represent any real
876 // construct, just a way to access the first level of the forest.
877 if (Node->Parent == nullptr)
878 return Modified;
879
880 // This node's parent is the root. Meaning this is a top-level construct.
881 // There can be multiple exists, but all are guaranteed to exit at most 1
882 // construct since we are at first level.
883 if (Node->Parent->Header == nullptr)
884 return Modified;
885
886 // Health check for the structure.
887 assert(Node->Header && Node->Merge);
888 assert(Node->Parent->Header && Node->Parent->Merge);
889
890 BlockSet ConstructBlocks = getConstructBlocks(S, Node);
891 auto Edges = getExitsFrom(ConstructBlocks, *Node->Header);
892
893 // No edges exiting the construct.
894 if (Edges.size() < 1)
895 return Modified;
896
897 bool HasBadEdge = Node->Merge == Node->Parent->Merge ||
898 Node->Merge == Node->Parent->Continue;
899 // BasicBlock *Target = Edges[0].second;
900 for (auto &[Src, Dst] : Edges) {
901 // - Breaking from a selection construct: S is a selection construct, S is
902 // the innermost structured
903 // control-flow construct containing A, and B is the merge block for S
904 // - Breaking from the innermost loop: S is the innermost loop construct
905 // containing A,
906 // and B is the merge block for S
907 if (Node->Merge == Dst)
908 continue;
909
910 // Entering the innermost loop’s continue construct: S is the innermost
911 // loop construct containing A, and B is the continue target for S
912 if (Node->Continue == Dst)
913 continue;
914
915 // TODO: what about cases branching to another case in the switch? Seems
916 // to work, but need to double check.
917 HasBadEdge = true;
918 }
919
920 if (!HasBadEdge)
921 return Modified;
922
923 // Create a single exit node gathering all exit edges.
924 BasicBlock *NewExit = S.createSingleExitNode(Node->Header, Edges);
925
926 // Fixup this construct's merge node to point to the new exit.
927 // Note: this algorithm fixes inner-most divergence construct first. So
928 // recursive structures sharing a single merge node are fixed from the
929 // inside toward the outside.
930 auto MergeInstructions = getMergeInstructions(*Node->Header);
931 assert(MergeInstructions.size() == 1);
932 Instruction *I = MergeInstructions[0];
933 BlockAddress *BA = cast<BlockAddress>(I->getOperand(0));
934 if (BA->getBasicBlock() == Node->Merge) {
935 auto MergeAddress = BlockAddress::get(NewExit->getParent(), NewExit);
936 I->setOperand(0, MergeAddress);
937 }
938
939 // Clean up of the possible dangling BockAddr operands to prevent MIR
940 // comments about "address of removed block taken".
941 if (!BA->isConstantUsed())
942 BA->destroyConstant();
943
944 Node->Merge = NewExit;
945 // Regenerate the dom trees.
946 S.invalidate();
947 return true;
948 }

960 // Simplify branches when possible:
961 // - if the 2 sides of a conditional branch are the same, transforms it to an
962 // unconditional branch.
963 // - if a switch has only 2 distinct successors, converts it to a conditional
964 // branch.
965 bool simplifyBranches(Function &F) {
966 bool Modified = false;
967
968 for (BasicBlock &BB : F) {
969 SwitchInst *SI = dyn_cast<SwitchInst>(BB.getTerminator());
970 if (!SI)
971 continue;
972 if (SI->getNumCases() > 1)
973 continue;
974
975 Modified = true;
976 IRBuilder<> Builder(&BB);
977 Builder.SetInsertPoint(SI);
978
979 if (SI->getNumCases() == 0) {
980 Builder.CreateBr(SI->getDefaultDest());
981 } else {
982 Value *Condition =
983 Builder.CreateCmp(CmpInst::ICMP_EQ, SI->getCondition(),
984 SI->case_begin()->getCaseValue());
985 Builder.CreateCondBr(Condition, SI->case_begin()->getCaseSuccessor(),
986 SI->getDefaultDest());
987 }
988 SI->eraseFromParent();
989 }
990
991 return Modified;
992 }

994 // Makes sure every case target in |F| is unique. If 2 cases branch to the
995 // same basic block, one of the targets is updated so it jumps to a new basic
996 // block ending with a single unconditional branch to the original target.
997 bool splitSwitchCases(Function &F) {
998 bool Modified = false;
999
1000 for (BasicBlock &BB : F) {
1001 SwitchInst *SI = dyn_cast<SwitchInst>(BB.getTerminator());
1002 if (!SI)
1003 continue;
1004
1005 BlockSet Seen;
1006 Seen.insert(SI->getDefaultDest());
1007
1008 auto It = SI->case_begin();
1009 while (It != SI->case_end()) {
1010 BasicBlock *Target = It->getCaseSuccessor();
1011 if (Seen.count(Target) == 0) {
1012 Seen.insert(Target);
1013 ++It;
1014 continue;
1015 }
1016
1017 Modified = true;
1018 BasicBlock *NewTarget =
1019 BasicBlock::Create(F.getContext(), "new.sw.case", &F);
1020 IRBuilder<> Builder(NewTarget);
1021 Builder.CreateBr(Target);
1022 SI->addCase(It->getCaseValue(), NewTarget);
1023 It = SI->removeCase(It);
1024 }
1025 }
1026
1027 return Modified;
1028 }

1077 bool addHeaderToRemainingDivergentDAG(Function &F) {
1078 bool Modified = false;
1079
1080 auto MergeBlocks = getMergeBlocks(F);
1081 auto ContinueBlocks = getContinueBlocks(F);
1082 auto HeaderBlocks = getHeaderBlocks(F);
1083
1084 DomTreeBuilder::BBDomTree DT;
1085 DomTreeBuilder::BBPostDomTree PDT;
1086 PDT.recalculate(F);
1087 DT.recalculate(F);
1088
1089 for (BasicBlock &BB : F) {
1090 if (HeaderBlocks.count(&BB) != 0)
1091 continue;
1092 if (succ_size(&BB) < 2)
1093 continue;
1094
1095 size_t CandidateEdges = 0;
1096 for (BasicBlock *Successor : successors(&BB)) {
1097 if (MergeBlocks.count(Successor) != 0 ||
1098 ContinueBlocks.count(Successor) != 0)
1099 continue;
1100 if (HeaderBlocks.count(Successor) != 0)
1101 continue;
1102 CandidateEdges += 1;
1103 }
1104
1105 if (CandidateEdges <= 1)
1106 continue;
1107
1108 BasicBlock *Header = &BB;
1109 BasicBlock *Merge = PDT.getNode(&BB)->getIDom()->getBlock();
1110
1111 bool HasBadBlock = false;
1112 visit(*Header, [&](const BasicBlock *Node) {
1113 if (DT.dominates(Header, Node))
1114 return false;
1115 if (PDT.dominates(Merge, Node))
1116 return false;
1117 if (Node == Header || Node == Merge)
1118 return true;
1119
1120 HasBadBlock |= MergeBlocks.count(Node) != 0 ||
1121 ContinueBlocks.count(Node) != 0 ||
1122 HeaderBlocks.count(Node) != 0;
1123 return !HasBadBlock;
1124 });
1125
1126 if (HasBadBlock)
1127 continue;
1128
1129 Modified = true;
1130 Instruction *SplitInstruction = Merge->getTerminator();
1131 if (isMergeInstruction(SplitInstruction->getPrevNode()))
1132 SplitInstruction = SplitInstruction->getPrevNode();
1133 BasicBlock *NewMerge =
1134 Merge->splitBasicBlockBefore(SplitInstruction, "new.merge");
1135
1136 IRBuilder<> Builder(Header);
1137 Builder.SetInsertPoint(Header->getTerminator());
1138
1139 auto MergeAddress = BlockAddress::get(NewMerge->getParent(), NewMerge);
1140 SmallVector<Value *, 1> Args = {MergeAddress};
1141 Builder.CreateIntrinsic(Intrinsic::spv_selection_merge, {}, {Args});
1142 }

1153
1154 virtual bool runOnFunction(Function &F) override {
1155 bool Modified = false;
1156
1157 // In LLVM, Switches are allowed to have several cases branching to the same
1158 // basic block. This is allowed in SPIR-V, but can make structurizing SPIR-V
1159 // harder, so first remove edge cases.
1160 Modified |= splitSwitchCases(F);
1161
1162 // LLVM allows conditional branches to have both side jumping to the same
1163 // block. It also allows switched to have a single default, or just one
1164 // case. Cleaning this up now.
1165 Modified |= simplifyBranches(F);
1166
1167 // At this state, we should have a reducible CFG with cycles.
1168 // STEP 1: Adding OpLoopMerge instructions to loop headers.
1169 Modified |= addMergeForLoops(F);
1170
1171 // STEP 2: adding OpSelectionMerge to each node with an in-degree >= 2.
1172 Modified |= addMergeForNodesWithMultiplePredecessors(F);
1173
1174 // STEP 3:
1175 // Sort selection merge, the largest construct goes first.
1176 // This simplifies the next step.
1177 Modified |= sortSelectionMergeHeaders(F);
1178
1179 // STEP 4: As this stage, we can have a single basic block with multiple
1180 // OpLoopMerge/OpSelectionMerge instructions. Splitting this block so each
1181 // BB has a single merge instruction.
1182 Modified |= splitBlocksWithMultipleHeaders(F);
1183
1184 // STEP 5: In the previous steps, we added merge blocks the loops and
1185 // natural merge blocks (in-degree >= 2). What remains are conditions with
1186 // an exiting branch (return, unreachable). In such case, we must start from
1187 // the header, and add headers to divergent construct with no headers.
1188 Modified |= addMergeForDivergentBlocks(F);
1189
1190 // STEP 6: At this stage, we have several divergent construct defines by a
1191 // header and a merge block. But their boundaries have no constraints: a
1192 // construct exit could be outside of the parents' construct exit. Such
1193 // edges are called critical edges. What we need is to split those edges
1194 // into several parts. Each part exiting the parent's construct by its merge
1195 // block.
1196 Modified |= splitCriticalEdges(F);
1197
1198 // STEP 7: The previous steps possibly created a lot of "proxy" blocks.
1199 // Blocks with a single unconditional branch, used to create a valid
1200 // divergent construct tree. Some nodes are still requires (e.g: nodes
1201 // allowing a valid exit through the parent's merge block). But some are
1202 // left-overs of past transformations, and could cause actual validation
1203 // issues. E.g: the SPIR-V spec allows a construct to break to the parents
1204 // loop construct without an OpSelectionMerge, but this requires a straight
1205 // jump. If a proxy block lies between the conditional branch and the
1206 // parent's merge, the CFG is not valid.
1207 Modified |= removeUselessBlocks(F);
1208
1209 // STEP 8: Final fix-up steps: our tree boundaries are correct, but some
1210 // blocks are branching with no header. Those are often simple conditional
1211 // branches with 1 or 2 returning edges. Adding a header for those.
1212 Modified |= addHeaderToRemainingDivergentDAG(F);
1213
1214 // STEP 9: sort basic blocks to match both the LLVM & SPIR-V requirements.
1215 Modified |= sortBlocks(F);
1216
1217 return Modified;
1218 }
