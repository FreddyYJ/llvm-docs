23FunctionPassManager &
24PassRegistry::parseAndCreatePassPipeline(StringRef Pipeline) {
25 static constexpr const char EndToken = '\0';
26 // Add EndToken to the end to ease parsing.
27 std::string PipelineStr = std::string(Pipeline) + EndToken;
28 int FlagBeginIdx = 0;
29 // Start with a FunctionPassManager.
30 auto &InitialPM = static_cast<FunctionPassManager &>(
31 registerPass(std::make_unique<FunctionPassManager>("init-fpm")));
32
33 for (auto [Idx, C] : enumerate(PipelineStr)) {
34 // Keep moving Idx until we find the end of the pass name.
35 bool FoundDelim = C == EndToken || C == PassDelimToken;
36 if (!FoundDelim)
37 continue;
38 unsigned Sz = Idx - FlagBeginIdx;
39 std::string PassName(&PipelineStr[FlagBeginIdx], Sz);
40 FlagBeginIdx = Idx + 1;
41
42 // Get the pass that corresponds to PassName and add it to the pass manager.
43 auto *Pass = getPassByName(PassName);
44 if (Pass == nullptr) {
45 errs() << "Pass '" << PassName << "' not registered!\n";
46 exit(1);
47 }
48 // TODO: This is safe for now, but would require proper upcasting once we
49 // add more Pass sub-classes.
50 InitialPM.addPass(static_cast<FunctionPass *>(Pass));
51 }
52 return InitialPM;
53}
