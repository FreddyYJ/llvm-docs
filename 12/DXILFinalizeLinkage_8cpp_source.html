21static bool finalizeLinkage(Module &M) {
22 SmallPtrSet<Function *, 8> EntriesAndExports;
23
24 // Find all entry points and export functions
25 for (Function &EF : M.functions()) {
26 if (!EF.hasFnAttribute("hlsl.shader") && !EF.hasFnAttribute("hlsl.export"))
27 continue;
28 EntriesAndExports.insert(&EF);
29 }
30
31 for (Function &F : M.functions()) {
32 if (F.getLinkage() == GlobalValue::ExternalLinkage &&
33 !EntriesAndExports.contains(&F)) {
34 F.setLinkage(GlobalValue::InternalLinkage);
35 }
36 }
37
38 return false;
39}
