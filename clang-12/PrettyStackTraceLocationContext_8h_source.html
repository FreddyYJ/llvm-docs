18///
19/// Example:
20/// \code
21/// 0. Program arguments: ...

24/// #0 void inlined()

26/// 3. crash-trace.c:6:3: Error evaluating statement
27/// \endcode
28class PrettyStackTraceLocationContext : public llvm::PrettyStackTraceEntry {
