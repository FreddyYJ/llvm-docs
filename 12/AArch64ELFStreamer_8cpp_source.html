165/// Extend the generic ELFStreamer class so that it can emit mapping symbols at
166/// the appropriate points in the object files. These symbols are defined in the
167/// AArch64 ELF ABI:
168/// infocenter.arm.com/help/topic/com.arm.doc.ihi0056a/IHI0056A_aaelf64.pdf
169///
170/// In brief: $x or $d should be emitted at the start of each contiguous region
171/// of A64 code or data in a section. In practice, this emission does not rely
172/// on explicit assembler directives but on inherent properties of the
173/// directives doing the emission (e.g. ".byte" is data, "add x0, x0, x0" an
174/// instruction).
175///
176/// As a result this system is orthogonal to the DataRegion infrastructure used
177/// by MachO. Beware!
178class AArch64ELFStreamer : public MCELFStreamer {
179public:
180 friend AArch64TargetELFStreamer;
181 AArch64ELFStreamer(MCContext &Context, std::unique_ptr<MCAsmBackend> TAB,
182 std::unique_ptr<MCObjectWriter> OW,
183 std::unique_ptr<MCCodeEmitter> Emitter)
184 : MCELFStreamer(Context, std::move(TAB), std::move(OW),
185 std::move(Emitter)),
186 LastEMS(EMS_None) {
187 auto *TO = getContext().getTargetOptions();
188 ImplicitMapSyms = TO && TO->ImplicitMapSyms;
189 }
190
191 void changeSection(MCSection *Section, uint32_t Subsection = 0) override {
192 // Save the mapping symbol state for potential reuse when revisiting the
193 // section. When ImplicitMapSyms is true, the initial state is
194 // EMS_A64 for text sections and EMS_Data for the others.
195 LastMappingSymbols[getCurrentSection().first] = LastEMS;
196 auto It = LastMappingSymbols.find(Section);
197 if (It != LastMappingSymbols.end())
198 LastEMS = It->second;
199 else if (ImplicitMapSyms)
200 LastEMS = Section->isText() ? EMS_A64 : EMS_Data;
201 else
202 LastEMS = EMS_None;
203
204 MCELFStreamer::changeSection(Section, Subsection);
205 }
206
207 // Reset state between object emissions
208 void reset() override {
209 MCELFStreamer::reset();
210 LastMappingSymbols.clear();
211 LastEMS = EMS_None;
212 }
213
214 /// This function is the one used to emit instruction data into the ELF
215 /// streamer. We override it to add the appropriate mapping symbol if
216 /// necessary.
217 void emitInstruction(const MCInst &Inst,
218 const MCSubtargetInfo &STI) override {
219 emitA64MappingSymbol();
220 MCELFStreamer::emitInstruction(Inst, STI);
221 }
222
223 /// Emit a 32-bit value as an instruction. This is only used for the .inst
224 /// directive, EmitInstruction should be used in other cases.
225 void emitInst(uint32_t Inst) {
226 char Buffer[4];
227
228 // We can't just use EmitIntValue here, as that will emit a data mapping
229 // symbol, and swap the endianness on big-endian systems (instructions are
230 // always little-endian).
231 for (char &C : Buffer) {
232 C = uint8_t(Inst);
233 Inst >>= 8;
234 }
235
236 emitA64MappingSymbol();
237 MCELFStreamer::emitBytes(StringRef(Buffer, 4));
238 }
239
240 /// This is one of the functions used to emit data into an ELF section, so the
241 /// AArch64 streamer overrides it to add the appropriate mapping symbol ($d)
242 /// if necessary.
243 void emitBytes(StringRef Data) override {
244 emitDataMappingSymbol();
245 MCELFStreamer::emitBytes(Data);
246 }
247
248 /// This is one of the functions used to emit data into an ELF section, so the
249 /// AArch64 streamer overrides it to add the appropriate mapping symbol ($d)
250 /// if necessary.
251 void emitValueImpl(const MCExpr *Value, unsigned Size, SMLoc Loc) override {
252 emitDataMappingSymbol();
253 MCELFStreamer::emitValueImpl(Value, Size, Loc);
254 }
255
256 void emitFill(const MCExpr &NumBytes, uint64_t FillValue,
257 SMLoc Loc) override {
258 emitDataMappingSymbol();
259 MCObjectStreamer::emitFill(NumBytes, FillValue, Loc);
260 }
261
262private:
263 enum ElfMappingSymbol {
264 EMS_None,
265 EMS_A64,
266 EMS_Data
267 };
268
269 void emitDataMappingSymbol() {
270 if (LastEMS == EMS_Data)
271 return;
272 emitMappingSymbol("$d");
273 LastEMS = EMS_Data;
274 }
275
276 void emitA64MappingSymbol() {
277 if (LastEMS == EMS_A64)
278 return;
279 emitMappingSymbol("$x");
280 LastEMS = EMS_A64;
281 }
282
283 MCSymbol *emitMappingSymbol(StringRef Name) {
284 auto *Symbol = cast<MCSymbolELF>(getContext().createLocalSymbol(Name));
285 emitLabel(Symbol);
286 return Symbol;
287 }
288
289 DenseMap<const MCSection *, ElfMappingSymbol> LastMappingSymbols;
290 ElfMappingSymbol LastEMS;
291 bool ImplicitMapSyms;
292};
293} // end anonymous namespace

308void AArch64TargetELFStreamer::finish() {
309 AArch64TargetStreamer::finish();
310 AArch64ELFStreamer &S = getStreamer();
311 MCContext &Ctx = S.getContext();
312 auto &Asm = S.getAssembler();
313
314 // If ImplicitMapSyms is specified, ensure that text sections end with
315 // the A64 state while non-text sections end with the data state. When
316 // sections are combined by the linker, the subsequent section will start with
317 // the right state. The ending mapping symbol is added right after the last
318 // symbol relative to the section. When a dumb linker combines (.text.0; .word
319 // 0) and (.text.1; .word 0), the ending $x of .text.0 precedes the $d of
320 // .text.1, even if they have the same address.
321 if (S.ImplicitMapSyms) {
322 auto &Syms = Asm.getSymbols();
323 const size_t NumSyms = Syms.size();
324 DenseMap<MCSection *, std::pair<size_t, MCSymbol *>> EndMapSym;
325 for (MCSection &Sec : Asm) {
326 S.switchSection(&Sec);
327 if (S.LastEMS == (Sec.isText() ? AArch64ELFStreamer::EMS_Data
328 : AArch64ELFStreamer::EMS_A64))
329 EndMapSym.insert(
330 {&Sec, {NumSyms, S.emitMappingSymbol(Sec.isText() ? "$x" : "$d")}});
331 }
332 if (Syms.size() != NumSyms) {
333 SmallVector<const MCSymbol *, 0> NewSyms;
334 DenseMap<MCSection *, size_t> Cnt;
335 Syms.truncate(NumSyms);
336 // Find the last symbol index for each candidate section.
337 for (auto [I, Sym] : llvm::enumerate(Syms)) {
338 if (!Sym->isInSection())
339 continue;
340 auto It = EndMapSym.find(&Sym->getSection());
341 if (It != EndMapSym.end())
342 It->second.first = I;
343 }
344 SmallVector<size_t, 0> Idx;
345 for (auto [I, Sym] : llvm::enumerate(Syms)) {
346 NewSyms.push_back(Sym);
347 if (!Sym->isInSection())
348 continue;
349 auto It = EndMapSym.find(&Sym->getSection());
350 // If `Sym` is the last symbol relative to the section, add the ending
351 // mapping symbol after `Sym`.
352 if (It != EndMapSym.end() && I == It->second.first) {
353 NewSyms.push_back(It->second.second);
354 Idx.push_back(I);
355 }
356 }
357 Syms = std::move(NewSyms);
358 // F.second holds the number of symbols added before the FILE symbol.
359 // Take into account the inserted mapping symbols.
360 for (auto &F : S.getWriter().getFileNames())
361 F.second += llvm::lower_bound(Idx, F.second) - Idx.begin();
362 }
363 }
364
365 MCSectionELF *MemtagSec = nullptr;
366 for (const MCSymbol &Symbol : Asm.symbols()) {
367 const auto &Sym = cast<MCSymbolELF>(Symbol);
368 if (Sym.isMemtag()) {
369 MemtagSec = Ctx.getELFSection(".memtag.globals.static",
370 ELF::SHT_AARCH64_MEMTAG_GLOBALS_STATIC, 0);
371 break;
372 }
373 }
374 if (!MemtagSec)
375 return;
376
377 // switchSection registers the section symbol and invalidates symbols(). We
378 // need a separate symbols() loop.
379 S.switchSection(MemtagSec);
380 const auto *Zero = MCConstantExpr::create(0, Ctx);
381 for (const MCSymbol &Symbol : Asm.symbols()) {
382 const auto &Sym = cast<MCSymbolELF>(Symbol);
383 if (!Sym.isMemtag())
384 continue;
385 auto *SRE = MCSymbolRefExpr::create(&Sym, MCSymbolRefExpr::VK_None, Ctx);
386 (void)S.emitRelocDirective(*Zero, "BFD_RELOC_NONE", SRE, SMLoc(),
387 *Ctx.getSubtargetInfo());
388 }
389}
