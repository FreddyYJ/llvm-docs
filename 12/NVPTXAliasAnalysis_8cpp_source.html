56static unsigned getAddressSpace(const Value *V, unsigned MaxLookup) {
57 // Find the first non-generic address space traversing the UD chain.
58 // It is undefined behaviour if a pointer belongs to more than one
59 // non-overlapping address spaces along a valid execution path.
60 auto GetAS = [](const Value *V) -> unsigned {
61 if (const auto *PTy = dyn_cast<PointerType>(V->getType()))
62 return PTy->getAddressSpace();
63 return ADDRESS_SPACE_GENERIC;
64 };
65 while (MaxLookup-- && GetAS(V) == ADDRESS_SPACE_GENERIC) {
66 const Value *NewV = getUnderlyingObject(V, 1);
67 if (NewV == V)
68 break;
69 V = NewV;
70 }
71 return GetAS(V);
72}

74static AliasResult::Kind getAliasResult(unsigned AS1, unsigned AS2) {
75 if ((AS1 == ADDRESS_SPACE_GENERIC) || (AS2 == ADDRESS_SPACE_GENERIC))
76 return AliasResult::MayAlias;
77
78 // PTX s6.4.1.1. Generic Addressing:
79 // A generic address maps to global memory unless it falls within
80 // the window for const, local, or shared memory. The Kernel
81 // Function Parameters (.param) window is contained within the
82 // .global window.
83 //
84 // Therefore a global pointer may alias with a param pointer on some
85 // GPUs via addrspacecast(param->generic->global) when cvta.param
86 // instruction is used (PTX 7.7+ and SM_70+).
87 //
88 // TODO: cvta.param is not yet supported. We need to change aliasing
89 // rules once it is added.
90
91 return (AS1 == AS2 ? AliasResult::MayAlias : AliasResult::NoAlias);
92}

94AliasResult NVPTXAAResult::alias(const MemoryLocation &Loc1,
95 const MemoryLocation &Loc2, AAQueryInfo &AAQI,
96 const Instruction *) {
97 unsigned AS1 = getAddressSpace(Loc1.Ptr, TraverseAddressSpacesLimit);
98 unsigned AS2 = getAddressSpace(Loc2.Ptr, TraverseAddressSpacesLimit);
99
100 return getAliasResult(AS1, AS2);
101}

106static bool isConstOrParam(unsigned AS) {
107 return AS == AddressSpace::ADDRESS_SPACE_CONST ||
108 AS == AddressSpace::ADDRESS_SPACE_PARAM;
