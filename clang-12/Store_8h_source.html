293inline StoreRef::StoreRef(Store store, StoreManager & smgr)
294 : store(store), mgr(smgr) {
295 if (store)
296 mgr.incrementReferenceCount(store);
297}
298
299inline StoreRef::StoreRef(const StoreRef &sr)
300 : store(sr.store), mgr(sr.mgr)
301{
302 if (store)
303 mgr.incrementReferenceCount(store);
304}
305
306inline StoreRef::~StoreRef() {
307 if (store)
308 mgr.decrementReferenceCount(store);
309}
310
311inline StoreRef &StoreRef::operator=(StoreRef const &newStore) {
312 assert(&newStore.mgr == &mgr);
313 if (store != newStore.store) {
314 mgr.incrementReferenceCount(newStore.store);
315 mgr.decrementReferenceCount(store);
316 store = newStore.getStore();
317 }
318 return *this;
319}
