54StructType *StructType::get(Context &Ctx, ArrayRef<Type *> Elements,
55 bool IsPacked) {
56 SmallVector<llvm::Type *> LLVMElements;
57 LLVMElements.reserve(Elements.size());
58 for (Type *Elm : Elements)
59 LLVMElements.push_back(Elm->LLVMTy);
60 return cast<StructType>(
61 Ctx.getType(llvm::StructType::get(Ctx.LLVMCtx, LLVMElements, IsPacked)));
62}
