26/// The legality outcome is represented by a class rather than an enum class
27/// because in some cases the legality checks are expensive and look for a
28/// particular instruction that can be passed along to the vectorizer to avoid
29/// repeating the same expensive computation.
30class LegalityResult {
31protected:
32 LegalityResultID ID;
33 /// Only Legality can create LegalityResults.
34 LegalityResult(LegalityResultID ID) : ID(ID) {}
35 friend class LegalityAnalysis;
36
37public:
38 LegalityResultID getSubclassID() const { return ID; }
39};
40
41class Widen final : public LegalityResult {
42 friend class LegalityAnalysis;
43 Widen() : LegalityResult(LegalityResultID::Widen) {}
44
45public:
46 static bool classof(const LegalityResult *From) {
47 return From->getSubclassID() == LegalityResultID::Widen;
48 }
49};
