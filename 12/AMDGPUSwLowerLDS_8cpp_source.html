9// This pass lowers the local data store, LDS, uses in kernel and non-kernel
10// functions in module to use dynamically allocated global memory.
11// Packed LDS Layout is emulated in the global memory.
12// The lowered memory instructions from LDS to global memory are then
13// instrumented for address sanitizer, to catch addressing errors.
14// This pass only work when address sanitizer has been enabled and has
15// instrumented the IR. It identifies that IR has been instrumented using
16// "nosanitize_address" module flag.
17//
18// Replacement of Kernel LDS accesses:
19// For a kernel, LDS access can be static or dynamic which are direct
20// (accessed within kernel) and indirect (accessed through non-kernels).
21// All these LDS accesses corresponding to kernel will be packed together,
22// where all static LDS accesses will be allocated first and then dynamic
23// LDS follows. The total size with alignment is calculated. A new LDS global
24// will be created for the kernel called "SW LDS" and it will have the
25// attribute "amdgpu-lds-size" attached with value of the size calculated.
26// All the LDS accesses in the module will be replaced by GEP with offset
27// into the "Sw LDS".
28// A new "llvm.amdgcn.<kernel>.dynlds" is created per kernel accessing
29// the dynamic LDS. This will be marked used by kernel and will have
30// MD_absolue_symbol metadata set to total static LDS size, Since dynamic
31// LDS allocation starts after all static LDS allocation.
32//
33// A device global memory equal to the total LDS size will be allocated.
34// At the prologue of the kernel, a single work-item from the
35// work-group, does a "malloc" and stores the pointer of the
36// allocation in "SW LDS".
37//
38// To store the offsets corresponding to all LDS accesses, another global
39// variable is created which will be called "SW LDS metadata" in this pass.
40// - SW LDS Global:
41// It is LDS global of ptr type with name
42// "llvm.amdgcn.sw.lds.<kernel-name>".
43// - Metadata Global:
44// It is of struct type, with n members. n equals the number of LDS
45// globals accessed by the kernel(direct and indirect). Each member of
46// struct is another struct of type {i32, i32, i32}. First member
47// corresponds to offset, second member corresponds to size of LDS global
48// being replaced and third represents the total aligned size. It will
49// have name "llvm.amdgcn.sw.lds.<kernel-name>.md". This global will have
50// an intializer with static LDS related offsets and sizes initialized.
51// But for dynamic LDS related entries, offsets will be intialized to
52// previous static LDS allocation end offset. Sizes for them will be zero
53// initially. These dynamic LDS offset and size values will be updated
54// within the kernel, since kernel can read the dynamic LDS size
55// allocation done at runtime with query to "hidden_dynamic_lds_size"
56// hidden kernel argument.
57//
58// At the epilogue of kernel, allocated memory would be made free by the same
59// single work-item.
60//
61// Replacement of non-kernel LDS accesses:
62// Multiple kernels can access the same non-kernel function.
63// All the kernels accessing LDS through non-kernels are sorted and
64// assigned a kernel-id. All the LDS globals accessed by non-kernels
65// are sorted. This information is used to build two tables:
66// - Base table:
67// Base table will have single row, with elements of the row
68// placed as per kernel ID. Each element in the row corresponds
69// to ptr of "SW LDS" variable created for that kernel.
70// - Offset table:
71// Offset table will have multiple rows and columns.
72// Rows are assumed to be from 0 to (n-1). n is total number
73// of kernels accessing the LDS through non-kernels.
74// Each row will have m elements. m is the total number of
75// unique LDS globals accessed by all non-kernels.
76// Each element in the row correspond to the ptr of
77// the replacement of LDS global done by that particular kernel.
78// A LDS variable in non-kernel will be replaced based on the information
79// from base and offset tables. Based on kernel-id query, ptr of "SW
80// LDS" for that corresponding kernel is obtained from base table.
81// The Offset into the base "SW LDS" is obtained from
82// corresponding element in offset table. With this information, replacement
83// value is obtained.

242SetVector<Function *> AMDGPUSwLowerLDS::getOrderedIndirectLDSAccessingKernels(
243 SetVector<Function *> &Kernels) {
244 // Sort the non-kernels accessing LDS based on their name.
245 // Also assign a kernel ID metadata based on the sorted order.
246 LLVMContext &Ctx = M.getContext();
247 if (Kernels.size() > UINT32_MAX) {
248 report_fatal_error("Unimplemented SW LDS lowering for > 2**32 kernels");
249 }
250 SetVector<Function *> OrderedKernels =
251 sortByName(std::vector<Function *>(Kernels.begin(), Kernels.end()));
252 for (size_t i = 0; i < Kernels.size(); i++) {
253 Metadata *AttrMDArgs[1] = {
254 ConstantAsMetadata::get(IRB.getInt32(i)),
255 };
256 Function *Func = OrderedKernels[i];
257 Func->setMetadata("llvm.amdgcn.lds.kernel.id",
258 MDNode::get(Ctx, AttrMDArgs));
259 }
260 return OrderedKernels;
261}

263void AMDGPUSwLowerLDS::getNonKernelsWithLDSArguments(const CallGraph &CG) {
264 // Among the kernels accessing LDS, get list of
265 // Non-kernels to which a call is made and a ptr
266 // to addrspace(3) is passed as argument.
267 for (auto &K : FuncLDSAccessInfo.KernelToLDSParametersMap) {
268 Function *Func = K.first;
269 const CallGraphNode *CGN = CG[Func];
270 if (!CGN)
271 continue;
272 for (auto &I : *CGN) {
273 CallGraphNode *CallerCGN = I.second;
274 Function *CalledFunc = CallerCGN->getFunction();
275 if (!CalledFunc || CalledFunc->isDeclaration())
276 continue;
277 if (AMDGPU::isKernelLDS(CalledFunc))
278 continue;
279 for (auto AI = CalledFunc->arg_begin(), E = CalledFunc->arg_end();
280 AI != E; ++AI) {
281 Type *ArgTy = (*AI).getType();
282 if (!ArgTy->isPointerTy())
283 continue;
284 if (ArgTy->getPointerAddressSpace() != AMDGPUAS::LOCAL_ADDRESS)
285 continue;
286 FuncLDSAccessInfo.NonKernelsWithLDSArgument.insert(CalledFunc);
287 // Also add the Calling function to KernelsWithIndirectLDSAccess list
288 // so that base table of LDS is generated.
289 FuncLDSAccessInfo.KernelsWithIndirectLDSAccess.insert(Func);
290 }
291 }
292 }

311static void recordLDSAbsoluteAddress(Module &M, GlobalVariable *GV,
312 uint32_t Address) {
313 // Write the specified address into metadata where it can be retrieved by
314 // the assembler. Format is a half open range, [Address Address+1)
315 LLVMContext &Ctx = M.getContext();
316 auto *IntTy = M.getDataLayout().getIntPtrType(Ctx, AMDGPUAS::LOCAL_ADDRESS);
317 MDBuilder MDB(Ctx);
318 MDNode *MetadataNode = MDB.createRange(ConstantInt::get(IntTy, Address),
319 ConstantInt::get(IntTy, Address + 1));
320 GV->setMetadata(LLVMContext::MD_absolute_symbol, MetadataNode);
321}

393void AMDGPUSwLowerLDS::populateSwMetadataGlobal(Function *Func) {
394 // Create new metadata global for every kernel and initialize the
395 // start offsets and sizes corresponding to each LDS accesses.
396 auto &LDSParams = FuncLDSAccessInfo.KernelToLDSParametersMap[Func];
397 auto &Ctx = M.getContext();
398 auto &DL = M.getDataLayout();
399 std::vector<Type *> Items;
400 Type *Int32Ty = IRB.getInt32Ty();
401 std::vector<Constant *> Initializers;
402 Align MaxAlignment(1);
403 auto UpdateMaxAlignment = [&MaxAlignment, &DL](GlobalVariable *GV) {
404 Align GVAlign = AMDGPU::getAlign(DL, GV);
405 MaxAlignment = std::max(MaxAlignment, GVAlign);
406 };
407
408 for (GlobalVariable *GV : LDSParams.DirectAccess.StaticLDSGlobals)
409 UpdateMaxAlignment(GV);
410
411 for (GlobalVariable *GV : LDSParams.DirectAccess.DynamicLDSGlobals)
412 UpdateMaxAlignment(GV);
413
414 for (GlobalVariable *GV : LDSParams.IndirectAccess.StaticLDSGlobals)
415 UpdateMaxAlignment(GV);
416
417 for (GlobalVariable *GV : LDSParams.IndirectAccess.DynamicLDSGlobals)
418 UpdateMaxAlignment(GV);
419
420 //{StartOffset, AlignedSizeInBytes}
421 SmallString<128> MDItemStr;
422 raw_svector_ostream MDItemOS(MDItemStr);
423 MDItemOS << "llvm.amdgcn.sw.lds." << Func->getName() << ".md.item";
424
425 StructType *LDSItemTy =
426 StructType::create(Ctx, {Int32Ty, Int32Ty, Int32Ty}, MDItemOS.str());
427 uint32_t &MallocSize = LDSParams.MallocSize;
428 SetVector<GlobalVariable *> UniqueLDSGlobals;
429 int AsanScale = AsanInfo.Scale;
430 auto buildInitializerForSwLDSMD =
431 [&](SetVector<GlobalVariable *> &LDSGlobals) {
432 for (auto &GV : LDSGlobals) {
433 if (is_contained(UniqueLDSGlobals, GV))
434 continue;
435 UniqueLDSGlobals.insert(GV);
436
437 Type *Ty = GV->getValueType();
438 const uint64_t SizeInBytes = DL.getTypeAllocSize(Ty);
439 Items.push_back(LDSItemTy);
440 Constant *ItemStartOffset = ConstantInt::get(Int32Ty, MallocSize);
441 Constant *SizeInBytesConst = ConstantInt::get(Int32Ty, SizeInBytes);
442 // Get redzone size corresponding a size.
443 const uint64_t RightRedzoneSize =
444 AMDGPU::getRedzoneSizeForGlobal(AsanScale, SizeInBytes);
445 // Update MallocSize with current size and redzone size.
446 MallocSize += SizeInBytes;
447 if (!AMDGPU::isDynamicLDS(*GV))
448 LDSParams.RedzoneOffsetAndSizeVector.emplace_back(MallocSize,
449 RightRedzoneSize);
450 MallocSize += RightRedzoneSize;
451 // Align current size plus redzone.
452 uint64_t AlignedSize =
453 alignTo(SizeInBytes + RightRedzoneSize, MaxAlignment);
454 Constant *AlignedSizeInBytesConst =
455 ConstantInt::get(Int32Ty, AlignedSize);
456 // Align MallocSize
457 MallocSize = alignTo(MallocSize, MaxAlignment);
458 Constant *InitItem =
459 ConstantStruct::get(LDSItemTy, {ItemStartOffset, SizeInBytesConst,
460 AlignedSizeInBytesConst});
461 Initializers.push_back(InitItem);
462 }
463 };
464 SetVector<GlobalVariable *> SwLDSVector;
465 SwLDSVector.insert(LDSParams.SwLDS);
466 buildInitializerForSwLDSMD(SwLDSVector);
467 buildInitializerForSwLDSMD(LDSParams.DirectAccess.StaticLDSGlobals);
468 buildInitializerForSwLDSMD(LDSParams.IndirectAccess.StaticLDSGlobals);
469 buildInitializerForSwLDSMD(LDSParams.DirectAccess.DynamicLDSGlobals);
470 buildInitializerForSwLDSMD(LDSParams.IndirectAccess.DynamicLDSGlobals);
471
472 // Update the LDS size used by the kernel.
473 Type *Ty = LDSParams.SwLDS->getValueType();
474 const uint64_t SizeInBytes = DL.getTypeAllocSize(Ty);
475 uint64_t AlignedSize = alignTo(SizeInBytes, MaxAlignment);
476 LDSParams.LDSSize = AlignedSize;
477 SmallString<128> MDTypeStr;
478 raw_svector_ostream MDTypeOS(MDTypeStr);
479 MDTypeOS << "llvm.amdgcn.sw.lds." << Func->getName() << ".md.type";
480 StructType *MetadataStructType =
481 StructType::create(Ctx, Items, MDTypeOS.str());
482 SmallString<128> MDStr;
483 raw_svector_ostream MDOS(MDStr);
484 MDOS << "llvm.amdgcn.sw.lds." << Func->getName() << ".md";
485 LDSParams.SwLDSMetadata = new GlobalVariable(
486 M, MetadataStructType, false, GlobalValue::InternalLinkage,
487 PoisonValue::get(MetadataStructType), MDOS.str(), nullptr,
488 GlobalValue::NotThreadLocal, AMDGPUAS::GLOBAL_ADDRESS, false);
489 Constant *data = ConstantStruct::get(MetadataStructType, Initializers);
490 LDSParams.SwLDSMetadata->setInitializer(data);
491 assert(LDSParams.SwLDS);
492 // Set the alignment to MaxAlignment for SwLDS.
493 LDSParams.SwLDS->setAlignment(MaxAlignment);
494 if (LDSParams.SwDynLDS)
495 LDSParams.SwDynLDS->setAlignment(MaxAlignment);
496 GlobalValue::SanitizerMetadata MD;
497 MD.NoAddress = true;
498 LDSParams.SwLDSMetadata->setSanitizerMetadata(MD);
499 return;
500}

502void AMDGPUSwLowerLDS::populateLDSToReplacementIndicesMap(Function *Func) {
503 // Fill the corresponding LDS replacement indices for each LDS access
504 // related to this kernel.
505 auto &LDSParams = FuncLDSAccessInfo.KernelToLDSParametersMap[Func];
506 SetVector<GlobalVariable *> UniqueLDSGlobals;

523 PopulateIndices(LDSParams.DirectAccess.DynamicLDSGlobals, Idx);
524 PopulateIndices(LDSParams.IndirectAccess.DynamicLDSGlobals, Idx);
525 return;

543void AMDGPUSwLowerLDS::replaceKernelLDSAccesses(Function *Func) {
544 auto &LDSParams = FuncLDSAccessInfo.KernelToLDSParametersMap[Func];
545 GlobalVariable *SwLDS = LDSParams.SwLDS;
546 assert(SwLDS);
547 GlobalVariable *SwLDSMetadata = LDSParams.SwLDSMetadata;
548 assert(SwLDSMetadata);
549 StructType *SwLDSMetadataStructType =
550 cast<StructType>(SwLDSMetadata->getValueType());
551 Type *Int32Ty = IRB.getInt32Ty();
552 auto &IndirectAccess = LDSParams.IndirectAccess;
553 auto &DirectAccess = LDSParams.DirectAccess;
554 // Replace all uses of LDS global in this Function with a Replacement.
555 SetVector<GlobalVariable *> UniqueLDSGlobals;
556 auto ReplaceLDSGlobalUses = [&](SetVector<GlobalVariable *> &LDSGlobals) {
557 for (auto &GV : LDSGlobals) {
558 // Do not generate instructions if LDS access is in non-kernel
559 // i.e indirect-access.
560 if ((IndirectAccess.StaticLDSGlobals.contains(GV) ||
561 IndirectAccess.DynamicLDSGlobals.contains(GV)) &&
562 (!DirectAccess.StaticLDSGlobals.contains(GV) &&
563 !DirectAccess.DynamicLDSGlobals.contains(GV)))
564 continue;
565 if (is_contained(UniqueLDSGlobals, GV))
566 continue;
567 UniqueLDSGlobals.insert(GV);
568 auto &Indices = LDSParams.LDSToReplacementIndicesMap[GV];
569 assert(Indices.size() == 3);
570 Constant *GEPIdx[] = {ConstantInt::get(Int32Ty, Indices[0]),
571 ConstantInt::get(Int32Ty, Indices[1]),
572 ConstantInt::get(Int32Ty, Indices[2])};
573 Constant *GEP = ConstantExpr::getGetElementPtr(
574 SwLDSMetadataStructType, SwLDSMetadata, GEPIdx, true);
575 Value *Offset = IRB.CreateLoad(Int32Ty, GEP);
576 Value *BasePlusOffset =
577 IRB.CreateInBoundsGEP(IRB.getInt8Ty(), SwLDS, {Offset});
578 LLVM_DEBUG(GV->printAsOperand(dbgs() << "Sw LDS Lowering, Replacing LDS ",
579 false));
580 replacesUsesOfGlobalInFunction(Func, GV, BasePlusOffset);
581 }
582 };
583 ReplaceLDSGlobalUses(DirectAccess.StaticLDSGlobals);
584 ReplaceLDSGlobalUses(IndirectAccess.StaticLDSGlobals);
585 ReplaceLDSGlobalUses(DirectAccess.DynamicLDSGlobals);
586 ReplaceLDSGlobalUses(IndirectAccess.DynamicLDSGlobals);
587}

589void AMDGPUSwLowerLDS::updateMallocSizeForDynamicLDS(
590 Function *Func, Value **CurrMallocSize, Value *HiddenDynLDSSize,
591 SetVector<GlobalVariable *> &DynamicLDSGlobals) {
592 auto &LDSParams = FuncLDSAccessInfo.KernelToLDSParametersMap[Func];
593 Type *Int32Ty = IRB.getInt32Ty();
594
595 GlobalVariable *SwLDS = LDSParams.SwLDS;
596 GlobalVariable *SwLDSMetadata = LDSParams.SwLDSMetadata;
597 assert(SwLDS && SwLDSMetadata);
598 StructType *MetadataStructType =
599 cast<StructType>(SwLDSMetadata->getValueType());
600 unsigned MaxAlignment = SwLDS->getAlignment();
601 Value *MaxAlignValue = IRB.getInt32(MaxAlignment);
602 Value *MaxAlignValueMinusOne = IRB.getInt32(MaxAlignment - 1);
603
604 for (GlobalVariable *DynGV : DynamicLDSGlobals) {
605 auto &Indices = LDSParams.LDSToReplacementIndicesMap[DynGV];
606 // Update the Offset metadata.
607 Constant *Index0 = ConstantInt::get(Int32Ty, 0);
608 Constant *Index1 = ConstantInt::get(Int32Ty, Indices[1]);
609
610 Constant *Index2Offset = ConstantInt::get(Int32Ty, 0);
611 auto *GEPForOffset = IRB.CreateInBoundsGEP(
612 MetadataStructType, SwLDSMetadata, {Index0, Index1, Index2Offset});
613
614 IRB.CreateStore(*CurrMallocSize, GEPForOffset);
615 // Update the size and Aligned Size metadata.
616 Constant *Index2Size = ConstantInt::get(Int32Ty, 1);
617 auto *GEPForSize = IRB.CreateInBoundsGEP(MetadataStructType, SwLDSMetadata,
618 {Index0, Index1, Index2Size});
619
620 Value *CurrDynLDSSize = IRB.CreateLoad(Int32Ty, HiddenDynLDSSize);
621 IRB.CreateStore(CurrDynLDSSize, GEPForSize);
622 Constant *Index2AlignedSize = ConstantInt::get(Int32Ty, 2);
623 auto *GEPForAlignedSize = IRB.CreateInBoundsGEP(
624 MetadataStructType, SwLDSMetadata, {Index0, Index1, Index2AlignedSize});
625
626 Value *AlignedDynLDSSize =
627 IRB.CreateAdd(CurrDynLDSSize, MaxAlignValueMinusOne);
628 AlignedDynLDSSize = IRB.CreateUDiv(AlignedDynLDSSize, MaxAlignValue);
629 AlignedDynLDSSize = IRB.CreateMul(AlignedDynLDSSize, MaxAlignValue);
630 IRB.CreateStore(AlignedDynLDSSize, GEPForAlignedSize);
631
632 // Update the Current Malloc Size
633 *CurrMallocSize = IRB.CreateAdd(*CurrMallocSize, AlignedDynLDSSize);
634 }

679void AMDGPUSwLowerLDS::translateLDSMemoryOperationsToGlobalMemory(
680 Function *Func, Value *LoadMallocPtr,
681 SetVector<Instruction *> &LDSInstructions) {
682 LLVM_DEBUG(dbgs() << "Translating LDS memory operations to global memory : "
683 << Func->getName());
684 for (Instruction *Inst : LDSInstructions) {
685 IRB.SetInsertPoint(Inst);
686 if (LoadInst *LI = dyn_cast<LoadInst>(Inst)) {
687 Value *LIOperand = LI->getPointerOperand();
688 Value *Replacement =
689 getTranslatedGlobalMemoryGEPOfLDSPointer(LoadMallocPtr, LIOperand);
690 LoadInst *NewLI = IRB.CreateAlignedLoad(LI->getType(), Replacement,
691 LI->getAlign(), LI->isVolatile());
692 NewLI->setAtomic(LI->getOrdering(), LI->getSyncScopeID());
693 AsanInfo.Instructions.insert(NewLI);
694 LI->replaceAllUsesWith(NewLI);
695 LI->eraseFromParent();
696 } else if (StoreInst *SI = dyn_cast<StoreInst>(Inst)) {
697 Value *SIOperand = SI->getPointerOperand();
698 Value *Replacement =
699 getTranslatedGlobalMemoryGEPOfLDSPointer(LoadMallocPtr, SIOperand);
700 StoreInst *NewSI = IRB.CreateAlignedStore(
701 SI->getValueOperand(), Replacement, SI->getAlign(), SI->isVolatile());
702 NewSI->setAtomic(SI->getOrdering(), SI->getSyncScopeID());
703 AsanInfo.Instructions.insert(NewSI);
704 SI->replaceAllUsesWith(NewSI);
705 SI->eraseFromParent();
706 } else if (AtomicRMWInst *RMW = dyn_cast<AtomicRMWInst>(Inst)) {
707 Value *RMWPtrOperand = RMW->getPointerOperand();
708 Value *RMWValOperand = RMW->getValOperand();
709 Value *Replacement = getTranslatedGlobalMemoryGEPOfLDSPointer(
710 LoadMallocPtr, RMWPtrOperand);
711 AtomicRMWInst *NewRMW = IRB.CreateAtomicRMW(
712 RMW->getOperation(), Replacement, RMWValOperand, RMW->getAlign(),
713 RMW->getOrdering(), RMW->getSyncScopeID());
714 NewRMW->setVolatile(RMW->isVolatile());
715 AsanInfo.Instructions.insert(NewRMW);
716 RMW->replaceAllUsesWith(NewRMW);
717 RMW->eraseFromParent();
718 } else if (AtomicCmpXchgInst *XCHG = dyn_cast<AtomicCmpXchgInst>(Inst)) {
719 Value *XCHGPtrOperand = XCHG->getPointerOperand();
720 Value *Replacement = getTranslatedGlobalMemoryGEPOfLDSPointer(
721 LoadMallocPtr, XCHGPtrOperand);
722 AtomicCmpXchgInst *NewXCHG = IRB.CreateAtomicCmpXchg(
723 Replacement, XCHG->getCompareOperand(), XCHG->getNewValOperand(),
724 XCHG->getAlign(), XCHG->getSuccessOrdering(),
725 XCHG->getFailureOrdering(), XCHG->getSyncScopeID());
726 NewXCHG->setVolatile(XCHG->isVolatile());
727 AsanInfo.Instructions.insert(NewXCHG);
728 XCHG->replaceAllUsesWith(NewXCHG);
729 XCHG->eraseFromParent();
730 } else
731 report_fatal_error("Unimplemented LDS lowering instruction");
732 }

757void AMDGPUSwLowerLDS::lowerKernelLDSAccesses(Function *Func,
758 DomTreeUpdater &DTU) {
759 LLVM_DEBUG(dbgs() << "Sw Lowering Kernel LDS for : " << Func->getName());
760 auto &LDSParams = FuncLDSAccessInfo.KernelToLDSParametersMap[Func];
761 auto &Ctx = M.getContext();
762 auto *PrevEntryBlock = &Func->getEntryBlock();
763 SetVector<Instruction *> LDSInstructions;
764 getLDSMemoryInstructions(Func, LDSInstructions);
765
766 // Create malloc block.
767 auto *MallocBlock = BasicBlock::Create(Ctx, "Malloc", Func, PrevEntryBlock);
768
769 // Create WIdBlock block which has instructions related to selection of
770 // {0,0,0} indiex work item in the work group.
771 auto *WIdBlock = BasicBlock::Create(Ctx, "WId", Func, MallocBlock);
772 IRB.SetInsertPoint(WIdBlock, WIdBlock->begin());
773 DebugLoc FirstDL =
774 getOrCreateDebugLoc(&*PrevEntryBlock->begin(), Func->getSubprogram());
775 IRB.SetCurrentDebugLocation(FirstDL);
776 Value *WIdx = IRB.CreateIntrinsic(Intrinsic::amdgcn_workitem_id_x, {}, {});
777 Value *WIdy = IRB.CreateIntrinsic(Intrinsic::amdgcn_workitem_id_y, {}, {});
778 Value *WIdz = IRB.CreateIntrinsic(Intrinsic::amdgcn_workitem_id_z, {}, {});
779 Value *XYOr = IRB.CreateOr(WIdx, WIdy);
780 Value *XYZOr = IRB.CreateOr(XYOr, WIdz);
781 Value *WIdzCond = IRB.CreateICmpEQ(XYZOr, IRB.getInt32(0));
782
783 // All work items will branch to PrevEntryBlock except {0,0,0} index
784 // work item which will branch to malloc block.
785 IRB.CreateCondBr(WIdzCond, MallocBlock, PrevEntryBlock);
786
787 // Malloc block
788 IRB.SetInsertPoint(MallocBlock, MallocBlock->begin());
789
790 // If Dynamic LDS globals are accessed by the kernel,
791 // Get the size of dyn lds from hidden dyn_lds_size kernel arg.
792 // Update the corresponding metadata global entries for this dyn lds global.
793 GlobalVariable *SwLDS = LDSParams.SwLDS;
794 GlobalVariable *SwLDSMetadata = LDSParams.SwLDSMetadata;
795 assert(SwLDS && SwLDSMetadata);
796 StructType *MetadataStructType =
797 cast<StructType>(SwLDSMetadata->getValueType());
798 uint32_t MallocSize = 0;
799 Value *CurrMallocSize;
800 Type *Int32Ty = IRB.getInt32Ty();
801 Type *Int64Ty = IRB.getInt64Ty();
802
803 SetVector<GlobalVariable *> UniqueLDSGlobals;
804 auto GetUniqueLDSGlobals = [&](SetVector<GlobalVariable *> &LDSGlobals) {
805 for (auto &GV : LDSGlobals) {
806 if (is_contained(UniqueLDSGlobals, GV))
807 continue;
808 UniqueLDSGlobals.insert(GV);
809 }
810 };
811
812 GetUniqueLDSGlobals(LDSParams.DirectAccess.StaticLDSGlobals);
813 GetUniqueLDSGlobals(LDSParams.IndirectAccess.StaticLDSGlobals);
814 unsigned NumStaticLDS = 1 + UniqueLDSGlobals.size();
815 UniqueLDSGlobals.clear();
816
817 if (NumStaticLDS) {
818 auto *GEPForEndStaticLDSOffset =
819 IRB.CreateInBoundsGEP(MetadataStructType, SwLDSMetadata,
820 {ConstantInt::get(Int32Ty, 0),
821 ConstantInt::get(Int32Ty, NumStaticLDS - 1),
822 ConstantInt::get(Int32Ty, 0)});
823
824 auto *GEPForEndStaticLDSSize =
825 IRB.CreateInBoundsGEP(MetadataStructType, SwLDSMetadata,
826 {ConstantInt::get(Int32Ty, 0),
827 ConstantInt::get(Int32Ty, NumStaticLDS - 1),
828 ConstantInt::get(Int32Ty, 2)});
829
830 Value *EndStaticLDSOffset =
831 IRB.CreateLoad(Int32Ty, GEPForEndStaticLDSOffset);
832 Value *EndStaticLDSSize = IRB.CreateLoad(Int32Ty, GEPForEndStaticLDSSize);
833 CurrMallocSize = IRB.CreateAdd(EndStaticLDSOffset, EndStaticLDSSize);
834 } else
835 CurrMallocSize = IRB.getInt32(MallocSize);
836
837 if (LDSParams.SwDynLDS) {
838 if (!(AMDGPU::getAMDHSACodeObjectVersion(M) >= AMDGPU::AMDHSA_COV5))
839 report_fatal_error(
840 "Dynamic LDS size query is only supported for CO V5 and later.");
841 // Get size from hidden dyn_lds_size argument of kernel
842 Value *ImplicitArg =
843 IRB.CreateIntrinsic(Intrinsic::amdgcn_implicitarg_ptr, {}, {});
844 Value *HiddenDynLDSSize = IRB.CreateInBoundsGEP(
845 ImplicitArg->getType(), ImplicitArg,
846 {ConstantInt::get(Int64Ty, COV5_HIDDEN_DYN_LDS_SIZE_ARG)});
847 UniqueLDSGlobals.clear();
848 GetUniqueLDSGlobals(LDSParams.DirectAccess.DynamicLDSGlobals);
849 GetUniqueLDSGlobals(LDSParams.IndirectAccess.DynamicLDSGlobals);
850 updateMallocSizeForDynamicLDS(Func, &CurrMallocSize, HiddenDynLDSSize,
851 UniqueLDSGlobals);
852 }
853
854 CurrMallocSize = IRB.CreateZExt(CurrMallocSize, Int64Ty);
855
856 // Create a call to malloc function which does device global memory allocation
857 // with size equals to all LDS global accesses size in this kernel.
858 Value *ReturnAddress =
859 IRB.CreateIntrinsic(Intrinsic::returnaddress, {}, {IRB.getInt32(0)});
860 FunctionCallee MallocFunc = M.getOrInsertFunction(
861 StringRef("__asan_malloc_impl"),
862 FunctionType::get(Int64Ty, {Int64Ty, Int64Ty}, false));
863 Value *RAPtrToInt = IRB.CreatePtrToInt(ReturnAddress, Int64Ty);
864 Value *MallocCall = IRB.CreateCall(MallocFunc, {CurrMallocSize, RAPtrToInt});
865
866 Value *MallocPtr =
867 IRB.CreateIntToPtr(MallocCall, IRB.getPtrTy(AMDGPUAS::GLOBAL_ADDRESS));
868
869 // Create store of malloc to new global
870 IRB.CreateStore(MallocPtr, SwLDS);
871
872 // Create calls to __asan_poison_region to poison redzones.
873 poisonRedzones(Func, MallocPtr);
874
875 // Create branch to PrevEntryBlock
876 IRB.CreateBr(PrevEntryBlock);
877
878 // Create wave-group barrier at the starting of Previous entry block
879 Type *Int1Ty = IRB.getInt1Ty();
880 IRB.SetInsertPoint(PrevEntryBlock, PrevEntryBlock->begin());
881 auto *XYZCondPhi = IRB.CreatePHI(Int1Ty, 2, "xyzCond");
882 XYZCondPhi->addIncoming(IRB.getInt1(0), WIdBlock);
883 XYZCondPhi->addIncoming(IRB.getInt1(1), MallocBlock);
884
885 IRB.CreateIntrinsic(Intrinsic::amdgcn_s_barrier, {}, {});
886
887 // Load malloc pointer from Sw LDS.
888 Value *LoadMallocPtr =
889 IRB.CreateLoad(IRB.getPtrTy(AMDGPUAS::GLOBAL_ADDRESS), SwLDS);
890
891 // Replace All uses of LDS globals with new LDS pointers.
892 replaceKernelLDSAccesses(Func);
893
894 // Replace Memory Operations on LDS with corresponding
895 // global memory pointers.
896 translateLDSMemoryOperationsToGlobalMemory(Func, LoadMallocPtr,
897 LDSInstructions);
898
899 auto *CondFreeBlock = BasicBlock::Create(Ctx, "CondFree", Func);
900 auto *FreeBlock = BasicBlock::Create(Ctx, "Free", Func);
901 auto *EndBlock = BasicBlock::Create(Ctx, "End", Func);
902 for (BasicBlock &BB : *Func) {
903 if (!BB.empty()) {
904 if (ReturnInst *RI = dyn_cast<ReturnInst>(&BB.back())) {
905 RI->eraseFromParent();
906 IRB.SetInsertPoint(&BB, BB.end());
907 IRB.CreateBr(CondFreeBlock);
908 }
909 }
910 }
911
912 // Cond Free Block
913 IRB.SetInsertPoint(CondFreeBlock, CondFreeBlock->begin());
914 IRB.CreateIntrinsic(Intrinsic::amdgcn_s_barrier, {}, {});
915 IRB.CreateCondBr(XYZCondPhi, FreeBlock, EndBlock);
916
917 // Free Block
918 IRB.SetInsertPoint(FreeBlock, FreeBlock->begin());
919
920 // Free the previously allocate device global memory.
921 FunctionCallee AsanFreeFunc = M.getOrInsertFunction(
922 StringRef("__asan_free_impl"),
923 FunctionType::get(IRB.getVoidTy(), {Int64Ty, Int64Ty}, false));
924 Value *ReturnAddr = IRB.CreateCall(
925 Intrinsic::getDeclaration(&M, Intrinsic::returnaddress), IRB.getInt32(0));
926 Value *RAPToInt = IRB.CreatePtrToInt(ReturnAddr, Int64Ty);
927 Value *MallocPtrToInt = IRB.CreatePtrToInt(LoadMallocPtr, Int64Ty);
928 IRB.CreateCall(AsanFreeFunc, {MallocPtrToInt, RAPToInt});
929
930 IRB.CreateBr(EndBlock);
931
932 // End Block
933 IRB.SetInsertPoint(EndBlock, EndBlock->begin());
934 IRB.CreateRetVoid();
935 // Update the DomTree with corresponding links to basic blocks.
936 DTU.applyUpdates({{DominatorTree::Insert, WIdBlock, MallocBlock},
937 {DominatorTree::Insert, MallocBlock, PrevEntryBlock},
938 {DominatorTree::Insert, CondFreeBlock, FreeBlock},
939 {DominatorTree::Insert, FreeBlock, EndBlock}});
940}

943 Function *Func, SetVector<GlobalVariable *> &Variables) {
944 Type *Int32Ty = IRB.getInt32Ty();

967 SwLDSMetadata, GEPIdx, true);

1007void AMDGPUSwLowerLDS::buildNonKernelLDSOffsetTable(
1008 NonKernelLDSParameters &NKLDSParams) {
1009 // Offset table will have multiple rows and columns.
1010 // Rows are assumed to be from 0 to (n-1). n is total number
1011 // of kernels accessing the LDS through non-kernels.
1012 // Each row will have m elements. m is the total number of
1013 // unique LDS globals accessed by non-kernels.

1015 // the replacement of LDS global done by that particular kernel.

1019 return;

1024 ArrayType::get(IRB.getPtrTy(AMDGPUAS::GLOBAL_ADDRESS), NumberVariables);

1027 ArrayType::get(KernelOffsetsType, NumberKernels);

1029 for (size_t i = 0; i < NumberKernels; i++) {
1030 Function *Func = Kernels[i];
1031 overallConstantExprElts[i] =
1032 getAddressesOfVariablesInKernel(Func, Variables);

1035 ConstantArray::get(AllKernelsOffsetsType, overallConstantExprElts);
1036 NKLDSParams.LDSOffsetTable = new GlobalVariable(
1037 M, AllKernelsOffsetsType, true, GlobalValue::InternalLinkage, Init,
1038 "llvm.amdgcn.sw.lds.offset.table", nullptr, GlobalValue::NotThreadLocal,

1042 NKLDSParams.LDSOffsetTable->setSanitizerMetadata(MD);

1045void AMDGPUSwLowerLDS::lowerNonKernelLDSAccesses(
1046 Function *Func, SetVector<GlobalVariable *> &LDSGlobals,
1047 NonKernelLDSParameters &NKLDSParams) {
1048 // Replace LDS access in non-kernel with replacement queried from
1049 // Base table and offset from offset table.
1050 LLVM_DEBUG(dbgs() << "Sw LDS lowering, lower non-kernel access for : "
1051 << Func->getName());
1052 auto InsertAt = Func->getEntryBlock().getFirstNonPHIOrDbgOrAlloca();
1053 IRB.SetInsertPoint(InsertAt);
1054
1055 // Get LDS memory instructions.
1056 SetVector<Instruction *> LDSInstructions;
1057 getLDSMemoryInstructions(Func, LDSInstructions);
1058
1059 Function *Decl =
1060 Intrinsic::getDeclaration(&M, Intrinsic::amdgcn_lds_kernel_id, {});
1061 auto *KernelId = IRB.CreateCall(Decl, {});
1062 GlobalVariable *LDSBaseTable = NKLDSParams.LDSBaseTable;
1063 GlobalVariable *LDSOffsetTable = NKLDSParams.LDSOffsetTable;
1064 auto &OrdereLDSGlobals = NKLDSParams.OrdereLDSGlobals;
1065 Value *BaseGEP = IRB.CreateInBoundsGEP(
1066 LDSBaseTable->getValueType(), LDSBaseTable, {IRB.getInt32(0), KernelId});
1067 Value *BaseLoad =

1127 bool Changed = false;
1128
1129 CallGraph CG = CallGraph(M);
1130
1131 Changed |= eliminateConstantExprUsesOfLDSFromAllInstructions(M);
1132
1133 // Get all the direct and indirect access of LDS for all the kernels.
1134 LDSUsesInfoTy LDSUsesInfo = getTransitiveUsesOfLDS(CG, M);
1135
1136 // Utility to group LDS access into direct, indirect, static and dynamic.
1137 auto PopulateKernelStaticDynamicLDS = [&](FunctionVariableMap &LDSAccesses,
1138 bool DirectAccess) {
1139 for (auto &K : LDSAccesses) {
1140 Function *F = K.first;
1141 if (!F || K.second.empty())
1142 continue;
1143
1144 assert(isKernelLDS(F));
1145 if (!F->hasFnAttribute(Attribute::SanitizeAddress))
1146 continue;
1147
1148 // Only inserts if key isn't already in the map.
1149 FuncLDSAccessInfo.KernelToLDSParametersMap.insert(
1150 {F, KernelLDSParameters()});
1151
1152 auto &LDSParams = FuncLDSAccessInfo.KernelToLDSParametersMap[F];
1153 if (!DirectAccess)
1154 FuncLDSAccessInfo.KernelsWithIndirectLDSAccess.insert(F);
1155 for (GlobalVariable *GV : K.second) {
1156 if (!DirectAccess) {
1157 if (AMDGPU::isDynamicLDS(*GV))
1158 LDSParams.IndirectAccess.DynamicLDSGlobals.insert(GV);
1159 else
1160 LDSParams.IndirectAccess.StaticLDSGlobals.insert(GV);
1161 FuncLDSAccessInfo.AllNonKernelLDSAccess.insert(GV);
1162 } else {
1163 if (AMDGPU::isDynamicLDS(*GV))
1164 LDSParams.DirectAccess.DynamicLDSGlobals.insert(GV);
1165 else
1166 LDSParams.DirectAccess.StaticLDSGlobals.insert(GV);
1167 }
1168 }
1169 }
1170 };
1171
1172 PopulateKernelStaticDynamicLDS(LDSUsesInfo.direct_access, true);
1173 PopulateKernelStaticDynamicLDS(LDSUsesInfo.indirect_access, false);
1174
1175 // Get address sanitizer scale.
1176 initAsanInfo();
1177
1178 for (auto &K : FuncLDSAccessInfo.KernelToLDSParametersMap) {
1179 Function *Func = K.first;
1180 auto &LDSParams = FuncLDSAccessInfo.KernelToLDSParametersMap[Func];
1181 if (LDSParams.DirectAccess.StaticLDSGlobals.empty() &&
1182 LDSParams.DirectAccess.DynamicLDSGlobals.empty() &&
1183 LDSParams.IndirectAccess.StaticLDSGlobals.empty() &&
1184 LDSParams.IndirectAccess.DynamicLDSGlobals.empty()) {
1185 Changed = false;
1186 } else {
1187 removeFnAttrFromReachable(CG, Func,
1188 {"amdgpu-no-workitem-id-x",
1189 "amdgpu-no-workitem-id-y",
1190 "amdgpu-no-workitem-id-z"});
1191 reorderStaticDynamicIndirectLDSSet(LDSParams);
1192 buildSwLDSGlobal(Func);
1193 buildSwDynLDSGlobal(Func);
1194 populateSwMetadataGlobal(Func);
1195 populateSwLDSAttributeAndMetadata(Func);
1196 populateLDSToReplacementIndicesMap(Func);
1197 DomTreeUpdater DTU(DTCallback(*Func),
1198 DomTreeUpdater::UpdateStrategy::Lazy);
1199 lowerKernelLDSAccesses(Func, DTU);
1200 Changed = true;
1201 }
1202 }
1203
1204 // Get the Uses of LDS from non-kernels.
1205 getUsesOfLDSByNonKernels();
1206
1207 // Get non-kernels with LDS ptr as argument and called by kernels.
1208 getNonKernelsWithLDSArguments(CG);
1209
1210 if (!FuncLDSAccessInfo.NonKernelToLDSAccessMap.empty() ||
1211 !FuncLDSAccessInfo.NonKernelsWithLDSArgument.empty()) {
1212 NonKernelLDSParameters NKLDSParams;
1213 NKLDSParams.OrderedKernels = getOrderedIndirectLDSAccessingKernels(
1214 FuncLDSAccessInfo.KernelsWithIndirectLDSAccess);
1215 NKLDSParams.OrdereLDSGlobals = getOrderedNonKernelAllLDSGlobals(
1216 FuncLDSAccessInfo.AllNonKernelLDSAccess);
1217 buildNonKernelLDSBaseTable(NKLDSParams);
1218 buildNonKernelLDSOffsetTable(NKLDSParams);
1219 for (auto &K : FuncLDSAccessInfo.NonKernelToLDSAccessMap) {
1220 Function *Func = K.first;
1221 DenseSet<GlobalVariable *> &LDSGlobals = K.second;
1222 SetVector<GlobalVariable *> OrderedLDSGlobals = sortByName(
1223 std::vector<GlobalVariable *>(LDSGlobals.begin(), LDSGlobals.end()));
1224 lowerNonKernelLDSAccesses(Func, OrderedLDSGlobals, NKLDSParams);
1225 }
1226 for (Function *Func : FuncLDSAccessInfo.NonKernelsWithLDSArgument) {
1227 auto &K = FuncLDSAccessInfo.NonKernelToLDSAccessMap;
1228 if (K.find(Func) != K.end())
1229 continue;
1230 SetVector<llvm::GlobalVariable *> Vec;
1231 lowerNonKernelLDSAccesses(Func, Vec, NKLDSParams);
1232 }
1233 Changed = true;
1234 }
1235
1236 if (!Changed)
1237 return Changed;
1238
1239 for (auto &GV : make_early_inc_range(M.globals())) {
1240 if (AMDGPU::isLDSVariableToLower(GV)) {
1241 // probably want to remove from used lists
1242 GV.removeDeadConstantUsers();
1243 if (GV.use_empty())
1244 GV.eraseFromParent();
1245 }
1246 }
1247
1248 if (AsanInstrumentLDS) {
1249 SmallVector<InterestingMemoryOperand, 16> OperandsToInstrument;
1250 for (Instruction *Inst : AsanInfo.Instructions) {
1251 SmallVector<InterestingMemoryOperand, 1> InterestingOperands;
1252 getInterestingMemoryOperands(M, Inst, InterestingOperands);
1253 for (auto &Operand : InterestingOperands) {
1254 OperandsToInstrument.push_back(Operand);
1255 }
1256 }
1257 for (auto &Operand : OperandsToInstrument) {
1258 Value *Addr = Operand.getPtr();
1259 instrumentAddress(M, IRB, Operand.getInsn(), Operand.getInsn(), Addr,
1260 Operand.Alignment.valueOrOne(), Operand.TypeStoreSize,
1261 Operand.IsWrite, nullptr, false, false, AsanInfo.Scale,
1262 AsanInfo.Offset);
1263 Changed = true;
1264 }
1265 }
1266
1267 return Changed;
1268}
