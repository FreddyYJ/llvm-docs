9// In normal operation of Clang, the FrontendAction's lifecycle both creates
10// and destroys the AST, and code should operate on it during callbacks in
11// between (e.g. via ASTConsumer).
12//
13// For tests it is often more convenient to parse an AST from code, and keep it
14// alive as a normal local object, with assertions as straight-line code.
15// TestAST provides such an interface.
16// (ASTUnit can be used for this purpose, but is a production library with
17// broad scope and complicated API).
18//
19//===----------------------------------------------------------------------===//

33/// Specifies a virtual source file to be parsed as part of a test.
34struct TestInputs {
35 TestInputs() = default;
36 TestInputs(StringRef Code) : Code(Code) {}
37
38 /// The source code of the input file to be parsed.
39 std::string Code;
40
41 /// The language to parse as.
42 /// This affects the -x and -std flags used, and the filename.
43 TestLanguage Language = TestLanguage::Lang_OBJCXX;
44
45 /// Extra argv to pass to clang -cc1.
46 std::vector<std::string> ExtraArgs = {};
47
48 /// Extra virtual files that are available to be #included.
49 /// Keys are plain filenames ("foo.h"), values are file content.
50 llvm::StringMap<std::string> ExtraFiles = {};
51
52 /// By default, error diagnostics during parsing are reported as gtest errors.
53 /// To suppress this, set ErrorOK or include "error-ok" in a comment in Code.
54 /// In either case, all diagnostics appear in TestAST::diagnostics().
55 bool ErrorOK = false;
56};

58/// The result of parsing a file specified by TestInputs.
59///
60/// The ASTContext, Sema etc are valid as long as this object is alive.
61class TestAST {
62public:
63 /// Constructing a TestAST parses the virtual file.
64 ///
65 /// To keep tests terse, critical errors (e.g. invalid flags) are reported as
66 /// unit test failures with ADD_FAILURE() and produce an empty ASTContext,
67 /// Sema etc. This frees the test code from handling these explicitly.
68 TestAST(const TestInputs &);
69 TestAST(StringRef Code) : TestAST(TestInputs(Code)) {}
70 TestAST(TestAST &&M);
71 TestAST &operator=(TestAST &&);
72 ~TestAST();
73
74 /// Provides access to the AST context and other parts of Clang.
75
76 ASTContext &context() { return Clang->getASTContext(); }
77 Sema &sema() { return Clang->getSema(); }
78 SourceManager &sourceManager() { return Clang->getSourceManager(); }
79 FileManager &fileManager() { return Clang->getFileManager(); }
80 Preprocessor &preprocessor() { return Clang->getPreprocessor(); }
81
82 /// Returns diagnostics emitted during parsing.
83 /// (By default, errors cause test failures, see TestInputs::ErrorOK).
84 llvm::ArrayRef<StoredDiagnostic> diagnostics() { return Diagnostics; }
85
86private:
87 void clear();
88 std::unique_ptr<FrontendAction> Action;
89 std::unique_ptr<CompilerInstance> Clang;
90 std::vector<StoredDiagnostic> Diagnostics;
91};
