65GCNSubtarget &GCNSubtarget::initializeSubtargetDependencies(const Triple &TT,
66 StringRef GPU,
67 StringRef FS) {
68 // Determine default and user-specified characteristics
69 //
70 // We want to be able to turn these off, but making this a subtarget feature
71 // for SI has the unhelpful behavior that it unsets everything else if you
72 // disable it.
73 //
74 // Similarly we want enable-prt-strict-null to be on by default and not to
75 // unset everything else if it is disabled
76
77 SmallString<256> FullFS("+promote-alloca,+load-store-opt,+enable-ds128,");
78
79 // Turn on features that HSA ABI requires. Also turn on FlatForGlobal by
80 // default
81 if (isAmdHsaOS())
82 FullFS += "+flat-for-global,+unaligned-access-mode,+trap-handler,";
83
84 FullFS += "+enable-prt-strict-null,"; // This is overridden by a disable in FS
85
86 // Disable mutually exclusive bits.
87 if (FS.contains_insensitive("+wavefrontsize")) {
88 if (!FS.contains_insensitive("wavefrontsize16"))
89 FullFS += "-wavefrontsize16,";
90 if (!FS.contains_insensitive("wavefrontsize32"))
91 FullFS += "-wavefrontsize32,";
92 if (!FS.contains_insensitive("wavefrontsize64"))
93 FullFS += "-wavefrontsize64,";
94 }
95
96 FullFS += FS;
97
98 ParseSubtargetFeatures(GPU, /*TuneCPU*/ GPU, FullFS);
99
100 // Implement the "generic" processors, which acts as the default when no
101 // generation features are enabled (e.g for -mcpu=''). HSA OS defaults to
102 // the first amdgcn target that supports flat addressing. Other OSes defaults
103 // to the first amdgcn target.
104 if (Gen == AMDGPUSubtarget::INVALID) {
105 Gen = TT.getOS() == Triple::AMDHSA ? AMDGPUSubtarget::SEA_ISLANDS
106 : AMDGPUSubtarget::SOUTHERN_ISLANDS;
107 }
108
109 if (!hasFeature(AMDGPU::FeatureWavefrontSize32) &&
110 !hasFeature(AMDGPU::FeatureWavefrontSize64)) {
111 // If there is no default wave size it must be a generation before gfx10,
112 // these have FeatureWavefrontSize64 in their definition already. For gfx10+
113 // set wave32 as a default.
114 ToggleFeature(AMDGPU::FeatureWavefrontSize32);
115 }
116
117 // We don't support FP64 for EG/NI atm.
118 assert(!hasFP64() || (getGeneration() >= AMDGPUSubtarget::SOUTHERN_ISLANDS));
119
120 // Targets must either support 64-bit offsets for MUBUF instructions, and/or
121 // support flat operations, otherwise they cannot access a 64-bit global
122 // address space
123 assert(hasAddr64() || hasFlat());
124 // Unless +-flat-for-global is specified, turn on FlatForGlobal for targets
125 // that do not support ADDR64 variants of MUBUF instructions. Such targets
126 // cannot use a 64 bit offset with a MUBUF instruction to access the global
127 // address space
128 if (!hasAddr64() && !FS.contains("flat-for-global") && !FlatForGlobal) {
129 ToggleFeature(AMDGPU::FeatureFlatForGlobal);
130 FlatForGlobal = true;
131 }
132 // Unless +-flat-for-global is specified, use MUBUF instructions for global
133 // address space access if flat operations are not available.
134 if (!hasFlat() && !FS.contains("flat-for-global") && FlatForGlobal) {
135 ToggleFeature(AMDGPU::FeatureFlatForGlobal);
136 FlatForGlobal = false;
137 }
138
139 // Set defaults if needed.
140 if (MaxPrivateElementSize == 0)
141 MaxPrivateElementSize = 4;
142
143 if (LDSBankCount == 0)
144 LDSBankCount = 32;
145
146 if (TT.getArch() == Triple::amdgcn && AddressableLocalMemorySize == 0)
147 AddressableLocalMemorySize = 32768;
148
149 LocalMemorySize = AddressableLocalMemorySize;
150 if (AMDGPU::isGFX10Plus(*this) &&
151 !getFeatureBits().test(AMDGPU::FeatureCuMode))
152 LocalMemorySize *= 2;
153
154 // Don't crash on invalid devices.
155 if (WavefrontSizeLog2 == 0)
156 WavefrontSizeLog2 = 5;
157
158 HasFminFmaxLegacy = getGeneration() < AMDGPUSubtarget::VOLCANIC_ISLANDS;
159 HasSMulHi = getGeneration() >= AMDGPUSubtarget::GFX9;
160
161 TargetID.setTargetIDFromFeaturesString(FS);
162
163 LLVM_DEBUG(dbgs() << "xnack setting for subtarget: "
164 << TargetID.getXnackSetting() << '\n');
165 LLVM_DEBUG(dbgs() << "sramecc setting for subtarget: "
166 << TargetID.getSramEccSetting() << '\n');
167
168 return *this;
169}

329void GCNSubtarget::overrideSchedPolicy(MachineSchedPolicy &Policy,
330 unsigned NumRegionInstrs) const {
331 // Track register pressure so the scheduler can try to decrease
332 // pressure once register usage is above the threshold defined by
333 // SIRegisterInfo::getRegPressureSetLimit()
334 Policy.ShouldTrackPressure = true;
335
336 // Enabling both top down and bottom up scheduling seems to give us less
337 // register spills than just using one of these approaches on its own.
338 Policy.OnlyTopDown = false;
339 Policy.OnlyBottomUp = false;
340
341 // Enabling ShouldTrackLaneMasks crashes the SI Machine Scheduler.
342 if (!enableSIScheduler())
343 Policy.ShouldTrackLaneMasks = true;
344}

377GCNSubtarget::getBaseReservedNumSGPRs(const bool HasFlatScratch) const {
378 if (getGeneration() >= AMDGPUSubtarget::GFX10)
379 return 2; // VCC. FLAT_SCRATCH and XNACK are no longer in SGPRs.
380
381 if (HasFlatScratch || HasArchitectedFlatScratch) {
382 if (getGeneration() >= AMDGPUSubtarget::VOLCANIC_ISLANDS)
383 return 6; // FLAT_SCRATCH, XNACK, VCC (in that order).
384 if (getGeneration() == AMDGPUSubtarget::SEA_ISLANDS)
385 return 4; // FLAT_SCRATCH, VCC (in that order).
386 }
387
388 if (isXNACKEnabled())
389 return 4; // XNACK, VCC (in that order).
390 return 2; // VCC.
391}

418unsigned GCNSubtarget::getBaseMaxNumSGPRs(
419 const Function &F, std::pair<unsigned, unsigned> WavesPerEU,
420 unsigned PreloadedSGPRs, unsigned ReservedNumSGPRs) const {
421 // Compute maximum number of SGPRs function can use using default/requested
422 // minimum number of waves per execution unit.
423 unsigned MaxNumSGPRs = getMaxNumSGPRs(WavesPerEU.first, false);
424 unsigned MaxAddressableNumSGPRs = getMaxNumSGPRs(WavesPerEU.first, true);
425
426 // Check if maximum number of SGPRs was explicitly requested using
427 // "amdgpu-num-sgpr" attribute.
428 if (F.hasFnAttribute("amdgpu-num-sgpr")) {
429 unsigned Requested =
430 F.getFnAttributeAsParsedInteger("amdgpu-num-sgpr", MaxNumSGPRs);
431
432 // Make sure requested value does not violate subtarget's specifications.
433 if (Requested && (Requested <= ReservedNumSGPRs))
434 Requested = 0;
435
436 // If more SGPRs are required to support the input user/system SGPRs,
437 // increase to accommodate them.
438 //
439 // FIXME: This really ends up using the requested number of SGPRs + number
440 // of reserved special registers in total. Theoretically you could re-use
441 // the last input registers for these special registers, but this would
442 // require a lot of complexity to deal with the weird aliasing.
443 unsigned InputNumSGPRs = PreloadedSGPRs;
444 if (Requested && Requested < InputNumSGPRs)
445 Requested = InputNumSGPRs;
446
447 // Make sure requested value is compatible with values implied by
448 // default/requested minimum/maximum number of waves per execution unit.
449 if (Requested && Requested > getMaxNumSGPRs(WavesPerEU.first, false))
450 Requested = 0;
451 if (WavesPerEU.second && Requested &&
452 Requested < getMinNumSGPRs(WavesPerEU.second))
453 Requested = 0;
454
455 if (Requested)
456 MaxNumSGPRs = Requested;
457 }
458
459 if (hasSGPRInitBug())
460 MaxNumSGPRs = AMDGPU::IsaInfo::FIXED_NUM_SGPRS_FOR_INIT_BUG;
461
462 return std::min(MaxNumSGPRs - ReservedNumSGPRs, MaxAddressableNumSGPRs);
463}

542void GCNSubtarget::adjustSchedDependency(
543 SUnit *Def, int DefOpIdx, SUnit *Use, int UseOpIdx, SDep &Dep,
544 const TargetSchedModel *SchedModel) const {
545 if (Dep.getKind() != SDep::Kind::Data || !Dep.getReg() || !Def->isInstr() ||
546 !Use->isInstr())
547 return;
548
549 MachineInstr *DefI = Def->getInstr();
550 MachineInstr *UseI = Use->getInstr();
551
552 if (DefI->isBundle()) {
553 const SIRegisterInfo *TRI = getRegisterInfo();
554 auto Reg = Dep.getReg();
555 MachineBasicBlock::const_instr_iterator I(DefI->getIterator());
556 MachineBasicBlock::const_instr_iterator E(DefI->getParent()->instr_end());
557 unsigned Lat = 0;
558 for (++I; I != E && I->isBundledWithPred(); ++I) {
559 if (I->modifiesRegister(Reg, TRI))
560 Lat = InstrInfo.getInstrLatency(getInstrItineraryData(), *I);
561 else if (Lat)
562 --Lat;
563 }
564 Dep.setLatency(Lat);
565 } else if (UseI->isBundle()) {
566 const SIRegisterInfo *TRI = getRegisterInfo();

588struct FillMFMAShadowMutation : ScheduleDAGMutation {
589 const SIInstrInfo *TII;
590
591 ScheduleDAGMI *DAG;
592
593 FillMFMAShadowMutation(const SIInstrInfo *tii) : TII(tii) {}
594
595 bool isSALU(const SUnit *SU) const {
596 const MachineInstr *MI = SU->getInstr();
597 return MI && TII->isSALU(*MI) && !MI->isTerminator();
598 }
599
600 bool isVALU(const SUnit *SU) const {
601 const MachineInstr *MI = SU->getInstr();
602 return MI && TII->isVALU(*MI);
603 }
604
605 // Link as many SALU instructions in chain as possible. Return the size
606 // of the chain. Links up to MaxChain instructions.
607 unsigned linkSALUChain(SUnit *From, SUnit *To, unsigned MaxChain,
608 SmallPtrSetImpl<SUnit *> &Visited) const {
609 SmallVector<SUnit *, 8> Worklist({To});
610 unsigned Linked = 0;
611
612 while (!Worklist.empty() && MaxChain-- > 0) {
613 SUnit *SU = Worklist.pop_back_val();
614 if (!Visited.insert(SU).second)
615 continue;
616
617 LLVM_DEBUG(dbgs() << "Inserting edge from\n"; DAG->dumpNode(*From);
618 dbgs() << "to\n"; DAG->dumpNode(*SU); dbgs() << '\n');
619
620 if (SU != From && From != &DAG->ExitSU && DAG->canAddEdge(SU, From))
621 if (DAG->addEdge(SU, SDep(From, SDep::Artificial)))
622 ++Linked;
623
624 for (SDep &SI : From->Succs) {
625 SUnit *SUv = SI.getSUnit();
626 if (SUv != From && SU != &DAG->ExitSU && isVALU(SUv) &&
627 DAG->canAddEdge(SUv, SU))
628 DAG->addEdge(SUv, SDep(SU, SDep::Artificial));
629 }
630
631 for (SDep &SI : SU->Succs) {
632 SUnit *Succ = SI.getSUnit();
633 if (Succ != SU && isSALU(Succ))
634 Worklist.push_back(Succ);
635 }
636 }
637
638 return Linked;
639 }
640
641 void apply(ScheduleDAGInstrs *DAGInstrs) override {
642 const GCNSubtarget &ST = DAGInstrs->MF.getSubtarget<GCNSubtarget>();
643 if (!ST.hasMAIInsts())
644 return;
645 DAG = static_cast<ScheduleDAGMI *>(DAGInstrs);
646 const TargetSchedModel *TSchedModel = DAGInstrs->getSchedModel();
647 if (!TSchedModel || DAG->SUnits.empty())
648 return;
649
650 // Scan for MFMA long latency instructions and try to add a dependency
651 // of available SALU instructions to give them a chance to fill MFMA
652 // shadow. That is desirable to fill MFMA shadow with SALU instructions
653 // rather than VALU to prevent power consumption bursts and throttle.
654 auto LastSALU = DAG->SUnits.begin();
655 auto E = DAG->SUnits.end();
656 SmallPtrSet<SUnit *, 32> Visited;
657 for (SUnit &SU : DAG->SUnits) {
658 MachineInstr &MAI = *SU.getInstr();
659 if (!TII->isMAI(MAI) ||
660 MAI.getOpcode() == AMDGPU::V_ACCVGPR_WRITE_B32_e64 ||
661 MAI.getOpcode() == AMDGPU::V_ACCVGPR_READ_B32_e64)
662 continue;
663
664 unsigned Lat = TSchedModel->computeInstrLatency(&MAI) - 1;
665
666 LLVM_DEBUG(dbgs() << "Found MFMA: "; DAG->dumpNode(SU);
667 dbgs() << "Need " << Lat
668 << " instructions to cover latency.\n");
669
670 // Find up to Lat independent scalar instructions as early as
671 // possible such that they can be scheduled after this MFMA.
672 for (; Lat && LastSALU != E; ++LastSALU) {
673 if (Visited.count(&*LastSALU))
674 continue;
675
676 if (&SU == &DAG->ExitSU || &SU == &*LastSALU || !isSALU(&*LastSALU) ||
677 !DAG->canAddEdge(&*LastSALU, &SU))
678 continue;
679
680 Lat -= linkSALUChain(&SU, &*LastSALU, Lat, Visited);
681 }
682 }
683 }
684};
