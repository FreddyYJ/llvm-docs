39void BottomUpVec::vectorizeRec(ArrayRef<Value *> Bndl) {
40 auto LegalityRes = Legality.canVectorize(Bndl);
41 switch (LegalityRes.getSubclassID()) {
42 case LegalityResultID::Widen: {
43 auto *I = cast<Instruction>(Bndl[0]);
44 for (auto OpIdx : seq<unsigned>(I->getNumOperands())) {
45 auto OperandBndl = getOperand(Bndl, OpIdx);
46 vectorizeRec(OperandBndl);
47 }
48 break;
49 }
50 }
51}
