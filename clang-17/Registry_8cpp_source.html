61void RegistryMaps::registerMatcher(
62 StringRef MatcherName, std::unique_ptr<MatcherDescriptor> Callback) {
63 assert(!Constructors.contains(MatcherName));
64 Constructors[MatcherName] = std::move(Callback);
65}

67#define REGISTER_MATCHER(name) \
68 registerMatcher(#name, internal::makeMatcherAutoMarshall( \
69 ::clang::ast_matchers::name, #name));
70
71#define REGISTER_MATCHER_OVERLOAD(name) \
72 registerMatcher(#name, \
73 std::make_unique<internal::OverloadedMatcherDescriptor>(name##Callbacks))
74
75#define SPECIFIC_MATCHER_OVERLOAD(name, Id) \
76 static_cast<::clang::ast_matchers::name##_Type##Id>( \
77 ::clang::ast_matchers::name)
78
79#define MATCHER_OVERLOAD_ENTRY(name, Id) \
80 internal::makeMatcherAutoMarshall(SPECIFIC_MATCHER_OVERLOAD(name, Id), \
81 #name)
82
83#define REGISTER_OVERLOADED_2(name) \
84 do { \
85 std::unique_ptr<MatcherDescriptor> name##Callbacks[] = { \
86 MATCHER_OVERLOAD_ENTRY(name, 0), \
87 MATCHER_OVERLOAD_ENTRY(name, 1)}; \
88 REGISTER_MATCHER_OVERLOAD(name); \
89 } while (false)
90
91#define REGISTER_REGEX_MATCHER(name) \
92 registerMatcher(#name, internal::makeMatcherRegexMarshall(name, name))

629std::optional<MatcherCtor> Registry::lookupMatcherCtor(StringRef MatcherName) {

651std::vector<ArgKind> Registry::getAcceptedCompletionTypes(
652 ArrayRef<std::pair<MatcherCtor, unsigned>> Context) {
653 ASTNodeKind InitialTypes[] = {
654 ASTNodeKind::getFromNodeKind<Decl>(),
655 ASTNodeKind::getFromNodeKind<QualType>(),
656 ASTNodeKind::getFromNodeKind<Type>(),
657 ASTNodeKind::getFromNodeKind<Stmt>(),
658 ASTNodeKind::getFromNodeKind<NestedNameSpecifier>(),
659 ASTNodeKind::getFromNodeKind<NestedNameSpecifierLoc>(),
660 ASTNodeKind::getFromNodeKind<TypeLoc>()};
661
662 // Starting with the above seed of acceptable top-level matcher types, compute
663 // the acceptable type set for the argument indicated by each context element.
664 std::set<ArgKind> TypeSet;
665 for (auto IT : InitialTypes) {
666 TypeSet.insert(ArgKind::MakeMatcherArg(IT));
667 }
668 for (const auto &CtxEntry : Context) {
669 MatcherCtor Ctor = CtxEntry.first;
670 unsigned ArgNumber = CtxEntry.second;
671 std::vector<ArgKind> NextTypeSet;
672 for (const ArgKind &Kind : TypeSet) {
673 if (Kind.getArgKind() == Kind.AK_Matcher &&
674 Ctor->isConvertibleTo(Kind.getMatcherKind()) &&
675 (Ctor->isVariadic() || ArgNumber < Ctor->getNumArgs()))
676 Ctor->getArgKinds(Kind.getMatcherKind(), ArgNumber, NextTypeSet);
677 }
678 TypeSet.clear();
679 TypeSet.insert(NextTypeSet.begin(), NextTypeSet.end());
680 }
681 return std::vector<ArgKind>(TypeSet.begin(), TypeSet.end());
682}

684std::vector<MatcherCompletion>
685Registry::getMatcherCompletions(ArrayRef<ArgKind> AcceptedTypes) {
686 std::vector<MatcherCompletion> Completions;
687
688 // Search the registry for acceptable matchers.
689 for (const auto &M : RegistryData->constructors()) {
690 const MatcherDescriptor& Matcher = *M.getValue();
691 StringRef Name = M.getKey();
692
693 std::set<ASTNodeKind> RetKinds;
694 unsigned NumArgs = Matcher.isVariadic() ? 1 : Matcher.getNumArgs();
695 bool IsPolymorphic = Matcher.isPolymorphic();
696 std::vector<std::vector<ArgKind>> ArgsKinds(NumArgs);
697 unsigned MaxSpecificity = 0;
698 bool NodeArgs = false;
699 for (const ArgKind& Kind : AcceptedTypes) {
700 if (Kind.getArgKind() != Kind.AK_Matcher &&
701 Kind.getArgKind() != Kind.AK_Node) {
702 continue;
703 }
704
705 if (Kind.getArgKind() == Kind.AK_Node) {
706 NodeArgs = true;
707 unsigned Specificity;
708 ASTNodeKind LeastDerivedKind;
709 if (Matcher.isConvertibleTo(Kind.getNodeKind(), &Specificity,
710 &LeastDerivedKind)) {
711 if (MaxSpecificity < Specificity)
712 MaxSpecificity = Specificity;
713 RetKinds.insert(LeastDerivedKind);
714 for (unsigned Arg = 0; Arg != NumArgs; ++Arg)
715 Matcher.getArgKinds(Kind.getNodeKind(), Arg, ArgsKinds[Arg]);
716 if (IsPolymorphic)
717 break;
718 }
719 } else {
720 unsigned Specificity;
721 ASTNodeKind LeastDerivedKind;
722 if (Matcher.isConvertibleTo(Kind.getMatcherKind(), &Specificity,
723 &LeastDerivedKind)) {
724 if (MaxSpecificity < Specificity)
725 MaxSpecificity = Specificity;
726 RetKinds.insert(LeastDerivedKind);
727 for (unsigned Arg = 0; Arg != NumArgs; ++Arg)
728 Matcher.getArgKinds(Kind.getMatcherKind(), Arg, ArgsKinds[Arg]);
729 if (IsPolymorphic)
730 break;
731 }
732 }
733 }
734
735 if (!RetKinds.empty() && MaxSpecificity > 0) {
736 std::string Decl;
737 llvm::raw_string_ostream OS(Decl);
738
739 std::string TypedText = std::string(Name);
740
741 if (NodeArgs) {
742 OS << Name;
743 } else {
744
745 if (IsPolymorphic) {
746 OS << "Matcher<T> " << Name << "(Matcher<T>";
747 } else {
748 OS << "Matcher<" << RetKinds << "> " << Name << "(";
749 for (const std::vector<ArgKind> &Arg : ArgsKinds) {
750 if (&Arg != &ArgsKinds[0])
751 OS << ", ";
752
753 bool FirstArgKind = true;
754 std::set<ASTNodeKind> MatcherKinds;
755 // Two steps. First all non-matchers, then matchers only.
756 for (const ArgKind &AK : Arg) {
757 if (AK.getArgKind() == ArgKind::AK_Matcher) {
758 MatcherKinds.insert(AK.getMatcherKind());
759 } else {
760 if (!FirstArgKind)
761 OS << "|";
762 FirstArgKind = false;
763 OS << AK.asString();
764 }
765 }
766 if (!MatcherKinds.empty()) {
767 if (!FirstArgKind) OS << "|";
768 OS << "Matcher<" << MatcherKinds << ">";
769 }
770 }
771 }
772 if (Matcher.isVariadic())
773 OS << "...";
774 OS << ")";
775
776 TypedText += "(";
777 if (ArgsKinds.empty())
778 TypedText += ")";
779 else if (ArgsKinds[0][0].getArgKind() == ArgKind::AK_String)
780 TypedText += "\"";
781 }
782
783 Completions.emplace_back(TypedText, OS.str(), MaxSpecificity);
784 }
785 }
786
787 return Completions;
788}
