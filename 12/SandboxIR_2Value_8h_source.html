31/// Iterator for the `Use` edges of a Value's users.
32/// \Returns a `Use` when dereferenced.
33class UserUseIterator {
34 sandboxir::Use Use;
35 /// Don't let the user create a non-empty UserUseIterator.
36 UserUseIterator(const class Use &Use) : Use(Use) {}
37 friend class Value; // For constructor
38
39public:
40 using difference_type = std::ptrdiff_t;
41 using value_type = sandboxir::Use;
42 using pointer = value_type *;
43 using reference = value_type &;
44 using iterator_category = std::input_iterator_tag;
45
46 UserUseIterator() = default;
47 value_type operator*() const { return Use; }
48 UserUseIterator &operator++();
49 bool operator==(const UserUseIterator &Other) const {
50 return Use == Other.Use;
51 }
52 bool operator!=(const UserUseIterator &Other) const {
53 return !(*this == Other);
54 }
55 const sandboxir::Use &getUse() const { return Use; }
56};

225 /// Return true if this value has N uses or more.
226 /// This is logically equivalent to getNumUses() >= N.
227 /// WARNING: This can be expensive, as it is linear to the number of users.
228 bool hasNUsesOrMore(unsigned Num) const {
229 unsigned Cnt = 0;
230 for (auto It = use_begin(), ItE = use_end(); It != ItE; ++It) {
231 if (++Cnt >= Num)
232 return true;
233 }
234 return false;

236 /// Return true if this Value has exactly N uses.

238 unsigned Cnt = 0;
