8//
9// This file is the component of SandboxIR that tracks all changes made to its
10// state, such that we can revert the state when needed.
11//
12// Tracking changes
13// ----------------
14// The user needs to call `Tracker::save()` to enable tracking changes
15// made to SandboxIR. From that point on, any change made to SandboxIR, will
16// automatically create a change tracking object and register it with the
17// tracker. IR-change objects are subclasses of `IRChangeBase` and get
18// registered with the `Tracker::track()` function. The change objects
19// are saved in the order they are registered with the tracker and are stored in
20// the `Tracker::Changes` vector. All of this is done transparently to
21// the user.
22//
23// Reverting changes
24// -----------------
25// Calling `Tracker::revert()` will restore the state saved when
26// `Tracker::save()` was called. Internally this goes through the
27// change objects in `Tracker::Changes` in reverse order, calling their
28// `IRChangeBase::revert()` function one by one.
29//
30// Accepting changes
31// -----------------
32// The user needs to either revert or accept changes before the tracker object
33// is destroyed. This is enforced in the tracker's destructor.
34// This is the job of `Tracker::accept()`. Internally this will go
35// through the change objects in `Tracker::Changes` in order, calling
36// `IRChangeBase::accept()`.
37//
38//===----------------------------------------------------------------------===//

216/// This class can be used for tracking most instruction setters.
217/// The two template arguments are:
218/// - GetterFn: The getter member function pointer (e.g., `&Foo::get`)
219/// - SetterFn: The setter member function pointer (e.g., `&Foo::set`)
220/// Upon construction, it saves a copy of the original value by calling the
221/// getter function. Revert sets the value back to the one saved, using the
222/// setter function provided.
223///
224/// Example:
225/// Tracker.track(std::make_unique<
226/// GenericSetter<&FooInst::get, &FooInst::set>>(I, Tracker));
227///
228template <auto GetterFn, auto SetterFn>
229class GenericSetter final : public IRChangeBase {
230 /// Traits for getting the class type from GetterFn type.
231 template <typename> struct GetClassTypeFromGetter;
232 template <typename RetT, typename ClassT>
233 struct GetClassTypeFromGetter<RetT (ClassT::*)() const> {
234 using ClassType = ClassT;
235 };
236 using InstrT = typename GetClassTypeFromGetter<decltype(GetterFn)>::ClassType;
237 using SavedValT = std::invoke_result_t<decltype(GetterFn), InstrT>;
238 InstrT *I;
239 SavedValT OrigVal;
240
241public:
242 GenericSetter(InstrT *I) : I(I), OrigVal((I->*GetterFn)()) {}
243 void revert(Tracker &Tracker) final { (I->*SetterFn)(OrigVal); }
244 void accept() final {}
245#ifndef NDEBUG
246 void dump(raw_ostream &OS) const final { OS << "GenericSetter"; }
247 LLVM_DUMP_METHOD void dump() const final {
248 dump(dbgs());
249 dbgs() << "\n";
250 }
251#endif
252};

389/// The tracker collects all the change objects and implements the main API for
390/// saving / reverting / accepting.
391class Tracker {
392public:
393 enum class TrackerState {
394 Disabled, ///> Tracking is disabled
395 Record, ///> Tracking changes
396 };
397
398private:
399 /// The list of changes that are being tracked.
400 SmallVector<std::unique_ptr<IRChangeBase>> Changes;
401 /// The current state of the tracker.
402 TrackerState State = TrackerState::Disabled;
403 Context &Ctx;
404
405public:
406#ifndef NDEBUG
407 /// Helps catch bugs where we are creating new change objects while in the
408 /// middle of creating other change objects.
409 bool InMiddleOfCreatingChange = false;
410#endif // NDEBUG
411
412 explicit Tracker(Context &Ctx) : Ctx(Ctx) {}
413 ~Tracker();
414 Context &getContext() const { return Ctx; }
415 /// Record \p Change and take ownership. This is the main function used to
416 /// track Sandbox IR changes.
417 void track(std::unique_ptr<IRChangeBase> &&Change) {
418 assert(State == TrackerState::Record && "The tracker should be tracking!");
419#ifndef NDEBUG
420 assert(!InMiddleOfCreatingChange &&
421 "We are in the middle of creating another change!");
422 if (isTracking())
423 InMiddleOfCreatingChange = true;
424#endif // NDEBUG
425 Changes.push_back(std::move(Change));
426
427#ifndef NDEBUG
428 InMiddleOfCreatingChange = false;
429#endif
430 }
431 /// A convenience wrapper for `track()` that constructs and tracks the Change
432 /// object if tracking is enabled. \Returns true if tracking is enabled.
433 template <typename ChangeT, typename... ArgsT>
434 bool emplaceIfTracking(ArgsT... Args) {
435 if (!isTracking())
436 return false;
437 track(std::make_unique<ChangeT>(Args...));
438 return true;
439 }
440 /// \Returns true if the tracker is recording changes.
441 bool isTracking() const { return State == TrackerState::Record; }
442 /// \Returns the current state of the tracker.
443 TrackerState getState() const { return State; }
444 /// Turns on IR tracking.
445 void save();
446 /// Stops tracking and accept changes.
447 void accept();
448 /// Stops tracking and reverts to saved state.
449 void revert();
450
451#ifndef NDEBUG
452 void dump(raw_ostream &OS) const;
453 LLVM_DUMP_METHOD void dump() const;
454 friend raw_ostream &operator<<(raw_ostream &OS, const Tracker &Tracker) {
455 Tracker.dump(OS);
456 return OS;
457 }
458#endif // NDEBUG
459};
