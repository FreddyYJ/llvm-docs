22/// Counts the number of bits in the source operand having a value of 1.
23///
24/// \headerfile <x86intrin.h>
25///
26/// This intrinsic corresponds to the <c> POPCNT </c> instruction.
27///
28/// \param __A
29/// An unsigned 32-bit integer operand.

31/// source operand.
