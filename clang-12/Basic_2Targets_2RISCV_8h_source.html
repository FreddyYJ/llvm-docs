103class LLVM_LIBRARY_VISIBILITY RISCV32TargetInfo : public RISCVTargetInfo {

105 RISCV32TargetInfo(const llvm::Triple &Triple, const TargetOptions &Opts)

107 IntPtrType = SignedInt;

112
113 bool setABI(const std::string &Name) override {
114 if (Name == "ilp32" || Name == "ilp32f" || Name == "ilp32d") {

116 return true;

118 return false;
119 }

121 bool isValidCPUName(StringRef Name) const override;

123 bool isValidTuneCPUName(StringRef Name) const override;
124 void fillValidTuneCPUList(SmallVectorImpl<StringRef> &Values) const override;
125
126 void setMaxAtomicWidth() override {

128
