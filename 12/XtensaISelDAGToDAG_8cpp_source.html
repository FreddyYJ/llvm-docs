40 // For load/store instructions generate (base+offset) pair from
41 // memory address. The offset must be a multiple of scale argument.
42 bool selectMemRegAddr(SDValue Addr, SDValue &Base, SDValue &Offset,
43 int Scale) {
44 EVT ValTy = Addr.getValueType();
45
46 // if Address is FI, get the TargetFrameIndex.
47 if (FrameIndexSDNode *FIN = dyn_cast<FrameIndexSDNode>(Addr)) {
48 Base = CurDAG->getTargetFrameIndex(FIN->getIndex(), ValTy);
49 Offset = CurDAG->getTargetConstant(0, SDLoc(Addr), ValTy);
50
51 return true;
52 }
53
54 if (TM.isPositionIndependent()) {
55 DiagnosticInfoUnsupported Diag(CurDAG->getMachineFunction().getFunction(),
56 "PIC relocations are not supported ",
57 Addr.getDebugLoc());
58 CurDAG->getContext()->diagnose(Diag);
59 }
60
61 if ((Addr.getOpcode() == ISD::TargetExternalSymbol ||
62 Addr.getOpcode() == ISD::TargetGlobalAddress))
63 return false;
64
65 // Addresses of the form FI+const
66 bool Valid = false;
67 if (CurDAG->isBaseWithConstantOffset(Addr)) {
68 ConstantSDNode *CN = dyn_cast<ConstantSDNode>(Addr.getOperand(1));
69 int64_t OffsetVal = CN->getSExtValue();
70
71 Valid = isValidAddrOffset(Scale, OffsetVal);
72
73 if (Valid) {
74 // If the first operand is a FI, get the TargetFI Node.
75 if (FrameIndexSDNode *FIN =
76 dyn_cast<FrameIndexSDNode>(Addr.getOperand(0)))
77 Base = CurDAG->getTargetFrameIndex(FIN->getIndex(), ValTy);
78 else
79 Base = Addr.getOperand(0);
80
81 Offset =
82 CurDAG->getTargetConstant(CN->getZExtValue(), SDLoc(Addr), ValTy);
83 return true;
84 }
85 }
86
87 // Last case
88 Base = Addr;
89 Offset = CurDAG->getTargetConstant(0, SDLoc(Addr), Addr.getValueType());
90 return true;
91 }
92
93 bool selectMemRegAddrISH1(SDValue Addr, SDValue &Base, SDValue &Offset) {
94 return selectMemRegAddr(Addr, Base, Offset, 1);
95 }
96
97 bool selectMemRegAddrISH2(SDValue Addr, SDValue &Base, SDValue &Offset) {
98 return selectMemRegAddr(Addr, Base, Offset, 2);
99 }
100
101 bool selectMemRegAddrISH4(SDValue Addr, SDValue &Base, SDValue &Offset) {
102 return selectMemRegAddr(Addr, Base, Offset, 4);
103 }

130void XtensaDAGToDAGISel::Select(SDNode *Node) {
131 SDLoc DL(Node);
132 EVT VT = Node->getValueType(0);
133
134 // If we have a custom node, we already have selected!
135 if (Node->isMachineOpcode()) {
136 Node->setNodeId(-1);
137 return;
138 }
139
140 switch (Node->getOpcode()) {
141 case ISD::SHL: {
142 SDValue N0 = Node->getOperand(0);
143 SDValue N1 = Node->getOperand(1);
144 auto *C = dyn_cast<ConstantSDNode>(N1);
145 // If C is constant in range [1..31] then we can generate SLLI
146 // instruction using pattern matching, otherwise generate SLL.
147 if (!C || C->isZero()) {
148 SDNode *SSL = CurDAG->getMachineNode(Xtensa::SSL, DL, MVT::Glue, N1);
149 SDNode *SLL =
150 CurDAG->getMachineNode(Xtensa::SLL, DL, VT, N0, SDValue(SSL, 0));
151 ReplaceNode(Node, SLL);
152 return;
153 }
154 break;
155 }
156 case ISD::SRL: {
157 SDValue N0 = Node->getOperand(0);
158 SDValue N1 = Node->getOperand(1);
159 auto *C = dyn_cast<ConstantSDNode>(N1);
160
161 // If C is constant then we can generate SRLI
162 // instruction using pattern matching or EXTUI, otherwise generate SRL.
163 if (C) {
164 if (isUInt<4>(C->getZExtValue()))
165 break;
166 unsigned ShAmt = C->getZExtValue();
167 SDNode *EXTUI = CurDAG->getMachineNode(
168 Xtensa::EXTUI, DL, VT, N0, CurDAG->getTargetConstant(ShAmt, DL, VT),
169 CurDAG->getTargetConstant(32 - ShAmt, DL, VT));
170 ReplaceNode(Node, EXTUI);
171 return;
172 }
173
174 SDNode *SSR = CurDAG->getMachineNode(Xtensa::SSR, DL, MVT::Glue, N1);
175 SDNode *SRL =
176 CurDAG->getMachineNode(Xtensa::SRL, DL, VT, N0, SDValue(SSR, 0));
177 ReplaceNode(Node, SRL);
178 return;
179 }
180 case ISD::SRA: {
181 SDValue N0 = Node->getOperand(0);
182 SDValue N1 = Node->getOperand(1);
183 auto *C = dyn_cast<ConstantSDNode>(N1);
184 // If C is constant then we can generate SRAI
185 // instruction using pattern matching, otherwise generate SRA.
186 if (!C) {
187 SDNode *SSR = CurDAG->getMachineNode(Xtensa::SSR, DL, MVT::Glue, N1);
188 SDNode *SRA =
189 CurDAG->getMachineNode(Xtensa::SRA, DL, VT, N0, SDValue(SSR, 0));
190 ReplaceNode(Node, SRA);
191 return;
192 }
193 break;
194 }
195 case XtensaISD::SRCL: {
196 SDValue N0 = Node->getOperand(0);
197 SDValue N1 = Node->getOperand(1);
198 SDValue N2 = Node->getOperand(2);
199 SDNode *SSL = CurDAG->getMachineNode(Xtensa::SSL, DL, MVT::Glue, N2);
200 SDNode *SRC =
201 CurDAG->getMachineNode(Xtensa::SRC, DL, VT, N0, N1, SDValue(SSL, 0));
202 ReplaceNode(Node, SRC);
203 return;
204 }
205 case XtensaISD::SRCR: {
206 SDValue N0 = Node->getOperand(0);
207 SDValue N1 = Node->getOperand(1);
208 SDValue N2 = Node->getOperand(2);
209 SDNode *SSR = CurDAG->getMachineNode(Xtensa::SSR, DL, MVT::Glue, N2);
210 SDNode *SRC =
211 CurDAG->getMachineNode(Xtensa::SRC, DL, VT, N0, N1, SDValue(SSR, 0));
212 ReplaceNode(Node, SRC);
213 return;
214 }
215 }
216
217 SelectCode(Node);
218}
