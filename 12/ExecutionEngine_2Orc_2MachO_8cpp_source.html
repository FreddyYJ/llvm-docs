21static std::string objDesc(const MemoryBufferRef &Obj, const Triple &TT,

30template <typename HeaderType>
31static Error checkMachORelocatableObject(MemoryBufferRef Obj,
32 bool SwapEndianness, const Triple &TT,
33 bool ObjIsSlice) {
34 StringRef Data = Obj.getBuffer();
35
36 HeaderType Hdr;
37 memcpy(&Hdr, Data.data(), sizeof(HeaderType));
38
39 if (SwapEndianness)
40 swapStruct(Hdr);
41
42 if (Hdr.filetype != MachO::MH_OBJECT)
43 return make_error<StringError>(objDesc(Obj, TT, ObjIsSlice) +
44 " is not a MachO relocatable object",
45 inconvertibleErrorCode());
46
47 auto ObjArch = object::MachOObjectFile::getArch(Hdr.cputype, Hdr.cpusubtype);
48 if (ObjArch != TT.getArch())
49 return make_error<StringError>(
50 objDesc(Obj, TT, ObjIsSlice) + Triple::getArchTypeName(ObjArch) +
51 ", cannot be loaded into " + TT.str() + " process",
52 inconvertibleErrorCode());
53
54 return Error::success();
55}

57Error checkMachORelocatableObject(MemoryBufferRef Obj, const Triple &TT,
58 bool ObjIsSlice) {
59 StringRef Data = Obj.getBuffer();
60
61 if (Data.size() < 4)
62 return make_error<StringError>(
63 objDesc(Obj, TT, ObjIsSlice) +
64 " is not a valid MachO relocatable object file (truncated header)",
65 inconvertibleErrorCode());
66
67 uint32_t Magic;
68 memcpy(&Magic, Data.data(), sizeof(uint32_t));
69
70 switch (Magic) {
71 case MachO::MH_MAGIC:
72 case MachO::MH_CIGAM:
73 return checkMachORelocatableObject<MachO::mach_header>(
74 std::move(Obj), Magic == MachO::MH_CIGAM, TT, ObjIsSlice);
75 case MachO::MH_MAGIC_64:
76 case MachO::MH_CIGAM_64:
77 return checkMachORelocatableObject<MachO::mach_header_64>(
78 std::move(Obj), Magic == MachO::MH_CIGAM_64, TT, ObjIsSlice);
79 default:
80 return make_error<StringError>(
81 objDesc(Obj, TT, ObjIsSlice) +
82 " is not a valid MachO relocatable object (bad magic value)",
83 inconvertibleErrorCode());
84 }
85}

87Expected<std::unique_ptr<MemoryBuffer>>
88checkMachORelocatableObject(std::unique_ptr<MemoryBuffer> Obj, const Triple &TT,

91 checkMachORelocatableObject(Obj->getMemBufferRef(), TT, ObjIsSlice))

96Expected<std::pair<std::unique_ptr<MemoryBuffer>, LinkableFileKind>>
97loadMachORelocatableObject(StringRef Path, const Triple &TT, LoadArchives LA,
98 std::optional<StringRef> IdentifierOverride) {
99 assert((TT.getObjectFormat() == Triple::UnknownObjectFormat ||
100 TT.getObjectFormat() == Triple::MachO) &&
101 "TT must specify MachO or Unknown object format");
102
103 if (!IdentifierOverride)
104 IdentifierOverride = Path;
105
106 Expected<sys::fs::file_t> FDOrErr =
107 sys::fs::openNativeFileForRead(Path, sys::fs::OF_None);
108 if (!FDOrErr)
109 return createFileError(Path, FDOrErr.takeError());
110 sys::fs::file_t FD = *FDOrErr;
111 auto CloseFile = make_scope_exit([&]() { sys::fs::closeFile(FD); });
112
113 auto Buf =
114 MemoryBuffer::getOpenFile(FD, *IdentifierOverride, /*FileSize=*/-1);
115 if (!Buf)
116 return make_error<StringError>(
117 StringRef("Could not load MachO object at path ") + Path,
118 Buf.getError());
119
120 switch (identify_magic((*Buf)->getBuffer())) {
121 case file_magic::macho_object: {
122 auto CheckedObj = checkMachORelocatableObject(std::move(*Buf), TT, false);
123 if (!CheckedObj)
124 return CheckedObj.takeError();
125 return std::make_pair(std::move(*CheckedObj),
126 LinkableFileKind::RelocatableObject);
127 }
128 case file_magic::macho_universal_binary:
129 return loadLinkableSliceFromMachOUniversalBinary(FD, std::move(*Buf), TT,
130 LoadArchives::Never, Path,
131 *IdentifierOverride);
132 default:
133 return make_error<StringError>(
134 Path + " does not contain a relocatable object file compatible with " +
135 TT.str(),
136 inconvertibleErrorCode());
137 }
138}

140Expected<std::pair<std::unique_ptr<MemoryBuffer>, LinkableFileKind>>
141loadLinkableSliceFromMachOUniversalBinary(sys::fs::file_t FD,
142 std::unique_ptr<MemoryBuffer> UBBuf,
143 const Triple &TT, LoadArchives LA,
144 StringRef UBPath,
145 StringRef Identifier) {
146
147 auto UniversalBin =
148 object::MachOUniversalBinary::create(UBBuf->getMemBufferRef());
149 if (!UniversalBin)
150 return UniversalBin.takeError();
151
152 auto SliceRange = getMachOSliceRangeForTriple(**UniversalBin, TT);
153 if (!SliceRange)
154 return SliceRange.takeError();
155
156 auto Buf = MemoryBuffer::getOpenFileSlice(FD, Identifier, SliceRange->second,
157 SliceRange->first);
158 if (!Buf)
159 return make_error<StringError>(
160 "Could not load " + TT.getArchName() +
161 " slice of MachO universal binary at path " + UBPath,
162 Buf.getError());
163
164 switch (identify_magic((*Buf)->getBuffer())) {
165 case file_magic::archive:
166 if (LA != LoadArchives::Never)
167 return std::make_pair(std::move(*Buf), LinkableFileKind::Archive);
168 break;
169 case file_magic::macho_object: {
170 if (LA != LoadArchives::Required) {
171 auto CheckedObj = checkMachORelocatableObject(std::move(*Buf), TT, true);
172 if (!CheckedObj)
173 return CheckedObj.takeError();
174 return std::make_pair(std::move(*CheckedObj),
175 LinkableFileKind::RelocatableObject);
176 }
177 break;
178 }
179 default:
180 break;
181 }
182
183 auto FT = [&] {
184 switch (LA) {
185 case LoadArchives::Never:
186 return "a mach-o relocatable object file";
187 case LoadArchives::Allowed:
188 return "a mach-o relocatable object file or archive";
189 case LoadArchives::Required:
190 return "an archive";
191 }
192 llvm_unreachable("Unknown LoadArchives enum");
193 };
194
195 return make_error<StringError>(TT.getArchName() + " slice of " + UBPath +
196 " does not contain " + FT(),
197 inconvertibleErrorCode());
198}

200Expected<std::pair<size_t, size_t>>
201getMachOSliceRangeForTriple(object::MachOUniversalBinary &UB,
202 const Triple &TT) {
203
204 for (const auto &Obj : UB.objects()) {
205 auto ObjTT = Obj.getTriple();
206 if (ObjTT.getArch() == TT.getArch() &&
207 ObjTT.getSubArch() == TT.getSubArch() &&
208 (TT.getVendor() == Triple::UnknownVendor ||
209 ObjTT.getVendor() == TT.getVendor())) {
210 // We found a match. Return the range for the slice.
211 return std::make_pair(Obj.getOffset(), Obj.getSize());
212 }
213 }
214
215 return make_error<StringError>(Twine("Universal binary ") + UB.getFileName() +
216 " does not contain a slice for " +
217 TT.str(),
218 inconvertibleErrorCode());
219}

221Expected<std::pair<size_t, size_t>>
222getMachOSliceRangeForTriple(MemoryBufferRef UBBuf, const Triple &TT) {
223
224 auto UB = object::MachOUniversalBinary::create(UBBuf);
225 if (!UB)
226 return UB.takeError();
227
228 return getMachOSliceRangeForTriple(**UB, TT);
229}
