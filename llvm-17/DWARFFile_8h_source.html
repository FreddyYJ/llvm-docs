22/// This class represents DWARF information for source file
23/// and it's address map.
24///
25/// May be used asynchroniously for reading.
26class DWARFFile {
27public:
28 using UnloadCallbackTy = std::function<void(StringRef FileName)>;
29
30 DWARFFile(StringRef Name, std::unique_ptr<DWARFContext> Dwarf,
31 std::unique_ptr<AddressesMap> Addresses,
32 const std::vector<std::string> &Warnings,
33 UnloadCallbackTy UnloadFunc = nullptr)
34 : FileName(Name), Dwarf(std::move(Dwarf)),
35 Addresses(std::move(Addresses)), Warnings(Warnings),
36 UnloadFunc(UnloadFunc) {
37 if (this->Dwarf)
38 Endianess = this->Dwarf->isLittleEndian() ? support::endianness::little
39 : support::endianness::big;
40 }
41
42 /// Object file name.
43 StringRef FileName;
44
45 /// Source DWARF information.
46 std::unique_ptr<DWARFContext> Dwarf;
47
48 /// Helpful address information(list of valid address ranges, relocations).
49 std::unique_ptr<AddressesMap> Addresses;
50
51 /// Warnings for object file.
52 const std::vector<std::string> &Warnings;
53
54 /// Endiannes of source DWARF information.
55 support::endianness Endianess = support::endianness::little;
56
57 /// Callback to the module keeping object file to unload.
58 UnloadCallbackTy UnloadFunc;
59
60 /// Unloads object file and corresponding AddressesMap and Dwarf Context.
61 void unload() {
62 Addresses.reset();
63 Dwarf.reset();
64
65 if (UnloadFunc)
66 UnloadFunc(FileName);
67 }
68};
