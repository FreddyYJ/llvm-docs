34static std::string getCGDataErrString(cgdata_error Err,
35 const std::string &ErrMsg = "") {
36 std::string Msg;
37 raw_string_ostream OS(Msg);
38
39 switch (Err) {
40 case cgdata_error::success:
41 OS << "success";
42 break;
43 case cgdata_error::eof:
44 OS << "end of File";
45 break;
46 case cgdata_error::bad_magic:
47 OS << "invalid codegen data (bad magic)";
48 break;
49 case cgdata_error::bad_header:
50 OS << "invalid codegen data (file header is corrupt)";
51 break;
52 case cgdata_error::empty_cgdata:
53 OS << "empty codegen data";
54 break;
55 case cgdata_error::malformed:
56 OS << "malformed codegen data";
57 break;
58 case cgdata_error::unsupported_version:
59 OS << "unsupported codegen data version";
60 break;
61 }
62
63 // If optional error message is not empty, append it to the message.
64 if (!ErrMsg.empty())
65 OS << ": " << ErrMsg;
66
67 return OS.str();
68}

119std::string getCodeGenDataSectionName(CGDataSectKind CGSK,
120 Triple::ObjectFormatType OF,
121 bool AddSegmentInfo) {
122 std::string SectName;
123
124 if (OF == Triple::MachO && AddSegmentInfo)
125 SectName = CodeGenDataSectNamePrefix[CGSK];
126
127 if (OF == Triple::COFF)
128 SectName += CodeGenDataSectNameCoff[CGSK];
129 else
130 SectName += CodeGenDataSectNameCommon[CGSK];
131
132 return SectName;
133}

138CodeGenData &CodeGenData::getInstance() {
139 std::call_once(CodeGenData::OnceFlag, []() {
140 Instance = std::unique_ptr<CodeGenData>(new CodeGenData());
141
142 if (CodeGenDataGenerate)
143 Instance->EmitCGData = true;
144 else if (!CodeGenDataUsePath.empty()) {
145 // Initialize the global CGData if the input file name is given.
146 // We do not error-out when failing to parse the input file.
147 // Instead, just emit an warning message and fall back as if no CGData
148 // were available.
149 auto FS = vfs::getRealFileSystem();
150 auto ReaderOrErr = CodeGenDataReader::create(CodeGenDataUsePath, *FS);
151 if (Error E = ReaderOrErr.takeError()) {
152 warn(std::move(E), CodeGenDataUsePath);
153 return;
154 }
155 // Publish each CGData based on the data type in the header.
156 auto Reader = ReaderOrErr->get();
157 if (Reader->hasOutlinedHashTree())
158 Instance->publishOutlinedHashTree(Reader->releaseOutlinedHashTree());
159 }
160 });
161 return *(Instance.get());
162}

166Expected<Header> Header::readFromBuffer(const unsigned char *Curr) {
167 using namespace support;
168
169 static_assert(std::is_standard_layout_v<llvm::IndexedCGData::Header>,
170 "The header should be standard layout type since we use offset "
171 "of fields to read.");
172 Header H;
173 H.Magic = endian::readNext<uint64_t, endianness::little, unaligned>(Curr);
174 if (H.Magic != IndexedCGData::Magic)
175 return make_error<CGDataError>(cgdata_error::bad_magic);
176 H.Version = endian::readNext<uint32_t, endianness::little, unaligned>(Curr);
177 if (H.Version > IndexedCGData::CGDataVersion::CurrentVersion)
178 return make_error<CGDataError>(cgdata_error::unsupported_version);
179 H.DataKind = endian::readNext<uint32_t, endianness::little, unaligned>(Curr);
180
181 switch (H.Version) {
182 // When a new field is added to the header add a case statement here to
183 // compute the size as offset of the new field + size of the new field. This
184 // relies on the field being added to the end of the list.
185 static_assert(IndexedCGData::CGDataVersion::CurrentVersion == Version1,
186 "Please update the size computation below if a new field has "
187 "been added to the header, if not add a case statement to "
188 "fall through to the latest version.");
189 case 1ull:
190 H.OutlinedHashTreeOffset =
191 endian::readNext<uint64_t, endianness::little, unaligned>(Curr);
192 }
193
194 return H;
195}
