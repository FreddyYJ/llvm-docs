9//
10// Arguments to kernel and device functions are passed via param space,
11// which imposes certain restrictions:
12// http://docs.nvidia.com/cuda/parallel-thread-execution/#state-spaces
13//
14// Kernel parameters are read-only and accessible only via ld.param
15// instruction, directly or via a pointer.
16//
17// Device function parameters are directly accessible via
18// ld.param/st.param, but taking the address of one returns a pointer
19// to a copy created in local space which *can't* be used with
20// ld.param/st.param.
21//
22// Copying a byval struct into local memory in IR allows us to enforce
23// the param space restrictions, gives the rest of IR a pointer w/o
24// param space restrictions, and gives us an opportunity to eliminate
25// the copy.
26//
27// Pointer arguments to kernel functions need more work to be lowered:
28//
29// 1. Convert non-byval pointer arguments of CUDA kernels to pointers in the
30// global address space. This allows later optimizations to emit
31// ld.global.*/st.global.* for accessing these pointer arguments. For
32// example,
33//
34// define void @foo(float* %input) {
35// %v = load float, float* %input, align 4
36// ...
37// }
38//
39// becomes
40//
41// define void @foo(float* %input) {
42// %input2 = addrspacecast float* %input to float addrspace(1)*
43// %input3 = addrspacecast float addrspace(1)* %input2 to float*
44// %v = load float, float* %input3, align 4
45// ...
46// }
47//
48// Later, NVPTXInferAddressSpaces will optimize it to
49//
50// define void @foo(float* %input) {
51// %input2 = addrspacecast float* %input to float addrspace(1)*
52// %v = load float, float addrspace(1)* %input2, align 4
53// ...
54// }
55//
56// 2. Convert byval kernel parameters to pointers in the param address space
57// (so that NVPTX emits ld/st.param). Convert pointers *within* a byval
58// kernel parameter to pointers in the global address space. This allows
59// NVPTX to emit ld/st.global.
60//
61// struct S {
62// int *x;
63// int *y;
64// };
65// __global__ void foo(S s) {
66// int *b = s.y;
67// // use b
68// }
69//
70// "b" points to the global address space. In the IR level,
71//
72// define void @foo(ptr byval %input) {
73// %b_ptr = getelementptr {ptr, ptr}, ptr %input, i64 0, i32 1
74// %b = load ptr, ptr %b_ptr
75// ; use %b
76// }
77//
78// becomes
79//
80// define void @foo({i32*, i32*}* byval %input) {
81// %b_param = addrspacecat ptr %input to ptr addrspace(101)
82// %b_ptr = getelementptr {ptr, ptr}, ptr addrspace(101) %b_param, i64 0, i32 1
83// %b = load ptr, ptr addrspace(101) %b_ptr
84// %b_global = addrspacecast ptr %b to ptr addrspace(1)
85// ; use %b_generic
86// }
87//
88// Create a local copy of kernel byval parameters used in a way that *might* mutate
89// the parameter, by storing it in an alloca. Mutations to "grid_constant" parameters
90// are undefined behaviour, and don't require local copies.
91//
92// define void @foo(ptr byval(%struct.s) align 4 %input) {
93// store i32 42, ptr %input
94// ret void
95// }
96//
97// becomes
98//
99// define void @foo(ptr byval(%struct.s) align 4 %input) #1 {
100// %input1 = alloca %struct.s, align 4
101// %input2 = addrspacecast ptr %input to ptr addrspace(101)
102// %input3 = load %struct.s, ptr addrspace(101) %input2, align 4
103// store %struct.s %input3, ptr %input1, align 4
104// store i32 42, ptr %input1, align 4
105// ret void
106// }
107//
108// If %input were passed to a device function, or written to memory,
109// conservatively assume that %input gets mutated, and create a local copy.
110//
111// Convert param pointers to grid_constant byval kernel parameters that are
112// passed into calls (device functions, intrinsics, inline asm), or otherwise
113// "escape" (into stores/ptrtoints) to the generic address space, using the
114// `nvvm.ptr.param.to.gen` intrinsic, so that NVPTX emits cvta.param
115// (available for sm70+)
116//
117// define void @foo(ptr byval(%struct.s) %input) {
118// ; %input is a grid_constant
119// %call = call i32 @escape(ptr %input)
120// ret void
121// }
122//
123// becomes
124//
125// define void @foo(ptr byval(%struct.s) %input) {
126// %input1 = addrspacecast ptr %input to ptr addrspace(101)
127// ; the following intrinsic converts pointer to generic. We don't use an addrspacecast
128// ; to prevent generic -> param -> generic from getting cancelled out
129// %input1.gen = call ptr @llvm.nvvm.ptr.param.to.gen.p0.p101(ptr addrspace(101) %input1)
130// %call = call i32 @escape(ptr %input1.gen)
131// ret void
132// }
133//
134// TODO: merge this pass with NVPTXInferAddressSpaces so that other passes don't
135// cancel the addrspacecast pair this pass emits.
136//===----------------------------------------------------------------------===//

203// =============================================================================
204// If the function had a byval struct ptr arg, say foo(%struct.x* byval %d),
205// and we can't guarantee that the only accesses are loads,
206// then add the following instructions to the first basic block:
207//
208// %temp = alloca %struct.x, align 8
209// %tempd = addrspacecast %struct.x* %d to %struct.x addrspace(101)*
210// %tv = load %struct.x addrspace(101)* %tempd
211// store %struct.x %tv, %struct.x* %temp, align 8
212//
213// The above code allocates some space in the stack and copies the incoming
214// struct from param space to local space.
215// Then replace all occurrences of %d by %temp.
216//
217// In case we know that all users are GEPs or Loads, replace them with the same
218// ones in parameter AS, so we can access them using ld.param.
219// =============================================================================
220
221// For Loads, replaces the \p OldUse of the pointer with a Use of the same
222// pointer in parameter AS.
223// For "escapes" (to memory, a function call, or a ptrtoint), cast the OldUse to
224// generic using cvta.param.
225static void convertToParamAS(Use *OldUse, Value *Param, bool HasCvtaParam,
226 bool IsGridConstant) {
227 Instruction *I = dyn_cast<Instruction>(OldUse->getUser());
228 assert(I && "OldUse must be in an instruction");
229 struct IP {
230 Use *OldUse;
231 Instruction *OldInstruction;
232 Value *NewParam;
233 };
234 SmallVector<IP> ItemsToConvert = {{OldUse, I, Param}};
235 SmallVector<Instruction *> InstructionsToDelete;
236
237 auto CloneInstInParamAS = [HasCvtaParam,
238 IsGridConstant](const IP &I) -> Value * {
239 if (auto *LI = dyn_cast<LoadInst>(I.OldInstruction)) {
240 LI->setOperand(0, I.NewParam);
241 return LI;
242 }
243 if (auto *GEP = dyn_cast<GetElementPtrInst>(I.OldInstruction)) {
244 SmallVector<Value *, 4> Indices(GEP->indices());
245 auto *NewGEP = GetElementPtrInst::Create(
246 GEP->getSourceElementType(), I.NewParam, Indices, GEP->getName(),
247 GEP->getIterator());
248 NewGEP->setIsInBounds(GEP->isInBounds());
249 return NewGEP;
250 }
251 if (auto *BC = dyn_cast<BitCastInst>(I.OldInstruction)) {
252 auto *NewBCType = PointerType::get(BC->getContext(), ADDRESS_SPACE_PARAM);
253 return BitCastInst::Create(BC->getOpcode(), I.NewParam, NewBCType,
254 BC->getName(), BC->getIterator());
255 }
256 if (auto *ASC = dyn_cast<AddrSpaceCastInst>(I.OldInstruction)) {
257 assert(ASC->getDestAddressSpace() == ADDRESS_SPACE_PARAM);
258 (void)ASC;
259 // Just pass through the argument, the old ASC is no longer needed.
260 return I.NewParam;
261 }
262 if (auto *MI = dyn_cast<MemTransferInst>(I.OldInstruction)) {
263 if (MI->getRawSource() == I.OldUse->get()) {
264 // convert to memcpy/memmove from param space.
265 IRBuilder<> Builder(I.OldInstruction);
266 Intrinsic::ID ID = MI->getIntrinsicID();
267
268 CallInst *B = Builder.CreateMemTransferInst(
269 ID, MI->getRawDest(), MI->getDestAlign(), I.NewParam,
270 MI->getSourceAlign(), MI->getLength(), MI->isVolatile());
271 for (unsigned I : {0, 1})
272 if (uint64_t Bytes = MI->getParamDereferenceableBytes(I))
273 B->addDereferenceableParamAttr(I, Bytes);
274 return B;
275 }
276 // We may be able to handle other cases if the argument is
277 // __grid_constant__
278 }
279
280 if (HasCvtaParam) {
281 auto GetParamAddrCastToGeneric =
282 [](Value *Addr, Instruction *OriginalUser) -> Value * {
283 PointerType *ReturnTy =
284 PointerType::get(OriginalUser->getContext(), ADDRESS_SPACE_GENERIC);
285 Function *CvtToGen = Intrinsic::getDeclaration(
286 OriginalUser->getModule(), Intrinsic::nvvm_ptr_param_to_gen,
287 {ReturnTy, PointerType::get(OriginalUser->getContext(),
288 ADDRESS_SPACE_PARAM)});
289
290 // Cast param address to generic address space
291 Value *CvtToGenCall =
292 CallInst::Create(CvtToGen, Addr, Addr->getName() + ".gen",
293 OriginalUser->getIterator());
294 return CvtToGenCall;
295 };
296 auto *ParamInGenericAS =
297 GetParamAddrCastToGeneric(I.NewParam, I.OldInstruction);
298
299 // phi/select could use generic arg pointers w/o __grid_constant__
300 if (auto *PHI = dyn_cast<PHINode>(I.OldInstruction)) {
301 for (auto [Idx, V] : enumerate(PHI->incoming_values())) {
302 if (V.get() == I.OldUse->get())
303 PHI->setIncomingValue(Idx, ParamInGenericAS);
304 }
305 }
306 if (auto *SI = dyn_cast<SelectInst>(I.OldInstruction)) {
307 if (SI->getTrueValue() == I.OldUse->get())
308 SI->setTrueValue(ParamInGenericAS);
309 if (SI->getFalseValue() == I.OldUse->get())
310 SI->setFalseValue(ParamInGenericAS);
311 }
312
313 // Escapes or writes can only use generic param pointers if
314 // __grid_constant__ is in effect.
315 if (IsGridConstant) {
316 if (auto *CI = dyn_cast<CallInst>(I.OldInstruction)) {
317 I.OldUse->set(ParamInGenericAS);
318 return CI;
319 }
320 if (auto *SI = dyn_cast<StoreInst>(I.OldInstruction)) {
321 // byval address is being stored, cast it to generic
322 if (SI->getValueOperand() == I.OldUse->get())
323 SI->setOperand(0, ParamInGenericAS);
324 return SI;
325 }
326 if (auto *PI = dyn_cast<PtrToIntInst>(I.OldInstruction)) {
327 if (PI->getPointerOperand() == I.OldUse->get())
328 PI->setOperand(0, ParamInGenericAS);
329 return PI;
330 }
331 // TODO: iIf we allow stores, we should allow memcpy/memset to
332 // parameter, too.
333 }
334 }
335
336 llvm_unreachable("Unsupported instruction");
337 };
338
339 while (!ItemsToConvert.empty()) {
340 IP I = ItemsToConvert.pop_back_val();
341 Value *NewInst = CloneInstInParamAS(I);
342
343 if (NewInst && NewInst != I.OldInstruction) {
344 // We've created a new instruction. Queue users of the old instruction to
345 // be converted and the instruction itself to be deleted. We can't delete
346 // the old instruction yet, because it's still in use by a load somewhere.
347 for (Use &U : I.OldInstruction->uses())
348 ItemsToConvert.push_back({&U, cast<Instruction>(U.getUser()), NewInst});
349
350 InstructionsToDelete.push_back(I.OldInstruction);
351 }
352 }
353
354 // Now we know that all argument loads are using addresses in parameter space
355 // and we can finally remove the old instructions in generic AS. Instructions
356 // scheduled for removal should be processed in reverse order so the ones
357 // closest to the load are deleted first. Otherwise they may still be in use.
358 // E.g if we have Value = Load(BitCast(GEP(arg))), InstructionsToDelete will
359 // have {GEP,BitCast}. GEP can't be deleted first, because it's still used by
360 // the BitCast.
361 for (Instruction *I : llvm::reverse(InstructionsToDelete))
362 I->eraseFromParent();
363}

365// Adjust alignment of arguments passed byval in .param address space. We can
366// increase alignment of such arguments in a way that ensures that we can
367// effectively vectorize their loads. We should also traverse all loads from
368// byval pointer and adjust their alignment, if those were using known offset.
369// Such alignment changes must be conformed with parameter store and load in
370// NVPTXTargetLowering::LowerCall.
371static void adjustByValArgAlignment(Argument *Arg, Value *ArgInParamAS,
372 const NVPTXTargetLowering *TLI) {
373 Function *Func = Arg->getParent();
374 Type *StructType = Arg->getParamByValType();
375 const DataLayout &DL = Func->getDataLayout();
376
377 uint64_t NewArgAlign =
378 TLI->getFunctionParamOptimizedAlign(Func, StructType, DL).value();
379 uint64_t CurArgAlign =
380 Arg->getAttribute(Attribute::Alignment).getValueAsInt();
381
382 if (CurArgAlign >= NewArgAlign)
383 return;
384
385 LLVM_DEBUG(dbgs() << "Try to use alignment " << NewArgAlign << " instead of "
386 << CurArgAlign << " for " << *Arg << '\n');
387
388 auto NewAlignAttr =
389 Attribute::get(Func->getContext(), Attribute::Alignment, NewArgAlign);
390 Arg->removeAttr(Attribute::Alignment);
391 Arg->addAttr(NewAlignAttr);
392
393 struct Load {
394 LoadInst *Inst;
395 uint64_t Offset;
396 };
397
398 struct LoadContext {
399 Value *InitialVal;
400 uint64_t Offset;
401 };
402
403 SmallVector<Load> Loads;
404 std::queue<LoadContext> Worklist;
405 Worklist.push({ArgInParamAS, 0});
406 bool IsGridConstant = isParamGridConstant(*Arg);
407
408 while (!Worklist.empty()) {
409 LoadContext Ctx = Worklist.front();
410 Worklist.pop();
411
412 for (User *CurUser : Ctx.InitialVal->users()) {
413 if (auto *I = dyn_cast<LoadInst>(CurUser)) {
414 Loads.push_back({I, Ctx.Offset});
415 continue;
416 }
417
418 if (auto *I = dyn_cast<BitCastInst>(CurUser)) {
419 Worklist.push({I, Ctx.Offset});
420 continue;
421 }
422
423 if (auto *I = dyn_cast<GetElementPtrInst>(CurUser)) {
424 APInt OffsetAccumulated =
425 APInt::getZero(DL.getIndexSizeInBits(ADDRESS_SPACE_PARAM));
426
427 if (!I->accumulateConstantOffset(DL, OffsetAccumulated))
428 continue;
429
430 uint64_t OffsetLimit = -1;
431 uint64_t Offset = OffsetAccumulated.getLimitedValue(OffsetLimit);
432 assert(Offset != OffsetLimit && "Expect Offset less than UINT64_MAX");
433
434 Worklist.push({I, Ctx.Offset + Offset});
435 continue;
436 }
437
438 // supported for grid_constant
439 if (IsGridConstant &&
440 (isa<CallInst>(CurUser) || isa<StoreInst>(CurUser) ||
441 isa<PtrToIntInst>(CurUser)))
442 continue;
443
444 llvm_unreachable("All users must be one of: load, "
445 "bitcast, getelementptr, call, store, ptrtoint");
446 }
447 }
448
449 for (Load &CurLoad : Loads) {
450 Align NewLoadAlign(std::gcd(NewArgAlign, CurLoad.Offset));
451 Align CurLoadAlign(CurLoad.Inst->getAlign());
452 CurLoad.Inst->setAlignment(std::max(NewLoadAlign, CurLoadAlign));
453 }
454}

456namespace {
457struct ArgUseChecker : PtrUseVisitor<ArgUseChecker> {
458 using Base = PtrUseVisitor<ArgUseChecker>;
459
460 bool IsGridConstant;
461 // Set of phi/select instructions using the Arg
462 SmallPtrSet<Instruction *, 4> Conditionals;
463
464 ArgUseChecker(const DataLayout &DL, bool IsGridConstant)
465 : PtrUseVisitor(DL), IsGridConstant(IsGridConstant) {}
466
467 PtrInfo visitArgPtr(Argument &A) {
468 assert(A.getType()->isPointerTy());
469 IntegerType *IntIdxTy = cast<IntegerType>(DL.getIndexType(A.getType()));
470 IsOffsetKnown = false;
471 Offset = APInt(IntIdxTy->getBitWidth(), 0);
472 PI.reset();
473 Conditionals.clear();
474
475 LLVM_DEBUG(dbgs() << "Checking Argument " << A << "\n");
476 // Enqueue the uses of this pointer.
477 enqueueUsers(A);
478
479 // Visit all the uses off the worklist until it is empty.
480 // Note that unlike PtrUseVisitor we intentionally do not track offsets.
481 // We're only interested in how we use the pointer.
482 while (!(Worklist.empty() || PI.isAborted())) {
483 UseToVisit ToVisit = Worklist.pop_back_val();
484 U = ToVisit.UseAndIsOffsetKnown.getPointer();
485 Instruction *I = cast<Instruction>(U->getUser());
486 if (isa<PHINode>(I) || isa<SelectInst>(I))
487 Conditionals.insert(I);
488 LLVM_DEBUG(dbgs() << "Processing " << *I << "\n");
489 Base::visit(I);
490 }
491 if (PI.isEscaped())
492 LLVM_DEBUG(dbgs() << "Argument pointer escaped: " << *PI.getEscapingInst()
493 << "\n");
494 else if (PI.isAborted())
495 LLVM_DEBUG(dbgs() << "Pointer use needs a copy: " << *PI.getAbortingInst()
496 << "\n");
497 LLVM_DEBUG(dbgs() << "Traversed " << Conditionals.size()
498 << " conditionals\n");
499 return PI;
500 }
501
502 void visitStoreInst(StoreInst &SI) {
503 // Storing the pointer escapes it.
504 if (U->get() == SI.getValueOperand())
505 return PI.setEscapedAndAborted(&SI);
506 // Writes to the pointer are UB w/ __grid_constant__, but do not force a
507 // copy.
508 if (!IsGridConstant)
509 return PI.setAborted(&SI);
510 }
511
512 void visitAddrSpaceCastInst(AddrSpaceCastInst &ASC) {
513 // ASC to param space are no-ops and do not need a copy
514 if (ASC.getDestAddressSpace() != ADDRESS_SPACE_PARAM)
515 return PI.setEscapedAndAborted(&ASC);
516 Base::visitAddrSpaceCastInst(ASC);
517 }
518
519 void visitPtrToIntInst(PtrToIntInst &I) {
520 if (IsGridConstant)
521 return;
522 Base::visitPtrToIntInst(I);
523 }
524 void visitPHINodeOrSelectInst(Instruction &I) {
525 assert(isa<PHINode>(I) || isa<SelectInst>(I));
526 }
527 // PHI and select just pass through the pointers.
528 void visitPHINode(PHINode &PN) { enqueueUsers(PN); }
529 void visitSelectInst(SelectInst &SI) { enqueueUsers(SI); }
530
531 void visitMemTransferInst(MemTransferInst &II) {
532 if (*U == II.getRawDest() && !IsGridConstant)
533 PI.setAborted(&II);
534 // memcpy/memmove are OK when the pointer is source. We can convert them to
535 // AS-specific memcpy.
536 }
537
538 void visitMemSetInst(MemSetInst &II) {
539 if (!IsGridConstant)
540 PI.setAborted(&II);
541 }
542}; // struct ArgUseChecker

545void NVPTXLowerArgs::handleByValParam(const NVPTXTargetMachine &TM,
546 Argument *Arg) {
547 Function *Func = Arg->getParent();
548 bool HasCvtaParam = TM.getSubtargetImpl(*Func)->hasCvtaParam();
549 bool IsGridConstant = HasCvtaParam && isParamGridConstant(*Arg);
550 const DataLayout &DL = Func->getDataLayout();
551 BasicBlock::iterator FirstInst = Func->getEntryBlock().begin();
552 Type *StructType = Arg->getParamByValType();
553 assert(StructType && "Missing byval type");
554
555 ArgUseChecker AUC(DL, IsGridConstant);
556 ArgUseChecker::PtrInfo PI = AUC.visitArgPtr(*Arg);
557 bool ArgUseIsReadOnly = !(PI.isEscaped() || PI.isAborted());
558 // Easy case, accessing parameter directly is fine.
559 if (ArgUseIsReadOnly && AUC.Conditionals.empty()) {
560 // Convert all loads and intermediate operations to use parameter AS and
561 // skip creation of a local copy of the argument.
562 SmallVector<Use *, 16> UsesToUpdate;
563 for (Use &U : Arg->uses())
564 UsesToUpdate.push_back(&U);
565
566 Value *ArgInParamAS = new AddrSpaceCastInst(
567 Arg, PointerType::get(StructType, ADDRESS_SPACE_PARAM), Arg->getName(),
568 FirstInst);
569 for (Use *U : UsesToUpdate)
570 convertToParamAS(U, ArgInParamAS, HasCvtaParam, IsGridConstant);
571 LLVM_DEBUG(dbgs() << "No need to copy or cast " << *Arg << "\n");
572
573 const auto *TLI =
574 cast<NVPTXTargetLowering>(TM.getSubtargetImpl()->getTargetLowering());
575
576 adjustByValArgAlignment(Arg, ArgInParamAS, TLI);
577
578 return;
579 }
580
581 // We can't access byval arg directly and need a pointer. on sm_70+ we have
582 // ability to take a pointer to the argument without making a local copy.
583 // However, we're still not allowed to write to it. If the user specified
584 // `__grid_constant__` for the argument, we'll consider escaped pointer as
585 // read-only.
586 unsigned AS = DL.getAllocaAddrSpace();
587 if (HasCvtaParam && (ArgUseIsReadOnly || IsGridConstant)) {
588 LLVM_DEBUG(dbgs() << "Using non-copy pointer to " << *Arg << "\n");
589 // Replace all argument pointer uses (which might include a device function
590 // call) with a cast to the generic address space using cvta.param
591 // instruction, which avoids a local copy.
592 IRBuilder<> IRB(&Func->getEntryBlock().front());
593
594 // Cast argument to param address space
595 auto *CastToParam = cast<AddrSpaceCastInst>(IRB.CreateAddrSpaceCast(
596 Arg, IRB.getPtrTy(ADDRESS_SPACE_PARAM), Arg->getName() + ".param"));
597
598 // Cast param address to generic address space. We do not use an
599 // addrspacecast to generic here, because, LLVM considers `Arg` to be in the
600 // generic address space, and a `generic -> param` cast followed by a `param
601 // -> generic` cast will be folded away. The `param -> generic` intrinsic
602 // will be correctly lowered to `cvta.param`.
603 Value *CvtToGenCall = IRB.CreateIntrinsic(
604 IRB.getPtrTy(ADDRESS_SPACE_GENERIC), Intrinsic::nvvm_ptr_param_to_gen,
605 CastToParam, nullptr, CastToParam->getName() + ".gen");
606
607 Arg->replaceAllUsesWith(CvtToGenCall);
608
609 // Do not replace Arg in the cast to param space
610 CastToParam->setOperand(0, Arg);
611 } else {
612 LLVM_DEBUG(dbgs() << "Creating a local copy of " << *Arg << "\n");
613 // Otherwise we have to create a temporary copy.
614 AllocaInst *AllocA =
615 new AllocaInst(StructType, AS, Arg->getName(), FirstInst);
616 // Set the alignment to alignment of the byval parameter. This is because,
617 // later load/stores assume that alignment, and we are going to replace
618 // the use of the byval parameter with this alloca instruction.
619 AllocA->setAlignment(Func->getParamAlign(Arg->getArgNo())
620 .value_or(DL.getPrefTypeAlign(StructType)));
621 Arg->replaceAllUsesWith(AllocA);
622
623 Value *ArgInParam = new AddrSpaceCastInst(
624 Arg, PointerType::get(Arg->getContext(), ADDRESS_SPACE_PARAM),
625 Arg->getName(), FirstInst);
626 // Be sure to propagate alignment to this load; LLVM doesn't know that NVPTX
627 // addrspacecast preserves alignment. Since params are constant, this load
628 // is definitely not volatile.
629 LoadInst *LI =
630 new LoadInst(StructType, ArgInParam, Arg->getName(),
631 /*isVolatile=*/false, AllocA->getAlign(), FirstInst);
632 new StoreInst(LI, AllocA, FirstInst);
633 }
634}

636void NVPTXLowerArgs::markPointerAsGlobal(Value *Ptr) {
637 if (Ptr->getType()->getPointerAddressSpace() != ADDRESS_SPACE_GENERIC)
638 return;
639
640 // Deciding where to emit the addrspacecast pair.
641 BasicBlock::iterator InsertPt;
642 if (Argument *Arg = dyn_cast<Argument>(Ptr)) {
643 // Insert at the functon entry if Ptr is an argument.
644 InsertPt = Arg->getParent()->getEntryBlock().begin();
645 } else {
646 // Insert right after Ptr if Ptr is an instruction.
647 InsertPt = ++cast<Instruction>(Ptr)->getIterator();
648 assert(InsertPt != InsertPt->getParent()->end() &&
649 "We don't call this function with Ptr being a terminator.");
650 }
651
652 Instruction *PtrInGlobal = new AddrSpaceCastInst(
653 Ptr, PointerType::get(Ptr->getContext(), ADDRESS_SPACE_GLOBAL),
654 Ptr->getName(), InsertPt);
655 Value *PtrInGeneric = new AddrSpaceCastInst(PtrInGlobal, Ptr->getType(),
656 Ptr->getName(), InsertPt);
657 // Replace with PtrInGeneric all uses of Ptr except PtrInGlobal.
658 Ptr->replaceAllUsesWith(PtrInGeneric);
659 PtrInGlobal->setOperand(0, Ptr);
660}

665bool NVPTXLowerArgs::runOnKernelFunction(const NVPTXTargetMachine &TM,
666 Function &F) {
667 // Copying of byval aggregates + SROA may result in pointers being loaded as
668 // integers, followed by intotoptr. We may want to mark those as global, too,
669 // but only if the loaded integer is used exclusively for conversion to a
670 // pointer with inttoptr.
671 auto HandleIntToPtr = [this](Value &V) {
672 if (llvm::all_of(V.users(), [](User *U) { return isa<IntToPtrInst>(U); })) {
673 SmallVector<User *, 16> UsersToUpdate(V.users());
674 for (User *U : UsersToUpdate)
675 markPointerAsGlobal(U);
676 }
677 };
678 if (TM.getDrvInterface() == NVPTX::CUDA) {
679 // Mark pointers in byval structs as global.
680 for (auto &B : F) {
681 for (auto &I : B) {
682 if (LoadInst *LI = dyn_cast<LoadInst>(&I)) {
683 if (LI->getType()->isPointerTy() || LI->getType()->isIntegerTy()) {
684 Value *UO = getUnderlyingObject(LI->getPointerOperand());
685 if (Argument *Arg = dyn_cast<Argument>(UO)) {
686 if (Arg->hasByValAttr()) {
687 // LI is a load from a pointer within a byval kernel parameter.
688 if (LI->getType()->isPointerTy())
689 markPointerAsGlobal(LI);
690 else
691 HandleIntToPtr(*LI);
692 }
693 }
694 }
695 }
696 }
697 }
698 }
699
700 LLVM_DEBUG(dbgs() << "Lowering kernel args of " << F.getName() << "\n");
701 for (Argument &Arg : F.args()) {
702 if (Arg.getType()->isPointerTy()) {
703 if (Arg.hasByValAttr())
704 handleByValParam(TM, &Arg);
705 else if (TM.getDrvInterface() == NVPTX::CUDA)
706 markPointerAsGlobal(&Arg);
707 } else if (Arg.getType()->isIntegerTy() &&
708 TM.getDrvInterface() == NVPTX::CUDA) {
709 HandleIntToPtr(Arg);
710 }
711 }
712 return true;
713}
714
715// Device functions only need to copy byval args into local memory.
716bool NVPTXLowerArgs::runOnDeviceFunction(const NVPTXTargetMachine &TM,
717 Function &F) {
718 LLVM_DEBUG(dbgs() << "Lowering function args of " << F.getName() << "\n");
719 for (Argument &Arg : F.args())
720 if (Arg.getType()->isPointerTy() && Arg.hasByValAttr())
721 handleByValParam(TM, &Arg);
722 return true;
723}

725bool NVPTXLowerArgs::runOnFunction(Function &F) {
726 auto &TM = getAnalysis<TargetPassConfig>().getTM<NVPTXTargetMachine>();
727
728 return isKernelFunction(F) ? runOnKernelFunction(TM, F)
729 : runOnDeviceFunction(TM, F);
730}
