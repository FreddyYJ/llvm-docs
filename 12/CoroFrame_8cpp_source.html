159// We cannot rely solely on natural alignment of a type when building a
160// coroutine frame and if the alignment specified on the Alloca instruction
161// differs from the natural alignment of the alloca type we will need to insert
162// padding.
163class FrameTypeBuilder {
164private:
165 struct Field {
166 uint64_t Size;
167 uint64_t Offset;
168 Type *Ty;
169 FieldIDType LayoutFieldIndex;
170 Align Alignment;
171 Align TyAlignment;
172 uint64_t DynamicAlignBuffer;
173 };

175 const DataLayout &DL;
176 LLVMContext &Context;
177 uint64_t StructSize = 0;
178 Align StructAlign;
179 bool IsFinished = false;
180
181 std::optional<Align> MaxFrameAlignment;
182

184 DenseMap<Value*, unsigned> FieldIndexByKey;
185
186public:
187 FrameTypeBuilder(LLVMContext &Context, const DataLayout &DL,
188 std::optional<Align> MaxFrameAlignment)
189 : DL(DL), Context(Context), MaxFrameAlignment(MaxFrameAlignment) {}

191 /// Add a field to this structure for the storage of an `alloca`
192 /// instruction.
193 [[nodiscard]] FieldIDType addFieldForAlloca(AllocaInst *AI,
194 bool IsHeader = false) {
195 Type *Ty = AI->getAllocatedType();
196
197 // Make an array type if this is a static array allocation.
198 if (AI->isArrayAllocation()) {
199 if (auto *CI = dyn_cast<ConstantInt>(AI->getArraySize()))
200 Ty = ArrayType::get(Ty, CI->getValue().getZExtValue());
201 else
202 report_fatal_error("Coroutines cannot handle non static allocas yet");
203 }
204
205 return addField(Ty, AI->getAlign(), IsHeader);
206 }
207
208 /// We want to put the allocas whose lifetime-ranges are not overlapped
209 /// into one slot of coroutine frame.
210 /// Consider the example at:https://bugs.llvm.org/show_bug.cgi?id=45566
211 ///
212 /// cppcoro::task<void> alternative_paths(bool cond) {
213 /// if (cond) {
214 /// big_structure a;
215 /// process(a);
216 /// co_await something();
217 /// } else {
218 /// big_structure b;
219 /// process2(b);
220 /// co_await something();
221 /// }
222 /// }
223 ///
224 /// We want to put variable a and variable b in the same slot to
225 /// reduce the size of coroutine frame.
226 ///
227 /// This function use StackLifetime algorithm to partition the AllocaInsts in
228 /// Spills to non-overlapped sets in order to put Alloca in the same
229 /// non-overlapped set into the same slot in the Coroutine Frame. Then add
230 /// field for the allocas in the same non-overlapped set by using the largest
231 /// type as the field type.
232 ///
233 /// Side Effects: Because We sort the allocas, the order of allocas in the
234 /// frame may be different with the order in the source code.
235 void addFieldForAllocas(const Function &F, FrameDataInfo &FrameData,
236 coro::Shape &Shape);
237
238 /// Add a field to this structure.
239 [[nodiscard]] FieldIDType addField(Type *Ty, MaybeAlign MaybeFieldAlignment,
240 bool IsHeader = false,
241 bool IsSpillOfValue = false) {
242 assert(!IsFinished && "adding fields to a finished builder");
243 assert(Ty && "must provide a type for a field");
244
245 // The field size is always the alloc size of the type.
246 uint64_t FieldSize = DL.getTypeAllocSize(Ty);
247
248 // For an alloca with size=0, we don't need to add a field and they
249 // can just point to any index in the frame. Use index 0.
250 if (FieldSize == 0) {
251 return 0;
252 }
253
254 // The field alignment might not be the type alignment, but we need
255 // to remember the type alignment anyway to build the type.
256 // If we are spilling values we don't need to worry about ABI alignment
257 // concerns.
258 Align ABIAlign = DL.getABITypeAlign(Ty);
259 Align TyAlignment = ABIAlign;
260 if (IsSpillOfValue && MaxFrameAlignment && *MaxFrameAlignment < ABIAlign)
261 TyAlignment = *MaxFrameAlignment;
262 Align FieldAlignment = MaybeFieldAlignment.value_or(TyAlignment);
263
264 // The field alignment could be bigger than the max frame case, in that case
265 // we request additional storage to be able to dynamically align the
266 // pointer.
267 uint64_t DynamicAlignBuffer = 0;
268 if (MaxFrameAlignment && (FieldAlignment > *MaxFrameAlignment)) {
269 DynamicAlignBuffer =
270 offsetToAlignment(MaxFrameAlignment->value(), FieldAlignment);
271 FieldAlignment = *MaxFrameAlignment;
272 FieldSize = FieldSize + DynamicAlignBuffer;
273 }
274
275 // Lay out header fields immediately.
276 uint64_t Offset;
277 if (IsHeader) {
278 Offset = alignTo(StructSize, FieldAlignment);
279 StructSize = Offset + FieldSize;
280
281 // Everything else has a flexible offset.
282 } else {
283 Offset = OptimizedStructLayoutField::FlexibleOffset;
284 }
285
286 Fields.push_back({FieldSize, Offset, Ty, 0, FieldAlignment, TyAlignment,
287 DynamicAlignBuffer});
288 return Fields.size() - 1;
289 }
290
291 /// Finish the layout and set the body on the given type.
292 void finish(StructType *Ty);
293
294 uint64_t getStructSize() const {
295 assert(IsFinished && "not yet finished!");
296 return StructSize;
297 }
298
299 Align getStructAlign() const {
300 assert(IsFinished && "not yet finished!");
301 return StructAlign;
302 }
303
304 FieldIDType getLayoutFieldIndex(FieldIDType Id) const {
305 assert(IsFinished && "not yet finished!");
306 return Fields[Id].LayoutFieldIndex;
307 }
308
309 Field getLayoutField(FieldIDType Id) const {
310 assert(IsFinished && "not yet finished!");
311 return Fields[Id];
312 }

316void FrameDataInfo::updateLayoutIndex(FrameTypeBuilder &B) {
317 auto Updater = [&](Value *I) {
318 auto Field = B.getLayoutField(getFieldIndex(I));
319 setFieldIndex(I, Field.LayoutFieldIndex);
320 setAlign(I, Field.Alignment);
321 uint64_t dynamicAlign =
322 Field.DynamicAlignBuffer
323 ? Field.DynamicAlignBuffer + Field.Alignment.value()
324 : 0;
325 setDynamicAlign(I, dynamicAlign);
326 setOffset(I, Field.Offset);
327 };
328 LayoutIndexUpdateStarted = true;
329 for (auto &S : Spills)
330 Updater(S.first);
331 for (const auto &A : Allocas)
332 Updater(A.Alloca);
333 LayoutIndexUpdateStarted = false;
334}

336void FrameTypeBuilder::addFieldForAllocas(const Function &F,
337 FrameDataInfo &FrameData,
338 coro::Shape &Shape) {
339 using AllocaSetType = SmallVector<AllocaInst *, 4>;
340 SmallVector<AllocaSetType, 4> NonOverlapedAllocas;
341
342 // We need to add field for allocas at the end of this function.
343 auto AddFieldForAllocasAtExit = make_scope_exit([&]() {
344 for (auto AllocaList : NonOverlapedAllocas) {
345 auto *LargestAI = *AllocaList.begin();
346 FieldIDType Id = addFieldForAlloca(LargestAI);
347 for (auto *Alloca : AllocaList)
348 FrameData.setFieldIndex(Alloca, Id);
349 }
350 });
351
352 if (!Shape.OptimizeFrame) {
353 for (const auto &A : FrameData.Allocas) {
354 AllocaInst *Alloca = A.Alloca;
355 NonOverlapedAllocas.emplace_back(AllocaSetType(1, Alloca));
356 }
357 return;
358 }
359
360 // Because there are paths from the lifetime.start to coro.end
361 // for each alloca, the liferanges for every alloca is overlaped
362 // in the blocks who contain coro.end and the successor blocks.
363 // So we choose to skip there blocks when we calculate the liferange
364 // for each alloca. It should be reasonable since there shouldn't be uses
365 // in these blocks and the coroutine frame shouldn't be used outside the
366 // coroutine body.
367 //
368 // Note that the user of coro.suspend may not be SwitchInst. However, this
369 // case seems too complex to handle. And it is harmless to skip these
370 // patterns since it just prevend putting the allocas to live in the same
371 // slot.
372 DenseMap<SwitchInst *, BasicBlock *> DefaultSuspendDest;
373 for (auto *CoroSuspendInst : Shape.CoroSuspends) {
374 for (auto *U : CoroSuspendInst->users()) {
375 if (auto *ConstSWI = dyn_cast<SwitchInst>(U)) {
376 auto *SWI = const_cast<SwitchInst *>(ConstSWI);
377 DefaultSuspendDest[SWI] = SWI->getDefaultDest();
378 SWI->setDefaultDest(SWI->getSuccessor(1));
379 }
380 }
381 }
382
383 auto ExtractAllocas = [&]() {
384 AllocaSetType Allocas;
385 Allocas.reserve(FrameData.Allocas.size());
386 for (const auto &A : FrameData.Allocas)
387 Allocas.push_back(A.Alloca);
388 return Allocas;
389 };
390 StackLifetime StackLifetimeAnalyzer(F, ExtractAllocas(),
391 StackLifetime::LivenessType::May);
392 StackLifetimeAnalyzer.run();
393 auto DoAllocasInterfere = [&](const AllocaInst *AI1, const AllocaInst *AI2) {
394 return StackLifetimeAnalyzer.getLiveRange(AI1).overlaps(
395 StackLifetimeAnalyzer.getLiveRange(AI2));
396 };
397 auto GetAllocaSize = [&](const coro::AllocaInfo &A) {
398 std::optional<TypeSize> RetSize = A.Alloca->getAllocationSize(DL);
399 assert(RetSize && "Variable Length Arrays (VLA) are not supported.\n");
400 assert(!RetSize->isScalable() && "Scalable vectors are not yet supported");
401 return RetSize->getFixedValue();
402 };
403 // Put larger allocas in the front. So the larger allocas have higher
404 // priority to merge, which can save more space potentially. Also each
405 // AllocaSet would be ordered. So we can get the largest Alloca in one
406 // AllocaSet easily.
407 sort(FrameData.Allocas, [&](const auto &Iter1, const auto &Iter2) {
408 return GetAllocaSize(Iter1) > GetAllocaSize(Iter2);
409 });
410 for (const auto &A : FrameData.Allocas) {
411 AllocaInst *Alloca = A.Alloca;
412 bool Merged = false;
413 // Try to find if the Alloca does not interfere with any existing
414 // NonOverlappedAllocaSet. If it is true, insert the alloca to that
415 // NonOverlappedAllocaSet.
416 for (auto &AllocaSet : NonOverlapedAllocas) {
417 assert(!AllocaSet.empty() && "Processing Alloca Set is not empty.\n");
418 bool NoInterference = none_of(AllocaSet, [&](auto Iter) {
419 return DoAllocasInterfere(Alloca, Iter);
420 });
421 // If the alignment of A is multiple of the alignment of B, the address
422 // of A should satisfy the requirement for aligning for B.
423 //
424 // There may be other more fine-grained strategies to handle the alignment
425 // infomation during the merging process. But it seems hard to handle
426 // these strategies and benefit little.
427 bool Alignable = [&]() -> bool {
428 auto *LargestAlloca = *AllocaSet.begin();
429 return LargestAlloca->getAlign().value() % Alloca->getAlign().value() ==
430 0;
431 }();
432 bool CouldMerge = NoInterference && Alignable;
433 if (!CouldMerge)
434 continue;
435 AllocaSet.push_back(Alloca);
436 Merged = true;
437 break;
438 }
439 if (!Merged) {
440 NonOverlapedAllocas.emplace_back(AllocaSetType(1, Alloca));
441 }
442 }
443 // Recover the default target destination for each Switch statement
444 // reserved.
445 for (auto SwitchAndDefaultDest : DefaultSuspendDest) {
446 SwitchInst *SWI = SwitchAndDefaultDest.first;
447 BasicBlock *DestBB = SwitchAndDefaultDest.second;
448 SWI->setDefaultDest(DestBB);
449 }
450 // This Debug Info could tell us which allocas are merged into one slot.
451 LLVM_DEBUG(for (auto &AllocaSet
452 : NonOverlapedAllocas) {
453 if (AllocaSet.size() > 1) {
454 dbgs() << "In Function:" << F.getName() << "\n";
455 dbgs() << "Find Union Set "
456 << "\n";
457 dbgs() << "\tAllocas are \n";
458 for (auto Alloca : AllocaSet)
459 dbgs() << "\t\t" << *Alloca << "\n";
460 }
461 });
462}

464void FrameTypeBuilder::finish(StructType *Ty) {
465 assert(!IsFinished && "already finished!");
466
467 // Prepare the optimal-layout field array.
468 // The Id in the layout field is a pointer to our Field for it.
469 SmallVector<OptimizedStructLayoutField, 8> LayoutFields;
470 LayoutFields.reserve(Fields.size());
471 for (auto &Field : Fields) {
472 LayoutFields.emplace_back(&Field, Field.Size, Field.Alignment,
473 Field.Offset);
474 }
475
476 // Perform layout.
477 auto SizeAndAlign = performOptimizedStructLayout(LayoutFields);
478 StructSize = SizeAndAlign.first;
479 StructAlign = SizeAndAlign.second;
480
481 auto getField = [](const OptimizedStructLayoutField &LayoutField) -> Field & {
482 return *static_cast<Field *>(const_cast<void*>(LayoutField.Id));
483 };
484
485 // We need to produce a packed struct type if there's a field whose
486 // assigned offset isn't a multiple of its natural type alignment.
487 bool Packed = [&] {
488 for (auto &LayoutField : LayoutFields) {
489 auto &F = getField(LayoutField);
490 if (!isAligned(F.TyAlignment, LayoutField.Offset))
491 return true;
492 }
493 return false;
494 }();
495
496 // Build the struct body.
497 SmallVector<Type*, 16> FieldTypes;
498 FieldTypes.reserve(LayoutFields.size() * 3 / 2);
499 uint64_t LastOffset = 0;
500 for (auto &LayoutField : LayoutFields) {
501 auto &F = getField(LayoutField);
502
503 auto Offset = LayoutField.Offset;
504
505 // Add a padding field if there's a padding gap and we're either
506 // building a packed struct or the padding gap is more than we'd
507 // get from aligning to the field type's natural alignment.
508 assert(Offset >= LastOffset);
509 if (Offset != LastOffset) {
510 if (Packed || alignTo(LastOffset, F.TyAlignment) != Offset)
511 FieldTypes.push_back(ArrayType::get(Type::getInt8Ty(Context),
512 Offset - LastOffset));
513 }
514
515 F.Offset = Offset;
516 F.LayoutFieldIndex = FieldTypes.size();
517
518 FieldTypes.push_back(F.Ty);
519 if (F.DynamicAlignBuffer) {
520 FieldTypes.push_back(
521 ArrayType::get(Type::getInt8Ty(Context), F.DynamicAlignBuffer));
522 }
523 LastOffset = Offset + F.Size;
524 }
525
526 Ty->setBody(FieldTypes, Packed);
527
528#ifndef NDEBUG
529 // Check that the IR layout matches the offsets we expect.
530 auto Layout = DL.getStructLayout(Ty);
531 for (auto &F : Fields) {
532 assert(Ty->getElementType(F.LayoutFieldIndex) == F.Ty);
533 assert(Layout->getElementOffset(F.LayoutFieldIndex) == F.Offset);
534 }
535#endif
536
537 IsFinished = true;
538}

558/// Create name for Type. It uses MDString to store new created string to
559/// avoid memory leak.
560static StringRef solveTypeName(Type *Ty) {
561 if (Ty->isIntegerTy()) {
562 // The longest name in common may be '__int_128', which has 9 bits.
563 SmallString<16> Buffer;
564 raw_svector_ostream OS(Buffer);
565 OS << "__int_" << cast<IntegerType>(Ty)->getBitWidth();
566 auto *MDName = MDString::get(Ty->getContext(), OS.str());
567 return MDName->getString();
568 }
569
570 if (Ty->isFloatingPointTy()) {
571 if (Ty->isFloatTy())
572 return "__float_";
573 if (Ty->isDoubleTy())
574 return "__double_";
575 return "__floating_type_";
576 }
577
578 if (Ty->isPointerTy())
579 return "PointerType";
580
581 if (Ty->isStructTy()) {
582 if (!cast<StructType>(Ty)->hasName())
583 return "__LiteralStructType_";
584
585 auto Name = Ty->getStructName();
586
587 SmallString<16> Buffer(Name);
588 for (auto &Iter : Buffer)
589 if (Iter == '.' || Iter == ':')
590 Iter = '_';
591 auto *MDName = MDString::get(Ty->getContext(), Buffer.str());
592 return MDName->getString();
593 }
594
595 return "UnknownType";
596}

598static DIType *solveDIType(DIBuilder &Builder, Type *Ty,
599 const DataLayout &Layout, DIScope *Scope,
600 unsigned LineNum,
601 DenseMap<Type *, DIType *> &DITypeCache) {
602 if (DIType *DT = DITypeCache.lookup(Ty))
603 return DT;
604
605 StringRef Name = solveTypeName(Ty);
606
607 DIType *RetType = nullptr;
608
609 if (Ty->isIntegerTy()) {
610 auto BitWidth = cast<IntegerType>(Ty)->getBitWidth();
611 RetType = Builder.createBasicType(Name, BitWidth, dwarf::DW_ATE_signed,
612 llvm::DINode::FlagArtificial);
613 } else if (Ty->isFloatingPointTy()) {
614 RetType = Builder.createBasicType(Name, Layout.getTypeSizeInBits(Ty),
615 dwarf::DW_ATE_float,
616 llvm::DINode::FlagArtificial);
617 } else if (Ty->isPointerTy()) {
618 // Construct PointerType points to null (aka void *) instead of exploring
619 // pointee type to avoid infinite search problem. For example, we would be
620 // in trouble if we traverse recursively:
621 //
622 // struct Node {
623 // Node* ptr;
624 // };
625 RetType =
626 Builder.createPointerType(nullptr, Layout.getTypeSizeInBits(Ty),
627 Layout.getABITypeAlign(Ty).value() * CHAR_BIT,
628 /*DWARFAddressSpace=*/std::nullopt, Name);
629 } else if (Ty->isStructTy()) {
630 auto *DIStruct = Builder.createStructType(
631 Scope, Name, Scope->getFile(), LineNum, Layout.getTypeSizeInBits(Ty),
632 Layout.getPrefTypeAlign(Ty).value() * CHAR_BIT,
633 llvm::DINode::FlagArtificial, nullptr, llvm::DINodeArray());
634
635 auto *StructTy = cast<StructType>(Ty);
636 SmallVector<Metadata *, 16> Elements;
637 for (unsigned I = 0; I < StructTy->getNumElements(); I++) {
638 DIType *DITy = solveDIType(Builder, StructTy->getElementType(I), Layout,
639 Scope, LineNum, DITypeCache);
640 assert(DITy);
641 Elements.push_back(Builder.createMemberType(
642 Scope, DITy->getName(), Scope->getFile(), LineNum,
643 DITy->getSizeInBits(), DITy->getAlignInBits(),
644 Layout.getStructLayout(StructTy)->getElementOffsetInBits(I),
645 llvm::DINode::FlagArtificial, DITy));
646 }
647
648 Builder.replaceArrays(DIStruct, Builder.getOrCreateArray(Elements));
649
650 RetType = DIStruct;
651 } else {
652 LLVM_DEBUG(dbgs() << "Unresolved Type: " << *Ty << "\n");
653 TypeSize Size = Layout.getTypeSizeInBits(Ty);
654 auto *CharSizeType = Builder.createBasicType(
655 Name, 8, dwarf::DW_ATE_unsigned_char, llvm::DINode::FlagArtificial);
656
657 if (Size <= 8)
658 RetType = CharSizeType;
659 else {
660 if (Size % 8 != 0)
661 Size = TypeSize::getFixed(Size + 8 - (Size % 8));
662
663 RetType = Builder.createArrayType(
664 Size, Layout.getPrefTypeAlign(Ty).value(), CharSizeType,
665 Builder.getOrCreateArray(Builder.getOrCreateSubrange(0, Size / 8)));
666 }
667 }
668
669 DITypeCache.insert({Ty, RetType});
670 return RetType;
671}

673/// Build artificial debug info for C++ coroutine frames to allow users to
674/// inspect the contents of the frame directly
675///
676/// Create Debug information for coroutine frame with debug name "__coro_frame".
677/// The debug information for the fields of coroutine frame is constructed from
678/// the following way:
679/// 1. For all the value in the Frame, we search the use of dbg.declare to find
680/// the corresponding debug variables for the value. If we can find the
681/// debug variable, we can get full and accurate debug information.
682/// 2. If we can't get debug information in step 1 and 2, we could only try to
683/// build the DIType by Type. We did this in solveDIType. We only handle
684/// integer, float, double, integer type and struct type for now.
685static void buildFrameDebugInfo(Function &F, coro::Shape &Shape,
686 FrameDataInfo &FrameData) {
687 DISubprogram *DIS = F.getSubprogram();
688 // If there is no DISubprogram for F, it implies the Function are not compiled
689 // with debug info. So we also don't need to generate debug info for the frame
690 // neither.
691 if (!DIS || !DIS->getUnit() ||
692 !dwarf::isCPlusPlus(
693 (dwarf::SourceLanguage)DIS->getUnit()->getSourceLanguage()) ||
694 DIS->getUnit()->getEmissionKind() != DICompileUnit::DebugEmissionKind::FullDebug)
695 return;
696
697 assert(Shape.ABI == coro::ABI::Switch &&
698 "We could only build debug infomation for C++ coroutine now.\n");
699
700 DIBuilder DBuilder(*F.getParent(), /*AllowUnresolved*/ false);
701
702 assert(Shape.getPromiseAlloca() &&
703 "Coroutine with switch ABI should own Promise alloca");
704
705 DIFile *DFile = DIS->getFile();
706 unsigned LineNum = DIS->getLine();
707
708 DICompositeType *FrameDITy = DBuilder.createStructType(
709 DIS->getUnit(), Twine(F.getName() + ".coro_frame_ty").str(),
710 DFile, LineNum, Shape.FrameSize * 8,
711 Shape.FrameAlign.value() * 8, llvm::DINode::FlagArtificial, nullptr,
712 llvm::DINodeArray());
713 StructType *FrameTy = Shape.FrameTy;
714 SmallVector<Metadata *, 16> Elements;
715 DataLayout Layout = F.getDataLayout();
716
717 DenseMap<Value *, DILocalVariable *> DIVarCache;
718 cacheDIVar(FrameData, DIVarCache);
719
720 unsigned ResumeIndex = coro::Shape::SwitchFieldIndex::Resume;
721 unsigned DestroyIndex = coro::Shape::SwitchFieldIndex::Destroy;
722 unsigned IndexIndex = Shape.SwitchLowering.IndexField;
723
724 DenseMap<unsigned, StringRef> NameCache;
725 NameCache.insert({ResumeIndex, "__resume_fn"});
726 NameCache.insert({DestroyIndex, "__destroy_fn"});
727 NameCache.insert({IndexIndex, "__coro_index"});
728
729 Type *ResumeFnTy = FrameTy->getElementType(ResumeIndex),
730 *DestroyFnTy = FrameTy->getElementType(DestroyIndex),
731 *IndexTy = FrameTy->getElementType(IndexIndex);
732
733 DenseMap<unsigned, DIType *> TyCache;
734 TyCache.insert(
735 {ResumeIndex, DBuilder.createPointerType(
736 nullptr, Layout.getTypeSizeInBits(ResumeFnTy))});
737 TyCache.insert(
738 {DestroyIndex, DBuilder.createPointerType(
739 nullptr, Layout.getTypeSizeInBits(DestroyFnTy))});
740
741 /// FIXME: If we fill the field `SizeInBits` with the actual size of
742 /// __coro_index in bits, then __coro_index wouldn't show in the debugger.
743 TyCache.insert({IndexIndex, DBuilder.createBasicType(
744 "__coro_index",
745 (Layout.getTypeSizeInBits(IndexTy) < 8)
746 ? 8
747 : Layout.getTypeSizeInBits(IndexTy),
748 dwarf::DW_ATE_unsigned_char)});
749
750 for (auto *V : FrameData.getAllDefs()) {
751 if (!DIVarCache.contains(V))
752 continue;
753
754 auto Index = FrameData.getFieldIndex(V);
755
756 NameCache.insert({Index, DIVarCache[V]->getName()});
757 TyCache.insert({Index, DIVarCache[V]->getType()});
758 }
759
760 // Cache from index to (Align, Offset Pair)
761 DenseMap<unsigned, std::pair<unsigned, unsigned>> OffsetCache;
762 // The Align and Offset of Resume function and Destroy function are fixed.
763 OffsetCache.insert({ResumeIndex, {8, 0}});
764 OffsetCache.insert({DestroyIndex, {8, 8}});
765 OffsetCache.insert(
766 {IndexIndex,
767 {Shape.SwitchLowering.IndexAlign, Shape.SwitchLowering.IndexOffset}});
768
769 for (auto *V : FrameData.getAllDefs()) {
770 auto Index = FrameData.getFieldIndex(V);
771
772 OffsetCache.insert(
773 {Index, {FrameData.getAlign(V).value(), FrameData.getOffset(V)}});
774 }
775
776 DenseMap<Type *, DIType *> DITypeCache;
777 // This counter is used to avoid same type names. e.g., there would be
778 // many i32 and i64 types in one coroutine. And we would use i32_0 and
779 // i32_1 to avoid the same type. Since it makes no sense the name of the
780 // fields confilicts with each other.
781 unsigned UnknownTypeNum = 0;
782 for (unsigned Index = 0; Index < FrameTy->getNumElements(); Index++) {
783 if (!OffsetCache.contains(Index))
784 continue;
785
786 std::string Name;
787 uint64_t SizeInBits;
788 uint32_t AlignInBits;
789 uint64_t OffsetInBits;
790 DIType *DITy = nullptr;
791
792 Type *Ty = FrameTy->getElementType(Index);
793 assert(Ty->isSized() && "We can't handle type which is not sized.\n");
794 SizeInBits = Layout.getTypeSizeInBits(Ty).getFixedValue();
795 AlignInBits = OffsetCache[Index].first * 8;
796 OffsetInBits = OffsetCache[Index].second * 8;
797
798 if (NameCache.contains(Index)) {
799 Name = NameCache[Index].str();
800 DITy = TyCache[Index];
801 } else {
802 DITy = solveDIType(DBuilder, Ty, Layout, FrameDITy, LineNum, DITypeCache);
803 assert(DITy && "SolveDIType shouldn't return nullptr.\n");
804 Name = DITy->getName().str();
805 Name += "_" + std::to_string(UnknownTypeNum);
806 UnknownTypeNum++;
807 }
808
809 Elements.push_back(DBuilder.createMemberType(
810 FrameDITy, Name, DFile, LineNum, SizeInBits, AlignInBits, OffsetInBits,
811 llvm::DINode::FlagArtificial, DITy));
812 }
813
814 DBuilder.replaceArrays(FrameDITy, DBuilder.getOrCreateArray(Elements));
815
816 auto *FrameDIVar =
817 DBuilder.createAutoVariable(DIS, "__coro_frame", DFile, LineNum,
818 FrameDITy, true, DINode::FlagArtificial);
819
820 // Subprogram would have ContainedNodes field which records the debug
821 // variables it contained. So we need to add __coro_frame to the
822 // ContainedNodes of it.
823 //
824 // If we don't add __coro_frame to the RetainedNodes, user may get
825 // `no symbol __coro_frame in context` rather than `__coro_frame`
826 // is optimized out, which is more precise.
827 auto RetainedNodes = DIS->getRetainedNodes();
828 SmallVector<Metadata *, 32> RetainedNodesVec(RetainedNodes.begin(),
829 RetainedNodes.end());
830 RetainedNodesVec.push_back(FrameDIVar);
831 DIS->replaceOperandWith(7, (MDTuple::get(F.getContext(), RetainedNodesVec)));
832
833 // Construct the location for the frame debug variable. The column number
834 // is fake but it should be fine.
835 DILocation *DILoc =
836 DILocation::get(DIS->getContext(), LineNum, /*Column=*/1, DIS);
837 assert(FrameDIVar->isValidLocationForIntrinsic(DILoc));
838
839 if (UseNewDbgInfoFormat) {
840 DbgVariableRecord *NewDVR =
841 new DbgVariableRecord(ValueAsMetadata::get(Shape.FramePtr), FrameDIVar,
842 DBuilder.createExpression(), DILoc,
843 DbgVariableRecord::LocationType::Declare);
844 BasicBlock::iterator It = Shape.getInsertPtAfterFramePtr();
845 It->getParent()->insertDbgRecordBefore(NewDVR, It);
846 } else {
847 DBuilder.insertDeclare(Shape.FramePtr, FrameDIVar,
848 DBuilder.createExpression(), DILoc,
849 &*Shape.getInsertPtAfterFramePtr());
850 }
851}

853// Build a struct that will keep state for an active coroutine.
854// struct f.frame {
855// ResumeFnTy ResumeFnAddr;
856// ResumeFnTy DestroyFnAddr;
857// ... promise (if present) ...
858// int ResumeIndex;
859// ... spills ...
860// };
861static StructType *buildFrameType(Function &F, coro::Shape &Shape,
862 FrameDataInfo &FrameData) {
863 LLVMContext &C = F.getContext();
864 const DataLayout &DL = F.getDataLayout();
865 StructType *FrameTy = [&] {
866 SmallString<32> Name(F.getName());
867 Name.append(".Frame");
868 return StructType::create(C, Name);
869 }();
870
871 // We will use this value to cap the alignment of spilled values.
872 std::optional<Align> MaxFrameAlignment;
873 if (Shape.ABI == coro::ABI::Async)
874 MaxFrameAlignment = Shape.AsyncLowering.getContextAlignment();
875 FrameTypeBuilder B(C, DL, MaxFrameAlignment);
876
877 AllocaInst *PromiseAlloca = Shape.getPromiseAlloca();
878 std::optional<FieldIDType> SwitchIndexFieldId;
879
880 if (Shape.ABI == coro::ABI::Switch) {
881 auto *FnPtrTy = PointerType::getUnqual(C);
882
883 // Add header fields for the resume and destroy functions.
884 // We can rely on these being perfectly packed.
885 (void)B.addField(FnPtrTy, std::nullopt, /*header*/ true);
886 (void)B.addField(FnPtrTy, std::nullopt, /*header*/ true);
887
888 // PromiseAlloca field needs to be explicitly added here because it's
889 // a header field with a fixed offset based on its alignment. Hence it
890 // needs special handling and cannot be added to FrameData.Allocas.
891 if (PromiseAlloca)
892 FrameData.setFieldIndex(
893 PromiseAlloca, B.addFieldForAlloca(PromiseAlloca, /*header*/ true));
894
895 // Add a field to store the suspend index. This doesn't need to
896 // be in the header.
897 unsigned IndexBits = std::max(1U, Log2_64_Ceil(Shape.CoroSuspends.size()));
898 Type *IndexType = Type::getIntNTy(C, IndexBits);
899
900 SwitchIndexFieldId = B.addField(IndexType, std::nullopt);
901 } else {
902 assert(PromiseAlloca == nullptr && "lowering doesn't support promises");
903 }
904
905 // Because multiple allocas may own the same field slot,
906 // we add allocas to field here.
907 B.addFieldForAllocas(F, FrameData, Shape);
908 // Add PromiseAlloca to Allocas list so that
909 // 1. updateLayoutIndex could update its index after
910 // `performOptimizedStructLayout`
911 // 2. it is processed in insertSpills.
912 if (Shape.ABI == coro::ABI::Switch && PromiseAlloca)
913 // We assume that the promise alloca won't be modified before
914 // CoroBegin and no alias will be create before CoroBegin.
915 FrameData.Allocas.emplace_back(
916 PromiseAlloca, DenseMap<Instruction *, std::optional<APInt>>{}, false);
917 // Create an entry for every spilled value.
918 for (auto &S : FrameData.Spills) {
919 Type *FieldType = S.first->getType();
920 // For byval arguments, we need to store the pointed value in the frame,
921 // instead of the pointer itself.
922 if (const Argument *A = dyn_cast<Argument>(S.first))
923 if (A->hasByValAttr())
924 FieldType = A->getParamByValType();
925 FieldIDType Id = B.addField(FieldType, std::nullopt, false /*header*/,
926 true /*IsSpillOfValue*/);
927 FrameData.setFieldIndex(S.first, Id);
928 }
929
930 B.finish(FrameTy);
931 FrameData.updateLayoutIndex(B);
932 Shape.FrameAlign = B.getStructAlign();
933 Shape.FrameSize = B.getStructSize();
934
935 switch (Shape.ABI) {
936 case coro::ABI::Switch: {
937 // In the switch ABI, remember the switch-index field.
938 auto IndexField = B.getLayoutField(*SwitchIndexFieldId);
939 Shape.SwitchLowering.IndexField = IndexField.LayoutFieldIndex;
940 Shape.SwitchLowering.IndexAlign = IndexField.Alignment.value();
941 Shape.SwitchLowering.IndexOffset = IndexField.Offset;
942
943 // Also round the frame size up to a multiple of its alignment, as is
944 // generally expected in C/C++.
945 Shape.FrameSize = alignTo(Shape.FrameSize, Shape.FrameAlign);
946 break;
947 }
948
949 // In the retcon ABI, remember whether the frame is inline in the storage.
950 case coro::ABI::Retcon:
951 case coro::ABI::RetconOnce: {
952 auto Id = Shape.getRetconCoroId();
953 Shape.RetconLowering.IsFrameInlineInStorage
954 = (B.getStructSize() <= Id->getStorageSize() &&
955 B.getStructAlign() <= Id->getStorageAlignment());
956 break;
957 }
958 case coro::ABI::Async: {
959 Shape.AsyncLowering.FrameOffset =
960 alignTo(Shape.AsyncLowering.ContextHeaderSize, Shape.FrameAlign);
961 // Also make the final context size a multiple of the context alignment to
962 // make allocation easier for allocators.
963 Shape.AsyncLowering.ContextSize =
964 alignTo(Shape.AsyncLowering.FrameOffset + Shape.FrameSize,
965 Shape.AsyncLowering.getContextAlignment());
966 if (Shape.AsyncLowering.getContextAlignment() < Shape.FrameAlign) {
967 report_fatal_error(
968 "The alignment requirment of frame variables cannot be higher than "
969 "the alignment of the async function context");
970 }
971 break;
972 }
973 }
974
975 return FrameTy;
976}

978// Replace all alloca and SSA values that are accessed across suspend points
979// with GetElementPointer from coroutine frame + loads and stores. Create an
980// AllocaSpillBB that will become the new entry block for the resume parts of
981// the coroutine:
982//
983// %hdl = coro.begin(...)
984// whatever
985//
986// becomes:
987//
988// %hdl = coro.begin(...)
989// br label %AllocaSpillBB
990//
991// AllocaSpillBB:
992// ; geps corresponding to allocas that were moved to coroutine frame
993// br label PostSpill
994//
995// PostSpill:
996// whatever
997//
998//
999static void insertSpills(const FrameDataInfo &FrameData, coro::Shape &Shape) {
1000 LLVMContext &C = Shape.CoroBegin->getContext();
1001 Function *F = Shape.CoroBegin->getFunction();
1002 IRBuilder<> Builder(C);
1003 StructType *FrameTy = Shape.FrameTy;
1004 Value *FramePtr = Shape.FramePtr;
1005 DominatorTree DT(*F);
1006 SmallDenseMap<Argument *, AllocaInst *, 4> ArgToAllocaMap;
1007
1008 // Create a GEP with the given index into the coroutine frame for the original
1009 // value Orig. Appends an extra 0 index for array-allocas, preserving the
1010 // original type.
1011 auto GetFramePointer = [&](Value *Orig) -> Value * {
1012 FieldIDType Index = FrameData.getFieldIndex(Orig);
1013 SmallVector<Value *, 3> Indices = {
1014 ConstantInt::get(Type::getInt32Ty(C), 0),
1015 ConstantInt::get(Type::getInt32Ty(C), Index),
1016 };
1017
1018 if (auto *AI = dyn_cast<AllocaInst>(Orig)) {
1019 if (auto *CI = dyn_cast<ConstantInt>(AI->getArraySize())) {
1020 auto Count = CI->getValue().getZExtValue();
1021 if (Count > 1) {
1022 Indices.push_back(ConstantInt::get(Type::getInt32Ty(C), 0));
1023 }
1024 } else {
1025 report_fatal_error("Coroutines cannot handle non static allocas yet");
1026 }
1027 }
1028
1029 auto GEP = cast<GetElementPtrInst>(
1030 Builder.CreateInBoundsGEP(FrameTy, FramePtr, Indices));
1031 if (auto *AI = dyn_cast<AllocaInst>(Orig)) {
1032 if (FrameData.getDynamicAlign(Orig) != 0) {
1033 assert(FrameData.getDynamicAlign(Orig) == AI->getAlign().value());
1034 auto *M = AI->getModule();
1035 auto *IntPtrTy = M->getDataLayout().getIntPtrType(AI->getType());
1036 auto *PtrValue = Builder.CreatePtrToInt(GEP, IntPtrTy);
1037 auto *AlignMask =
1038 ConstantInt::get(IntPtrTy, AI->getAlign().value() - 1);
1039 PtrValue = Builder.CreateAdd(PtrValue, AlignMask);
1040 PtrValue = Builder.CreateAnd(PtrValue, Builder.CreateNot(AlignMask));
1041 return Builder.CreateIntToPtr(PtrValue, AI->getType());
1042 }
1043 // If the type of GEP is not equal to the type of AllocaInst, it implies
1044 // that the AllocaInst may be reused in the Frame slot of other
1045 // AllocaInst. So We cast GEP to the AllocaInst here to re-use
1046 // the Frame storage.
1047 //
1048 // Note: If we change the strategy dealing with alignment, we need to refine
1049 // this casting.
1050 if (GEP->getType() != Orig->getType())
1051 return Builder.CreateAddrSpaceCast(GEP, Orig->getType(),
1052 Orig->getName() + Twine(".cast"));
1053 }
1054 return GEP;
1055 };
1056
1057 for (auto const &E : FrameData.Spills) {
1058 Value *Def = E.first;
1059 auto SpillAlignment = Align(FrameData.getAlign(Def));
1060 // Create a store instruction storing the value into the
1061 // coroutine frame.
1062 BasicBlock::iterator InsertPt = coro::getSpillInsertionPt(Shape, Def, DT);
1063
1064 Type *ByValTy = nullptr;
1065 if (auto *Arg = dyn_cast<Argument>(Def)) {
1066 // If we're spilling an Argument, make sure we clear 'nocapture'
1067 // from the coroutine function.
1068 Arg->getParent()->removeParamAttr(Arg->getArgNo(), Attribute::NoCapture);
1069
1070 if (Arg->hasByValAttr())
1071 ByValTy = Arg->getParamByValType();
1072 }
1073
1074 auto Index = FrameData.getFieldIndex(Def);
1075 Builder.SetInsertPoint(InsertPt->getParent(), InsertPt);
1076 auto *G = Builder.CreateConstInBoundsGEP2_32(
1077 FrameTy, FramePtr, 0, Index, Def->getName() + Twine(".spill.addr"));
1078 if (ByValTy) {
1079 // For byval arguments, we need to store the pointed value in the frame,
1080 // instead of the pointer itself.
1081 auto *Value = Builder.CreateLoad(ByValTy, Def);
1082 Builder.CreateAlignedStore(Value, G, SpillAlignment);
1083 } else {
1084 Builder.CreateAlignedStore(Def, G, SpillAlignment);
1085 }
1086
1087 BasicBlock *CurrentBlock = nullptr;
1088 Value *CurrentReload = nullptr;
1089 for (auto *U : E.second) {
1090 // If we have not seen the use block, create a load instruction to reload
1091 // the spilled value from the coroutine frame. Populates the Value pointer
1092 // reference provided with the frame GEP.
1093 if (CurrentBlock != U->getParent()) {
1094 CurrentBlock = U->getParent();
1095 Builder.SetInsertPoint(CurrentBlock,
1096 CurrentBlock->getFirstInsertionPt());
1097
1098 auto *GEP = GetFramePointer(E.first);
1099 GEP->setName(E.first->getName() + Twine(".reload.addr"));
1100 if (ByValTy)
1101 CurrentReload = GEP;
1102 else
1103 CurrentReload = Builder.CreateAlignedLoad(
1104 FrameTy->getElementType(FrameData.getFieldIndex(E.first)), GEP,
1105 SpillAlignment, E.first->getName() + Twine(".reload"));
1106
1107 TinyPtrVector<DbgDeclareInst *> DIs = findDbgDeclares(Def);
1108 TinyPtrVector<DbgVariableRecord *> DVRs = findDVRDeclares(Def);
1109 // Try best to find dbg.declare. If the spill is a temp, there may not
1110 // be a direct dbg.declare. Walk up the load chain to find one from an
1111 // alias.
1112 if (F->getSubprogram()) {
1113 auto *CurDef = Def;
1114 while (DIs.empty() && DVRs.empty() && isa<LoadInst>(CurDef)) {
1115 auto *LdInst = cast<LoadInst>(CurDef);
1116 // Only consider ptr to ptr same type load.
1117 if (LdInst->getPointerOperandType() != LdInst->getType())
1118 break;
1119 CurDef = LdInst->getPointerOperand();
1120 if (!isa<AllocaInst, LoadInst>(CurDef))
1121 break;
1122 DIs = findDbgDeclares(CurDef);
1123 DVRs = findDVRDeclares(CurDef);
1124 }
1125 }

1127 auto SalvageOne = [&](auto *DDI) {
1128 bool AllowUnresolved = false;
1129 // This dbg.declare is preserved for all coro-split function
1130 // fragments. It will be unreachable in the main function, and
1131 // processed by coro::salvageDebugInfo() by CoroCloner.
1132 if (UseNewDbgInfoFormat) {
1133 DbgVariableRecord *NewDVR = new DbgVariableRecord(
1134 ValueAsMetadata::get(CurrentReload), DDI->getVariable(),
1135 DDI->getExpression(), DDI->getDebugLoc(),
1136 DbgVariableRecord::LocationType::Declare);
1137 Builder.GetInsertPoint()->getParent()->insertDbgRecordBefore(
1138 NewDVR, Builder.GetInsertPoint());
1139 } else {
1140 DIBuilder(*CurrentBlock->getParent()->getParent(), AllowUnresolved)
1141 .insertDeclare(CurrentReload, DDI->getVariable(),
1142 DDI->getExpression(), DDI->getDebugLoc(),
1143 &*Builder.GetInsertPoint());
1144 }
1145 // This dbg.declare is for the main function entry point. It
1146 // will be deleted in all coro-split functions.
1147 coro::salvageDebugInfo(ArgToAllocaMap, *DDI, false /*UseEntryValue*/);
1148 };
1149 for_each(DIs, SalvageOne);
1150 for_each(DVRs, SalvageOne);
1151 }
1152
1153 // If we have a single edge PHINode, remove it and replace it with a
1154 // reload from the coroutine frame. (We already took care of multi edge
1155 // PHINodes by normalizing them in the rewritePHIs function).
1156 if (auto *PN = dyn_cast<PHINode>(U)) {
1157 assert(PN->getNumIncomingValues() == 1 &&
1158 "unexpected number of incoming "
1159 "values in the PHINode");
1160 PN->replaceAllUsesWith(CurrentReload);
1161 PN->eraseFromParent();
1162 continue;
1163 }
1164
1165 // Replace all uses of CurrentValue in the current instruction with
1166 // reload.
1167 U->replaceUsesOfWith(Def, CurrentReload);
1168 // Instructions are added to Def's user list if the attached
1169 // debug records use Def. Update those now.
1170 for (DbgVariableRecord &DVR : filterDbgVars(U->getDbgRecordRange()))
1171 DVR.replaceVariableLocationOp(Def, CurrentReload, true);
1172 }
1173 }
1174
1175 BasicBlock *FramePtrBB = Shape.getInsertPtAfterFramePtr()->getParent();
1176
1177 auto SpillBlock = FramePtrBB->splitBasicBlock(
1178 Shape.getInsertPtAfterFramePtr(), "AllocaSpillBB");
1179 SpillBlock->splitBasicBlock(&SpillBlock->front(), "PostSpill");
1180 Shape.AllocaSpillBlock = SpillBlock;
1181
1182 // retcon and retcon.once lowering assumes all uses have been sunk.
1183 if (Shape.ABI == coro::ABI::Retcon || Shape.ABI == coro::ABI::RetconOnce ||
1184 Shape.ABI == coro::ABI::Async) {
1185 // If we found any allocas, replace all of their remaining uses with Geps.
1186 Builder.SetInsertPoint(SpillBlock, SpillBlock->begin());
1187 for (const auto &P : FrameData.Allocas) {
1188 AllocaInst *Alloca = P.Alloca;
1189 auto *G = GetFramePointer(Alloca);
1190
1191 // We are not using ReplaceInstWithInst(P.first, cast<Instruction>(G))
1192 // here, as we are changing location of the instruction.
1193 G->takeName(Alloca);
1194 Alloca->replaceAllUsesWith(G);
1195 Alloca->eraseFromParent();
1196 }
1197 return;
1198 }
1199
1200 // If we found any alloca, replace all of their remaining uses with GEP
1201 // instructions. To remain debugbility, we replace the uses of allocas for
1202 // dbg.declares and dbg.values with the reload from the frame.
1203 // Note: We cannot replace the alloca with GEP instructions indiscriminately,
1204 // as some of the uses may not be dominated by CoroBegin.
1205 Builder.SetInsertPoint(Shape.AllocaSpillBlock,
1206 Shape.AllocaSpillBlock->begin());
1207 SmallVector<Instruction *, 4> UsersToUpdate;
1208 for (const auto &A : FrameData.Allocas) {
1209 AllocaInst *Alloca = A.Alloca;
1210 UsersToUpdate.clear();
1211 for (User *U : Alloca->users()) {
1212 auto *I = cast<Instruction>(U);
1213 if (DT.dominates(Shape.CoroBegin, I))
1214 UsersToUpdate.push_back(I);
1215 }
1216 if (UsersToUpdate.empty())
1217 continue;
1218 auto *G = GetFramePointer(Alloca);
1219 G->setName(Alloca->getName() + Twine(".reload.addr"));
1220
1221 SmallVector<DbgVariableIntrinsic *, 4> DIs;
1222 SmallVector<DbgVariableRecord *> DbgVariableRecords;
1223 findDbgUsers(DIs, Alloca, &DbgVariableRecords);
1224 for (auto *DVI : DIs)
1225 DVI->replaceUsesOfWith(Alloca, G);
1226 for (auto *DVR : DbgVariableRecords)
1227 DVR->replaceVariableLocationOp(Alloca, G);
1228
1229 for (Instruction *I : UsersToUpdate) {
1230 // It is meaningless to retain the lifetime intrinsics refer for the
1231 // member of coroutine frames and the meaningless lifetime intrinsics
1232 // are possible to block further optimizations.
1233 if (I->isLifetimeStartOrEnd()) {
1234 I->eraseFromParent();
1235 continue;
1236 }
1237
1238 I->replaceUsesOfWith(Alloca, G);
1239 }
1240 }
1241 Builder.SetInsertPoint(&*Shape.getInsertPtAfterFramePtr());
1242 for (const auto &A : FrameData.Allocas) {
1243 AllocaInst *Alloca = A.Alloca;
1244 if (A.MayWriteBeforeCoroBegin) {
1245 // isEscaped really means potentially modified before CoroBegin.
1246 if (Alloca->isArrayAllocation())
1247 report_fatal_error(
1248 "Coroutines cannot handle copying of array allocas yet");
1249
1250 auto *G = GetFramePointer(Alloca);
1251 auto *Value = Builder.CreateLoad(Alloca->getAllocatedType(), Alloca);
1252 Builder.CreateStore(Value, G);
1253 }
1254 // For each alias to Alloca created before CoroBegin but used after
1255 // CoroBegin, we recreate them after CoroBegin by applying the offset
1256 // to the pointer in the frame.
1257 for (const auto &Alias : A.Aliases) {
1258 auto *FramePtr = GetFramePointer(Alloca);
1259 auto &Value = *Alias.second;
1260 auto ITy = IntegerType::get(C, Value.getBitWidth());
1261 auto *AliasPtr =
1262 Builder.CreatePtrAdd(FramePtr, ConstantInt::get(ITy, Value));
1263 Alias.first->replaceUsesWithIf(
1264 AliasPtr, [&](Use &U) { return DT.dominates(Shape.CoroBegin, U); });
1265 }
1266 }

1268 // PromiseAlloca is not collected in FrameData.Allocas. So we don't handle
1269 // the case that the PromiseAlloca may have writes before CoroBegin in the
1270 // above codes. And it may be problematic in edge cases. See
1271 // https://github.com/llvm/llvm-project/issues/57861 for an example.
1272 if (Shape.ABI == coro::ABI::Switch && Shape.SwitchLowering.PromiseAlloca) {
1273 AllocaInst *PA = Shape.SwitchLowering.PromiseAlloca;
1274 // If there is memory accessing to promise alloca before CoroBegin;
1275 bool HasAccessingPromiseBeforeCB = llvm::any_of(PA->uses(), [&](Use &U) {
1276 auto *Inst = dyn_cast<Instruction>(U.getUser());
1277 if (!Inst || DT.dominates(Shape.CoroBegin, Inst))
1278 return false;
1279
1280 if (auto *CI = dyn_cast<CallInst>(Inst)) {
1281 // It is fine if the call wouldn't write to the Promise.
1282 // This is possible for @llvm.coro.id intrinsics, which
1283 // would take the promise as the second argument as a
1284 // marker.
1285 if (CI->onlyReadsMemory() ||
1286 CI->onlyReadsMemory(CI->getArgOperandNo(&U)))
1287 return false;
1288 return true;
1289 }

1291 return isa<StoreInst>(Inst) ||
1292 // It may take too much time to track the uses.
1293 // Be conservative about the case the use may escape.
1294 isa<GetElementPtrInst>(Inst) ||
1295 // There would always be a bitcast for the promise alloca
1296 // before we enabled Opaque pointers. And now given
1297 // opaque pointers are enabled by default. This should be
1298 // fine.
1299 isa<BitCastInst>(Inst);
1300 });
1301 if (HasAccessingPromiseBeforeCB) {
1302 Builder.SetInsertPoint(&*Shape.getInsertPtAfterFramePtr());
1303 auto *G = GetFramePointer(PA);
1304 auto *Value = Builder.CreateLoad(PA->getAllocatedType(), PA);
1305 Builder.CreateStore(Value, G);
1306 }

1310// Moves the values in the PHIs in SuccBB that correspong to PredBB into a new
1311// PHI in InsertedBB.
1312static void movePHIValuesToInsertedBlock(BasicBlock *SuccBB,
1313 BasicBlock *InsertedBB,
1314 BasicBlock *PredBB,
1315 PHINode *UntilPHI = nullptr) {
1316 auto *PN = cast<PHINode>(&SuccBB->front());

1318 int Index = PN->getBasicBlockIndex(InsertedBB);
1319 Value *V = PN->getIncomingValue(Index);
1320 PHINode *InputV = PHINode::Create(
1321 V->getType(), 1, V->getName() + Twine(".") + SuccBB->getName());

1325 PN = dyn_cast<PHINode>(PN->getNextNode());
1326 } while (PN != UntilPHI);

1329// Rewrites the PHI Nodes in a cleanuppad.
1330static void rewritePHIsForCleanupPad(BasicBlock *CleanupPadBB,
1331 CleanupPadInst *CleanupPad) {
1332 // For every incoming edge to a CleanupPad we will create a new block holding
1333 // all incoming values in single-value PHI nodes. We will then create another
1334 // block to act as a dispather (as all unwind edges for related EH blocks
1335 // must be the same).
1336 //
1337 // cleanuppad:
1338 // %2 = phi i32[%0, %catchswitch], [%1, %catch.1]
1339 // %3 = cleanuppad within none []
1340 //
1341 // It will create:
1342 //
1343 // cleanuppad.corodispatch
1344 // %2 = phi i8[0, %catchswitch], [1, %catch.1]
1345 // %3 = cleanuppad within none []
1346 // switch i8 % 2, label %unreachable
1347 // [i8 0, label %cleanuppad.from.catchswitch
1348 // i8 1, label %cleanuppad.from.catch.1]
1349 // cleanuppad.from.catchswitch:
1350 // %4 = phi i32 [%0, %catchswitch]
1351 // br %label cleanuppad
1352 // cleanuppad.from.catch.1:
1353 // %6 = phi i32 [%1, %catch.1]
1354 // br %label cleanuppad
1355 // cleanuppad:
1356 // %8 = phi i32 [%4, %cleanuppad.from.catchswitch],
1357 // [%6, %cleanuppad.from.catch.1]
1358
1359 // Unreachable BB, in case switching on an invalid value in the dispatcher.
1360 auto *UnreachBB = BasicBlock::Create(
1361 CleanupPadBB->getContext(), "unreachable", CleanupPadBB->getParent());
1362 IRBuilder<> Builder(UnreachBB);
1363 Builder.CreateUnreachable();
1364
1365 // Create a new cleanuppad which will be the dispatcher.
1366 auto *NewCleanupPadBB =
1367 BasicBlock::Create(CleanupPadBB->getContext(),
1368 CleanupPadBB->getName() + Twine(".corodispatch"),
1369 CleanupPadBB->getParent(), CleanupPadBB);
1370 Builder.SetInsertPoint(NewCleanupPadBB);
1371 auto *SwitchType = Builder.getInt8Ty();
1372 auto *SetDispatchValuePN =
1373 Builder.CreatePHI(SwitchType, pred_size(CleanupPadBB));
1374 CleanupPad->removeFromParent();
1375 CleanupPad->insertAfter(SetDispatchValuePN);
1376 auto *SwitchOnDispatch = Builder.CreateSwitch(SetDispatchValuePN, UnreachBB,
1377 pred_size(CleanupPadBB));
1378
1379 int SwitchIndex = 0;
1380 SmallVector<BasicBlock *, 8> Preds(predecessors(CleanupPadBB));
1381 for (BasicBlock *Pred : Preds) {
1382 // Create a new cleanuppad and move the PHI values to there.
1383 auto *CaseBB = BasicBlock::Create(CleanupPadBB->getContext(),
1384 CleanupPadBB->getName() +
1385 Twine(".from.") + Pred->getName(),
1386 CleanupPadBB->getParent(), CleanupPadBB);
1387 updatePhiNodes(CleanupPadBB, Pred, CaseBB);
1388 CaseBB->setName(CleanupPadBB->getName() + Twine(".from.") +
1389 Pred->getName());
1390 Builder.SetInsertPoint(CaseBB);
1391 Builder.CreateBr(CleanupPadBB);
1392 movePHIValuesToInsertedBlock(CleanupPadBB, CaseBB, NewCleanupPadBB);
1393
1394 // Update this Pred to the new unwind point.
1395 setUnwindEdgeTo(Pred->getTerminator(), NewCleanupPadBB);
1396
1397 // Setup the switch in the dispatcher.
1398 auto *SwitchConstant = ConstantInt::get(SwitchType, SwitchIndex);
1399 SetDispatchValuePN->addIncoming(SwitchConstant, Pred);
1400 SwitchOnDispatch->addCase(SwitchConstant, CaseBB);
1401 SwitchIndex++;
1402 }
1403}

1422static void rewritePHIs(BasicBlock &BB) {
1423 // For every incoming edge we will create a block holding all
1424 // incoming values in a single PHI nodes.
1425 //
1426 // loop:
1427 // %n.val = phi i32[%n, %entry], [%inc, %loop]
1428 //
1429 // It will create:
1430 //
1431 // loop.from.entry:
1432 // %n.loop.pre = phi i32 [%n, %entry]
1433 // br %label loop
1434 // loop.from.loop:
1435 // %inc.loop.pre = phi i32 [%inc, %loop]
1436 // br %label loop
1437 //
1438 // After this rewrite, further analysis will ignore any phi nodes with more
1439 // than one incoming edge.
1440
1441 // TODO: Simplify PHINodes in the basic block to remove duplicate
1442 // predecessors.
1443
1444 // Special case for CleanupPad: all EH blocks must have the same unwind edge
1445 // so we need to create an additional "dispatcher" block.
1446 if (auto *CleanupPad =
1447 dyn_cast_or_null<CleanupPadInst>(BB.getFirstNonPHI())) {
1448 SmallVector<BasicBlock *, 8> Preds(predecessors(&BB));
1449 for (BasicBlock *Pred : Preds) {
1450 if (CatchSwitchInst *CS =
1451 dyn_cast<CatchSwitchInst>(Pred->getTerminator())) {
1452 // CleanupPad with a CatchSwitch predecessor: therefore this is an
1453 // unwind destination that needs to be handle specially.
1454 assert(CS->getUnwindDest() == &BB);
1455 (void)CS;
1456 rewritePHIsForCleanupPad(&BB, CleanupPad);
1457 return;
1458 }
1459 }
1460 }
1461
1462 LandingPadInst *LandingPad = nullptr;
1463 PHINode *ReplPHI = nullptr;
1464 if ((LandingPad = dyn_cast_or_null<LandingPadInst>(BB.getFirstNonPHI()))) {
1465 // ehAwareSplitEdge will clone the LandingPad in all the edge blocks.
1466 // We replace the original landing pad with a PHINode that will collect the
1467 // results from all of them.
1468 ReplPHI = PHINode::Create(LandingPad->getType(), 1, "");
1469 ReplPHI->insertBefore(LandingPad->getIterator());
1470 ReplPHI->takeName(LandingPad);
1471 LandingPad->replaceAllUsesWith(ReplPHI);
1472 // We will erase the original landing pad at the end of this function after
1473 // ehAwareSplitEdge cloned it in the transition blocks.
1474 }
1475
1476 SmallVector<BasicBlock *, 8> Preds(predecessors(&BB));
1477 for (BasicBlock *Pred : Preds) {
1478 auto *IncomingBB = ehAwareSplitEdge(Pred, &BB, LandingPad, ReplPHI);
1479 IncomingBB->setName(BB.getName() + Twine(".from.") + Pred->getName());
1480
1481 // Stop the moving of values at ReplPHI, as this is either null or the PHI
1482 // that replaced the landing pad.
1483 movePHIValuesToInsertedBlock(&BB, IncomingBB, Pred, ReplPHI);
1484 }
1485
1486 if (LandingPad) {
1487 // Calls to ehAwareSplitEdge function cloned the original lading pad.
1488 // No longer need it.
1489 LandingPad->eraseFromParent();
1490 }
1491}

1525/// After we split the coroutine, will the given basic block be along
1526/// an obvious exit path for the resumption function?
1527static bool willLeaveFunctionImmediatelyAfter(BasicBlock *BB,
1528 unsigned depth = 3) {
1529 // If we've bottomed out our depth count, stop searching and assume
1530 // that the path might loop back.
1531 if (depth == 0) return false;
1532
1533 // If this is a suspend block, we're about to exit the resumption function.
1534 if (coro::isSuspendBlock(BB))
1535 return true;
1536
1537 // Recurse into the successors.
1538 for (auto *Succ : successors(BB)) {
1539 if (!willLeaveFunctionImmediatelyAfter(Succ, depth - 1))
1540 return false;
1541 }
1542
1543 // If none of the successors leads back in a loop, we're on an exit/abort.
1544 return true;
1545}

1547static bool localAllocaNeedsStackSave(CoroAllocaAllocInst *AI) {
1548 // Look for a free that isn't sufficiently obviously followed by
1549 // either a suspend or a termination, i.e. something that will leave
1550 // the coro resumption frame.
1551 for (auto *U : AI->users()) {
1552 auto FI = dyn_cast<CoroAllocaFreeInst>(U);
1553 if (!FI) continue;
1554
1555 if (!willLeaveFunctionImmediatelyAfter(FI->getParent()))
1556 return true;
1557 }
1558
1559 // If we never found one, we don't need a stack save.
1560 return false;
1561}

1563/// Turn each of the given local allocas into a normal (dynamic) alloca
1564/// instruction.
1565static void lowerLocalAllocas(ArrayRef<CoroAllocaAllocInst*> LocalAllocas,
1566 SmallVectorImpl<Instruction*> &DeadInsts) {
1567 for (auto *AI : LocalAllocas) {
1568 IRBuilder<> Builder(AI);
1569
1570 // Save the stack depth. Try to avoid doing this if the stackrestore
1571 // is going to immediately precede a return or something.
1572 Value *StackSave = nullptr;
1573 if (localAllocaNeedsStackSave(AI))
1574 StackSave = Builder.CreateStackSave();
1575
1576 // Allocate memory.
1577 auto Alloca = Builder.CreateAlloca(Builder.getInt8Ty(), AI->getSize());
1578 Alloca->setAlignment(AI->getAlignment());
1579
1580 for (auto *U : AI->users()) {
1581 // Replace gets with the allocation.
1582 if (isa<CoroAllocaGetInst>(U)) {
1583 U->replaceAllUsesWith(Alloca);
1584
1585 // Replace frees with stackrestores. This is safe because
1586 // alloca.alloc is required to obey a stack discipline, although we
1587 // don't enforce that structurally.
1588 } else {
1589 auto FI = cast<CoroAllocaFreeInst>(U);
1590 if (StackSave) {
1591 Builder.SetInsertPoint(FI);
1592 Builder.CreateStackRestore(StackSave);
1593 }
1594 }
1595 DeadInsts.push_back(cast<Instruction>(U));
1596 }
1597
1598 DeadInsts.push_back(AI);
1599 }

1632/// then put in back in the alloca afterwards.
1633///
1634/// Returns an address that will stand in for the swifterror slot
1635/// until splitting.
1636static Value *emitSetAndGetSwiftErrorValueAround(Instruction *Call,
1637 AllocaInst *Alloca,
1638 coro::Shape &Shape) {
1639 auto ValueTy = Alloca->getAllocatedType();
1640 IRBuilder<> Builder(Call);
1641
1642 // Load the current value from the alloca and set it as the
1643 // swifterror value.
1644 auto ValueBeforeCall = Builder.CreateLoad(ValueTy, Alloca);
1645 auto Addr = emitSetSwiftErrorValue(Builder, ValueBeforeCall, Shape);
1646
1647 // Move to after the call. Since swifterror only has a guaranteed
1648 // value on normal exits, we can ignore implicit and explicit unwind
1649 // edges.

1651 Builder.SetInsertPoint(Call->getNextNode());

1654 Builder.SetInsertPoint(Invoke->getNormalDest()->getFirstNonPHIOrDbg());
1655 }

1657 // Get the current swifterror value and store it to the alloca.
1658 auto ValueAfterCall = emitGetSwiftErrorValue(Builder, ValueTy, Shape);

1664/// Eliminate a formerly-swifterror alloca by inserting the get/set
1665/// intrinsics and attempting to MemToReg the alloca away.
1666static void eliminateSwiftErrorAlloca(Function &F, AllocaInst *Alloca,
1667 coro::Shape &Shape) {
1668 for (Use &Use : llvm::make_early_inc_range(Alloca->uses())) {
1669 // swifterror values can only be used in very specific ways.
1670 // We take advantage of that here.
1671 auto User = Use.getUser();
1672 if (isa<LoadInst>(User) || isa<StoreInst>(User))
1673 continue;
1674
1675 assert(isa<CallInst>(User) || isa<InvokeInst>(User));
1676 auto Call = cast<Instruction>(User);
1677
1678 auto Addr = emitSetAndGetSwiftErrorValueAround(Call, Alloca, Shape);
1679
1680 // Use the returned slot address as the call argument.
1681 Use.set(Addr);
1682 }
1683
1684 // All the uses should be loads and stores now.
1685 assert(isAllocaPromotable(Alloca));
1686}

1688/// "Eliminate" a swifterror argument by reducing it to the alloca case
1689/// and then loading and storing in the prologue and epilog.
1690///
1691/// The argument keeps the swifterror flag.
1692static void eliminateSwiftErrorArgument(Function &F, Argument &Arg,
1693 coro::Shape &Shape,
1694 SmallVectorImpl<AllocaInst*> &AllocasToPromote) {
1695 IRBuilder<> Builder(F.getEntryBlock().getFirstNonPHIOrDbg());
1696
1697 auto ArgTy = cast<PointerType>(Arg.getType());
1698 auto ValueTy = PointerType::getUnqual(F.getContext());
1699
1700 // Reduce to the alloca case:
1701
1702 // Create an alloca and replace all uses of the arg with it.
1703 auto Alloca = Builder.CreateAlloca(ValueTy, ArgTy->getAddressSpace());
1704 Arg.replaceAllUsesWith(Alloca);
1705
1706 // Set an initial value in the alloca. swifterror is always null on entry.
1707 auto InitialValue = Constant::getNullValue(ValueTy);
1708 Builder.CreateStore(InitialValue, Alloca);
1709
1710 // Find all the suspends in the function and save and restore around them.
1711 for (auto *Suspend : Shape.CoroSuspends) {
1712 (void) emitSetAndGetSwiftErrorValueAround(Suspend, Alloca, Shape);
1713 }
1714
1715 // Find all the coro.ends in the function and restore the error value.
1716 for (auto *End : Shape.CoroEnds) {
1717 Builder.SetInsertPoint(End);
1718 auto FinalValue = Builder.CreateLoad(ValueTy, Alloca);
1719 (void) emitSetSwiftErrorValue(Builder, FinalValue, Shape);
1720 }
1721
1722 // Now we can use the alloca logic.
1723 AllocasToPromote.push_back(Alloca);
1724 eliminateSwiftErrorAlloca(F, Alloca, Shape);
1725}

1727/// Eliminate all problematic uses of swifterror arguments and allocas
1728/// from the function. We'll fix them up later when splitting the function.
1729static void eliminateSwiftError(Function &F, coro::Shape &Shape) {
1730 SmallVector<AllocaInst*, 4> AllocasToPromote;
1731
1732 // Look for a swifterror argument.
1733 for (auto &Arg : F.args()) {
1734 if (!Arg.hasSwiftErrorAttr()) continue;
1735
1736 eliminateSwiftErrorArgument(F, Arg, Shape, AllocasToPromote);
1737 break;
1738 }
1739
1740 // Look for swifterror allocas.
1741 for (auto &Inst : F.getEntryBlock()) {

1743 if (!Alloca || !Alloca->isSwiftError()) continue;

1746 Alloca->setSwiftError(false);
1747
1748 AllocasToPromote.push_back(Alloca);
1749 eliminateSwiftErrorAlloca(F, Alloca, Shape);
1750 }
1751
1752 // If we have any allocas to promote, compute a dominator tree and
1753 // promote them en masse.
1754 if (!AllocasToPromote.empty()) {
1755 DominatorTree DT(F);
1756 PromoteMemToReg(AllocasToPromote, DT);
1757 }

1760/// For each local variable that all of its user are only used inside one of
1761/// suspended region, we sink their lifetime.start markers to the place where
1762/// after the suspend block. Doing so minimizes the lifetime of each variable,
1763/// hence minimizing the amount of data we end up putting on the frame.
1764static void sinkLifetimeStartMarkers(Function &F, coro::Shape &Shape,
1765 SuspendCrossingInfo &Checker,
1766 const DominatorTree &DT) {
1767 if (F.hasOptNone())
1768 return;
1769
1770 // Collect all possible basic blocks which may dominate all uses of allocas.
1771 SmallPtrSet<BasicBlock *, 4> DomSet;
1772 DomSet.insert(&F.getEntryBlock());
1773 for (auto *CSI : Shape.CoroSuspends) {
1774 BasicBlock *SuspendBlock = CSI->getParent();
1775 assert(coro::isSuspendBlock(SuspendBlock) &&
1776 SuspendBlock->getSingleSuccessor() &&
1777 "should have split coro.suspend into its own block");
1778 DomSet.insert(SuspendBlock->getSingleSuccessor());
1779 }
1780
1781 for (Instruction &I : instructions(F)) {
1782 AllocaInst* AI = dyn_cast<AllocaInst>(&I);
1783 if (!AI)
1784 continue;
1785
1786 for (BasicBlock *DomBB : DomSet) {
1787 bool Valid = true;
1788 SmallVector<Instruction *, 1> Lifetimes;
1789
1790 auto isLifetimeStart = [](Instruction* I) {
1791 if (auto* II = dyn_cast<IntrinsicInst>(I))
1792 return II->getIntrinsicID() == Intrinsic::lifetime_start;
1793 return false;
1794 };
1795
1796 auto collectLifetimeStart = [&](Instruction *U, AllocaInst *AI) {
1797 if (isLifetimeStart(U)) {
1798 Lifetimes.push_back(U);
1799 return true;
1800 }
1801 if (!U->hasOneUse() || U->stripPointerCasts() != AI)
1802 return false;
1803 if (isLifetimeStart(U->user_back())) {
1804 Lifetimes.push_back(U->user_back());
1805 return true;
1806 }
1807 return false;
1808 };
1809
1810 for (User *U : AI->users()) {
1811 Instruction *UI = cast<Instruction>(U);
1812 // For all users except lifetime.start markers, if they are all
1813 // dominated by one of the basic blocks and do not cross
1814 // suspend points as well, then there is no need to spill the
1815 // instruction.
1816 if (!DT.dominates(DomBB, UI->getParent()) ||
1817 Checker.isDefinitionAcrossSuspend(DomBB, UI)) {
1818 // Skip lifetime.start, GEP and bitcast used by lifetime.start
1819 // markers.
1820 if (collectLifetimeStart(UI, AI))
1821 continue;
1822 Valid = false;
1823 break;
1824 }
1825 }
1826 // Sink lifetime.start markers to dominate block when they are
1827 // only used outside the region.
1828 if (Valid && Lifetimes.size() != 0) {
1829 auto *NewLifetime = Lifetimes[0]->clone();
1830 NewLifetime->replaceUsesOfWith(NewLifetime->getOperand(1), AI);
1831 NewLifetime->insertBefore(DomBB->getTerminator());
1832
1833 // All the outsided lifetime.start markers are no longer necessary.
1834 for (Instruction *S : Lifetimes)
1835 S->eraseFromParent();
1836
1837 break;
1838 }
1839 }
1840 }

1843static std::optional<std::pair<Value &, DIExpression &>>
1844salvageDebugInfoImpl(SmallDenseMap<Argument *, AllocaInst *, 4> &ArgToAllocaMap,
1845 bool UseEntryValue, Function *F, Value *Storage,
1846 DIExpression *Expr, bool SkipOutermostLoad) {
1847 IRBuilder<> Builder(F->getContext());
1848 auto InsertPt = F->getEntryBlock().getFirstInsertionPt();
1849 while (isa<IntrinsicInst>(InsertPt))
1850 ++InsertPt;
1851 Builder.SetInsertPoint(&F->getEntryBlock(), InsertPt);
1852
1853 while (auto *Inst = dyn_cast_or_null<Instruction>(Storage)) {
1854 if (auto *LdInst = dyn_cast<LoadInst>(Inst)) {
1855 Storage = LdInst->getPointerOperand();
1856 // FIXME: This is a heuristic that works around the fact that
1857 // LLVM IR debug intrinsics cannot yet distinguish between
1858 // memory and value locations: Because a dbg.declare(alloca) is
1859 // implicitly a memory location no DW_OP_deref operation for the
1860 // last direct load from an alloca is necessary. This condition
1861 // effectively drops the *last* DW_OP_deref in the expression.
1862 if (!SkipOutermostLoad)
1863 Expr = DIExpression::prepend(Expr, DIExpression::DerefBefore);
1864 } else if (auto *StInst = dyn_cast<StoreInst>(Inst)) {
1865 Storage = StInst->getValueOperand();
1866 } else {
1867 SmallVector<uint64_t, 16> Ops;
1868 SmallVector<Value *, 0> AdditionalValues;
1869 Value *Op = llvm::salvageDebugInfoImpl(
1870 *Inst, Expr ? Expr->getNumLocationOperands() : 0, Ops,
1871 AdditionalValues);
1872 if (!Op || !AdditionalValues.empty()) {
1873 // If salvaging failed or salvaging produced more than one location
1874 // operand, give up.
1875 break;
1876 }
1877 Storage = Op;
1878 Expr = DIExpression::appendOpsToArg(Expr, Ops, 0, /*StackValue*/ false);
1879 }
1880 SkipOutermostLoad = false;
1881 }
1882 if (!Storage)
1883 return std::nullopt;
1884
1885 auto *StorageAsArg = dyn_cast<Argument>(Storage);
1886 const bool IsSwiftAsyncArg =
1887 StorageAsArg && StorageAsArg->hasAttribute(Attribute::SwiftAsync);
1888
1889 // Swift async arguments are described by an entry value of the ABI-defined
1890 // register containing the coroutine context.
1891 // Entry values in variadic expressions are not supported.
1892 if (IsSwiftAsyncArg && UseEntryValue && !Expr->isEntryValue() &&
1893 Expr->isSingleLocationExpression())
1894 Expr = DIExpression::prepend(Expr, DIExpression::EntryValue);
1895
1896 // If the coroutine frame is an Argument, store it in an alloca to improve
1897 // its availability (e.g. registers may be clobbered).
1898 // Avoid this if the value is guaranteed to be available through other means
1899 // (e.g. swift ABI guarantees).
1900 if (StorageAsArg && !IsSwiftAsyncArg) {
1901 auto &Cached = ArgToAllocaMap[StorageAsArg];
1902 if (!Cached) {
1903 Cached = Builder.CreateAlloca(Storage->getType(), 0, nullptr,
1904 Storage->getName() + ".debug");
1905 Builder.CreateStore(Storage, Cached);
1906 }
1907 Storage = Cached;
1908 // FIXME: LLVM lacks nuanced semantics to differentiate between
1909 // memory and direct locations at the IR level. The backend will
1910 // turn a dbg.declare(alloca, ..., DIExpression()) into a memory
1911 // location. Thus, if there are deref and offset operations in the
1912 // expression, we need to add a DW_OP_deref at the *start* of the
1913 // expression to first load the contents of the alloca before
1914 // adjusting it with the expression.
1915 Expr = DIExpression::prepend(Expr, DIExpression::DerefBefore);
1916 }
1917
1918 Expr = Expr->foldConstantMath();
1919 return {{*Storage, *Expr}};
1920}

2008void coro::normalizeCoroutine(Function &F, coro::Shape &Shape,
2009 TargetTransformInfo &TTI) {
2010 // Don't eliminate swifterror in async functions that won't be split.
2011 if (Shape.ABI != coro::ABI::Async || !Shape.CoroSuspends.empty())
2012 eliminateSwiftError(F, Shape);
2013
2014 if (Shape.ABI == coro::ABI::Switch &&
2015 Shape.SwitchLowering.PromiseAlloca) {
2016 Shape.getSwitchCoroId()->clearPromise();
2017 }
2018
2019 // Make sure that all coro.save, coro.suspend and the fallthrough coro.end
2020 // intrinsics are in their own blocks to simplify the logic of building up
2021 // SuspendCrossing data.
2022 for (auto *CSI : Shape.CoroSuspends) {
2023 if (auto *Save = CSI->getCoroSave())
2024 splitAround(Save, "CoroSave");
2025 splitAround(CSI, "CoroSuspend");
2026 }
2027
2028 // Put CoroEnds into their own blocks.
2029 for (AnyCoroEndInst *CE : Shape.CoroEnds) {
2030 splitAround(CE, "CoroEnd");
2031
2032 // Emit the musttail call function in a new block before the CoroEnd.
2033 // We do this here so that the right suspend crossing info is computed for
2034 // the uses of the musttail call function call. (Arguments to the coro.end
2035 // instructions would be ignored)
2036 if (auto *AsyncEnd = dyn_cast<CoroAsyncEndInst>(CE)) {
2037 auto *MustTailCallFn = AsyncEnd->getMustTailCallFunction();
2038 if (!MustTailCallFn)
2039 continue;
2040 IRBuilder<> Builder(AsyncEnd);
2041 SmallVector<Value *, 8> Args(AsyncEnd->args());
2042 auto Arguments = ArrayRef<Value *>(Args).drop_front(3);
2043 auto *Call = coro::createMustTailCall(
2044 AsyncEnd->getDebugLoc(), MustTailCallFn, TTI, Arguments, Builder);
2045 splitAround(Call, "MustTailCall.Before.CoroEnd");
2046 }
2047 }
2048
2049 // Later code makes structural assumptions about single predecessors phis e.g
2050 // that they are not live across a suspend point.
2051 cleanupSinglePredPHIs(F);
2052
2053 // Transforms multi-edge PHI Nodes, so that any value feeding into a PHI will
2054 // never have its definition separated from the PHI by the suspend point.
2055 rewritePHIs(F);
2056}

2091 Shape.FrameTy = buildFrameType(F, Shape, FrameData);
