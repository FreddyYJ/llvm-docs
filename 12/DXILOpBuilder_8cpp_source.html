60static const char *getOverloadTypeName(OverloadKind Kind) {
61 switch (Kind) {
62 case OverloadKind::HALF:
63 return "f16";
64 case OverloadKind::FLOAT:
65 return "f32";
66 case OverloadKind::DOUBLE:
67 return "f64";
68 case OverloadKind::I1:
69 return "i1";
70 case OverloadKind::I8:
71 return "i8";
72 case OverloadKind::I16:
73 return "i16";
74 case OverloadKind::I32:
75 return "i32";
76 case OverloadKind::I64:
77 return "i64";
78 case OverloadKind::VOID:
79 case OverloadKind::UNDEFINED:
80 return "void";
81 case OverloadKind::ObjectType:
82 case OverloadKind::UserDefineType:
83 break;
84 }
85 llvm_unreachable("invalid overload type for name");
86}

88static OverloadKind getOverloadKind(Type *Ty) {
89 if (!Ty)
90 return OverloadKind::VOID;
91
92 Type::TypeID T = Ty->getTypeID();
93 switch (T) {
94 case Type::VoidTyID:
95 return OverloadKind::VOID;
96 case Type::HalfTyID:
97 return OverloadKind::HALF;
98 case Type::FloatTyID:
99 return OverloadKind::FLOAT;
100 case Type::DoubleTyID:
101 return OverloadKind::DOUBLE;
102 case Type::IntegerTyID: {
103 IntegerType *ITy = cast<IntegerType>(Ty);
104 unsigned Bits = ITy->getBitWidth();
105 switch (Bits) {
106 case 1:
107 return OverloadKind::I1;
108 case 8:
109 return OverloadKind::I8;
110 case 16:
111 return OverloadKind::I16;
112 case 32:
113 return OverloadKind::I32;
114 case 64:
115 return OverloadKind::I64;
116 default:
117 llvm_unreachable("invalid overload type");
118 return OverloadKind::VOID;
119 }
120 }
121 case Type::PointerTyID:
122 return OverloadKind::UserDefineType;
123 case Type::StructTyID: {
124 // TODO: This is a hack. As described in DXILEmitter.cpp, we need to rework
125 // how we're handling overloads and remove the `OverloadKind` proxy enum.
126 StructType *ST = cast<StructType>(Ty);
127 return getOverloadKind(ST->getElementType(0));
128 }
129 default:
130 return OverloadKind::UNDEFINED;
131 }
132}

134static std::string getTypeName(OverloadKind Kind, Type *Ty) {
135 if (Kind < OverloadKind::UserDefineType) {
136 return getOverloadTypeName(Kind);
137 } else if (Kind == OverloadKind::UserDefineType) {
138 StructType *ST = cast<StructType>(Ty);
139 return ST->getStructName().str();
140 } else if (Kind == OverloadKind::ObjectType) {
141 StructType *ST = cast<StructType>(Ty);
142 return ST->getStructName().str();
143 } else {
144 std::string Str;
145 raw_string_ostream OS(Str);
146 Ty->print(OS);
147 return OS.str();
148 }
149}

172static std::string constructOverloadName(OverloadKind Kind, Type *Ty,
173 const OpCodeProperty &Prop) {
174 if (Kind == OverloadKind::VOID) {
175 return (Twine(DXILOpNamePrefix) + getOpCodeClassName(Prop)).str();
176 }
177 return (Twine(DXILOpNamePrefix) + getOpCodeClassName(Prop) + "." +
178 getTypeName(Kind, Ty))
179 .str();
180}

182static std::string constructOverloadTypeName(OverloadKind Kind,
183 StringRef TypeName) {
184 if (Kind == OverloadKind::VOID)
185 return TypeName.str();
186
187 assert(Kind < OverloadKind::UserDefineType && "invalid overload kind");
188 return (Twine(TypeName) + getOverloadTypeName(Kind)).str();

323/// Construct DXIL function type. This is the type of a function with
324/// the following prototype
325/// OverloadType dx.op.<opclass>.<return-type>(int opcode, <param types>)
326/// <param-types> are constructed from types in Prop.
327static FunctionType *getDXILOpFunctionType(dxil::OpCode OpCode,
328 LLVMContext &Context,
329 Type *OverloadTy) {
330
331 switch (OpCode) {
332#define DXIL_OP_FUNCTION_TYPE(OpCode, RetType, ...) \
333 case OpCode: \
334 return FunctionType::get( \
335 getTypeFromOpParamType(RetType, Context, OverloadTy), \
336 getArgTypesFromOpParamTypes({__VA_ARGS__}, Context, OverloadTy), \
337 /*isVarArg=*/false);
338#include "DXILOperation.inc"
339 }
340 llvm_unreachable("Invalid OpCode?");
341}

343/// Get index of the property from PropList valid for the most recent
344/// DXIL version not greater than DXILVer.
345/// PropList is expected to be sorted in ascending order of DXIL version.
346template <typename T>
347static std::optional<size_t> getPropIndex(ArrayRef<T> PropList,
348 const VersionTuple DXILVer) {
349 size_t Index = PropList.size() - 1;
350 for (auto Iter = PropList.rbegin(); Iter != PropList.rend();
351 Iter++, Index--) {
352 const T &Prop = *Iter;
353 if (VersionTuple(Prop.DXILVersion.Major, Prop.DXILVersion.Minor) <=
354 DXILVer) {
355 return Index;
356 }
357 }
358 return std::nullopt;
359}

387Expected<CallInst *> DXILOpBuilder::tryCreateOp(dxil::OpCode OpCode,
388 ArrayRef<Value *> Args,
389 const Twine &Name,
390 Type *RetTy) {
391 const OpCodeProperty *Prop = getOpCodeProperty(OpCode);
392
393 Type *OverloadTy = nullptr;
394 if (Prop->OverloadParamIndex == 0) {
395 if (!RetTy)
396 return makeOpError(OpCode, "Op overloaded on unknown return type");
397 OverloadTy = RetTy;
398 } else if (Prop->OverloadParamIndex > 0) {
399 // The index counts including the return type
400 unsigned ArgIndex = Prop->OverloadParamIndex - 1;
401 if (static_cast<unsigned>(ArgIndex) >= Args.size())
402 return makeOpError(OpCode, "Wrong number of arguments");
403 OverloadTy = Args[ArgIndex]->getType();
404 }
405
406 FunctionType *DXILOpFT =
407 getDXILOpFunctionType(OpCode, M.getContext(), OverloadTy);
408
409 std::optional<size_t> OlIndexOrErr =
410 getPropIndex(ArrayRef(Prop->Overloads), DXILVersion);
411 if (!OlIndexOrErr.has_value())
412 return makeOpError(OpCode, Twine("No valid overloads for DXIL version ") +
413 DXILVersion.getAsString());
414
415 uint16_t ValidTyMask = Prop->Overloads[*OlIndexOrErr].ValidTys;
416
417 OverloadKind Kind = getOverloadKind(OverloadTy);
418
419 // Check if the operation supports overload types and OverloadTy is valid
420 // per the specified types for the operation
421 if ((ValidTyMask != OverloadKind::UNDEFINED) &&
422 (ValidTyMask & (uint16_t)Kind) == 0)
423 return makeOpError(OpCode, "Invalid overload type");
424
425 // Perform necessary checks to ensure Opcode is valid in the targeted shader
426 // kind
427 std::optional<size_t> StIndexOrErr =
428 getPropIndex(ArrayRef(Prop->Stages), DXILVersion);
429 if (!StIndexOrErr.has_value())
430 return makeOpError(OpCode, Twine("No valid stage for DXIL version ") +
431 DXILVersion.getAsString());
432
433 uint16_t ValidShaderKindMask = Prop->Stages[*StIndexOrErr].ValidStages;
434
435 // Ensure valid shader stage properties are specified
436 if (ValidShaderKindMask == ShaderKind::removed)
437 return makeOpError(OpCode, "Operation has been removed");
438
439 // Shader stage need not be validated since getShaderKindEnum() fails
440 // for unknown shader stage.
441
442 // Verify the target shader stage is valid for the DXIL operation
443 ShaderKind ModuleStagekind = getShaderKindEnum(ShaderStage);
444 if (!(ValidShaderKindMask & ModuleStagekind))
445 return makeOpError(OpCode, "Invalid stage");
446
447 std::string DXILFnName = constructOverloadName(Kind, OverloadTy, *Prop);
448 FunctionCallee DXILFn = M.getOrInsertFunction(DXILFnName, DXILOpFT);
449
450 // We need to inject the opcode as the first argument.
451 SmallVector<Value *> OpArgs;
452 OpArgs.push_back(IRB.getInt32(llvm::to_underlying(OpCode)));
453 OpArgs.append(Args.begin(), Args.end());
454
455 return IRB.CreateCall(DXILFn, OpArgs, Name);
456}
