30// RematGraph is used to construct a DAG for rematerializable instructions
31// When the constructor is invoked with a candidate instruction (which is
32// materializable) it builds a DAG of materializable instructions from that
33// point.
34// Typically, for each instruction identified as re-materializable across a
35// suspend point, a RematGraph will be created.
36struct RematGraph {
37 // Each RematNode in the graph contains the edges to instructions providing
38 // operands in the current node.
39 struct RematNode {
40 Instruction *Node;
41 SmallVector<RematNode *> Operands;
42 RematNode() = default;
43 RematNode(Instruction *V) : Node(V) {}
44 };
45
46 RematNode *EntryNode;
47 using RematNodeMap =
48 SmallMapVector<Instruction *, std::unique_ptr<RematNode>, 8>;
49 RematNodeMap Remats;
50 const std::function<bool(Instruction &)> &MaterializableCallback;
51 SuspendCrossingInfo &Checker;
52
53 RematGraph(const std::function<bool(Instruction &)> &MaterializableCallback,
54 Instruction *I, SuspendCrossingInfo &Checker)
55 : MaterializableCallback(MaterializableCallback), Checker(Checker) {
56 std::unique_ptr<RematNode> FirstNode = std::make_unique<RematNode>(I);
57 EntryNode = FirstNode.get();
58 std::deque<std::unique_ptr<RematNode>> WorkList;
59 addNode(std::move(FirstNode), WorkList, cast<User>(I));
60 while (WorkList.size()) {
61 std::unique_ptr<RematNode> N = std::move(WorkList.front());
62 WorkList.pop_front();
63 addNode(std::move(N), WorkList, cast<User>(I));
64 }
65 }
66
67 void addNode(std::unique_ptr<RematNode> NUPtr,
68 std::deque<std::unique_ptr<RematNode>> &WorkList,
69 User *FirstUse) {
70 RematNode *N = NUPtr.get();
71 if (Remats.count(N->Node))
72 return;
73
74 // We haven't see this node yet - add to the list
75 Remats[N->Node] = std::move(NUPtr);
76 for (auto &Def : N->Node->operands()) {
77 Instruction *D = dyn_cast<Instruction>(Def.get());
78 if (!D || !MaterializableCallback(*D) ||
79 !Checker.isDefinitionAcrossSuspend(*D, FirstUse))
80 continue;
81
82 if (Remats.count(D)) {
83 // Already have this in the graph
84 N->Operands.push_back(Remats[D].get());
85 continue;
86 }
87
88 bool NoMatch = true;
89 for (auto &I : WorkList) {
90 if (I->Node == D) {
91 NoMatch = false;
92 N->Operands.push_back(I.get());
93 break;
94 }
95 }
96 if (NoMatch) {
97 // Create a new node
98 std::unique_ptr<RematNode> ChildNode = std::make_unique<RematNode>(D);
99 N->Operands.push_back(ChildNode.get());
100 WorkList.push_back(std::move(ChildNode));
101 }
102 }
103 }

145// For each instruction identified as materializable across the suspend point,
146// and its associated DAG of other rematerializable instructions,
147// recreate the DAG of instructions after the suspend point.
148static void rewriteMaterializableInstructions(
149 const SmallMapVector<Instruction *, std::unique_ptr<RematGraph>, 8>
150 &AllRemats) {
151 // This has to be done in 2 phases
152 // Do the remats and record the required defs to be replaced in the
153 // original use instructions
154 // Once all the remats are complete, replace the uses in the final
155 // instructions with the new defs
156 typedef struct {
157 Instruction *Use;
158 Instruction *Def;
159 Instruction *Remat;
160 } ProcessNode;
161
162 SmallVector<ProcessNode> FinalInstructionsToProcess;
163
164 for (const auto &E : AllRemats) {
165 Instruction *Use = E.first;
166 Instruction *CurrentMaterialization = nullptr;
167 RematGraph *RG = E.second.get();
168 ReversePostOrderTraversal<RematGraph *> RPOT(RG);
169 SmallVector<Instruction *> InstructionsToProcess;
170
171 // If the target use is actually a suspend instruction then we have to
172 // insert the remats into the end of the predecessor (there should only be
173 // one). This is so that suspend blocks always have the suspend instruction
174 // as the first instruction.
175 auto InsertPoint = &*Use->getParent()->getFirstInsertionPt();
176 if (isa<AnyCoroSuspendInst>(Use)) {
177 BasicBlock *SuspendPredecessorBlock =
178 Use->getParent()->getSinglePredecessor();
179 assert(SuspendPredecessorBlock && "malformed coro suspend instruction");
180 InsertPoint = SuspendPredecessorBlock->getTerminator();
181 }
182
183 // Note: skip the first instruction as this is the actual use that we're
184 // rematerializing everything for.
185 auto I = RPOT.begin();
186 ++I;
187 for (; I != RPOT.end(); ++I) {
188 Instruction *D = (*I)->Node;
189 CurrentMaterialization = D->clone();
190 CurrentMaterialization->setName(D->getName());
191 CurrentMaterialization->insertBefore(InsertPoint);
192 InsertPoint = CurrentMaterialization;
193
194 // Replace all uses of Def in the instructions being added as part of this
195 // rematerialization group
196 for (auto &I : InstructionsToProcess)
197 I->replaceUsesOfWith(D, CurrentMaterialization);
198
199 // Don't replace the final use at this point as this can cause problems
200 // for other materializations. Instead, for any final use that uses a
201 // define that's being rematerialized, record the replace values
202 for (unsigned i = 0, E = Use->getNumOperands(); i != E; ++i)
203 if (Use->getOperand(i) == D) // Is this operand pointing to oldval?
204 FinalInstructionsToProcess.push_back(
205 {Use, D, CurrentMaterialization});
206
207 InstructionsToProcess.push_back(CurrentMaterialization);
208 }
209 }
210
211 // Finally, replace the uses with the defines that we've just rematerialized
212 for (auto &R : FinalInstructionsToProcess) {
213 if (auto *PN = dyn_cast<PHINode>(R.Use)) {
214 assert(PN->getNumIncomingValues() == 1 && "unexpected number of incoming "
215 "values in the PHINode");
216 PN->replaceAllUsesWith(R.Remat);
217 PN->eraseFromParent();
218 continue;
219 }
220 R.Use->replaceUsesOfWith(R.Def, R.Remat);
221 }
222}

224/// Default materializable callback
225// Check for instructions that we can recreate on resume as opposed to spill
226// the result into a coroutine frame.
227bool llvm::coro::defaultMaterializable(Instruction &V) {
228 return (isa<CastInst>(&V) || isa<GetElementPtrInst>(&V) ||
229 isa<BinaryOperator>(&V) || isa<CmpInst>(&V) || isa<SelectInst>(&V));
230}

248void coro::doRematerializations(
249 Function &F, SuspendCrossingInfo &Checker,
250 std::function<bool(Instruction &)> IsMaterializable) {
251 if (F.hasOptNone())
252 return;
253
254 coro::SpillInfo Spills;
255
256 // See if there are materializable instructions across suspend points
257 // We record these as the starting point to also identify materializable
258 // defs of uses in these operations
259 for (Instruction &I : instructions(F)) {
260 if (!IsMaterializable(I))
261 continue;
262 for (User *U : I.users())
263 if (Checker.isDefinitionAcrossSuspend(I, U))
264 Spills[&I].push_back(cast<Instruction>(U));
265 }
266
267 // Process each of the identified rematerializable instructions
268 // and add predecessor instructions that can also be rematerialized.
269 // This is actually a graph of instructions since we could potentially
270 // have multiple uses of a def in the set of predecessor instructions.
271 // The approach here is to maintain a graph of instructions for each bottom
272 // level instruction - where we have a unique set of instructions (nodes)
273 // and edges between them. We then walk the graph in reverse post-dominator
274 // order to insert them past the suspend point, but ensure that ordering is
275 // correct. We also rely on CSE removing duplicate defs for remats of
276 // different instructions with a def in common (rather than maintaining more
277 // complex graphs for each suspend point)
278
279 // We can do this by adding new nodes to the list for each suspend
280 // point. Then using standard GraphTraits to give a reverse post-order
281 // traversal when we insert the nodes after the suspend
282 SmallMapVector<Instruction *, std::unique_ptr<RematGraph>, 8> AllRemats;
283 for (auto &E : Spills) {
284 for (Instruction *U : E.second) {
285 // Don't process a user twice (this can happen if the instruction uses
286 // more than one rematerializable def)
287 if (AllRemats.count(U))
288 continue;
289
290 // Constructor creates the whole RematGraph for the given Use
291 auto RematUPtr =
292 std::make_unique<RematGraph>(IsMaterializable, U, Checker);
293
294 LLVM_DEBUG(dbgs() << "***** Next remat group *****\n";
295 ReversePostOrderTraversal<RematGraph *> RPOT(RematUPtr.get());
296 for (auto I = RPOT.begin(); I != RPOT.end();
297 ++I) { (*I)->Node->dump(); } dbgs()
298 << "\n";);
299
300 AllRemats[U] = std::move(RematUPtr);
301 }
302 }
303
304 // Rewrite materializable instructions to be materialized at the use
305 // point.
306 LLVM_DEBUG(dumpRemats("Materializations", AllRemats));
307 rewriteMaterializableInstructions(AllRemats);
308}
