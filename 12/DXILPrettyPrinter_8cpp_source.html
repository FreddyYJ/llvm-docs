23static StringRef getRCName(dxil::ResourceClass RC) {
24 switch (RC) {
25 case dxil::ResourceClass::SRV:
26 return "SRV";
27 case dxil::ResourceClass::UAV:
28 return "UAV";
29 case dxil::ResourceClass::CBuffer:
30 return "cbuffer";
31 case dxil::ResourceClass::Sampler:
32 return "sampler";
33 }
34 llvm_unreachable("covered switch");
35}

51static StringRef getFormatName(const dxil::ResourceInfo &RI) {
52 if (RI.isTyped()) {
53 switch (RI.getTyped().ElementTy) {
54 case dxil::ElementType::I1:

56 case dxil::ElementType::I16:
57 return "i16";
58 case dxil::ElementType::U16:
59 return "u16";
60 case dxil::ElementType::I32:
61 return "i32";
62 case dxil::ElementType::U32:
63 return "u32";
64 case dxil::ElementType::I64:
65 return "i64";
66 case dxil::ElementType::U64:
67 return "u64";
68 case dxil::ElementType::F16:
69 return "f16";
70 case dxil::ElementType::F32:
71 return "f32";
72 case dxil::ElementType::F64:
73 return "f64";
74 case dxil::ElementType::SNormF16:
75 return "snorm_f16";
76 case dxil::ElementType::UNormF16:
77 return "unorm_f16";
78 case dxil::ElementType::SNormF32:
79 return "snorm_f32";
80 case dxil::ElementType::UNormF32:
81 return "unorm_f32";
82 case dxil::ElementType::SNormF64:
83 return "snorm_f64";
84 case dxil::ElementType::UNormF64:
85 return "unorm_f64";
86 case dxil::ElementType::PackedS8x32:
87 return "p32i8";
88 case dxil::ElementType::PackedU8x32:
89 return "p32u8";
90 case dxil::ElementType::Invalid:
91 llvm_unreachable("Invalid ElementType");
92 }
93 llvm_unreachable("Unhandled ElementType");
94 } else if (RI.isStruct())
95 return "struct";
96 else if (RI.isCBuffer() || RI.isSampler())
97 return "NA";
98 return "byte";

141struct FormatResourceDimension
142 : public llvm::FormatAdapter<const dxil::ResourceInfo &> {
143 explicit FormatResourceDimension(const dxil::ResourceInfo &RI)
144 : llvm::FormatAdapter<const dxil::ResourceInfo &>(RI) {}
145
146 void format(llvm::raw_ostream &OS, StringRef Style) override {
147 dxil::ResourceKind RK = Item.getResourceKind();
148 switch (RK) {
149 default: {
150 OS << getTextureDimName(RK);
151 if (Item.isMultiSample())
152 OS << Item.getMultiSample().Count;
153 break;
154 }
155 case dxil::ResourceKind::RawBuffer:
156 case dxil::ResourceKind::StructuredBuffer:
157 if (!Item.isUAV())
158 OS << "r/o";
159 else if (Item.getUAV().HasCounter)
160 OS << "r/w+cnt";
161 else
162 OS << "r/w";
163 break;
164 case dxil::ResourceKind::TypedBuffer:
165 OS << "buf";
166 break;
167 case dxil::ResourceKind::RTAccelerationStructure:
168 // TODO: dxc would print "ras" here. Can/should this happen?
169 llvm_unreachable("RTAccelerationStructure printing is not implemented");
170 }
171 }
172};

174struct FormatBindingID
175 : public llvm::FormatAdapter<const dxil::ResourceInfo &> {
176 explicit FormatBindingID(const dxil::ResourceInfo &RI)
177 : llvm::FormatAdapter<const dxil::ResourceInfo &>(RI) {}

182 }

185struct FormatBindingLocation
186 : public llvm::FormatAdapter<const dxil::ResourceInfo &> {
187 explicit FormatBindingLocation(const dxil::ResourceInfo &RI)
188 : llvm::FormatAdapter<const dxil::ResourceInfo &>(RI) {}
189
190 void format(llvm::raw_ostream &OS, StringRef Style) override {
191 const auto &Binding = Item.getBinding();
192 OS << getRCPrefix(Item.getResourceClass()) << Binding.LowerBound;
193 if (Binding.Space)
194 OS << ",space" << Binding.Space;
195 }
196};

214static void prettyPrintResources(raw_ostream &OS, const DXILResourceMap &DRM,
215 const dxil::Resources &MDResources) {
216 // Column widths are arbitrary but match the widths DXC uses.
217 OS << ";\n; Resource Bindings:\n;\n";
218 OS << formatv("; {0,-30} {1,10} {2,7} {3,11} {4,7} {5,14} {6,9}\n", "Name",
219 "Type", "Format", "Dim", "ID", "HLSL Bind", "Count");
220 OS << formatv(
221 "; {0,-+30} {1,-+10} {2,-+7} {3,-+11} {4,-+7} {5,-+14} {6,-+9}\n", "", "",
222 "", "", "", "", "");
223
224 // TODO: Do we want to sort these by binding or something like that?
225 for (const dxil::ResourceInfo &RI : DRM) {
226 dxil::ResourceClass RC = RI.getResourceClass();
227 assert((RC != dxil::ResourceClass::CBuffer || !MDResources.hasCBuffers()) &&
228 "Old and new cbuffer representations can't coexist");
229 assert((RC != dxil::ResourceClass::UAV || !MDResources.hasUAVs()) &&
230 "Old and new UAV representations can't coexist");
231
232 StringRef Name(RI.getName());
233 StringRef Type(getRCName(RC));
234 StringRef Format(getFormatName(RI));
235 FormatResourceDimension Dim(RI);
236 FormatBindingID ID(RI);
237 FormatBindingLocation Bind(RI);
238 FormatBindingSize Count(RI);
239 OS << formatv("; {0,-30} {1,10} {2,7} {3,11} {4,7} {5,14} {6,9}\n", Name,
240 Type, Format, Dim, ID, Bind, Count);
241 }
242
243 if (MDResources.hasCBuffers())
244 MDResources.printCBuffers(OS);
245 if (MDResources.hasUAVs())
246 MDResources.printUAVs(OS);
247
248 OS << ";\n";
249}
