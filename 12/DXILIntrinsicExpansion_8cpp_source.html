36static bool isIntrinsicExpansion(Function &F) {
37 switch (F.getIntrinsicID()) {
38 case Intrinsic::abs:
39 case Intrinsic::atan2:
40 case Intrinsic::exp:
41 case Intrinsic::log:
42 case Intrinsic::log10:
43 case Intrinsic::pow:
44 case Intrinsic::dx_all:
45 case Intrinsic::dx_any:
46 case Intrinsic::dx_clamp:
47 case Intrinsic::dx_uclamp:
48 case Intrinsic::dx_lerp:
49 case Intrinsic::dx_length:
50 case Intrinsic::dx_normalize:
51 case Intrinsic::dx_fdot:
52 case Intrinsic::dx_sdot:
53 case Intrinsic::dx_udot:
54 case Intrinsic::dx_sign:
55 case Intrinsic::dx_step:
56 return true;
57 }
58 return false;
59}

61static Value *expandAbs(CallInst *Orig) {
62 Value *X = Orig->getOperand(0);
63 IRBuilder<> Builder(Orig);
64 Type *Ty = X->getType();
65 Type *EltTy = Ty->getScalarType();
66 Constant *Zero = Ty->isVectorTy()
67 ? ConstantVector::getSplat(
68 ElementCount::getFixed(
69 cast<FixedVectorType>(Ty)->getNumElements()),
70 ConstantInt::get(EltTy, 0))
71 : ConstantInt::get(EltTy, 0);
72 auto *V = Builder.CreateSub(Zero, X);
73 return Builder.CreateIntrinsic(Ty, Intrinsic::smax, {X, V}, nullptr,
74 "dx.max");
75}

77// Create appropriate DXIL float dot intrinsic for the given A and B operands
78// The appropriate opcode will be determined by the size of the operands
79// The dot product is placed in the position indicated by Orig
80static Value *expandFloatDotIntrinsic(CallInst *Orig, Value *A, Value *B) {
81 Type *ATy = A->getType();
82 [[maybe_unused]] Type *BTy = B->getType();
83 assert(ATy->isVectorTy() && BTy->isVectorTy());
84
85 IRBuilder<> Builder(Orig);
86
87 auto *AVec = dyn_cast<FixedVectorType>(ATy);
88
89 assert(ATy->getScalarType()->isFloatingPointTy());
90
91 Intrinsic::ID DotIntrinsic = Intrinsic::dx_dot4;
92 switch (AVec->getNumElements()) {
93 case 2:
94 DotIntrinsic = Intrinsic::dx_dot2;
95 break;
96 case 3:
97 DotIntrinsic = Intrinsic::dx_dot3;
98 break;
99 case 4:
100 DotIntrinsic = Intrinsic::dx_dot4;
101 break;
102 default:
103 report_fatal_error(
104 Twine("Invalid dot product input vector: length is outside 2-4"),
105 /* gen_crash_diag=*/false);
106 return nullptr;
107 }
108 return Builder.CreateIntrinsic(ATy->getScalarType(), DotIntrinsic,
109 ArrayRef<Value *>{A, B}, nullptr, "dot");
110}

120// Expand integer dot product to multiply and add ops
121static Value *expandIntegerDotIntrinsic(CallInst *Orig,
122 Intrinsic::ID DotIntrinsic) {
123 assert(DotIntrinsic == Intrinsic::dx_sdot ||
124 DotIntrinsic == Intrinsic::dx_udot);
125 Value *A = Orig->getOperand(0);
126 Value *B = Orig->getOperand(1);
127 Type *ATy = A->getType();
128 [[maybe_unused]] Type *BTy = B->getType();
129 assert(ATy->isVectorTy() && BTy->isVectorTy());
130
131 IRBuilder<> Builder(Orig);
132
133 auto *AVec = dyn_cast<FixedVectorType>(ATy);
134
135 assert(ATy->getScalarType()->isIntegerTy());
136
137 Value *Result;
138 Intrinsic::ID MadIntrinsic = DotIntrinsic == Intrinsic::dx_sdot
139 ? Intrinsic::dx_imad
140 : Intrinsic::dx_umad;
141 Value *Elt0 = Builder.CreateExtractElement(A, (uint64_t)0);
142 Value *Elt1 = Builder.CreateExtractElement(B, (uint64_t)0);
143 Result = Builder.CreateMul(Elt0, Elt1);
144 for (unsigned I = 1; I < AVec->getNumElements(); I++) {
145 Elt0 = Builder.CreateExtractElement(A, I);
146 Elt1 = Builder.CreateExtractElement(B, I);
147 Result = Builder.CreateIntrinsic(Result->getType(), MadIntrinsic,
148 ArrayRef<Value *>{Elt0, Elt1, Result},
149 nullptr, "dx.mad");
150 }
151 return Result;
152}

154static Value *expandExpIntrinsic(CallInst *Orig) {
155 Value *X = Orig->getOperand(0);
156 IRBuilder<> Builder(Orig);
157 Type *Ty = X->getType();
158 Type *EltTy = Ty->getScalarType();
159 Constant *Log2eConst =
160 Ty->isVectorTy() ? ConstantVector::getSplat(
161 ElementCount::getFixed(
162 cast<FixedVectorType>(Ty)->getNumElements()),
163 ConstantFP::get(EltTy, numbers::log2ef))
164 : ConstantFP::get(EltTy, numbers::log2ef);
165 Value *NewX = Builder.CreateFMul(Log2eConst, X);
166 auto *Exp2Call =
167 Builder.CreateIntrinsic(Ty, Intrinsic::exp2, {NewX}, nullptr, "dx.exp2");
168 Exp2Call->setTailCall(Orig->isTailCall());
169 Exp2Call->setAttributes(Orig->getAttributes());
170 return Exp2Call;
171}

173static Value *expandAnyOrAllIntrinsic(CallInst *Orig,
174 Intrinsic::ID intrinsicId) {
175 Value *X = Orig->getOperand(0);
176 IRBuilder<> Builder(Orig);
177 Type *Ty = X->getType();
178 Type *EltTy = Ty->getScalarType();
179
180 auto ApplyOp = [&Builder](Intrinsic::ID IntrinsicId, Value *Result,
181 Value *Elt) {
182 if (IntrinsicId == Intrinsic::dx_any)
183 return Builder.CreateOr(Result, Elt);
184 assert(IntrinsicId == Intrinsic::dx_all);
185 return Builder.CreateAnd(Result, Elt);
186 };
187
188 Value *Result = nullptr;
189 if (!Ty->isVectorTy()) {
190 Result = EltTy->isFloatingPointTy()
191 ? Builder.CreateFCmpUNE(X, ConstantFP::get(EltTy, 0))
192 : Builder.CreateICmpNE(X, ConstantInt::get(EltTy, 0));
193 } else {
194 auto *XVec = dyn_cast<FixedVectorType>(Ty);
195 Value *Cond =
196 EltTy->isFloatingPointTy()
197 ? Builder.CreateFCmpUNE(
198 X, ConstantVector::getSplat(
199 ElementCount::getFixed(XVec->getNumElements()),
200 ConstantFP::get(EltTy, 0)))
201 : Builder.CreateICmpNE(
202 X, ConstantVector::getSplat(
203 ElementCount::getFixed(XVec->getNumElements()),
204 ConstantInt::get(EltTy, 0)));
205 Result = Builder.CreateExtractElement(Cond, (uint64_t)0);
206 for (unsigned I = 1; I < XVec->getNumElements(); I++) {
207 Value *Elt = Builder.CreateExtractElement(Cond, I);
208 Result = ApplyOp(intrinsicId, Result, Elt);
209 }
210 }
211 return Result;
212}

214static Value *expandLengthIntrinsic(CallInst *Orig) {
215 Value *X = Orig->getOperand(0);
216 IRBuilder<> Builder(Orig);
217 Type *Ty = X->getType();
218 Type *EltTy = Ty->getScalarType();
219
220 // Though dx.length does work on scalar type, we can optimize it to just emit
221 // fabs, in CGBuiltin.cpp. We shouldn't see a scalar type here because
222 // CGBuiltin.cpp should have emitted a fabs call.
223 Value *Elt = Builder.CreateExtractElement(X, (uint64_t)0);
224 auto *XVec = dyn_cast<FixedVectorType>(Ty);
225 unsigned XVecSize = XVec->getNumElements();
226 if (!(Ty->isVectorTy() && XVecSize > 1))
227 report_fatal_error(Twine("Invalid input type for length intrinsic"),
228 /* gen_crash_diag=*/false);
229
230 Value *Sum = Builder.CreateFMul(Elt, Elt);
231 for (unsigned I = 1; I < XVecSize; I++) {
232 Elt = Builder.CreateExtractElement(X, I);
233 Value *Mul = Builder.CreateFMul(Elt, Elt);
234 Sum = Builder.CreateFAdd(Sum, Mul);
235 }
236 return Builder.CreateIntrinsic(EltTy, Intrinsic::sqrt, ArrayRef<Value *>{Sum},
237 nullptr, "elt.sqrt");
238}

272// Use dot product of vector operand with itself to calculate the length.
273// Divide the vector by that length to normalize it.
274static Value *expandNormalizeIntrinsic(CallInst *Orig) {
275 Value *X = Orig->getOperand(0);
276 Type *Ty = Orig->getType();
277 Type *EltTy = Ty->getScalarType();
278 IRBuilder<> Builder(Orig);
279
280 auto *XVec = dyn_cast<FixedVectorType>(Ty);
281 if (!XVec) {
282 if (auto *constantFP = dyn_cast<ConstantFP>(X)) {
283 const APFloat &fpVal = constantFP->getValueAPF();
284 if (fpVal.isZero())
285 report_fatal_error(Twine("Invalid input scalar: length is zero"),
286 /* gen_crash_diag=*/false);
287 }
288 return Builder.CreateFDiv(X, X);
289 }
290
291 Value *DotProduct = expandFloatDotIntrinsic(Orig, X, X);
292
293 // verify that the length is non-zero
294 // (if the dot product is non-zero, then the length is non-zero)
295 if (auto *constantFP = dyn_cast<ConstantFP>(DotProduct)) {
296 const APFloat &fpVal = constantFP->getValueAPF();
297 if (fpVal.isZero())
298 report_fatal_error(Twine("Invalid input vector: length is zero"),
299 /* gen_crash_diag=*/false);
300 }
301
302 Value *Multiplicand = Builder.CreateIntrinsic(EltTy, Intrinsic::dx_rsqrt,
303 ArrayRef<Value *>{DotProduct},
304 nullptr, "dx.rsqrt");
305
306 Value *MultiplicandVec =
307 Builder.CreateVectorSplat(XVec->getNumElements(), Multiplicand);
308 return Builder.CreateFMul(X, MultiplicandVec);
309}

311static Value *expandAtan2Intrinsic(CallInst *Orig) {
312 Value *Y = Orig->getOperand(0);
313 Value *X = Orig->getOperand(1);
314 Type *Ty = X->getType();
315 IRBuilder<> Builder(Orig);
316 Builder.setFastMathFlags(Orig->getFastMathFlags());
317
318 Value *Tan = Builder.CreateFDiv(Y, X);
319
320 CallInst *Atan =
321 Builder.CreateIntrinsic(Ty, Intrinsic::atan, {Tan}, nullptr, "Elt.Atan");
322 Atan->setTailCall(Orig->isTailCall());
323 Atan->setAttributes(Orig->getAttributes());
324
325 // Modify atan result based on https://en.wikipedia.org/wiki/Atan2.
326 Constant *Pi = ConstantFP::get(Ty, llvm::numbers::pi);
327 Constant *HalfPi = ConstantFP::get(Ty, llvm::numbers::pi / 2);
328 Constant *NegHalfPi = ConstantFP::get(Ty, -llvm::numbers::pi / 2);
329 Constant *Zero = ConstantFP::get(Ty, 0);
330 Value *AtanAddPi = Builder.CreateFAdd(Atan, Pi);
331 Value *AtanSubPi = Builder.CreateFSub(Atan, Pi);
332
333 // x > 0 -> atan.
334 Value *Result = Atan;
335 Value *XLt0 = Builder.CreateFCmpOLT(X, Zero);
336 Value *XEq0 = Builder.CreateFCmpOEQ(X, Zero);
337 Value *YGe0 = Builder.CreateFCmpOGE(Y, Zero);
338 Value *YLt0 = Builder.CreateFCmpOLT(Y, Zero);
339
340 // x < 0, y >= 0 -> atan + pi.
341 Value *XLt0AndYGe0 = Builder.CreateAnd(XLt0, YGe0);
342 Result = Builder.CreateSelect(XLt0AndYGe0, AtanAddPi, Result);
343
344 // x < 0, y < 0 -> atan - pi.
345 Value *XLt0AndYLt0 = Builder.CreateAnd(XLt0, YLt0);
346 Result = Builder.CreateSelect(XLt0AndYLt0, AtanSubPi, Result);
347
348 // x == 0, y < 0 -> -pi/2
349 Value *XEq0AndYLt0 = Builder.CreateAnd(XEq0, YLt0);
350 Result = Builder.CreateSelect(XEq0AndYLt0, NegHalfPi, Result);
351
352 // x == 0, y > 0 -> pi/2
353 Value *XEq0AndYGe0 = Builder.CreateAnd(XEq0, YGe0);
354 Result = Builder.CreateSelect(XEq0AndYGe0, HalfPi, Result);
355
356 return Result;
357}

376static Value *expandStepIntrinsic(CallInst *Orig) {
377
378 Value *X = Orig->getOperand(0);
379 Value *Y = Orig->getOperand(1);
380 Type *Ty = X->getType();
381 IRBuilder<> Builder(Orig);
382
383 Constant *One = ConstantFP::get(Ty->getScalarType(), 1.0);
384 Constant *Zero = ConstantFP::get(Ty->getScalarType(), 0.0);
385 Value *Cond = Builder.CreateFCmpOLT(Y, X);
386
387 if (Ty != Ty->getScalarType()) {
388 auto *XVec = dyn_cast<FixedVectorType>(Ty);
389 One = ConstantVector::getSplat(
390 ElementCount::getFixed(XVec->getNumElements()), One);
391 Zero = ConstantVector::getSplat(
392 ElementCount::getFixed(XVec->getNumElements()), Zero);
393 }
394
395 return Builder.CreateSelect(Cond, Zero, One);
396}

437static Value *expandSignIntrinsic(CallInst *Orig) {
438 Value *X = Orig->getOperand(0);
439 Type *Ty = X->getType();
440 Type *ScalarTy = Ty->getScalarType();
441 Type *RetTy = Orig->getType();
442 Constant *Zero = Constant::getNullValue(Ty);
443
444 IRBuilder<> Builder(Orig);
445
446 Value *GT;
447 Value *LT;
448 if (ScalarTy->isFloatingPointTy()) {
449 GT = Builder.CreateFCmpOLT(Zero, X);
450 LT = Builder.CreateFCmpOLT(X, Zero);
451 } else {
452 assert(ScalarTy->isIntegerTy());
453 GT = Builder.CreateICmpSLT(Zero, X);
454 LT = Builder.CreateICmpSLT(X, Zero);
455 }
456
457 Value *ZextGT = Builder.CreateZExt(GT, RetTy);
458 Value *ZextLT = Builder.CreateZExt(LT, RetTy);
459
460 return Builder.CreateSub(ZextGT, ZextLT);
461}

463static bool expandIntrinsic(Function &F, CallInst *Orig) {
464 Value *Result = nullptr;
465 Intrinsic::ID IntrinsicId = F.getIntrinsicID();
466 switch (IntrinsicId) {
467 case Intrinsic::abs:
468 Result = expandAbs(Orig);
469 break;
470 case Intrinsic::atan2:
471 Result = expandAtan2Intrinsic(Orig);
472 break;
473 case Intrinsic::exp:
474 Result = expandExpIntrinsic(Orig);
475 break;
476 case Intrinsic::log:
477 Result = expandLogIntrinsic(Orig);
478 break;
479 case Intrinsic::log10:
480 Result = expandLog10Intrinsic(Orig);
481 break;
482 case Intrinsic::pow:
483 Result = expandPowIntrinsic(Orig);
484 break;
485 case Intrinsic::dx_all:
486 case Intrinsic::dx_any:
487 Result = expandAnyOrAllIntrinsic(Orig, IntrinsicId);
488 break;
489 case Intrinsic::dx_uclamp:
490 case Intrinsic::dx_clamp:
491 Result = expandClampIntrinsic(Orig, IntrinsicId);
492 break;
493 case Intrinsic::dx_lerp:
494 Result = expandLerpIntrinsic(Orig);
495 break;
496 case Intrinsic::dx_length:
497 Result = expandLengthIntrinsic(Orig);
498 break;
499 case Intrinsic::dx_normalize:
500 Result = expandNormalizeIntrinsic(Orig);
501 break;
502 case Intrinsic::dx_fdot:
503 Result = expandFloatDotIntrinsic(Orig);
504 break;
505 case Intrinsic::dx_sdot:
506 case Intrinsic::dx_udot:
507 Result = expandIntegerDotIntrinsic(Orig, IntrinsicId);
508 break;
509 case Intrinsic::dx_sign:
510 Result = expandSignIntrinsic(Orig);
511 break;
512 case Intrinsic::dx_step:
513 Result = expandStepIntrinsic(Orig);
514 }
515 if (Result) {
516 Orig->replaceAllUsesWith(Result);
517 Orig->eraseFromParent();
518 return true;
519 }
520 return false;
521}

523static bool expansionIntrinsics(Module &M) {
524 for (auto &F : make_early_inc_range(M.functions())) {
525 if (!isIntrinsicExpansion(F))
526 continue;
527 bool IntrinsicExpanded = false;
528 for (User *U : make_early_inc_range(F.users())) {
529 auto *IntrinsicCall = dyn_cast<CallInst>(U);
530 if (!IntrinsicCall)
531 continue;
532 IntrinsicExpanded = expandIntrinsic(F, IntrinsicCall);
533 }
534 if (F.user_empty() && IntrinsicExpanded)
535 F.eraseFromParent();
536 }
537 return true;
538}
