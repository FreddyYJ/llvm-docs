24MCSymbol *MCResourceInfo::getSymbol(StringRef FuncName, ResourceInfoKind RIK,
25 MCContext &OutContext) {
26 auto GOCS = [FuncName, &OutContext](StringRef Suffix) {
27 return OutContext.getOrCreateSymbol(FuncName + Twine(Suffix));
28 };
29 switch (RIK) {
30 case RIK_NumVGPR:
31 return GOCS(".num_vgpr");
32 case RIK_NumAGPR:
33 return GOCS(".num_agpr");
34 case RIK_NumSGPR:
35 return GOCS(".numbered_sgpr");
36 case RIK_PrivateSegSize:
37 return GOCS(".private_seg_size");
38 case RIK_UsesVCC:
39 return GOCS(".uses_vcc");
40 case RIK_UsesFlatScratch:
41 return GOCS(".uses_flat_scratch");
42 case RIK_HasDynSizedStack:
43 return GOCS(".has_dyn_sized_stack");
44 case RIK_HasRecursion:
45 return GOCS(".has_recursion");
46 case RIK_HasIndirectCall:
47 return GOCS(".has_indirect_call");
48 }
49 llvm_unreachable("Unexpected ResourceInfoKind.");
50}

58void MCResourceInfo::assignMaxRegs(MCContext &OutContext) {
59 // Assign expression to get the max register use to the max_num_Xgpr symbol.
60 MCSymbol *MaxVGPRSym = getMaxVGPRSymbol(OutContext);
61 MCSymbol *MaxAGPRSym = getMaxAGPRSymbol(OutContext);
62 MCSymbol *MaxSGPRSym = getMaxSGPRSymbol(OutContext);
63
64 auto assignMaxRegSym = [&OutContext](MCSymbol *Sym, int32_t RegCount) {
65 const MCExpr *MaxExpr = MCConstantExpr::create(RegCount, OutContext);
66 Sym->setVariableValue(MaxExpr);
67 };
68
69 assignMaxRegSym(MaxVGPRSym, MaxVGPR);
70 assignMaxRegSym(MaxAGPRSym, MaxAGPR);
71 assignMaxRegSym(MaxSGPRSym, MaxSGPR);
72}

92void MCResourceInfo::assignResourceInfoExpr(
93 int64_t LocalValue, ResourceInfoKind RIK, AMDGPUMCExpr::VariantKind Kind,
94 const MachineFunction &MF, const SmallVectorImpl<const Function *> &Callees,
95 MCContext &OutContext) {
96 const MCConstantExpr *LocalConstExpr =
97 MCConstantExpr::create(LocalValue, OutContext);
98 const MCExpr *SymVal = LocalConstExpr;
99 if (!Callees.empty()) {
100 SmallVector<const MCExpr *, 8> ArgExprs;
101 // Avoid recursive symbol assignment.
102 SmallPtrSet<const Function *, 8> Seen;
103 ArgExprs.push_back(LocalConstExpr);
104 const Function &F = MF.getFunction();
105 Seen.insert(&F);
106
107 for (const Function *Callee : Callees) {
108 if (!Seen.insert(Callee).second)
109 continue;
110 MCSymbol *CalleeValSym = getSymbol(Callee->getName(), RIK, OutContext);
111 ArgExprs.push_back(MCSymbolRefExpr::create(CalleeValSym, OutContext));
112 }
113 SymVal = AMDGPUMCExpr::create(Kind, ArgExprs, OutContext);
114 }
115 MCSymbol *Sym = getSymbol(MF.getName(), RIK, OutContext);
116 Sym->setVariableValue(SymVal);
117}

119void MCResourceInfo::gatherResourceInfo(
120 const MachineFunction &MF,
121 const AMDGPUResourceUsageAnalysis::SIFunctionResourceInfo &FRI,
122 MCContext &OutContext) {
123 // Worst case VGPR use for non-hardware-entrypoints.
124 MCSymbol *MaxVGPRSym = getMaxVGPRSymbol(OutContext);
125 MCSymbol *MaxAGPRSym = getMaxAGPRSymbol(OutContext);
126 MCSymbol *MaxSGPRSym = getMaxSGPRSymbol(OutContext);
127
128 if (!AMDGPU::isEntryFunctionCC(MF.getFunction().getCallingConv())) {
129 addMaxVGPRCandidate(FRI.NumVGPR);
130 addMaxAGPRCandidate(FRI.NumAGPR);
131 addMaxSGPRCandidate(FRI.NumExplicitSGPR);
132 }
133
134 auto SetMaxReg = [&](MCSymbol *MaxSym, int32_t numRegs,
135 ResourceInfoKind RIK) {
136 if (!FRI.HasIndirectCall) {
137 assignResourceInfoExpr(numRegs, RIK, AMDGPUMCExpr::AGVK_Max, MF,
138 FRI.Callees, OutContext);
139 } else {
140 const MCExpr *SymRef = MCSymbolRefExpr::create(MaxSym, OutContext);
141 MCSymbol *LocalNumSym = getSymbol(MF.getName(), RIK, OutContext);
142 const MCExpr *MaxWithLocal = AMDGPUMCExpr::createMax(
143 {MCConstantExpr::create(numRegs, OutContext), SymRef}, OutContext);
144 LocalNumSym->setVariableValue(MaxWithLocal);
145 }
146 };
147
148 SetMaxReg(MaxVGPRSym, FRI.NumVGPR, RIK_NumVGPR);
149 SetMaxReg(MaxAGPRSym, FRI.NumAGPR, RIK_NumAGPR);
150 SetMaxReg(MaxSGPRSym, FRI.NumExplicitSGPR, RIK_NumSGPR);
151
152 {
153 // The expression for private segment size should be: FRI.PrivateSegmentSize
154 // + max(FRI.Callees, FRI.CalleeSegmentSize)
155 SmallVector<const MCExpr *, 8> ArgExprs;
156 if (FRI.CalleeSegmentSize)
157 ArgExprs.push_back(
158 MCConstantExpr::create(FRI.CalleeSegmentSize, OutContext));
159
160 if (!FRI.HasIndirectCall) {
161 for (const Function *Callee : FRI.Callees) {
162 MCSymbol *calleeValSym =
163 getSymbol(Callee->getName(), RIK_PrivateSegSize, OutContext);
164 ArgExprs.push_back(MCSymbolRefExpr::create(calleeValSym, OutContext));
165 }
166 }
167 const MCExpr *localConstExpr =
168 MCConstantExpr::create(FRI.PrivateSegmentSize, OutContext);
169 if (!ArgExprs.empty()) {
170 const AMDGPUMCExpr *transitiveExpr =
171 AMDGPUMCExpr::createMax(ArgExprs, OutContext);
172 localConstExpr =
173 MCBinaryExpr::createAdd(localConstExpr, transitiveExpr, OutContext);
174 }
175 getSymbol(MF.getName(), RIK_PrivateSegSize, OutContext)
176 ->setVariableValue(localConstExpr);
177 }
178
179 auto SetToLocal = [&](int64_t LocalValue, ResourceInfoKind RIK) {
180 MCSymbol *Sym = getSymbol(MF.getName(), RIK, OutContext);
181 Sym->setVariableValue(MCConstantExpr::create(LocalValue, OutContext));
182 };
183
184 if (!FRI.HasIndirectCall) {
185 assignResourceInfoExpr(FRI.UsesVCC, ResourceInfoKind::RIK_UsesVCC,
186 AMDGPUMCExpr::AGVK_Or, MF, FRI.Callees, OutContext);
187 assignResourceInfoExpr(FRI.UsesFlatScratch,
188 ResourceInfoKind::RIK_UsesFlatScratch,
189 AMDGPUMCExpr::AGVK_Or, MF, FRI.Callees, OutContext);
190 assignResourceInfoExpr(FRI.HasDynamicallySizedStack,
191 ResourceInfoKind::RIK_HasDynSizedStack,
192 AMDGPUMCExpr::AGVK_Or, MF, FRI.Callees, OutContext);
193 assignResourceInfoExpr(FRI.HasRecursion, ResourceInfoKind::RIK_HasRecursion,
194 AMDGPUMCExpr::AGVK_Or, MF, FRI.Callees, OutContext);
195 assignResourceInfoExpr(FRI.HasIndirectCall,
196 ResourceInfoKind::RIK_HasIndirectCall,
197 AMDGPUMCExpr::AGVK_Or, MF, FRI.Callees, OutContext);
198 } else {
199 SetToLocal(FRI.UsesVCC, ResourceInfoKind::RIK_UsesVCC);
200 SetToLocal(FRI.UsesFlatScratch, ResourceInfoKind::RIK_UsesFlatScratch);
201 SetToLocal(FRI.HasDynamicallySizedStack,
202 ResourceInfoKind::RIK_HasDynSizedStack);
203 SetToLocal(FRI.HasRecursion, ResourceInfoKind::RIK_HasRecursion);
204 SetToLocal(FRI.HasIndirectCall, ResourceInfoKind::RIK_HasIndirectCall);
205 }
206}
