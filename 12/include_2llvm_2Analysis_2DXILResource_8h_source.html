27class ResourceInfo {
28public:
29 struct ResourceBinding {
30 uint32_t RecordID;
31 uint32_t Space;
32 uint32_t LowerBound;
33 uint32_t Size;
34
35 bool operator==(const ResourceBinding &RHS) const {
36 return std::tie(RecordID, Space, LowerBound, Size) ==
37 std::tie(RHS.RecordID, RHS.Space, RHS.LowerBound, RHS.Size);
38 }
39 bool operator!=(const ResourceBinding &RHS) const {
40 return !(*this == RHS);
41 }
42 bool operator<(const ResourceBinding &RHS) const {
43 return std::tie(RecordID, Space, LowerBound, Size) <
44 std::tie(RHS.RecordID, RHS.Space, RHS.LowerBound, RHS.Size);
45 }
46 };
47
48 struct UAVInfo {
49 bool GloballyCoherent;
50 bool HasCounter;
51 bool IsROV;
52
53 bool operator==(const UAVInfo &RHS) const {
54 return std::tie(GloballyCoherent, HasCounter, IsROV) ==
55 std::tie(RHS.GloballyCoherent, RHS.HasCounter, RHS.IsROV);
56 }
57 bool operator!=(const UAVInfo &RHS) const { return !(*this == RHS); }
58 bool operator<(const UAVInfo &RHS) const {
59 return std::tie(GloballyCoherent, HasCounter, IsROV) <
60 std::tie(RHS.GloballyCoherent, RHS.HasCounter, RHS.IsROV);
61 }
62 };
63
64 struct StructInfo {
65 uint32_t Stride;
66 // Note: we store an integer here rather than using `MaybeAlign` because in
67 // GCC 7 MaybeAlign isn't trivial so having one in this union would delete
68 // our move constructor.
69 // See https://www.open-std.org/jtc1/sc22/wg21/docs/papers/2018/p0602r4.html
70 uint32_t AlignLog2;
71
72 bool operator==(const StructInfo &RHS) const {
73 return std::tie(Stride, AlignLog2) == std::tie(RHS.Stride, RHS.AlignLog2);
74 }
75 bool operator!=(const StructInfo &RHS) const { return !(*this == RHS); }
76 bool operator<(const StructInfo &RHS) const {
77 return std::tie(Stride, AlignLog2) < std::tie(RHS.Stride, RHS.AlignLog2);
78 }
79 };
80
81 struct TypedInfo {
82 dxil::ElementType ElementTy;
83 uint32_t ElementCount;
84
85 bool operator==(const TypedInfo &RHS) const {
86 return std::tie(ElementTy, ElementCount) ==
87 std::tie(RHS.ElementTy, RHS.ElementCount);
88 }
89 bool operator!=(const TypedInfo &RHS) const { return !(*this == RHS); }
90 bool operator<(const TypedInfo &RHS) const {
91 return std::tie(ElementTy, ElementCount) <
92 std::tie(RHS.ElementTy, RHS.ElementCount);
93 }
94 };
95
96 struct MSInfo {
97 uint32_t Count;
98
99 bool operator==(const MSInfo &RHS) const { return Count == RHS.Count; }
100 bool operator!=(const MSInfo &RHS) const { return !(*this == RHS); }
101 bool operator<(const MSInfo &RHS) const { return Count < RHS.Count; }
102 };
103
104 struct FeedbackInfo {
105 dxil::SamplerFeedbackType Type;
106
107 bool operator==(const FeedbackInfo &RHS) const { return Type == RHS.Type; }
108 bool operator!=(const FeedbackInfo &RHS) const { return !(*this == RHS); }
109 bool operator<(const FeedbackInfo &RHS) const { return Type < RHS.Type; }
110 };
