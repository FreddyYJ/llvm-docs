82 void swapOperandsInternal(unsigned OpIdxA, unsigned OpIdxB) {
83 assert(OpIdxA < getNumOperands() && "OpIdxA out of bounds!");

85 auto UseA = getOperandUse(OpIdxA);
86 auto UseB = getOperandUse(OpIdxB);
