75static NamedMDNode *emitResourceMetadata(Module &M, const DXILResourceMap &DRM,
76 const dxil::Resources &MDResources) {
77 LLVMContext &Context = M.getContext();
78
79 SmallVector<Metadata *> SRVs, UAVs, CBufs, Smps;
80 for (const ResourceInfo &RI : DRM.srvs())
81 SRVs.push_back(RI.getAsMetadata(Context));
82 for (const ResourceInfo &RI : DRM.uavs())
83 UAVs.push_back(RI.getAsMetadata(Context));
84 for (const ResourceInfo &RI : DRM.cbuffers())
85 CBufs.push_back(RI.getAsMetadata(Context));
86 for (const ResourceInfo &RI : DRM.samplers())
87 Smps.push_back(RI.getAsMetadata(Context));
88
89 Metadata *SRVMD = SRVs.empty() ? nullptr : MDNode::get(Context, SRVs);
90 Metadata *UAVMD = UAVs.empty() ? nullptr : MDNode::get(Context, UAVs);
91 Metadata *CBufMD = CBufs.empty() ? nullptr : MDNode::get(Context, CBufs);
92 Metadata *SmpMD = Smps.empty() ? nullptr : MDNode::get(Context, Smps);
93 bool HasResources = !DRM.empty();
94
95 if (MDResources.hasUAVs()) {
96 assert(!UAVMD && "Old and new UAV representations can't coexist");
97 UAVMD = MDResources.writeUAVs(M);
98 HasResources = true;
99 }
100
101 if (MDResources.hasCBuffers()) {
102 assert(!CBufMD && "Old and new cbuffer representations can't coexist");
103 CBufMD = MDResources.writeCBuffers(M);
104 HasResources = true;
105 }
106
107 if (!HasResources)
108 return nullptr;
109
110 NamedMDNode *ResourceMD = M.getOrInsertNamedMetadata("dx.resources");
111 ResourceMD->addOperand(
112 MDNode::get(M.getContext(), {SRVMD, UAVMD, CBufMD, SmpMD}));
113
114 return ResourceMD;
115}

117static StringRef getShortShaderStage(Triple::EnvironmentType Env) {
118 switch (Env) {
119 case Triple::Pixel:
120 return "ps";
121 case Triple::Vertex:
122 return "vs";
123 case Triple::Geometry:
124 return "gs";
125 case Triple::Hull:
126 return "hs";
127 case Triple::Domain:
128 return "ds";
129 case Triple::Compute:
130 return "cs";
131 case Triple::Library:
132 return "lib";
133 case Triple::Mesh:
134 return "ms";
135 case Triple::Amplification:
136 return "as";
137 default:
138 break;
139 }
140 llvm_unreachable("Unsupported environment for DXIL generation.");
141}

147static SmallVector<Metadata *>
148getTagValueAsMetadata(EntryPropsTag Tag, uint64_t Value, LLVMContext &Ctx) {
149 SmallVector<Metadata *> MDVals;
150 MDVals.emplace_back(ConstantAsMetadata::get(
151 ConstantInt::get(Type::getInt32Ty(Ctx), static_cast<int>(Tag))));
152 switch (Tag) {
153 case EntryPropsTag::ShaderFlags:
154 MDVals.emplace_back(ConstantAsMetadata::get(
155 ConstantInt::get(Type::getInt64Ty(Ctx), Value)));
156 break;
157 case EntryPropsTag::ShaderKind:
158 MDVals.emplace_back(ConstantAsMetadata::get(
159 ConstantInt::get(Type::getInt32Ty(Ctx), Value)));
160 break;
161 case EntryPropsTag::GSState:
162 case EntryPropsTag::DSState:
163 case EntryPropsTag::HSState:
164 case EntryPropsTag::NumThreads:
165 case EntryPropsTag::AutoBindingSpace:
166 case EntryPropsTag::RayPayloadSize:
167 case EntryPropsTag::RayAttribSize:
168 case EntryPropsTag::MSState:
169 case EntryPropsTag::ASStateTag:
170 case EntryPropsTag::WaveSize:
171 case EntryPropsTag::EntryRootSig:
172 llvm_unreachable("NYI: Unhandled entry property tag");
173 }
174 return MDVals;
175}

177static MDTuple *
178getEntryPropAsMetadata(const EntryProperties &EP, uint64_t EntryShaderFlags,
179 const Triple::EnvironmentType ShaderProfile) {
180 SmallVector<Metadata *> MDVals;
181 LLVMContext &Ctx = EP.Entry->getContext();
182 if (EntryShaderFlags != 0)
183 MDVals.append(getTagValueAsMetadata(EntryPropsTag::ShaderFlags,
184 EntryShaderFlags, Ctx));
185
186 if (EP.Entry != nullptr) {
187 // FIXME: support more props.
188 // See https://github.com/llvm/llvm-project/issues/57948.
189 // Add shader kind for lib entries.
190 if (ShaderProfile == Triple::EnvironmentType::Library &&
191 EP.ShaderStage != Triple::EnvironmentType::Library)
192 MDVals.append(getTagValueAsMetadata(EntryPropsTag::ShaderKind,
193 getShaderStage(EP.ShaderStage), Ctx));
194
195 if (EP.ShaderStage == Triple::EnvironmentType::Compute) {
196 MDVals.emplace_back(ConstantAsMetadata::get(ConstantInt::get(
197 Type::getInt32Ty(Ctx), static_cast<int>(EntryPropsTag::NumThreads))));
198 Metadata *NumThreadVals[] = {ConstantAsMetadata::get(ConstantInt::get(
199 Type::getInt32Ty(Ctx), EP.NumThreadsX)),
200 ConstantAsMetadata::get(ConstantInt::get(
201 Type::getInt32Ty(Ctx), EP.NumThreadsY)),
202 ConstantAsMetadata::get(ConstantInt::get(
203 Type::getInt32Ty(Ctx), EP.NumThreadsZ))};
204 MDVals.emplace_back(MDNode::get(Ctx, NumThreadVals));
205 }
206 }
207 if (MDVals.empty())
208 return nullptr;
209 return MDNode::get(Ctx, MDVals);
210}

212MDTuple *constructEntryMetadata(const Function *EntryFn, MDTuple *Signatures,
213 MDNode *Resources, MDTuple *Properties,
214 LLVMContext &Ctx) {
215 // Each entry point metadata record specifies:
216 // * reference to the entry point function global symbol
217 // * unmangled name
218 // * list of signatures
219 // * list of resources
220 // * list of tag-value pairs of shader capabilities and other properties
221 Metadata *MDVals[5];
222 MDVals[0] =
223 EntryFn ? ValueAsMetadata::get(const_cast<Function *>(EntryFn)) : nullptr;
224 MDVals[1] = MDString::get(Ctx, EntryFn ? EntryFn->getName() : "");
225 MDVals[2] = Signatures;
226 MDVals[3] = Resources;
227 MDVals[4] = Properties;
228 return MDNode::get(Ctx, MDVals);
229}

241static void emitValidatorVersionMD(Module &M, const ModuleMetadataInfo &MMDI) {
242 if (MMDI.ValidatorVersion.empty())
243 return;
244
245 LLVMContext &Ctx = M.getContext();
246 IRBuilder<> IRB(Ctx);
247 Metadata *MDVals[2];
248 MDVals[0] =
249 ConstantAsMetadata::get(IRB.getInt32(MMDI.ValidatorVersion.getMajor()));
250 MDVals[1] = ConstantAsMetadata::get(
251 IRB.getInt32(MMDI.ValidatorVersion.getMinor().value_or(0)));
252 NamedMDNode *ValVerNode = M.getOrInsertNamedMetadata("dx.valver");
253 // Set validator version obtained from DXIL Metadata Analysis pass
254 ValVerNode->clearOperands();
255 ValVerNode->addOperand(MDNode::get(Ctx, MDVals));
256}

258static void emitShaderModelVersionMD(Module &M,
259 const ModuleMetadataInfo &MMDI) {
260 LLVMContext &Ctx = M.getContext();
261 IRBuilder<> IRB(Ctx);
262 Metadata *SMVals[3];
263 VersionTuple SM = MMDI.ShaderModelVersion;
264 SMVals[0] = MDString::get(Ctx, getShortShaderStage(MMDI.ShaderProfile));
265 SMVals[1] = ConstantAsMetadata::get(IRB.getInt32(SM.getMajor()));
266 SMVals[2] = ConstantAsMetadata::get(IRB.getInt32(SM.getMinor().value_or(0)));
267 NamedMDNode *SMMDNode = M.getOrInsertNamedMetadata("dx.shaderModel");
268 SMMDNode->addOperand(MDNode::get(Ctx, SMVals));
269}

283static MDTuple *emitTopLevelLibraryNode(Module &M, MDNode *RMD,
284 uint64_t ShaderFlags) {
285 LLVMContext &Ctx = M.getContext();
286 MDTuple *Properties = nullptr;
287 if (ShaderFlags != 0) {
288 SmallVector<Metadata *> MDVals;
289 // FIXME: ShaderFlagsAnalysis pass needs to collect and provide
290 // ShaderFlags for each entry function. Currently, ShaderFlags value
291 // provided by ShaderFlagsAnalysis pass is created by walking *all* the
292 // function instructions of the module. Is it is correct to use this value
293 // for metadata of the empty library entry?
294 MDVals.append(
295 getTagValueAsMetadata(EntryPropsTag::ShaderFlags, ShaderFlags, Ctx));
296 Properties = MDNode::get(Ctx, MDVals);
297 }
298 // Library has an entry metadata with resource table metadata and all other
299 // MDNodes as null.
300 return constructEntryMetadata(nullptr, nullptr, RMD, Properties, Ctx);
301}

303static void translateMetadata(Module &M, const DXILResourceMap &DRM,
304 const Resources &MDResources,
305 const ComputedShaderFlags &ShaderFlags,
306 const ModuleMetadataInfo &MMDI) {
307 LLVMContext &Ctx = M.getContext();
308 IRBuilder<> IRB(Ctx);
309 SmallVector<MDNode *> EntryFnMDNodes;
310
311 emitValidatorVersionMD(M, MMDI);
312 emitShaderModelVersionMD(M, MMDI);
313 emitDXILVersionTupleMD(M, MMDI);
314 NamedMDNode *NamedResourceMD = emitResourceMetadata(M, DRM, MDResources);
315 auto *ResourceMD =
316 (NamedResourceMD != nullptr) ? NamedResourceMD->getOperand(0) : nullptr;
317 // FIXME: Add support to construct Signatures
318 // See https://github.com/llvm/llvm-project/issues/57928
319 MDTuple *Signatures = nullptr;
320
321 if (MMDI.ShaderProfile == Triple::EnvironmentType::Library)
322 EntryFnMDNodes.emplace_back(
323 emitTopLevelLibraryNode(M, ResourceMD, ShaderFlags));
324 else if (MMDI.EntryPropertyVec.size() > 1) {
325 M.getContext().diagnose(DiagnosticInfoTranslateMD(
326 M, "Non-library shader: One and only one entry expected"));
327 }
328
329 for (const EntryProperties &EntryProp : MMDI.EntryPropertyVec) {
330 // FIXME: ShaderFlagsAnalysis pass needs to collect and provide
331 // ShaderFlags for each entry function. For now, assume shader flags value
332 // of entry functions being compiled for lib_* shader profile viz.,
333 // EntryPro.Entry is 0.
334 uint64_t EntryShaderFlags =
335 (MMDI.ShaderProfile == Triple::EnvironmentType::Library) ? 0
336 : ShaderFlags;
337 if (MMDI.ShaderProfile != Triple::EnvironmentType::Library) {
338 if (EntryProp.ShaderStage != MMDI.ShaderProfile) {
339 M.getContext().diagnose(DiagnosticInfoTranslateMD(
340 M,
341 "Shader stage '" +
342 Twine(getShortShaderStage(EntryProp.ShaderStage) +
343 "' for entry '" + Twine(EntryProp.Entry->getName()) +
344 "' different from specified target profile '" +
345 Twine(Triple::getEnvironmentTypeName(MMDI.ShaderProfile) +
346 "'"))));
347 }
348 }
349 EntryFnMDNodes.emplace_back(emitEntryMD(EntryProp, Signatures, ResourceMD,
350 EntryShaderFlags,
351 MMDI.ShaderProfile));
352 }
353
354 NamedMDNode *EntryPointsNamedMD =
355 M.getOrInsertNamedMetadata("dx.entryPoints");
356 for (auto *Entry : EntryFnMDNodes)
357 EntryPointsNamedMD->addOperand(Entry);
358}

360PreservedAnalyses DXILTranslateMetadata::run(Module &M,
361 ModuleAnalysisManager &MAM) {
362 const DXILResourceMap &DRM = MAM.getResult<DXILResourceAnalysis>(M);
363 const dxil::Resources &MDResources = MAM.getResult<DXILResourceMDAnalysis>(M);
364 const ComputedShaderFlags &ShaderFlags =
365 MAM.getResult<ShaderFlagsAnalysis>(M);
366 const dxil::ModuleMetadataInfo MMDI = MAM.getResult<DXILMetadataAnalysis>(M);
367
368 translateMetadata(M, DRM, MDResources, ShaderFlags, MMDI);
369
370 return PreservedAnalyses::all();
371}
