334bool ResourceInfo::operator==(const ResourceInfo &RHS) const {
335 if (std::tie(Symbol, Name, Binding, RC, Kind) !=
336 std::tie(RHS.Symbol, RHS.Name, RHS.Binding, RHS.RC, RHS.Kind))
337 return false;
338 if (isCBuffer() && RHS.isCBuffer() && CBufferSize != RHS.CBufferSize)
339 return false;
340 if (isSampler() && RHS.isSampler() && SamplerTy != RHS.SamplerTy)
341 return false;
342 if (isUAV() && RHS.isUAV() && UAVFlags != RHS.UAVFlags)
343 return false;
344 if (isStruct() && RHS.isStruct() && Struct != RHS.Struct)
345 return false;
346 if (isFeedback() && RHS.isFeedback() && Feedback != RHS.Feedback)
347 return false;
348 if (isTyped() && RHS.isTyped() && Typed != RHS.Typed)
349 return false;
350 if (isMultiSample() && RHS.isMultiSample() && MultiSample != RHS.MultiSample)
351 return false;
352 return true;
353}

355bool ResourceInfo::operator<(const ResourceInfo &RHS) const {
356 // Skip the symbol to avoid non-determinism, and the name to keep a consistent
357 // ordering even when we strip reflection data.
358 if (std::tie(Binding, RC, Kind) < std::tie(RHS.Binding, RHS.RC, RHS.Kind))
359 return true;
360 if (isCBuffer() && RHS.isCBuffer() && CBufferSize < RHS.CBufferSize)
361 return true;
362 if (isSampler() && RHS.isSampler() && SamplerTy < RHS.SamplerTy)
363 return true;
364 if (isUAV() && RHS.isUAV() && UAVFlags < RHS.UAVFlags)
365 return true;
366 if (isStruct() && RHS.isStruct() && Struct < RHS.Struct)
367 return true;
368 if (isFeedback() && RHS.isFeedback() && Feedback < RHS.Feedback)
369 return true;
370 if (isTyped() && RHS.isTyped() && Typed < RHS.Typed)
371 return true;
372 if (isMultiSample() && RHS.isMultiSample() && MultiSample < RHS.MultiSample)
373 return true;
374 return false;
375}

377MDTuple *ResourceInfo::getAsMetadata(LLVMContext &Ctx) const {
378 SmallVector<Metadata *, 11> MDVals;
379
380 Type *I32Ty = Type::getInt32Ty(Ctx);
381 Type *I1Ty = Type::getInt1Ty(Ctx);
382 auto getIntMD = [&I32Ty](uint32_t V) {
383 return ConstantAsMetadata::get(
384 Constant::getIntegerValue(I32Ty, APInt(32, V)));
385 };
386 auto getBoolMD = [&I1Ty](uint32_t V) {
387 return ConstantAsMetadata::get(
388 Constant::getIntegerValue(I1Ty, APInt(1, V)));
389 };
390
391 MDVals.push_back(getIntMD(Binding.RecordID));
392 MDVals.push_back(ValueAsMetadata::get(Symbol));
393 MDVals.push_back(MDString::get(Ctx, Name));
394 MDVals.push_back(getIntMD(Binding.Space));
395 MDVals.push_back(getIntMD(Binding.LowerBound));
396 MDVals.push_back(getIntMD(Binding.Size));
397
398 if (isCBuffer()) {
399 MDVals.push_back(getIntMD(CBufferSize));
400 MDVals.push_back(nullptr);
401 } else if (isSampler()) {
402 MDVals.push_back(getIntMD(llvm::to_underlying(SamplerTy)));
403 MDVals.push_back(nullptr);
404 } else {
405 MDVals.push_back(getIntMD(llvm::to_underlying(Kind)));
406
407 if (isUAV()) {
408 MDVals.push_back(getBoolMD(UAVFlags.GloballyCoherent));
409 MDVals.push_back(getBoolMD(UAVFlags.HasCounter));
410 MDVals.push_back(getBoolMD(UAVFlags.IsROV));
411 } else {
412 // All SRVs include sample count in the metadata, but it's only meaningful
413 // for multi-sampled textured. Also, UAVs can be multisampled in SM6.7+,
414 // but this just isn't reflected in the metadata at all.
415 uint32_t SampleCount = isMultiSample() ? MultiSample.Count : 0;
416 MDVals.push_back(getIntMD(SampleCount));
417 }
418
419 // Further properties are attached to a metadata list of tag-value pairs.
420 SmallVector<Metadata *> Tags;
421 if (isStruct()) {
422 Tags.push_back(
423 getIntMD(llvm::to_underlying(ExtPropTags::StructuredBufferStride)));
424 Tags.push_back(getIntMD(Struct.Stride));
425 } else if (isTyped()) {
426 Tags.push_back(getIntMD(llvm::to_underlying(ExtPropTags::ElementType)));
427 Tags.push_back(getIntMD(llvm::to_underlying(Typed.ElementTy)));
428 } else if (isFeedback()) {
429 Tags.push_back(
430 getIntMD(llvm::to_underlying(ExtPropTags::SamplerFeedbackKind)));
431 Tags.push_back(getIntMD(llvm::to_underlying(Feedback.Type)));
432 }
433 MDVals.push_back(Tags.empty() ? nullptr : MDNode::get(Ctx, Tags));
434 }
435
436 return MDNode::get(Ctx, MDVals);
437}

439std::pair<uint32_t, uint32_t> ResourceInfo::getAnnotateProps() const {
440 uint32_t ResourceKind = llvm::to_underlying(Kind);
441 uint32_t AlignLog2 = isStruct() ? Struct.AlignLog2 : 0;
442 bool IsUAV = isUAV();
443 bool IsROV = IsUAV && UAVFlags.IsROV;
444 bool IsGloballyCoherent = IsUAV && UAVFlags.GloballyCoherent;
445 uint8_t SamplerCmpOrHasCounter = 0;
446 if (IsUAV)
447 SamplerCmpOrHasCounter = UAVFlags.HasCounter;
448 else if (isSampler())
449 SamplerCmpOrHasCounter = SamplerTy == SamplerType::Comparison;
450
451 // TODO: Document this format. Currently the only reference is the
452 // implementation of dxc's DxilResourceProperties struct.
453 uint32_t Word0 = 0;
454 Word0 |= ResourceKind & 0xFF;
455 Word0 |= (AlignLog2 & 0xF) << 8;
456 Word0 |= (IsUAV & 1) << 12;
457 Word0 |= (IsROV & 1) << 13;
458 Word0 |= (IsGloballyCoherent & 1) << 14;
459 Word0 |= (SamplerCmpOrHasCounter & 1) << 15;
460
461 uint32_t Word1 = 0;
462 if (isStruct())
463 Word1 = Struct.Stride;
464 else if (isCBuffer())
465 Word1 = CBufferSize;
466 else if (isFeedback())
467 Word1 = llvm::to_underlying(Feedback.Type);
468 else if (isTyped()) {
469 uint32_t CompType = llvm::to_underlying(Typed.ElementTy);
470 uint32_t CompCount = Typed.ElementCount;
471 uint32_t SampleCount = isMultiSample() ? MultiSample.Count : 0;
472
473 Word1 |= (CompType & 0xFF) << 0;
474 Word1 |= (CompCount & 0xFF) << 8;
475 Word1 |= (SampleCount & 0xFF) << 16;
476 }
477
478 return {Word0, Word1};
479}

523static dxil::ElementType toDXILElementType(Type *Ty, bool IsSigned) {
524 // TODO: Handle unorm, snorm, and packed.
525 Ty = Ty->getScalarType();
526
527 if (Ty->isIntegerTy()) {
528 switch (Ty->getIntegerBitWidth()) {
529 case 16:
530 return IsSigned ? ElementType::I16 : ElementType::U16;
531 case 32:
532 return IsSigned ? ElementType::I32 : ElementType::U32;
533 case 64:
534 return IsSigned ? ElementType::I64 : ElementType::U64;
535 case 1:
536 default:
537 return ElementType::Invalid;
538 }
539 } else if (Ty->isFloatTy()) {
540 return ElementType::F32;
541 } else if (Ty->isDoubleTy()) {
542 return ElementType::F64;
543 } else if (Ty->isHalfTy()) {
544 return ElementType::F16;
545 }
546
547 return ElementType::Invalid;
548}

560 void diagnoseHandle(CallInst *CI, const Twine &Msg,
561 DiagnosticSeverity Severity = DS_Error) {

567 Context.diagnose(Diag);

570 ResourceInfo *mapBufferType(CallInst *CI, TargetExtType *HandleTy,
571 bool IsTyped) {
572 if (HandleTy->getNumTypeParameters() != 1 ||
573 HandleTy->getNumIntParameters() != (IsTyped ? 3 : 2)) {
574 diagnoseHandle(CI, Twine("Invalid buffer target type"));
575 return nullptr;
576 }
577
578 Type *ElTy = HandleTy->getTypeParameter(0);
579 unsigned IsWriteable = HandleTy->getIntParameter(0);
580 unsigned IsROV = HandleTy->getIntParameter(1);
581 bool IsSigned = IsTyped && HandleTy->getIntParameter(2);
582
583 ResourceClass RC = IsWriteable ? ResourceClass::UAV : ResourceClass::SRV;
584 ResourceKind Kind;
585 if (IsTyped)
586 Kind = ResourceKind::TypedBuffer;
587 else if (ElTy->isIntegerTy(8))
588 Kind = ResourceKind::RawBuffer;
589 else
590 Kind = ResourceKind::StructuredBuffer;
591
592 // TODO: We need to lower to a typed pointer, can we smuggle the type
593 // through?
594 Value *Symbol = UndefValue::get(PointerType::getUnqual(Context));
595 // TODO: We don't actually keep track of the name right now...
596 StringRef Name = "";
597
598 // Note that we return a pointer into the vector's storage. This is okay as
599 // long as we don't add more elements until we're done with the pointer.
600 auto &Pair =
601 Resources.emplace_back(CI, ResourceInfo{RC, Kind, Symbol, Name});
602 ResourceInfo *RI = &Pair.second;
603
604 if (RI->isUAV())
605 // TODO: We need analysis for GloballyCoherent and HasCounter
606 RI->setUAV(false, false, IsROV);
607
608 if (RI->isTyped()) {
609 dxil::ElementType ET = toDXILElementType(ElTy, IsSigned);
610 uint32_t Count = 1;
611 if (auto *VTy = dyn_cast<FixedVectorType>(ElTy))
612 Count = VTy->getNumElements();
613 RI->setTyped(ET, Count);
614 } else if (RI->isStruct()) {
615 const DataLayout &DL = M.getDataLayout();
616
617 // This mimics what DXC does. Notably, we only ever set the alignment if
618 // the type is actually a struct type.
619 uint32_t Stride = DL.getTypeAllocSize(ElTy);
620 MaybeAlign Alignment;
621 if (auto *STy = dyn_cast<StructType>(ElTy))
622 Alignment = DL.getStructLayout(STy)->getAlignment();
623 RI->setStruct(Stride, Alignment);
624 }
625
626 return RI;
627 }

629 ResourceInfo *mapHandleIntrin(CallInst *CI) {
630 FunctionType *FTy = CI->getFunctionType();
631 Type *RetTy = FTy->getReturnType();
632 auto *HandleTy = dyn_cast<TargetExtType>(RetTy);
633 if (!HandleTy) {
634 diagnoseHandle(CI, "dx.handle.fromBinding requires target type");
635 return nullptr;
636 }
637
638 StringRef TypeName = HandleTy->getName();
639 if (TypeName == "dx.TypedBuffer") {
640 return mapBufferType(CI, HandleTy, /*IsTyped=*/true);
641 } else if (TypeName == "dx.RawBuffer") {
642 return mapBufferType(CI, HandleTy, /*IsTyped=*/false);
643 } else if (TypeName == "dx.CBuffer") {
644 // TODO: implement
645 diagnoseHandle(CI, "dx.CBuffer handles are not implemented yet");
646 return nullptr;
647 } else if (TypeName == "dx.Sampler") {
648 // TODO: implement
649 diagnoseHandle(CI, "dx.Sampler handles are not implemented yet");
650 return nullptr;
651 } else if (TypeName == "dx.Texture") {
652 // TODO: implement
653 diagnoseHandle(CI, "dx.Texture handles are not implemented yet");
654 return nullptr;
655 }
656
657 diagnoseHandle(CI, "Invalid target(dx) type");
658 return nullptr;
659 }

661 ResourceInfo *mapHandleFromBinding(CallInst *CI) {
662 assert(CI->getIntrinsicID() == Intrinsic::dx_handle_fromBinding &&
663 "Must be dx.handle.fromBinding intrinsic");
664
665 ResourceInfo *RI = mapHandleIntrin(CI);
666 if (!RI)
667 return nullptr;
668
669 uint32_t Space = cast<ConstantInt>(CI->getArgOperand(0))->getZExtValue();
670 uint32_t LowerBound =
671 cast<ConstantInt>(CI->getArgOperand(1))->getZExtValue();
672 uint32_t Size = cast<ConstantInt>(CI->getArgOperand(2))->getZExtValue();
673
674 // We use a binding ID of zero for now - these will be filled in later.
675 RI->bind(0U, Space, LowerBound, Size);
676
677 return RI;
678 }
679
680 DXILResourceMap mapResources() {
681 for (Function &F : M.functions()) {
682 if (!F.isDeclaration())
683 continue;
684 LLVM_DEBUG(dbgs() << "Function: " << F.getName() << "\n");
685 Intrinsic::ID ID = F.getIntrinsicID();
686 switch (ID) {
687 default:
688 // TODO: handle `dx.op` functions.
689 continue;
690 case Intrinsic::dx_handle_fromBinding:
691 for (User *U : F.users()) {
692 LLVM_DEBUG(dbgs() << " Visiting: " << *U << "\n");
693 if (CallInst *CI = dyn_cast<CallInst>(U))
694 mapHandleFromBinding(CI);
695 }
696 break;
697 }
698 }
699
700 return DXILResourceMap(std::move(Resources));
701 }
702};
703
704} // namespace
705
706DXILResourceMap::DXILResourceMap(
707 SmallVectorImpl<std::pair<CallInst *, dxil::ResourceInfo>> &&CIToRI) {
708 if (CIToRI.empty())
709 return;
710
711 llvm::stable_sort(CIToRI, [](auto &LHS, auto &RHS) {
712 // Sort by resource class first for grouping purposes, and then by the rest
713 // of the fields so that we can remove duplicates.
714 ResourceClass LRC = LHS.second.getResourceClass();
715 ResourceClass RRC = RHS.second.getResourceClass();
716 return std::tie(LRC, LHS.second) < std::tie(RRC, RHS.second);
717 });
718 for (auto [CI, RI] : CIToRI) {
719 if (Resources.empty() || RI != Resources.back())
720 Resources.push_back(RI);
721 CallMap[CI] = Resources.size() - 1;
722 }
723
724 unsigned Size = Resources.size();
725 // In DXC, Record ID is unique per resource type. Match that.
726 FirstUAV = FirstCBuffer = FirstSampler = Size;
727 uint32_t NextID = 0;
728 for (unsigned I = 0, E = Size; I != E; ++I) {
729 ResourceInfo &RI = Resources[I];
730 if (RI.isUAV() && FirstUAV == Size) {
731 FirstUAV = I;
732 NextID = 0;
733 } else if (RI.isCBuffer() && FirstCBuffer == Size) {
734 FirstCBuffer = I;
735 NextID = 0;
736 } else if (RI.isSampler() && FirstSampler == Size) {
737 FirstSampler = I;
738 NextID = 0;
739 }
740
741 // Adjust the resource binding to use the next ID.
742 const ResourceInfo::ResourceBinding &Binding = RI.getBinding();
743 RI.bind(NextID++, Binding.Space, Binding.LowerBound, Binding.Size);
744 }
745}

748 for (unsigned I = 0, E = Resources.size(); I != E; ++I) {
749 OS << "Binding " << I << ":\n";
750 Resources[I].print(OS);
751 OS << "\n";
752 }
753
754 for (const auto &[CI, Index] : CallMap) {
755 OS << "Call bound to " << Index << ":";
756 CI->print(OS);
757 OS << "\n";
758 }
