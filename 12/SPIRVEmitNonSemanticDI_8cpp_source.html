92bool SPIRVEmitNonSemanticDI::emitGlobalDI(MachineFunction &MF) {
93 // If this MachineFunction doesn't have any BB repeat procedure
94 // for the next
95 if (MF.begin() == MF.end()) {
96 IsGlobalDIEmitted = false;
97 return false;
98 }
99
100 // Required variables to get from metadata search
101 LLVMContext *Context;
102 SmallVector<SmallString<128>> FilePaths;
103 SmallVector<int64_t> LLVMSourceLanguages;
104 int64_t DwarfVersion = 0;
105 int64_t DebugInfoVersion = 0;
106 SmallPtrSet<DIBasicType *, 12> BasicTypes;
107 // Searching through the Module metadata to find nescessary
108 // information like DwarfVersion or SourceLanguage
109 {
110 const MachineModuleInfo &MMI =
111 getAnalysis<MachineModuleInfoWrapperPass>().getMMI();
112 const Module *M = MMI.getModule();
113 Context = &M->getContext();
114 const NamedMDNode *DbgCu = M->getNamedMetadata("llvm.dbg.cu");
115 if (!DbgCu)
116 return false;
117 for (const auto *Op : DbgCu->operands()) {
118 if (const auto *CompileUnit = dyn_cast<DICompileUnit>(Op)) {
119 DIFile *File = CompileUnit->getFile();
120 FilePaths.emplace_back();
121 sys::path::append(FilePaths.back(), File->getDirectory(),
122 File->getFilename());
123 LLVMSourceLanguages.push_back(CompileUnit->getSourceLanguage());
124 }
125 }
126 const NamedMDNode *ModuleFlags = M->getNamedMetadata("llvm.module.flags");
127 for (const auto *Op : ModuleFlags->operands()) {
128 const MDOperand &MaybeStrOp = Op->getOperand(1);
129 if (MaybeStrOp.equalsStr("Dwarf Version"))
130 DwarfVersion =
131 cast<ConstantInt>(
132 cast<ConstantAsMetadata>(Op->getOperand(2))->getValue())
133 ->getSExtValue();
134 else if (MaybeStrOp.equalsStr("Debug Info Version"))
135 DebugInfoVersion =
136 cast<ConstantInt>(
137 cast<ConstantAsMetadata>(Op->getOperand(2))->getValue())
138 ->getSExtValue();
139 }
140
141 // This traversal is the only supported way to access
142 // instruction related DI metadata like DIBasicType
143 for (auto &F : *M) {
144 for (auto &BB : F) {
145 for (auto &I : BB) {
146 for (DbgVariableRecord &DVR : filterDbgVars(I.getDbgRecordRange())) {
147 DILocalVariable *LocalVariable = DVR.getVariable();
148 if (auto *BasicType =
149 dyn_cast<DIBasicType>(LocalVariable->getType()))
150 BasicTypes.insert(BasicType);
151 }
152 }
153 }
154 }
155 }
156 // NonSemantic.Shader.DebugInfo.100 global DI instruction emitting
157 {
158 // Required LLVM variables for emitting logic
159 const SPIRVInstrInfo *TII = TM->getSubtargetImpl()->getInstrInfo();
160 const SPIRVRegisterInfo *TRI = TM->getSubtargetImpl()->getRegisterInfo();
161 const RegisterBankInfo *RBI = TM->getSubtargetImpl()->getRegBankInfo();
162 SPIRVGlobalRegistry *GR = TM->getSubtargetImpl()->getSPIRVGlobalRegistry();
163 MachineRegisterInfo &MRI = MF.getRegInfo();
164 MachineBasicBlock &MBB = *MF.begin();
165
166 // To correct placement of a OpLabel instruction during SPIRVAsmPrinter
167 // emission all new instructions needs to be placed after OpFunction
168 // and before first terminator
169 MachineIRBuilder MIRBuilder(MBB, MBB.getFirstTerminator());
170
171 const auto EmitOpString = [&](StringRef SR) {
172 const Register StrReg = MRI.createVirtualRegister(&SPIRV::IDRegClass);
173 MRI.setType(StrReg, LLT::scalar(32));
174 MachineInstrBuilder MIB = MIRBuilder.buildInstr(SPIRV::OpString);
175 MIB.addDef(StrReg);
176 addStringImm(SR, MIB);
177 return StrReg;
178 };
179
180 const SPIRVType *VoidTy =
181 GR->getOrCreateSPIRVType(Type::getVoidTy(*Context), MIRBuilder);
182
183 const auto EmitDIInstruction =
184 [&](SPIRV::NonSemanticExtInst::NonSemanticExtInst Inst,
185 std::initializer_list<Register> Registers) {
186 const Register InstReg =
187 MRI.createVirtualRegister(&SPIRV::IDRegClass);
188 MRI.setType(InstReg, LLT::scalar(32));
189 MachineInstrBuilder MIB =
190 MIRBuilder.buildInstr(SPIRV::OpExtInst)
191 .addDef(InstReg)
192 .addUse(GR->getSPIRVTypeID(VoidTy))
193 .addImm(static_cast<int64_t>(
194 SPIRV::InstructionSet::NonSemantic_Shader_DebugInfo_100))
195 .addImm(Inst);
196 for (auto Reg : Registers) {
197 MIB.addUse(Reg);
198 }
199 MIB.constrainAllUses(*TII, *TRI, *RBI);
200 GR->assignSPIRVTypeToVReg(VoidTy, InstReg, MF);
201 return InstReg;
202 };
203
204 const SPIRVType *I32Ty =
205 GR->getOrCreateSPIRVType(Type::getInt32Ty(*Context), MIRBuilder);
206
207 const Register DwarfVersionReg =
208 GR->buildConstantInt(DwarfVersion, MIRBuilder, I32Ty, false);
209 const Register DebugInfoVersionReg =
210 GR->buildConstantInt(DebugInfoVersion, MIRBuilder, I32Ty, false);
211
212 for (unsigned Idx = 0; Idx < LLVMSourceLanguages.size(); ++Idx) {
213 const Register FilePathStrReg = EmitOpString(FilePaths[Idx]);
214
215 const Register DebugSourceResIdReg = EmitDIInstruction(
216 SPIRV::NonSemanticExtInst::DebugSource, {FilePathStrReg});
217
218 SourceLanguage SpirvSourceLanguage = SourceLanguage::Unknown;
219 switch (LLVMSourceLanguages[Idx]) {
220 case dwarf::DW_LANG_OpenCL:
221 SpirvSourceLanguage = SourceLanguage::OpenCL_C;
222 break;
223 case dwarf::DW_LANG_OpenCL_CPP:
224 SpirvSourceLanguage = SourceLanguage::OpenCL_CPP;
225 break;
226 case dwarf::DW_LANG_CPP_for_OpenCL:
227 SpirvSourceLanguage = SourceLanguage::CPP_for_OpenCL;
228 break;
229 case dwarf::DW_LANG_GLSL:
230 SpirvSourceLanguage = SourceLanguage::GLSL;
231 break;
232 case dwarf::DW_LANG_HLSL:
233 SpirvSourceLanguage = SourceLanguage::HLSL;
234 break;
235 case dwarf::DW_LANG_SYCL:
236 SpirvSourceLanguage = SourceLanguage::SYCL;
237 break;
238 case dwarf::DW_LANG_Zig:
239 SpirvSourceLanguage = SourceLanguage::Zig;
240 break;
241 }
242
243 const Register SourceLanguageReg =
244 GR->buildConstantInt(SpirvSourceLanguage, MIRBuilder, I32Ty, false);
245
246 [[maybe_unused]]
247 const Register DebugCompUnitResIdReg =
248 EmitDIInstruction(SPIRV::NonSemanticExtInst::DebugCompilationUnit,
249 {DebugInfoVersionReg, DwarfVersionReg,
250 DebugSourceResIdReg, SourceLanguageReg});
251 }
252
253 // We aren't extracting any DebugInfoFlags now so we
254 // emitting zero to use as <id>Flags argument for DebugBasicType
255 const Register I32ZeroReg =
256 GR->buildConstantInt(0, MIRBuilder, I32Ty, false);
257
258 for (auto *BasicType : BasicTypes) {
259 const Register BasicTypeStrReg = EmitOpString(BasicType->getName());
260
261 const Register ConstIntBitwidthReg = GR->buildConstantInt(
262 BasicType->getSizeInBits(), MIRBuilder, I32Ty, false);
263
264 uint64_t AttributeEncoding = BaseTypeAttributeEncoding::Unspecified;
265 switch (BasicType->getEncoding()) {
266 case dwarf::DW_ATE_signed:
267 AttributeEncoding = BaseTypeAttributeEncoding::Signed;
268 break;
269 case dwarf::DW_ATE_unsigned:
270 AttributeEncoding = BaseTypeAttributeEncoding::Unsigned;
271 break;
272 case dwarf::DW_ATE_unsigned_char:
273 AttributeEncoding = BaseTypeAttributeEncoding::UnsignedChar;
274 break;
275 case dwarf::DW_ATE_signed_char:
276 AttributeEncoding = BaseTypeAttributeEncoding::SignedChar;
277 break;
278 case dwarf::DW_ATE_float:
279 AttributeEncoding = BaseTypeAttributeEncoding::Float;
280 break;
281 case dwarf::DW_ATE_boolean:
282 AttributeEncoding = BaseTypeAttributeEncoding::Boolean;
283 break;
284 case dwarf::DW_ATE_address:
285 AttributeEncoding = BaseTypeAttributeEncoding::Address;
286 }
287
288 const Register AttributeEncodingReg =
289 GR->buildConstantInt(AttributeEncoding, MIRBuilder, I32Ty, false);
290
291 [[maybe_unused]]
292 const Register BasicTypeReg =
293 EmitDIInstruction(SPIRV::NonSemanticExtInst::DebugTypeBasic,
294 {BasicTypeStrReg, ConstIntBitwidthReg,
295 AttributeEncodingReg, I32ZeroReg});
296 }
297 }
298 return true;
299}

301bool SPIRVEmitNonSemanticDI::runOnMachineFunction(MachineFunction &MF) {
302 bool Res = false;
303 // emitGlobalDI needs to be executed only once to avoid

307 Res = emitGlobalDI(MF);
308 }
309 return Res;
