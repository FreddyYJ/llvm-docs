9// This pass traverses through all the basic blocks in a function and converts
10// an indexed load/store with offset "0" to a absolute-set load/store
11// instruction as long as the use of the register in the new instruction
12// dominates the rest of the uses and there are more than 2 uses.

77bool HexagonGenMemAbsolute::runOnMachineFunction(MachineFunction &Fn) {
78 if (skipFunction(Fn.getFunction()))
79 return false;
80
81 TII = Fn.getSubtarget<HexagonSubtarget>().getInstrInfo();
82 MRI = &Fn.getRegInfo();
83 TRI = Fn.getRegInfo().getTargetRegisterInfo();
84
85 MachineDominatorTree &MDT =
86 getAnalysis<MachineDominatorTreeWrapperPass>().getDomTree();
87
88 // Loop over all of the basic blocks
89 for (MachineBasicBlock &MBB : Fn) {
90 // Traverse the basic block
91 for (MachineBasicBlock::iterator MII = MBB.begin(); MII != MBB.end();
92 ++MII) {
93 MachineInstr *MI = &*MII;
94 int Opc = MI->getOpcode();
95 if (Opc != Hexagon::CONST32 && Opc != Hexagon::A2_tfrsi)
96 continue;
97
98 const MachineOperand &MO = MI->getOperand(0);
99 if (!MO.isReg() || !MO.isDef())
100 continue;
101
102 unsigned DstReg = MO.getReg();
103 if (MRI->use_nodbg_empty(DstReg))
104 continue;
105
106 typedef MachineRegisterInfo::use_nodbg_iterator use_iterator;
107 use_iterator NextUseMI = MRI->use_nodbg_begin(DstReg);
108
109 MachineInstr *NextMI = NextUseMI->getParent();
110 int NextOpc = NextMI->getOpcode();
111 int NewOpc;
112 bool IsLoad = isValidIndexedLoad(NextOpc, NewOpc);
113
114 if (!IsLoad && !isValidIndexedStore(NextOpc, NewOpc))
115 continue;
116
117 // Base and Offset positions for load and store instructions
118 // Load R(dest), R(base), Imm -> R(dest) = mem(R(base) + Imm)
119 // Store R(base), Imm, R (src) -> mem(R(base) + Imm) = R(src)
120 unsigned BaseRegPos, ImmPos, RegPos;
121 if (!TII->getBaseAndOffsetPosition(*NextMI, BaseRegPos, ImmPos))
122 continue;
123 RegPos = IsLoad ? 0 : 2;
124
125 bool IsGlobal = MI->getOperand(1).isGlobal();
126 if (!MI->getOperand(1).isImm() && !IsGlobal)
127 continue;
128
129 const MachineOperand *BaseOp = nullptr;
130 int64_t Offset;
131 bool Scalable;
132 TII->getMemOperandWithOffset(*NextMI, BaseOp, Offset, Scalable, TRI);
133
134 // Ensure BaseOp is non-null and register type.
135 if (!BaseOp || !BaseOp->isReg())
136 continue;
137
138 if (Scalable)
139 continue;
140
141 unsigned BaseReg = BaseOp->getReg();
142 if ((DstReg != BaseReg) || (Offset != 0))
143 continue;
144
145 const MachineOperand &MO0 = NextMI->getOperand(RegPos);
146
147 if (!MO0.isReg())
148 continue;
149
150 unsigned LoadStoreReg = MO0.getReg();
151
152 // Store: Bail out if the src and base are same (def and use on same
153 // register).
154 if (LoadStoreReg == BaseReg)
155 continue;
156
157 // Insert the absolute-set instruction "I" only if the use of the
158 // BaseReg in "I" dominates the rest of the uses of BaseReg and if
159 // there are more than 2 uses of this BaseReg.
160 bool Dominates = true;
161 unsigned Counter = 0;
162 for (use_iterator I = NextUseMI, E = MRI->use_nodbg_end(); I != E; ++I) {
163 Counter++;
164 if (!MDT.dominates(NextMI, I->getParent()))
165 Dominates = false;
166 }
167
168 if ((!Dominates) || (Counter < 3))
169 continue;
170
171 // If we reach here, we have met all the conditions required for the
172 // replacement of the absolute instruction.
173 LLVM_DEBUG({
174 dbgs() << "Found a pair of instructions for absolute-set "
175 << (IsLoad ? "load" : "store") << "\n";
176 dbgs() << *MI;
177 dbgs() << *NextMI;
178 });
179 MachineBasicBlock *ParentBlock = NextMI->getParent();
180 MachineInstrBuilder MIB;
181 if (IsLoad) { // Insert absolute-set load instruction
182 ++HexagonNumLoadAbsConversions;
183 MIB = BuildMI(*ParentBlock, NextMI, NextMI->getDebugLoc(),
184 TII->get(NewOpc), LoadStoreReg)
185 .addReg(DstReg, RegState::Define);
186 } else { // Insert absolute-set store instruction
187 ++HexagonNumStoreAbsConversions;
188 MIB = BuildMI(*ParentBlock, NextMI, NextMI->getDebugLoc(),
189 TII->get(NewOpc), DstReg);
190 }
191
192 MachineOperand ImmOperand = MI->getOperand(1);
193 if (IsGlobal)
194 MIB.addGlobalAddress(ImmOperand.getGlobal(), ImmOperand.getOffset(),
195 ImmOperand.getTargetFlags());
196 else
197 MIB.addImm(ImmOperand.getImm());
198
199 if (IsLoad)
200 MIB->getOperand(0).setSubReg(MO0.getSubReg());
201 else
202 MIB.addReg(LoadStoreReg, 0, MO0.getSubReg());
203
204 LLVM_DEBUG(dbgs() << "Replaced with " << *MIB << "\n");
205 // Erase the instructions that got replaced.
206 MII = MBB.erase(MI);
207 --MII;
208 NextMI->getParent()->erase(NextMI);
209 }
210 }
211
212 return true;
213}

215bool HexagonGenMemAbsolute::isValidIndexedLoad(int &Opc, int &NewOpc) {
216
217 bool Result = true;
218 switch (Opc) {
219 case Hexagon::L2_loadrb_io:
220 NewOpc = Hexagon::L4_loadrb_ap;
221 break;
222 case Hexagon::L2_loadrh_io:
223 NewOpc = Hexagon::L4_loadrh_ap;
224 break;
225 case Hexagon::L2_loadri_io:
226 NewOpc = Hexagon::L4_loadri_ap;
227 break;
228 case Hexagon::L2_loadrd_io:
229 NewOpc = Hexagon::L4_loadrd_ap;
230 break;
231 case Hexagon::L2_loadruh_io:
232 NewOpc = Hexagon::L4_loadruh_ap;
233 break;
234 case Hexagon::L2_loadrub_io:
235 NewOpc = Hexagon::L4_loadrub_ap;
236 break;
237 default:
238 Result = false;
239 }
240
241 return Result;
242}

244bool HexagonGenMemAbsolute::isValidIndexedStore(int &Opc, int &NewOpc) {
245
246 bool Result = true;
247 switch (Opc) {
248 case Hexagon::S2_storerd_io:
249 NewOpc = Hexagon::S4_storerd_ap;
250 break;
251 case Hexagon::S2_storeri_io:
252 NewOpc = Hexagon::S4_storeri_ap;
253 break;
254 case Hexagon::S2_storerh_io:
255 NewOpc = Hexagon::S4_storerh_ap;
256 break;
257 case Hexagon::S2_storerb_io:
258 NewOpc = Hexagon::S4_storerb_ap;
259 break;
260 default:
261 Result = false;
262 }
263
264 return Result;
265}
