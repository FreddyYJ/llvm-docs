28// Provides two way mapping between the blocks and numbers.
29class BlockToIndexMapping {
30 SmallVector<BasicBlock *, 32> V;
31
32public:
33 size_t size() const { return V.size(); }
34
35 BlockToIndexMapping(Function &F) {
36 for (BasicBlock &BB : F)
37 V.push_back(&BB);
38 llvm::sort(V);
39 }
40
41 size_t blockToIndex(BasicBlock const *BB) const {
42 auto *I = llvm::lower_bound(V, BB);
43 assert(I != V.end() && *I == BB && "BasicBlockNumberng: Unknown block");
44 return I - V.begin();
45 }
46
47 BasicBlock *indexToBlock(unsigned Index) const { return V[Index]; }
48};

50// The SuspendCrossingInfo maintains data that allows to answer a question
51// whether given two BasicBlocks A and B there is a path from A to B that
52// passes through a suspend point.
53//
54// For every basic block 'i' it maintains a BlockData that consists of:
55// Consumes: a bit vector which contains a set of indices of blocks that can
56// reach block 'i'. A block can trivially reach itself.
57// Kills: a bit vector which contains a set of indices of blocks that can
58// reach block 'i' but there is a path crossing a suspend point
59// not repeating 'i' (path to 'i' without cycles containing 'i').
60// Suspend: a boolean indicating whether block 'i' contains a suspend point.
61// End: a boolean indicating whether block 'i' contains a coro.end intrinsic.
62// KillLoop: There is a path from 'i' to 'i' not otherwise repeating 'i' that
63// crosses a suspend point.
64//
65class SuspendCrossingInfo {
66 BlockToIndexMapping Mapping;
67
68 struct BlockData {
69 BitVector Consumes;
70 BitVector Kills;
71 bool Suspend = false;
72 bool End = false;
73 bool KillLoop = false;
74 bool Changed = false;
75 };
76 SmallVector<BlockData, 32> Block;
77
78 iterator_range<pred_iterator> predecessors(BlockData const &BD) const {
79 BasicBlock *BB = Mapping.indexToBlock(&BD - &Block[0]);
80 return llvm::predecessors(BB);
81 }
82
83 BlockData &getBlockData(BasicBlock *BB) {
84 return Block[Mapping.blockToIndex(BB)];
85 }

116 bool isDefinitionAcrossSuspend(BasicBlock *DefBB, User *U) const {
117 auto *I = cast<Instruction>(U);
118
119 // We rewrote PHINodes, so that only the ones with exactly one incoming
120 // value need to be analyzed.
121 if (auto *PN = dyn_cast<PHINode>(I))
122 if (PN->getNumIncomingValues() > 1)
123 return false;
124
125 BasicBlock *UseBB = I->getParent();
126
127 // As a special case, treat uses by an llvm.coro.suspend.retcon or an
128 // llvm.coro.suspend.async as if they were uses in the suspend's single
129 // predecessor: the uses conceptually occur before the suspend.
130 if (isa<CoroSuspendRetconInst>(I) || isa<CoroSuspendAsyncInst>(I)) {
131 UseBB = UseBB->getSinglePredecessor();
132 assert(UseBB && "should have split coro.suspend into its own block");
133 }
134
135 return hasPathCrossingSuspendPoint(DefBB, UseBB);
136 }
137
138 bool isDefinitionAcrossSuspend(Argument &A, User *U) const {
139 return isDefinitionAcrossSuspend(&A.getParent()->getEntryBlock(), U);
140 }
141
142 bool isDefinitionAcrossSuspend(Instruction &I, User *U) const {
143 auto *DefBB = I.getParent();
144
145 // As a special case, treat values produced by an llvm.coro.suspend.*
146 // as if they were defined in the single successor: the uses
147 // conceptually occur after the suspend.
148 if (isa<AnyCoroSuspendInst>(I)) {
149 DefBB = DefBB->getSingleSuccessor();
150 assert(DefBB && "should have split coro.suspend into its own block");
151 }
152
153 return isDefinitionAcrossSuspend(DefBB, U);
154 }
155
156 bool isDefinitionAcrossSuspend(Value &V, User *U) const {
157 if (auto *Arg = dyn_cast<Argument>(&V))
158 return isDefinitionAcrossSuspend(*Arg, U);
159 if (auto *Inst = dyn_cast<Instruction>(&V))
160 return isDefinitionAcrossSuspend(*Inst, U);
161
162 llvm_unreachable(
163 "Coroutine could only collect Argument and Instruction now.");
164 }
165
166 bool isDefinitionAcrossSuspend(Value &V) const {
167 if (auto *Arg = dyn_cast<Argument>(&V)) {
168 for (User *U : Arg->users())
169 if (isDefinitionAcrossSuspend(*Arg, U))
170 return true;
171 } else if (auto *Inst = dyn_cast<Instruction>(&V)) {
172 for (User *U : Inst->users())
173 if (isDefinitionAcrossSuspend(*Inst, U))
174 return true;
175 }
176
177 llvm_unreachable(
178 "Coroutine could only collect Argument and Instruction now.");
179 }
180};
