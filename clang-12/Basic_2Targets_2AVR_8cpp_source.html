288bool AVRTargetInfo::isValidCPUName(StringRef Name) const {
289 bool IsFamily =
290 llvm::find(ValidFamilyNames, Name) != std::end(ValidFamilyNames);
291
292 bool IsMCU =
293 llvm::find_if(AVRMcus, [&](const MCUInfo &Info) {
294 return Info.Name == Name;
295 }) != std::end(AVRMcus);
296 return IsFamily || IsMCU;
297}
298
299void AVRTargetInfo::fillValidCPUList(SmallVectorImpl<StringRef> &Values) const {
300 Values.append(std::begin(ValidFamilyNames), std::end(ValidFamilyNames));
301 for (const MCUInfo &Info : AVRMcus)
302 Values.push_back(Info.Name);
303}

305void AVRTargetInfo::getTargetDefines(const LangOptions &Opts,
306 MacroBuilder &Builder) const {
307 Builder.defineMacro("AVR");
308 Builder.defineMacro("__AVR");
309 Builder.defineMacro("__AVR__");
310 Builder.defineMacro("__ELF__");
311
312 if (!this->CPU.empty()) {
313 auto It = llvm::find_if(
314 AVRMcus, [&](const MCUInfo &Info) { return Info.Name == this->CPU; });
315
316 if (It != std::end(AVRMcus))
317 Builder.defineMacro(It->DefineName);
318 }
