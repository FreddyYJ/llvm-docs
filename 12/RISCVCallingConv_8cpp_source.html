20// Calling Convention Implementation.
21// The expectations for frontend ABI lowering vary from target to target.
22// Ideally, an LLVM frontend would be able to avoid worrying about many ABI
23// details, but this is a longer term goal. For now, we simply try to keep the
24// role of the frontend as simple and well-defined as possible. The rules can
25// be summarised as:
26// * Never split up large scalar arguments. We handle them here.
27// * If a hardfloat calling convention is being used, and the struct may be
28// passed in a pair of registers (fp+fp, int+fp), and both registers are
29// available, then pass as two separate arguments. If either the GPRs or FPRs
30// are exhausted, then pass according to the rule below.
31// * If a struct could never be passed in registers or directly in a stack
32// slot (as it is larger than 2*XLEN and the floating point rules don't
33// apply), then pass it using a pointer with the byval attribute.
34// * If a struct is less than 2*XLEN, then coerce to either a two-element
35// word-sized array or a 2*XLEN scalar (depending on alignment).
36// * The frontend can determine whether a struct is returned by reference or
37// not based on its size and fields. If it will be returned by reference, the
38// frontend must modify the prototype so a pointer with the sret annotation is
39// passed as the first argument. This is not necessary for large scalar
40// returns.
41// * Struct return values and varargs should be coerced to structs containing
42// register-size fields in the same situations they would be for fixed
43// arguments.
44

197// Pass a 2*XLEN argument that has been split into two XLEN values through
198// registers or the stack as necessary.
199static bool CC_RISCVAssign2XLen(unsigned XLen, CCState &State, CCValAssign VA1,
200 ISD::ArgFlagsTy ArgFlags1, unsigned ValNo2,
201 MVT ValVT2, MVT LocVT2,
202 ISD::ArgFlagsTy ArgFlags2, bool EABI) {
203 unsigned XLenInBytes = XLen / 8;
204 const RISCVSubtarget &STI =
205 State.getMachineFunction().getSubtarget<RISCVSubtarget>();
206 ArrayRef<MCPhysReg> ArgGPRs = RISCV::getArgGPRs(STI.getTargetABI());
207
208 if (MCRegister Reg = State.AllocateReg(ArgGPRs)) {
209 // At least one half can be passed via register.
210 State.addLoc(CCValAssign::getReg(VA1.getValNo(), VA1.getValVT(), Reg,
211 VA1.getLocVT(), CCValAssign::Full));
212 } else {
213 // Both halves must be passed on the stack, with proper alignment.
214 // TODO: To be compatible with GCC's behaviors, we force them to have 4-byte
215 // alignment. This behavior may be changed when RV32E/ILP32E is ratified.
216 Align StackAlign(XLenInBytes);
217 if (!EABI || XLen != 32)
218 StackAlign = std::max(StackAlign, ArgFlags1.getNonZeroOrigAlign());
219 State.addLoc(
220 CCValAssign::getMem(VA1.getValNo(), VA1.getValVT(),
221 State.AllocateStack(XLenInBytes, StackAlign),
222 VA1.getLocVT(), CCValAssign::Full));
223 State.addLoc(CCValAssign::getMem(
224 ValNo2, ValVT2, State.AllocateStack(XLenInBytes, Align(XLenInBytes)),
225 LocVT2, CCValAssign::Full));
226 return false;
227 }
228
229 if (MCRegister Reg = State.AllocateReg(ArgGPRs)) {
230 // The second half can also be passed via register.
231 State.addLoc(
232 CCValAssign::getReg(ValNo2, ValVT2, Reg, LocVT2, CCValAssign::Full));
233 } else {
234 // The second half is passed via the stack, without additional alignment.
235 State.addLoc(CCValAssign::getMem(
236 ValNo2, ValVT2, State.AllocateStack(XLenInBytes, Align(XLenInBytes)),
237 LocVT2, CCValAssign::Full));
238 }
239
240 return false;
241}

243static MCRegister allocateRVVReg(MVT ValVT, unsigned ValNo, CCState &State,
244 const RISCVTargetLowering &TLI) {
245 const TargetRegisterClass *RC = TLI.getRegClassFor(ValVT);
246 if (RC == &RISCV::VRRegClass) {
247 // Assign the first mask argument to V0.
248 // This is an interim calling convention and it may be changed in the
249 // future.
250 if (ValVT.getVectorElementType() == MVT::i1)
251 if (MCRegister Reg = State.AllocateReg(RISCV::V0))
252 return Reg;
253 return State.AllocateReg(ArgVRs);
254 }
255 if (RC == &RISCV::VRM2RegClass)
256 return State.AllocateReg(ArgVRM2s);
257 if (RC == &RISCV::VRM4RegClass)
258 return State.AllocateReg(ArgVRM4s);
259 if (RC == &RISCV::VRM8RegClass)
260 return State.AllocateReg(ArgVRM8s);
261 if (RC == &RISCV::VRN2M1RegClass)
262 return State.AllocateReg(ArgVRN2M1s);
263 if (RC == &RISCV::VRN3M1RegClass)
264 return State.AllocateReg(ArgVRN3M1s);
265 if (RC == &RISCV::VRN4M1RegClass)
266 return State.AllocateReg(ArgVRN4M1s);
267 if (RC == &RISCV::VRN5M1RegClass)
268 return State.AllocateReg(ArgVRN5M1s);
269 if (RC == &RISCV::VRN6M1RegClass)
270 return State.AllocateReg(ArgVRN6M1s);
271 if (RC == &RISCV::VRN7M1RegClass)
272 return State.AllocateReg(ArgVRN7M1s);
273 if (RC == &RISCV::VRN8M1RegClass)
274 return State.AllocateReg(ArgVRN8M1s);
275 if (RC == &RISCV::VRN2M2RegClass)
276 return State.AllocateReg(ArgVRN2M2s);
277 if (RC == &RISCV::VRN3M2RegClass)
278 return State.AllocateReg(ArgVRN3M2s);
279 if (RC == &RISCV::VRN4M2RegClass)
280 return State.AllocateReg(ArgVRN4M2s);
281 if (RC == &RISCV::VRN2M4RegClass)
282 return State.AllocateReg(ArgVRN2M4s);
283 llvm_unreachable("Unhandled register class for ValueType");
284}

286// Implements the RISC-V calling convention. Returns true upon failure.
287bool llvm::CC_RISCV(unsigned ValNo, MVT ValVT, MVT LocVT,
288 CCValAssign::LocInfo LocInfo, ISD::ArgFlagsTy ArgFlags,
289 CCState &State, bool IsFixed, bool IsRet, Type *OrigTy) {
290 const MachineFunction &MF = State.getMachineFunction();
291 const DataLayout &DL = MF.getDataLayout();
292 const RISCVSubtarget &Subtarget = MF.getSubtarget<RISCVSubtarget>();
293 const RISCVTargetLowering &TLI = *Subtarget.getTargetLowering();
294
295 unsigned XLen = Subtarget.getXLen();
296 MVT XLenVT = Subtarget.getXLenVT();
297
298 // Static chain parameter must not be passed in normal argument registers,
299 // so we assign t2 for it as done in GCC's __builtin_call_with_static_chain
300 if (ArgFlags.isNest()) {
301 if (MCRegister Reg = State.AllocateReg(RISCV::X7)) {
302 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
303 return false;
304 }
305 }
306
307 // Any return value split in to more than two values can't be returned
308 // directly. Vectors are returned via the available vector registers.
309 if (!LocVT.isVector() && IsRet && ValNo > 1)
310 return true;
311
312 // UseGPRForF16_F32 if targeting one of the soft-float ABIs, if passing a
313 // variadic argument, or if no F16/F32 argument registers are available.
314 bool UseGPRForF16_F32 = true;
315 // UseGPRForF64 if targeting soft-float ABIs or an FLEN=32 ABI, if passing a
316 // variadic argument, or if no F64 argument registers are available.
317 bool UseGPRForF64 = true;
318
319 RISCVABI::ABI ABI = Subtarget.getTargetABI();
320 switch (ABI) {
321 default:
322 llvm_unreachable("Unexpected ABI");
323 case RISCVABI::ABI_ILP32:
324 case RISCVABI::ABI_ILP32E:
325 case RISCVABI::ABI_LP64:
326 case RISCVABI::ABI_LP64E:
327 break;
328 case RISCVABI::ABI_ILP32F:
329 case RISCVABI::ABI_LP64F:
330 UseGPRForF16_F32 = !IsFixed;
331 break;
332 case RISCVABI::ABI_ILP32D:
333 case RISCVABI::ABI_LP64D:
334 UseGPRForF16_F32 = !IsFixed;
335 UseGPRForF64 = !IsFixed;
336 break;
337 }
338
339 if ((LocVT == MVT::f16 || LocVT == MVT::bf16) && !UseGPRForF16_F32) {
340 if (MCRegister Reg = State.AllocateReg(ArgFPR16s)) {
341 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
342 return false;
343 }
344 }
345
346 if (LocVT == MVT::f32 && !UseGPRForF16_F32) {
347 if (MCRegister Reg = State.AllocateReg(ArgFPR32s)) {
348 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
349 return false;
350 }
351 }
352
353 if (LocVT == MVT::f64 && !UseGPRForF64) {
354 if (MCRegister Reg = State.AllocateReg(ArgFPR64s)) {
355 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
356 return false;
357 }
358 }
359
360 if ((ValVT == MVT::f16 && Subtarget.hasStdExtZhinxmin())) {
361 if (MCRegister Reg = State.AllocateReg(getArgGPR16s(ABI))) {
362 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
363 return false;
364 }
365 }
366
367 ArrayRef<MCPhysReg> ArgGPRs = RISCV::getArgGPRs(ABI);
368
369 // Zfinx/Zdinx use GPR without a bitcast when possible.
370 if ((LocVT == MVT::f32 && XLen == 32 && Subtarget.hasStdExtZfinx()) ||
371 (LocVT == MVT::f64 && XLen == 64 && Subtarget.hasStdExtZdinx())) {
372 if (MCRegister Reg = State.AllocateReg(ArgGPRs)) {
373 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
374 return false;
375 }
376 }
377
378 // FP smaller than XLen, uses custom GPR.
379 if (LocVT == MVT::f16 || LocVT == MVT::bf16 ||
380 (LocVT == MVT::f32 && XLen == 64)) {
381 if (MCRegister Reg = State.AllocateReg(ArgGPRs)) {
382 LocVT = XLenVT;
383 State.addLoc(
384 CCValAssign::getCustomReg(ValNo, ValVT, Reg, LocVT, LocInfo));
385 return false;
386 }
387 }
388
389 // Bitcast FP to GPR if we can use a GPR register.
390 if ((XLen == 32 && LocVT == MVT::f32) || (XLen == 64 && LocVT == MVT::f64)) {
391 if (MCRegister Reg = State.AllocateReg(ArgGPRs)) {
392 LocVT = XLenVT;
393 LocInfo = CCValAssign::BCvt;
394 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
395 return false;
396 }
397 }
398
399 // If this is a variadic argument, the RISC-V calling convention requires
400 // that it is assigned an 'even' or 'aligned' register if it has 8-byte
401 // alignment (RV32) or 16-byte alignment (RV64). An aligned register should
402 // be used regardless of whether the original argument was split during
403 // legalisation or not. The argument will not be passed by registers if the
404 // original type is larger than 2*XLEN, so the register alignment rule does
405 // not apply.
406 // TODO: To be compatible with GCC's behaviors, we don't align registers
407 // currently if we are using ILP32E calling convention. This behavior may be
408 // changed when RV32E/ILP32E is ratified.
409 unsigned TwoXLenInBytes = (2 * XLen) / 8;
410 if (!IsFixed && ArgFlags.getNonZeroOrigAlign() == TwoXLenInBytes &&
411 DL.getTypeAllocSize(OrigTy) == TwoXLenInBytes &&
412 ABI != RISCVABI::ABI_ILP32E) {
413 unsigned RegIdx = State.getFirstUnallocated(ArgGPRs);
414 // Skip 'odd' register if necessary.
415 if (RegIdx != std::size(ArgGPRs) && RegIdx % 2 == 1)
416 State.AllocateReg(ArgGPRs);
417 }
418
419 SmallVectorImpl<CCValAssign> &PendingLocs = State.getPendingLocs();
420 SmallVectorImpl<ISD::ArgFlagsTy> &PendingArgFlags =
421 State.getPendingArgFlags();
422
423 assert(PendingLocs.size() == PendingArgFlags.size() &&
424 "PendingLocs and PendingArgFlags out of sync");
425
426 // Handle passing f64 on RV32D with a soft float ABI or when floating point
427 // registers are exhausted.
428 if (XLen == 32 && LocVT == MVT::f64) {
429 assert(PendingLocs.empty() && "Can't lower f64 if it is split");
430 // Depending on available argument GPRS, f64 may be passed in a pair of
431 // GPRs, split between a GPR and the stack, or passed completely on the
432 // stack. LowerCall/LowerFormalArguments/LowerReturn must recognise these
433 // cases.
434 MCRegister Reg = State.AllocateReg(ArgGPRs);
435 if (!Reg) {
436 int64_t StackOffset = State.AllocateStack(8, Align(8));
437 State.addLoc(
438 CCValAssign::getMem(ValNo, ValVT, StackOffset, LocVT, LocInfo));
439 return false;
440 }
441 LocVT = MVT::i32;
442 State.addLoc(CCValAssign::getCustomReg(ValNo, ValVT, Reg, LocVT, LocInfo));
443 MCRegister HiReg = State.AllocateReg(ArgGPRs);
444 if (HiReg) {
445 State.addLoc(
446 CCValAssign::getCustomReg(ValNo, ValVT, HiReg, LocVT, LocInfo));
447 } else {
448 int64_t StackOffset = State.AllocateStack(4, Align(4));
449 State.addLoc(
450 CCValAssign::getCustomMem(ValNo, ValVT, StackOffset, LocVT, LocInfo));
451 }
452 return false;
453 }
454
455 // Split arguments might be passed indirectly, so keep track of the pending
456 // values. Split vectors are passed via a mix of registers and indirectly, so
457 // treat them as we would any other argument.
458 if (ValVT.isScalarInteger() && (ArgFlags.isSplit() || !PendingLocs.empty())) {
459 LocVT = XLenVT;
460 LocInfo = CCValAssign::Indirect;
461 PendingLocs.push_back(
462 CCValAssign::getPending(ValNo, ValVT, LocVT, LocInfo));
463 PendingArgFlags.push_back(ArgFlags);
464 if (!ArgFlags.isSplitEnd()) {
465 return false;
466 }
467 }
468
469 // If the split argument only had two elements, it should be passed directly
470 // in registers or on the stack.
471 if (ValVT.isScalarInteger() && ArgFlags.isSplitEnd() &&
472 PendingLocs.size() <= 2) {
473 assert(PendingLocs.size() == 2 && "Unexpected PendingLocs.size()");
474 // Apply the normal calling convention rules to the first half of the
475 // split argument.
476 CCValAssign VA = PendingLocs[0];
477 ISD::ArgFlagsTy AF = PendingArgFlags[0];
478 PendingLocs.clear();
479 PendingArgFlags.clear();
480 return CC_RISCVAssign2XLen(
481 XLen, State, VA, AF, ValNo, ValVT, LocVT, ArgFlags,
482 ABI == RISCVABI::ABI_ILP32E || ABI == RISCVABI::ABI_LP64E);
483 }
484
485 // Allocate to a register if possible, or else a stack slot.
486 MCRegister Reg;
487 unsigned StoreSizeBytes = XLen / 8;
488 Align StackAlign = Align(XLen / 8);
489
490 if (ValVT.isVector() || ValVT.isRISCVVectorTuple()) {
491 Reg = allocateRVVReg(ValVT, ValNo, State, TLI);
492 if (Reg) {
493 // Fixed-length vectors are located in the corresponding scalable-vector
494 // container types.
495 if (ValVT.isFixedLengthVector()) {
496 LocVT = TLI.getContainerForFixedLengthVector(LocVT);
497 State.addLoc(
498 CCValAssign::getCustomReg(ValNo, ValVT, Reg, LocVT, LocInfo));
499 return false;
500 }
501 } else {
502 // For return values, the vector must be passed fully via registers or
503 // via the stack.
504 // FIXME: The proposed vector ABI only mandates v8-v15 for return values,
505 // but we're using all of them.
506 if (IsRet)
507 return true;
508 // Try using a GPR to pass the address
509 if ((Reg = State.AllocateReg(ArgGPRs))) {
510 LocVT = XLenVT;
511 LocInfo = CCValAssign::Indirect;
512 } else if (ValVT.isScalableVector()) {
513 LocVT = XLenVT;
514 LocInfo = CCValAssign::Indirect;
515 } else {
516 StoreSizeBytes = ValVT.getStoreSize();
517 // Align vectors to their element sizes, being careful for vXi1
518 // vectors.
519 StackAlign = MaybeAlign(ValVT.getScalarSizeInBits() / 8).valueOrOne();
520 }
521 }
522 } else {
523 Reg = State.AllocateReg(ArgGPRs);
524 }
525
526 int64_t StackOffset =
527 Reg ? 0 : State.AllocateStack(StoreSizeBytes, StackAlign);
528
529 // If we reach this point and PendingLocs is non-empty, we must be at the
530 // end of a split argument that must be passed indirectly.
531 if (!PendingLocs.empty()) {
532 assert(ArgFlags.isSplitEnd() && "Expected ArgFlags.isSplitEnd()");
533 assert(PendingLocs.size() > 2 && "Unexpected PendingLocs.size()");
534
535 for (auto &It : PendingLocs) {
536 if (Reg)
537 It.convertToReg(Reg);
538 else
539 It.convertToMem(StackOffset);
540 State.addLoc(It);
541 }
542 PendingLocs.clear();
543 PendingArgFlags.clear();
544 return false;
545 }
546
547 assert(((ValVT.isFloatingPoint() && !ValVT.isVector()) || LocVT == XLenVT ||
548 (TLI.getSubtarget().hasVInstructions() &&
549 (ValVT.isVector() || ValVT.isRISCVVectorTuple()))) &&
550 "Expected an XLenVT or vector types at this stage");
551
552 if (Reg) {
553 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
554 return false;
555 }
556
557 State.addLoc(CCValAssign::getMem(ValNo, ValVT, StackOffset, LocVT, LocInfo));
558 return false;
559}

561// FastCC has less than 1% performance improvement for some particular
562// benchmark. But theoretically, it may have benefit for some cases.
563bool llvm::CC_RISCV_FastCC(unsigned ValNo, MVT ValVT, MVT LocVT,
564 CCValAssign::LocInfo LocInfo,
565 ISD::ArgFlagsTy ArgFlags, CCState &State,
566 bool IsFixed, bool IsRet, Type *OrigTy) {
567 const MachineFunction &MF = State.getMachineFunction();
568 const RISCVSubtarget &Subtarget = MF.getSubtarget<RISCVSubtarget>();
569 const RISCVTargetLowering &TLI = *Subtarget.getTargetLowering();
570 RISCVABI::ABI ABI = Subtarget.getTargetABI();
571
572 if ((LocVT == MVT::f16 && Subtarget.hasStdExtZfhmin()) ||
573 (LocVT == MVT::bf16 && Subtarget.hasStdExtZfbfmin())) {
574 static const MCPhysReg FPR16List[] = {
575 RISCV::F10_H, RISCV::F11_H, RISCV::F12_H, RISCV::F13_H, RISCV::F14_H,
576 RISCV::F15_H, RISCV::F16_H, RISCV::F17_H, RISCV::F0_H, RISCV::F1_H,
577 RISCV::F2_H, RISCV::F3_H, RISCV::F4_H, RISCV::F5_H, RISCV::F6_H,
578 RISCV::F7_H, RISCV::F28_H, RISCV::F29_H, RISCV::F30_H, RISCV::F31_H};
579 if (MCRegister Reg = State.AllocateReg(FPR16List)) {
580 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
581 return false;
582 }
583 }
584
585 if (LocVT == MVT::f32 && Subtarget.hasStdExtF()) {
586 static const MCPhysReg FPR32List[] = {
587 RISCV::F10_F, RISCV::F11_F, RISCV::F12_F, RISCV::F13_F, RISCV::F14_F,
588 RISCV::F15_F, RISCV::F16_F, RISCV::F17_F, RISCV::F0_F, RISCV::F1_F,
589 RISCV::F2_F, RISCV::F3_F, RISCV::F4_F, RISCV::F5_F, RISCV::F6_F,
590 RISCV::F7_F, RISCV::F28_F, RISCV::F29_F, RISCV::F30_F, RISCV::F31_F};
591 if (MCRegister Reg = State.AllocateReg(FPR32List)) {
592 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
593 return false;
594 }
595 }
596
597 if (LocVT == MVT::f64 && Subtarget.hasStdExtD()) {
598 static const MCPhysReg FPR64List[] = {
599 RISCV::F10_D, RISCV::F11_D, RISCV::F12_D, RISCV::F13_D, RISCV::F14_D,
600 RISCV::F15_D, RISCV::F16_D, RISCV::F17_D, RISCV::F0_D, RISCV::F1_D,
601 RISCV::F2_D, RISCV::F3_D, RISCV::F4_D, RISCV::F5_D, RISCV::F6_D,
602 RISCV::F7_D, RISCV::F28_D, RISCV::F29_D, RISCV::F30_D, RISCV::F31_D};
603 if (MCRegister Reg = State.AllocateReg(FPR64List)) {
604 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
605 return false;
606 }
607 }
608
609 MVT XLenVT = Subtarget.getXLenVT();
610
611 // Check if there is an available GPRF16 before hitting the stack.
612 if ((LocVT == MVT::f16 && Subtarget.hasStdExtZhinxmin())) {
613 if (MCRegister Reg = State.AllocateReg(getFastCCArgGPRF16s(ABI))) {
614 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
615 return false;
616 }
617 }
618
619 // Check if there is an available GPR before hitting the stack.
620 if ((LocVT == MVT::f32 && Subtarget.hasStdExtZfinx()) ||
621 (LocVT == MVT::f64 && Subtarget.is64Bit() &&
622 Subtarget.hasStdExtZdinx())) {
623 if (MCRegister Reg = State.AllocateReg(getFastCCArgGPRs(ABI))) {
624 if (LocVT.getSizeInBits() != Subtarget.getXLen()) {
625 LocVT = XLenVT;
626 State.addLoc(
627 CCValAssign::getCustomReg(ValNo, ValVT, Reg, LocVT, LocInfo));
628 return false;
629 }
630 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
631 return false;
632 }
633 }
634
635 ArrayRef<MCPhysReg> ArgGPRs = getFastCCArgGPRs(ABI);
636
637 if (LocVT.isVector()) {
638 if (MCRegister Reg = allocateRVVReg(ValVT, ValNo, State, TLI)) {
639 // Fixed-length vectors are located in the corresponding scalable-vector
640 // container types.
641 if (LocVT.isFixedLengthVector()) {
642 LocVT = TLI.getContainerForFixedLengthVector(LocVT);
643 State.addLoc(
644 CCValAssign::getCustomReg(ValNo, ValVT, Reg, LocVT, LocInfo));
645 return false;
646 }
647 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
648 return false;
649 }
650
651 // Pass scalable vectors indirectly. Pass fixed vectors indirectly if we
652 // have a free GPR.
653 if (LocVT.isScalableVector() ||
654 State.getFirstUnallocated(ArgGPRs) != ArgGPRs.size()) {
655 LocInfo = CCValAssign::Indirect;
656 LocVT = XLenVT;
657 }
658 }
659
660 if (LocVT == XLenVT) {
661 if (MCRegister Reg = State.AllocateReg(getFastCCArgGPRs(ABI))) {
662 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
663 return false;
664 }
665 }
666
667 if (LocVT == XLenVT || LocVT == MVT::f16 || LocVT == MVT::bf16 ||
668 LocVT == MVT::f32 || LocVT == MVT::f64 || LocVT.isFixedLengthVector()) {
669 Align StackAlign = MaybeAlign(ValVT.getScalarSizeInBits() / 8).valueOrOne();
670 int64_t Offset = State.AllocateStack(LocVT.getStoreSize(), StackAlign);
671 State.addLoc(CCValAssign::getMem(ValNo, ValVT, Offset, LocVT, LocInfo));
672 return false;
673 }
674
675 return true; // CC didn't match.
676}

678bool llvm::CC_RISCV_GHC(unsigned ValNo, MVT ValVT, MVT LocVT,
679 CCValAssign::LocInfo LocInfo, ISD::ArgFlagsTy ArgFlags,
680 CCState &State) {
681 if (ArgFlags.isNest()) {
682 report_fatal_error(
683 "Attribute 'nest' is not supported in GHC calling convention");
684 }
685

687 RISCV::X9, RISCV::X18, RISCV::X19, RISCV::X20, RISCV::X21, RISCV::X22,
688 RISCV::X23, RISCV::X24, RISCV::X25, RISCV::X26, RISCV::X27};

700 State.getMachineFunction().getSubtarget<RISCVSubtarget>();
701
702 if (LocVT == MVT::f32 && Subtarget.hasStdExtF()) {
703 // Pass in STG registers: F1, ..., F6
704 // fs0 ... fs5
705 static const MCPhysReg FPR32List[] = {RISCV::F8_F, RISCV::F9_F,
706 RISCV::F18_F, RISCV::F19_F,
707 RISCV::F20_F, RISCV::F21_F};

709 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
710 return false;
711 }
712 }
713
714 if (LocVT == MVT::f64 && Subtarget.hasStdExtD()) {
715 // Pass in STG registers: D1, ..., D6
716 // fs6 ... fs11
717 static const MCPhysReg FPR64List[] = {RISCV::F22_D, RISCV::F23_D,
718 RISCV::F24_D, RISCV::F25_D,
719 RISCV::F26_D, RISCV::F27_D};
720 if (MCRegister Reg = State.AllocateReg(FPR64List)) {
721 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));

723 }
724 }

727 (LocVT == MVT::f64 && Subtarget.hasStdExtZdinx() &&

729 if (MCRegister Reg = State.AllocateReg(GPRList)) {
730 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Reg, LocVT, LocInfo));
731 return false;
732 }
733 }
734
735 report_fatal_error("No registers left in GHC calling convention");
736 return true;
