16std::unique_ptr<Value> Context::detachLLVMValue(llvm::Value *V) {
17 std::unique_ptr<Value> Erased;
18 auto It = LLVMValueToValueMap.find(V);
19 if (It != LLVMValueToValueMap.end()) {
20 auto *Val = It->second.release();
21 Erased = std::unique_ptr<Value>(Val);
22 LLVMValueToValueMap.erase(It);
23 }
24 return Erased;
25}

27std::unique_ptr<Value> Context::detach(Value *V) {
28 assert(V->getSubclassID() != Value::ClassID::Constant &&
29 "Can't detach a constant!");
30 assert(V->getSubclassID() != Value::ClassID::User && "Can't detach a user!");
31 return detachLLVMValue(V->Val);
32}

34Value *Context::registerValue(std::unique_ptr<Value> &&VPtr) {
35 assert(VPtr->getSubclassID() != Value::ClassID::User &&
36 "Can't register a user!");
37
38 // Track creation of instructions.
39 // Please note that we don't allow the creation of detached instructions,
40 // meaning that the instructions need to be inserted into a block upon
41 // creation. This is why the tracker class combines creation and insertion.
42 if (auto *I = dyn_cast<Instruction>(VPtr.get()))
43 getTracker().emplaceIfTracking<CreateAndInsertInst>(I);
44
45 Value *V = VPtr.get();
46 [[maybe_unused]] auto Pair =
47 LLVMValueToValueMap.insert({VPtr->Val, std::move(VPtr)});
48 assert(Pair.second && "Already exists!");
49 return V;
50}

52Value *Context::getOrCreateValueInternal(llvm::Value *LLVMV, llvm::User *U) {
53 auto Pair = LLVMValueToValueMap.insert({LLVMV, nullptr});
54 auto It = Pair.first;
55 if (!Pair.second)
56 return It->second.get();
57
58 if (auto *C = dyn_cast<llvm::Constant>(LLVMV)) {
59 switch (C->getValueID()) {
60 case llvm::Value::ConstantIntVal:
61 It->second = std::unique_ptr<ConstantInt>(
62 new ConstantInt(cast<llvm::ConstantInt>(C), *this));
63 return It->second.get();
64 case llvm::Value::ConstantFPVal:
65 It->second = std::unique_ptr<ConstantFP>(
66 new ConstantFP(cast<llvm::ConstantFP>(C), *this));
67 return It->second.get();
68 case llvm::Value::BlockAddressVal:
69 It->second = std::unique_ptr<BlockAddress>(
70 new BlockAddress(cast<llvm::BlockAddress>(C), *this));
71 return It->second.get();
72 case llvm::Value::ConstantTokenNoneVal:
73 It->second = std::unique_ptr<ConstantTokenNone>(
74 new ConstantTokenNone(cast<llvm::ConstantTokenNone>(C), *this));
75 return It->second.get();
76 case llvm::Value::ConstantAggregateZeroVal: {
77 auto *CAZ = cast<llvm::ConstantAggregateZero>(C);
78 It->second = std::unique_ptr<ConstantAggregateZero>(
79 new ConstantAggregateZero(CAZ, *this));
80 auto *Ret = It->second.get();
81 // Must create sandboxir for elements.
82 auto EC = CAZ->getElementCount();
83 if (EC.isFixed()) {
84 for (auto ElmIdx : seq<unsigned>(0, EC.getFixedValue()))
85 getOrCreateValueInternal(CAZ->getElementValue(ElmIdx), CAZ);
86 }
87 return Ret;
88 }
89 case llvm::Value::ConstantPointerNullVal:
90 It->second = std::unique_ptr<ConstantPointerNull>(
91 new ConstantPointerNull(cast<llvm::ConstantPointerNull>(C), *this));
92 return It->second.get();
93 case llvm::Value::PoisonValueVal:
94 It->second = std::unique_ptr<PoisonValue>(
95 new PoisonValue(cast<llvm::PoisonValue>(C), *this));
96 return It->second.get();
97 case llvm::Value::UndefValueVal:
98 It->second = std::unique_ptr<UndefValue>(
99 new UndefValue(cast<llvm::UndefValue>(C), *this));
100 return It->second.get();
101 case llvm::Value::DSOLocalEquivalentVal: {
102 auto *DSOLE = cast<llvm::DSOLocalEquivalent>(C);
103 It->second = std::unique_ptr<DSOLocalEquivalent>(
104 new DSOLocalEquivalent(DSOLE, *this));
105 auto *Ret = It->second.get();
106 getOrCreateValueInternal(DSOLE->getGlobalValue(), DSOLE);
107 return Ret;
108 }
109 case llvm::Value::ConstantArrayVal:
110 It->second = std::unique_ptr<ConstantArray>(
111 new ConstantArray(cast<llvm::ConstantArray>(C), *this));
112 break;
113 case llvm::Value::ConstantStructVal:
114 It->second = std::unique_ptr<ConstantStruct>(
115 new ConstantStruct(cast<llvm::ConstantStruct>(C), *this));
116 break;
117 case llvm::Value::ConstantVectorVal:
118 It->second = std::unique_ptr<ConstantVector>(
119 new ConstantVector(cast<llvm::ConstantVector>(C), *this));
120 break;
121 case llvm::Value::FunctionVal:
122 It->second = std::unique_ptr<Function>(
123 new Function(cast<llvm::Function>(C), *this));
124 break;
125 case llvm::Value::GlobalIFuncVal:
126 It->second = std::unique_ptr<GlobalIFunc>(
127 new GlobalIFunc(cast<llvm::GlobalIFunc>(C), *this));
128 break;
129 case llvm::Value::GlobalVariableVal:
130 It->second = std::unique_ptr<GlobalVariable>(
131 new GlobalVariable(cast<llvm::GlobalVariable>(C), *this));
132 break;
133 case llvm::Value::GlobalAliasVal:
134 It->second = std::unique_ptr<GlobalAlias>(
135 new GlobalAlias(cast<llvm::GlobalAlias>(C), *this));
136 break;
137 case llvm::Value::NoCFIValueVal:
138 It->second = std::unique_ptr<NoCFIValue>(
139 new NoCFIValue(cast<llvm::NoCFIValue>(C), *this));
140 break;
141 case llvm::Value::ConstantPtrAuthVal:
142 It->second = std::unique_ptr<ConstantPtrAuth>(
143 new ConstantPtrAuth(cast<llvm::ConstantPtrAuth>(C), *this));
144 break;
145 case llvm::Value::ConstantExprVal:
146 It->second = std::unique_ptr<ConstantExpr>(
147 new ConstantExpr(cast<llvm::ConstantExpr>(C), *this));
148 break;
149 default:
150 It->second = std::unique_ptr<Constant>(new Constant(C, *this));
151 break;
152 }
153 auto *NewC = It->second.get();
154 for (llvm::Value *COp : C->operands())
155 getOrCreateValueInternal(COp, C);
156 return NewC;
157 }
158 if (auto *Arg = dyn_cast<llvm::Argument>(LLVMV)) {
159 It->second = std::unique_ptr<Argument>(new Argument(Arg, *this));

418BasicBlock *Context::createBasicBlock(llvm::BasicBlock *LLVMBB) {
419 assert(getValue(LLVMBB) == nullptr && "Already exists!");
420 auto NewBBPtr = std::unique_ptr<BasicBlock>(new BasicBlock(LLVMBB, *this));
421 auto *BB = cast<BasicBlock>(registerValue(std::move(NewBBPtr)));
422 // Create SandboxIR for BB's body.
423 BB->buildBasicBlockFromLLVMIR(LLVMBB);
424 return BB;
425}

693Function *Context::createFunction(llvm::Function *F) {
694 assert(getValue(F) == nullptr && "Already exists!");
695 // Create the module if needed before we create the new sandboxir::Function.
696 // Note: this won't fully populate the module. The only globals that will be
697 // available will be the ones being used within the function.
698 getOrCreateModule(F->getParent());
699
700 auto NewFPtr = std::unique_ptr<Function>(new Function(F, *this));
701 auto *SBF = cast<Function>(registerValue(std::move(NewFPtr)));
702 // Create arguments.
703 for (auto &Arg : F->args())
704 getOrCreateArgument(&Arg);
705 // Create BBs.
706 for (auto &BB : *F)
707 createBasicBlock(&BB);
708 return SBF;
709}
710
711Module *Context::createModule(llvm::Module *LLVMM) {
712 auto *M = getOrCreateModule(LLVMM);
713 // Create the functions.
714 for (auto &LLVMF : *LLVMM)
715 createFunction(&LLVMF);
716 // Create globals.
717 for (auto &Global : LLVMM->globals())
718 getOrCreateValue(&Global);
719 // Create aliases.
720 for (auto &Alias : LLVMM->aliases())
721 getOrCreateValue(&Alias);
722 // Create ifuncs.
723 for (auto &IFunc : LLVMM->ifuncs())
724 getOrCreateValue(&IFunc);
725
726 return M;
727}
