32void LoongArchAsmPrinter::emitInstruction(const MachineInstr *MI) {
33 LoongArch_MC::verifyInstructionPredicates(
34 MI->getOpcode(), getSubtargetInfo().getFeatureBits());
35
36 // Do any auto-generated pseudo lowerings.
37 if (MCInst OutInst; lowerPseudoInstExpansion(MI, OutInst)) {
38 EmitToStreamer(*OutStreamer, OutInst);
39 return;
40 }
41
42 switch (MI->getOpcode()) {
43 case TargetOpcode::STATEPOINT:
44 LowerSTATEPOINT(*MI);
45 return;
46 case TargetOpcode::PATCHABLE_FUNCTION_ENTER:
47 LowerPATCHABLE_FUNCTION_ENTER(*MI);
48 return;
49 case TargetOpcode::PATCHABLE_FUNCTION_EXIT:
50 LowerPATCHABLE_FUNCTION_EXIT(*MI);
51 return;
52 case TargetOpcode::PATCHABLE_TAIL_CALL:
53 LowerPATCHABLE_TAIL_CALL(*MI);
54 return;
55 }
56
57 MCInst TmpInst;
58 if (!lowerLoongArchMachineInstrToMCInst(MI, TmpInst, *this))
59 EmitToStreamer(*OutStreamer, TmpInst);
60}

62bool LoongArchAsmPrinter::PrintAsmOperand(const MachineInstr *MI, unsigned OpNo,
63 const char *ExtraCode,
64 raw_ostream &OS) {
65 // First try the generic code, which knows about modifiers like 'c' and 'n'.
66 if (!AsmPrinter::PrintAsmOperand(MI, OpNo, ExtraCode, OS))
67 return false;
68
69 const MachineOperand &MO = MI->getOperand(OpNo);
70 if (ExtraCode && ExtraCode[0]) {
71 if (ExtraCode[1] != 0)
72 return true; // Unknown modifier.
73
74 switch (ExtraCode[0]) {
75 default:
76 return true; // Unknown modifier.
77 case 'z': // Print $zero register if zero, regular printing otherwise.
78 if (MO.isImm() && MO.getImm() == 0) {
79 OS << '$' << LoongArchInstPrinter::getRegisterName(LoongArch::R0);
80 return false;
81 }
82 break;
83 case 'w': // Print LSX registers.
84 if (MO.getReg().id() >= LoongArch::VR0 &&
85 MO.getReg().id() <= LoongArch::VR31)
86 break;
87 // The modifier is 'w' but the operand is not an LSX register; Report an
88 // unknown operand error.
89 return true;
90 case 'u': // Print LASX registers.
91 if (MO.getReg().id() >= LoongArch::XR0 &&
92 MO.getReg().id() <= LoongArch::XR31)
93 break;
94 // The modifier is 'u' but the operand is not an LASX register; Report an
95 // unknown operand error.
96 return true;
97 // TODO: handle other extra codes if any.
98 }
99 }
100
101 switch (MO.getType()) {
102 case MachineOperand::MO_Immediate:
103 OS << MO.getImm();
104 return false;
105 case MachineOperand::MO_Register:
106 OS << '$' << LoongArchInstPrinter::getRegisterName(MO.getReg());
107 return false;
108 case MachineOperand::MO_GlobalAddress:
109 PrintSymbolOperand(MO, OS);
110 return false;
111 default:
112 llvm_unreachable("not implemented");
113 }
114
115 return true;
116}

118bool LoongArchAsmPrinter::PrintAsmMemoryOperand(const MachineInstr *MI,
119 unsigned OpNo,
120 const char *ExtraCode,
121 raw_ostream &OS) {
122 // TODO: handle extra code.
123 if (ExtraCode)
124 return true;
125
126 // We only support memory operands like "Base + Offset", where base must be a
127 // register, and offset can be a register or an immediate value.
128 const MachineOperand &BaseMO = MI->getOperand(OpNo);
129 // Base address must be a register.
130 if (!BaseMO.isReg())
131 return true;
132 // Print the base address register.
133 OS << "$" << LoongArchInstPrinter::getRegisterName(BaseMO.getReg());
134 // Print the offset operand.
135 const MachineOperand &OffsetMO = MI->getOperand(OpNo + 1);
136 MCOperand MCO;
137 if (!lowerOperand(OffsetMO, MCO))
138 return true;
139 if (OffsetMO.isReg())
140 OS << ", $" << LoongArchInstPrinter::getRegisterName(OffsetMO.getReg());
141 else if (OffsetMO.isImm())
142 OS << ", " << OffsetMO.getImm();
143 else if (OffsetMO.isGlobal() || OffsetMO.isBlockAddress() ||
144 OffsetMO.isMCSymbol())
145 OS << ", " << *MCO.getExpr();
146 else
147 return true;
148
149 return false;
150}

152void LoongArchAsmPrinter::LowerSTATEPOINT(const MachineInstr &MI) {
153 StatepointOpers SOpers(&MI);
154 if (unsigned PatchBytes = SOpers.getNumPatchBytes()) {
155 assert(PatchBytes % 4 == 0 && "Invalid number of NOP bytes requested!");
156 emitNops(PatchBytes / 4);
157 } else {
158 // Lower call target and choose correct opcode.
159 const MachineOperand &CallTarget = SOpers.getCallTarget();
160 MCOperand CallTargetMCOp;
161 switch (CallTarget.getType()) {
162 case MachineOperand::MO_GlobalAddress:
163 case MachineOperand::MO_ExternalSymbol:
164 lowerOperand(CallTarget, CallTargetMCOp);
165 EmitToStreamer(*OutStreamer,
166 MCInstBuilder(LoongArch::BL).addOperand(CallTargetMCOp));
167 break;
168 case MachineOperand::MO_Immediate:
169 CallTargetMCOp = MCOperand::createImm(CallTarget.getImm());
170 EmitToStreamer(*OutStreamer,
171 MCInstBuilder(LoongArch::BL).addOperand(CallTargetMCOp));
172 break;
173 case MachineOperand::MO_Register:
174 CallTargetMCOp = MCOperand::createReg(CallTarget.getReg());
175 EmitToStreamer(*OutStreamer, MCInstBuilder(LoongArch::JIRL)
176 .addReg(LoongArch::R1)
177 .addOperand(CallTargetMCOp)
178 .addImm(0));
179 break;
180 default:
181 llvm_unreachable("Unsupported operand type in statepoint call target");
182 break;
183 }
184 }
185
186 auto &Ctx = OutStreamer->getContext();
187 MCSymbol *MILabel = Ctx.createTempSymbol();
188 OutStreamer->emitLabel(MILabel);
189 SM.recordStatepoint(*MILabel, MI);
190}

192void LoongArchAsmPrinter::LowerPATCHABLE_FUNCTION_ENTER(
193 const MachineInstr &MI) {
194 const Function &F = MF->getFunction();
195 if (F.hasFnAttribute("patchable-function-entry")) {
196 unsigned Num;
197 if (F.getFnAttribute("patchable-function-entry")
198 .getValueAsString()
199 .getAsInteger(10, Num))
200 return;
201 emitNops(Num);
202 return;
203 }
204
205 emitSled(MI, SledKind::FUNCTION_ENTER);
206}

216void LoongArchAsmPrinter::emitSled(const MachineInstr &MI, SledKind Kind) {
217 // For loongarch64 we want to emit the following pattern:
218 //
219 // .Lxray_sled_beginN:
220 // B .Lxray_sled_endN
221 // 11 NOPs (44 bytes)
222 // .Lxray_sled_endN:
223 //
224 // We need the extra bytes because at runtime they may be used for the
225 // actual pattern defined at compiler-rt/lib/xray/xray_loongarch64.cpp.
226 // The count here should be adjusted accordingly if the implementation
227 // changes.
228 const int8_t NoopsInSledCount = 11;
229 OutStreamer->emitCodeAlignment(Align(4), &getSubtargetInfo());
230 MCSymbol *BeginOfSled = OutContext.createTempSymbol("xray_sled_begin");
231 MCSymbol *EndOfSled = OutContext.createTempSymbol("xray_sled_end");
232 OutStreamer->emitLabel(BeginOfSled);
233 EmitToStreamer(*OutStreamer,
234 MCInstBuilder(LoongArch::B)
235 .addExpr(MCSymbolRefExpr::create(EndOfSled, OutContext)));
236 emitNops(NoopsInSledCount);
237 OutStreamer->emitLabel(EndOfSled);
238 recordSled(BeginOfSled, MI, Kind, 2);
239}
