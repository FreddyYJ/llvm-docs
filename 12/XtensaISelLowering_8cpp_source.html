161TargetLowering::ConstraintWeight
162XtensaTargetLowering::getSingleConstraintMatchWeight(
163 AsmOperandInfo &Info, const char *Constraint) const {
164 ConstraintWeight Weight = CW_Invalid;
165 Value *CallOperandVal = Info.CallOperandVal;
166 // If we don't have a value, we can't do a match,
167 // but allow it at the lowest weight.
168 if (!CallOperandVal)
169 return CW_Default;
170
171 Type *Ty = CallOperandVal->getType();
172
173 // Look at the constraint type.
174 switch (*Constraint) {
175 default:
176 Weight = TargetLowering::getSingleConstraintMatchWeight(Info, Constraint);
177 break;
178 case 'r':
179 if (Ty->isIntegerTy())
180 Weight = CW_Register;
181 break;
182 }
183 return Weight;
184}

219static bool CC_Xtensa_Custom(unsigned ValNo, MVT ValVT, MVT LocVT,
220 CCValAssign::LocInfo LocInfo,
221 ISD::ArgFlagsTy ArgFlags, CCState &State) {
222 static const MCPhysReg IntRegs[] = {Xtensa::A2, Xtensa::A3, Xtensa::A4,
223 Xtensa::A5, Xtensa::A6, Xtensa::A7};
224
225 if (ArgFlags.isByVal()) {
226 Align ByValAlign = ArgFlags.getNonZeroByValAlign();
227 unsigned ByValSize = ArgFlags.getByValSize();
228 if (ByValSize < 4) {
229 ByValSize = 4;
230 }
231 if (ByValAlign < Align(4)) {
232 ByValAlign = Align(4);
233 }
234 unsigned Offset = State.AllocateStack(ByValSize, ByValAlign);
235 State.addLoc(CCValAssign::getMem(ValNo, ValVT, Offset, LocVT, LocInfo));
236 // Mark all unused registers as allocated to avoid misuse
237 // of such registers.
238 while (State.AllocateReg(IntRegs))
239 ;
240 return false;
241 }
242
243 // Promote i8 and i16
244 if (LocVT == MVT::i8 || LocVT == MVT::i16) {
245 LocVT = MVT::i32;
246 if (ArgFlags.isSExt())
247 LocInfo = CCValAssign::SExt;
248 else if (ArgFlags.isZExt())
249 LocInfo = CCValAssign::ZExt;
250 else
251 LocInfo = CCValAssign::AExt;
252 }
253
254 unsigned Register;
255
256 Align OrigAlign = ArgFlags.getNonZeroOrigAlign();
257 bool needs64BitAlign = (ValVT == MVT::i32 && OrigAlign == Align(8));
258 bool needs128BitAlign = (ValVT == MVT::i32 && OrigAlign == Align(16));
259
260 if (ValVT == MVT::i32) {
261 Register = State.AllocateReg(IntRegs);
262 // If this is the first part of an i64 arg,
263 // the allocated register must be either A2, A4 or A6.
264 if (needs64BitAlign && (Register == Xtensa::A3 || Register == Xtensa::A5 ||
265 Register == Xtensa::A7))
266 Register = State.AllocateReg(IntRegs);
267 // arguments with 16byte alignment must be passed in the first register or
268 // passed via stack
269 if (needs128BitAlign && (Register != Xtensa::A2))
270 while ((Register = State.AllocateReg(IntRegs)))
271 ;
272 LocVT = MVT::i32;
273 } else if (ValVT == MVT::f64) {
274 // Allocate int register and shadow next int register.
275 Register = State.AllocateReg(IntRegs);
276 if (Register == Xtensa::A3 || Register == Xtensa::A5 ||
277 Register == Xtensa::A7)
278 Register = State.AllocateReg(IntRegs);
279 State.AllocateReg(IntRegs);
280 LocVT = MVT::i32;
281 } else {
282 report_fatal_error("Cannot handle this ValVT.");
283 }
284
285 if (!Register) {
286 unsigned Offset = State.AllocateStack(ValVT.getStoreSize(), OrigAlign);
287 State.addLoc(CCValAssign::getMem(ValNo, ValVT, Offset, LocVT, LocInfo));
288 } else {
289 State.addLoc(CCValAssign::getReg(ValNo, ValVT, Register, LocVT, LocInfo));
290 }
291
292 return false;
293}

300SDValue XtensaTargetLowering::LowerFormalArguments(
301 SDValue Chain, CallingConv::ID CallConv, bool IsVarArg,
302 const SmallVectorImpl<ISD::InputArg> &Ins, const SDLoc &DL,
303 SelectionDAG &DAG, SmallVectorImpl<SDValue> &InVals) const {
304 MachineFunction &MF = DAG.getMachineFunction();
305 MachineFrameInfo &MFI = MF.getFrameInfo();
306
307 // Used with vargs to acumulate store chains.
308 std::vector<SDValue> OutChains;
309
310 if (IsVarArg)
311 report_fatal_error("Var arg not supported by FormalArguments Lowering");
312
313 // Assign locations to all of the incoming arguments.
314 SmallVector<CCValAssign, 16> ArgLocs;
315 CCState CCInfo(CallConv, IsVarArg, DAG.getMachineFunction(), ArgLocs,
316 *DAG.getContext());
317
318 CCInfo.AnalyzeFormalArguments(Ins, CCAssignFnForCall(CallConv, IsVarArg));
319
320 for (unsigned i = 0, e = ArgLocs.size(); i != e; ++i) {
321 CCValAssign &VA = ArgLocs[i];
322 // Arguments stored on registers
323 if (VA.isRegLoc()) {
324 EVT RegVT = VA.getLocVT();
325 const TargetRegisterClass *RC;
326
327 if (RegVT == MVT::i32)
328 RC = &Xtensa::ARRegClass;
329 else
330 report_fatal_error("RegVT not supported by FormalArguments Lowering");
331
332 // Transform the arguments stored on
333 // physical registers into virtual ones
334 unsigned Register = MF.addLiveIn(VA.getLocReg(), RC);
335 SDValue ArgValue = DAG.getCopyFromReg(Chain, DL, Register, RegVT);
336
337 // If this is an 8 or 16-bit value, it has been passed promoted
338 // to 32 bits. Insert an assert[sz]ext to capture this, then
339 // truncate to the right size.
340 if (VA.getLocInfo() != CCValAssign::Full) {
341 unsigned Opcode = 0;
342 if (VA.getLocInfo() == CCValAssign::SExt)
343 Opcode = ISD::AssertSext;
344 else if (VA.getLocInfo() == CCValAssign::ZExt)
345 Opcode = ISD::AssertZext;
346 if (Opcode)
347 ArgValue = DAG.getNode(Opcode, DL, RegVT, ArgValue,
348 DAG.getValueType(VA.getValVT()));
349 ArgValue = DAG.getNode((VA.getValVT() == MVT::f32) ? ISD::BITCAST
350 : ISD::TRUNCATE,
351 DL, VA.getValVT(), ArgValue);
352 }
353
354 InVals.push_back(ArgValue);
355
356 } else {
357 assert(VA.isMemLoc());
358
359 EVT ValVT = VA.getValVT();
360
361 // The stack pointer offset is relative to the caller stack frame.
362 int FI = MFI.CreateFixedObject(ValVT.getStoreSize(), VA.getLocMemOffset(),
363 true);
364
365 if (Ins[VA.getValNo()].Flags.isByVal()) {
366 // Assume that in this case load operation is created
367 SDValue FIN = DAG.getFrameIndex(FI, MVT::i32);
368 InVals.push_back(FIN);
369 } else {
370 // Create load nodes to retrieve arguments from the stack
371 SDValue FIN =
372 DAG.getFrameIndex(FI, getFrameIndexTy(DAG.getDataLayout()));
373 InVals.push_back(DAG.getLoad(
374 ValVT, DL, Chain, FIN,
375 MachinePointerInfo::getFixedStack(DAG.getMachineFunction(), FI)));
376 }
377 }
378 }
379
380 // All stores are grouped in one node to allow the matching between
381 // the size of Ins and InVals. This only happens when on varg functions
382 if (!OutChains.empty()) {
383 OutChains.push_back(Chain);
384 Chain = DAG.getNode(ISD::TokenFactor, DL, MVT::Other, OutChains);
385 }
386
387 return Chain;
388}

390SDValue
391XtensaTargetLowering::LowerCall(CallLoweringInfo &CLI,
392 SmallVectorImpl<SDValue> &InVals) const {
393 SelectionDAG &DAG = CLI.DAG;
394 SDLoc &DL = CLI.DL;
395 SmallVector<ISD::OutputArg, 32> &Outs = CLI.Outs;
396 SmallVector<SDValue, 32> &OutVals = CLI.OutVals;
397 SmallVector<ISD::InputArg, 32> &Ins = CLI.Ins;
398 SDValue Chain = CLI.Chain;
399 SDValue Callee = CLI.Callee;
400 bool &IsTailCall = CLI.IsTailCall;
401 CallingConv::ID CallConv = CLI.CallConv;
402 bool IsVarArg = CLI.IsVarArg;
403
404 MachineFunction &MF = DAG.getMachineFunction();
405 EVT PtrVT = getPointerTy(DAG.getDataLayout());
406 const TargetFrameLowering *TFL = Subtarget.getFrameLowering();
407
408 // TODO: Support tail call optimization.
409 IsTailCall = false;
410
411 // Analyze the operands of the call, assigning locations to each operand.
412 SmallVector<CCValAssign, 16> ArgLocs;
413 CCState CCInfo(CallConv, IsVarArg, MF, ArgLocs, *DAG.getContext());
414
415 CCAssignFn *CC = CCAssignFnForCall(CallConv, IsVarArg);
416
417 CCInfo.AnalyzeCallOperands(Outs, CC);
418
419 // Get a count of how many bytes are to be pushed on the stack.
420 unsigned NumBytes = CCInfo.getStackSize();
421
422 Align StackAlignment = TFL->getStackAlign();
423 unsigned NextStackOffset = alignTo(NumBytes, StackAlignment);
424
425 Chain = DAG.getCALLSEQ_START(Chain, NextStackOffset, 0, DL);
426
427 // Copy argument values to their designated locations.
428 std::deque<std::pair<unsigned, SDValue>> RegsToPass;
429 SmallVector<SDValue, 8> MemOpChains;
430 SDValue StackPtr;
431 for (unsigned I = 0, E = ArgLocs.size(); I != E; ++I) {
432 CCValAssign &VA = ArgLocs[I];
433 SDValue ArgValue = OutVals[I];
434 ISD::ArgFlagsTy Flags = Outs[I].Flags;
435
436 if (VA.isRegLoc())
437 // Queue up the argument copies and emit them at the end.
438 RegsToPass.push_back(std::make_pair(VA.getLocReg(), ArgValue));
439 else if (Flags.isByVal()) {
440 assert(VA.isMemLoc());
441 assert(Flags.getByValSize() &&
442 "ByVal args of size 0 should have been ignored by front-end.");
443 assert(!IsTailCall &&
444 "Do not tail-call optimize if there is a byval argument.");
445
446 if (!StackPtr.getNode())
447 StackPtr = DAG.getCopyFromReg(Chain, DL, Xtensa::SP, PtrVT);
448 unsigned Offset = VA.getLocMemOffset();
449 SDValue Address = DAG.getNode(ISD::ADD, DL, PtrVT, StackPtr,
450 DAG.getIntPtrConstant(Offset, DL));
451 SDValue SizeNode = DAG.getConstant(Flags.getByValSize(), DL, MVT::i32);
452 SDValue Memcpy = DAG.getMemcpy(
453 Chain, DL, Address, ArgValue, SizeNode, Flags.getNonZeroByValAlign(),
454 /*isVolatile=*/false, /*AlwaysInline=*/false,
455 /*CI=*/nullptr, std::nullopt, MachinePointerInfo(), MachinePointerInfo());
456 MemOpChains.push_back(Memcpy);
457 } else {
458 assert(VA.isMemLoc() && "Argument not register or memory");
459
460 // Work out the address of the stack slot. Unpromoted ints and
461 // floats are passed as right-justified 8-byte values.
462 if (!StackPtr.getNode())
463 StackPtr = DAG.getCopyFromReg(Chain, DL, Xtensa::SP, PtrVT);
464 unsigned Offset = VA.getLocMemOffset();
465 SDValue Address = DAG.getNode(ISD::ADD, DL, PtrVT, StackPtr,
466 DAG.getIntPtrConstant(Offset, DL));
467
468 // Emit the store.
469 MemOpChains.push_back(
470 DAG.getStore(Chain, DL, ArgValue, Address, MachinePointerInfo()));
471 }
472 }
473
474 // Join the stores, which are independent of one another.
475 if (!MemOpChains.empty())
476 Chain = DAG.getNode(ISD::TokenFactor, DL, MVT::Other, MemOpChains);
477
478 // Build a sequence of copy-to-reg nodes, chained and glued together.
479 SDValue Glue;
480 for (unsigned I = 0, E = RegsToPass.size(); I != E; ++I) {
481 unsigned Reg = RegsToPass[I].first;
482 Chain = DAG.getCopyToReg(Chain, DL, Reg, RegsToPass[I].second, Glue);
483 Glue = Chain.getValue(1);
484 }
485 std::string name;
486 unsigned char TF = 0;
487
488 // Accept direct calls by converting symbolic call addresses to the
489 // associated Target* opcodes.
490 if (ExternalSymbolSDNode *E = dyn_cast<ExternalSymbolSDNode>(Callee)) {
491 name = E->getSymbol();
492 TF = E->getTargetFlags();
493 if (isPositionIndependent()) {
494 report_fatal_error("PIC relocations is not supported");
495 } else
496 Callee = DAG.getTargetExternalSymbol(E->getSymbol(), PtrVT, TF);
497 } else if (GlobalAddressSDNode *G = dyn_cast<GlobalAddressSDNode>(Callee)) {
498 const GlobalValue *GV = G->getGlobal();
499 name = GV->getName().str();
500 }
501
502 if ((!name.empty()) && isLongCall(name.c_str())) {
503 // Create a constant pool entry for the callee address
504 XtensaCP::XtensaCPModifier Modifier = XtensaCP::no_modifier;
505
506 XtensaConstantPoolValue *CPV = XtensaConstantPoolSymbol::Create(
507 *DAG.getContext(), name.c_str(), 0 /* XtensaCLabelIndex */, false,
508 Modifier);
509
510 // Get the address of the callee into a register
511 SDValue CPAddr = DAG.getTargetConstantPool(CPV, PtrVT, Align(4), 0, TF);
512 SDValue CPWrap = getAddrPCRel(CPAddr, DAG);
513 Callee = CPWrap;
514 }
515
516 // The first call operand is the chain and the second is the target address.
517 SmallVector<SDValue, 8> Ops;
518 Ops.push_back(Chain);
519 Ops.push_back(Callee);
520
521 // Add a register mask operand representing the call-preserved registers.
522 const TargetRegisterInfo *TRI = Subtarget.getRegisterInfo();
523 const uint32_t *Mask = TRI->getCallPreservedMask(MF, CallConv);
524 assert(Mask && "Missing call preserved mask for calling convention");
525 Ops.push_back(DAG.getRegisterMask(Mask));
526
527 // Add argument registers to the end of the list so that they are
528 // known live into the call.
529 for (unsigned I = 0, E = RegsToPass.size(); I != E; ++I) {
530 unsigned Reg = RegsToPass[I].first;
531 Ops.push_back(DAG.getRegister(Reg, RegsToPass[I].second.getValueType()));
532 }
533
534 // Glue the call to the argument copies, if any.
535 if (Glue.getNode())
536 Ops.push_back(Glue);
537
538 SDVTList NodeTys = DAG.getVTList(MVT::Other, MVT::Glue);
539 Chain = DAG.getNode(XtensaISD::CALL, DL, NodeTys, Ops);
540 Glue = Chain.getValue(1);
541
542 // Mark the end of the call, which is glued to the call itself.
543 Chain = DAG.getCALLSEQ_END(Chain, DAG.getConstant(NumBytes, DL, PtrVT, true),
544 DAG.getConstant(0, DL, PtrVT, true), Glue, DL);
545 Glue = Chain.getValue(1);
546
547 // Assign locations to each value returned by this call.
548 SmallVector<CCValAssign, 16> RetLocs;
549 CCState RetCCInfo(CallConv, IsVarArg, MF, RetLocs, *DAG.getContext());
550 RetCCInfo.AnalyzeCallResult(Ins, RetCC_Xtensa);
551
552 // Copy all of the result registers out of their specified physreg.
553 for (unsigned I = 0, E = RetLocs.size(); I != E; ++I) {
554 CCValAssign &VA = RetLocs[I];
555
556 // Copy the value out, gluing the copy to the end of the call sequence.
557 unsigned Reg = VA.getLocReg();
558 SDValue RetValue = DAG.getCopyFromReg(Chain, DL, Reg, VA.getLocVT(), Glue);
559 Chain = RetValue.getValue(1);
560 Glue = RetValue.getValue(2);
561
562 InVals.push_back(RetValue);
563 }
564 return Chain;
565}

575SDValue
576XtensaTargetLowering::LowerReturn(SDValue Chain, CallingConv::ID CallConv,
577 bool IsVarArg,
578 const SmallVectorImpl<ISD::OutputArg> &Outs,
579 const SmallVectorImpl<SDValue> &OutVals,
580 const SDLoc &DL, SelectionDAG &DAG) const {
581 if (IsVarArg)
582 report_fatal_error("VarArg not supported");
583
584 MachineFunction &MF = DAG.getMachineFunction();
585
586 // Assign locations to each returned value.
587 SmallVector<CCValAssign, 16> RetLocs;
588 CCState RetCCInfo(CallConv, IsVarArg, MF, RetLocs, *DAG.getContext());
589 RetCCInfo.AnalyzeReturn(Outs, RetCC_Xtensa);
590
591 SDValue Glue;
592 // Quick exit for void returns
593 if (RetLocs.empty())
594 return DAG.getNode(XtensaISD::RET, DL, MVT::Other, Chain);
595
596 // Copy the result values into the output registers.
597 SmallVector<SDValue, 4> RetOps;
598 RetOps.push_back(Chain);
599 for (unsigned I = 0, E = RetLocs.size(); I != E; ++I) {
600 CCValAssign &VA = RetLocs[I];
601 SDValue RetValue = OutVals[I];
602
603 // Make the return register live on exit.
604 assert(VA.isRegLoc() && "Can only return in registers!");
605
606 // Chain and glue the copies together.
607 unsigned Register = VA.getLocReg();
608 Chain = DAG.getCopyToReg(Chain, DL, Register, RetValue, Glue);
609 Glue = Chain.getValue(1);
610 RetOps.push_back(DAG.getRegister(Register, VA.getLocVT()));
611 }
612
613 // Update chain and glue.
614 RetOps[0] = Chain;
615 if (Glue.getNode())
616 RetOps.push_back(Glue);
617
618 return DAG.getNode(XtensaISD::RET, DL, MVT::Other, RetOps);
619}

861SDValue XtensaTargetLowering::LowerShiftLeftParts(SDValue Op,
862 SelectionDAG &DAG) const {
863 SDLoc DL(Op);
864 MVT VT = MVT::i32;
865 SDValue Lo = Op.getOperand(0), Hi = Op.getOperand(1);
866 SDValue Shamt = Op.getOperand(2);
867
868 // if Shamt - register size < 0: // Shamt < register size
869 // Lo = Lo << Shamt
870 // Hi = (Hi << Shamt) | (Lo >>u (register size - Shamt))
871 // else:
872 // Lo = 0
873 // Hi = Lo << (Shamt - register size)
874
875 SDValue MinusRegisterSize = DAG.getConstant(-32, DL, VT);
876 SDValue ShamtMinusRegisterSize =
877 DAG.getNode(ISD::ADD, DL, VT, Shamt, MinusRegisterSize);
878
879 SDValue LoTrue = DAG.getNode(ISD::SHL, DL, VT, Lo, Shamt);
880 SDValue HiTrue = DAG.getNode(XtensaISD::SRCL, DL, VT, Hi, Lo, Shamt);
881 SDValue Zero = DAG.getConstant(0, DL, VT);
882 SDValue HiFalse = DAG.getNode(ISD::SHL, DL, VT, Lo, ShamtMinusRegisterSize);
883
884 SDValue Cond = DAG.getSetCC(DL, VT, ShamtMinusRegisterSize, Zero, ISD::SETLT);
885 Lo = DAG.getNode(ISD::SELECT, DL, VT, Cond, LoTrue, Zero);
886 Hi = DAG.getNode(ISD::SELECT, DL, VT, Cond, HiTrue, HiFalse);
887
888 return DAG.getMergeValues({Lo, Hi}, DL);
889}

891SDValue XtensaTargetLowering::LowerShiftRightParts(SDValue Op,
892 SelectionDAG &DAG,
893 bool IsSRA) const {
894 SDLoc DL(Op);
895 SDValue Lo = Op.getOperand(0), Hi = Op.getOperand(1);
896 SDValue Shamt = Op.getOperand(2);
897 MVT VT = MVT::i32;
898
899 // SRA expansion:
900 // if Shamt - register size < 0: // Shamt < register size
901 // Lo = (Lo >>u Shamt) | (Hi << u (register size - Shamt))
902 // Hi = Hi >>s Shamt
903 // else:
904 // Lo = Hi >>s (Shamt - register size);
905 // Hi = Hi >>s (register size - 1)
906 //
907 // SRL expansion:
908 // if Shamt - register size < 0: // Shamt < register size
909 // Lo = (Lo >>u Shamt) | (Hi << u (register size - Shamt))
910 // Hi = Hi >>u Shamt
911 // else:
912 // Lo = Hi >>u (Shamt - register size);
913 // Hi = 0;
914
915 unsigned ShiftRightOp = IsSRA ? ISD::SRA : ISD::SRL;
916 SDValue MinusRegisterSize = DAG.getConstant(-32, DL, VT);
917 SDValue RegisterSizeMinus1 = DAG.getConstant(32 - 1, DL, VT);
918 SDValue ShamtMinusRegisterSize =
919 DAG.getNode(ISD::ADD, DL, VT, Shamt, MinusRegisterSize);
920
921 SDValue LoTrue = DAG.getNode(XtensaISD::SRCR, DL, VT, Hi, Lo, Shamt);
922 SDValue HiTrue = DAG.getNode(ShiftRightOp, DL, VT, Hi, Shamt);
923 SDValue Zero = DAG.getConstant(0, DL, VT);
924 SDValue LoFalse =
925 DAG.getNode(ShiftRightOp, DL, VT, Hi, ShamtMinusRegisterSize);
926 SDValue HiFalse;
927
928 if (IsSRA) {
929 HiFalse = DAG.getNode(ShiftRightOp, DL, VT, Hi, RegisterSizeMinus1);
930 } else {
931 HiFalse = Zero;
932 }
933
934 SDValue Cond = DAG.getSetCC(DL, VT, ShamtMinusRegisterSize, Zero, ISD::SETLT);
935 Lo = DAG.getNode(ISD::SELECT, DL, VT, Cond, LoTrue, LoFalse);
936 Hi = DAG.getNode(ISD::SELECT, DL, VT, Cond, HiTrue, HiFalse);
937
938 return DAG.getMergeValues({Lo, Hi}, DL);
939}

1040MachineBasicBlock *
1041XtensaTargetLowering::emitSelectCC(MachineInstr &MI,
1042 MachineBasicBlock *MBB) const {
1043 const TargetInstrInfo &TII = *Subtarget.getInstrInfo();
1044 DebugLoc DL = MI.getDebugLoc();
1045
1046 MachineOperand &LHS = MI.getOperand(1);
1047 MachineOperand &RHS = MI.getOperand(2);
1048 MachineOperand &TrueValue = MI.getOperand(3);
1049 MachineOperand &FalseValue = MI.getOperand(4);
1050 unsigned BrKind = MI.getOperand(5).getImm();
1051
1052 // To "insert" a SELECT_CC instruction, we actually have to insert
1053 // CopyMBB and SinkMBB blocks and add branch to MBB. We build phi
1054 // operation in SinkMBB like phi (TrueVakue,FalseValue), where TrueValue
1055 // is passed from MMB and FalseValue is passed from CopyMBB.
1056 // MBB
1057 // | \
1058 // | CopyMBB
1059 // | /
1060 // SinkMBB
1061 // The incoming instruction knows the
1062 // destination vreg to set, the condition code register to branch on, the
1063 // true/false values to select between, and a branch opcode to use.
1064 const BasicBlock *LLVM_BB = MBB->getBasicBlock();
1065 MachineFunction::iterator It = ++MBB->getIterator();
1066
1067 MachineFunction *F = MBB->getParent();
1068 MachineBasicBlock *CopyMBB = F->CreateMachineBasicBlock(LLVM_BB);
1069 MachineBasicBlock *SinkMBB = F->CreateMachineBasicBlock(LLVM_BB);
1070
1071 F->insert(It, CopyMBB);
1072 F->insert(It, SinkMBB);
1073
1074 // Transfer the remainder of MBB and its successor edges to SinkMBB.
1075 SinkMBB->splice(SinkMBB->begin(), MBB,
1076 std::next(MachineBasicBlock::iterator(MI)), MBB->end());
1077 SinkMBB->transferSuccessorsAndUpdatePHIs(MBB);
1078
1079 MBB->addSuccessor(CopyMBB);
1080 MBB->addSuccessor(SinkMBB);
1081
1082 BuildMI(MBB, DL, TII.get(BrKind))
1083 .addReg(LHS.getReg())
1084 .addReg(RHS.getReg())
1085 .addMBB(SinkMBB);
1086
1087 CopyMBB->addSuccessor(SinkMBB);
1088
1089 // SinkMBB:
1090 // %Result = phi [ %FalseValue, CopyMBB ], [ %TrueValue, MBB ]
1091 // ...
1092
1093 BuildMI(*SinkMBB, SinkMBB->begin(), DL, TII.get(Xtensa::PHI),
1094 MI.getOperand(0).getReg())
1095 .addReg(FalseValue.getReg())
1096 .addMBB(CopyMBB)
1097 .addReg(TrueValue.getReg())
1098 .addMBB(MBB);
1099
1100 MI.eraseFromParent(); // The pseudo instruction is gone now.
1101 return SinkMBB;
1102}
