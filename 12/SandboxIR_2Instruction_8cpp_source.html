26llvm::Instruction *Instruction::getTopmostLLVMInstruction() const {
27 Instruction *Prev = getPrevNode();
28 if (Prev == nullptr) {
29 // If at top of the BB, return the first BB instruction.
30 return &*cast<llvm::BasicBlock>(getParent()->Val)->begin();
31 }
32 // Else get the Previous sandbox IR instruction's bottom IR instruction and
33 // return its successor.
34 llvm::Instruction *PrevBotI = cast<llvm::Instruction>(Prev->Val);
35 return PrevBotI->getNextNode();
36}

43Instruction *Instruction::getNextNode() const {
44 assert(getParent() != nullptr && "Detached!");
45 assert(getIterator() != getParent()->end() && "Already at end!");
46 // `Val` is the bottom-most LLVM IR instruction. Get the next in the chain,
47 // and get the corresponding sandboxir Instruction that maps to it. This works
48 // even for SandboxIR Instructions that map to more than one LLVM Instruction.
49 auto *LLVMI = cast<llvm::Instruction>(Val);
50 assert(LLVMI->getParent() != nullptr && "LLVM IR instr is detached!");
51 auto *NextLLVMI = LLVMI->getNextNode();
52 auto *NextI = cast_or_null<Instruction>(Ctx.getValue(NextLLVMI));
53 if (NextI == nullptr)
54 return nullptr;
55 return NextI;
56}

58Instruction *Instruction::getPrevNode() const {
59 assert(getParent() != nullptr && "Detached!");
60 auto It = getIterator();
61 if (It != getParent()->begin())
62 return std::prev(getIterator()).get();
63 return nullptr;
64}

66void Instruction::removeFromParent() {
67 Ctx.getTracker().emplaceIfTracking<RemoveFromParent>(this);
68
69 // Detach all the LLVM IR instructions from their parent BB.
70 for (llvm::Instruction *I : getLLVMInstrs())
71 I->removeFromParent();

74void Instruction::eraseFromParent() {
75 assert(users().empty() && "Still connected to users, can't erase!");
76 std::unique_ptr<Value> Detached = Ctx.detach(this);
77 auto LLVMInstrs = getLLVMInstrs();
78
79 auto &Tracker = Ctx.getTracker();
80 if (Tracker.isTracking()) {
81 Tracker.track(std::make_unique<EraseFromParent>(std::move(Detached)));
82 // We don't actually delete the IR instruction, because then it would be
83 // impossible to bring it back from the dead at the same memory location.
84 // Instead we remove it from its BB and track its current location.
85 for (llvm::Instruction *I : LLVMInstrs)
86 I->removeFromParent();
87 // TODO: Multi-instructions need special treatment because some of the
88 // references are internal to the instruction.
89 for (llvm::Instruction *I : LLVMInstrs)
90 I->dropAllReferences();
91 } else {
92 // Erase in reverse to avoid erasing nstructions with attached uses.
93 for (llvm::Instruction *I : reverse(LLVMInstrs))
94 I->eraseFromParent();
95 }
96}

98void Instruction::moveBefore(BasicBlock &BB, const BBIterator &WhereIt) {
99 if (std::next(getIterator()) == WhereIt)
100 // Destination is same as origin, nothing to do.
101 return;
102
103 Ctx.getTracker().emplaceIfTracking<MoveInstr>(this);
104
105 auto *LLVMBB = cast<llvm::BasicBlock>(BB.Val);
106 llvm::BasicBlock::iterator It;
107 if (WhereIt == BB.end()) {
108 It = LLVMBB->end();
109 } else {
110 Instruction *WhereI = &*WhereIt;
111 It = WhereI->getTopmostLLVMInstruction()->getIterator();
112 }
113 // TODO: Move this to the verifier of sandboxir::Instruction.
114 assert(is_sorted(getLLVMInstrs(),
115 [](auto *I1, auto *I2) { return I1->comesBefore(I2); }) &&
116 "Expected program order!");
117 // Do the actual move in LLVM IR.
118 for (auto *I : getLLVMInstrs())
119 I->moveBefore(*LLVMBB, It);
120}

140void Instruction::insertInto(BasicBlock *BB, const BBIterator &WhereIt) {
141 llvm::BasicBlock *LLVMBB = cast<llvm::BasicBlock>(BB->Val);
142 llvm::Instruction *LLVMBeforeI;
143 llvm::BasicBlock::iterator LLVMBeforeIt;
144 Instruction *BeforeI;
145 if (WhereIt != BB->end()) {
146 BeforeI = &*WhereIt;
147 LLVMBeforeI = BeforeI->getTopmostLLVMInstruction();
148 LLVMBeforeIt = LLVMBeforeI->getIterator();
149 } else {
150 BeforeI = nullptr;
151 LLVMBeforeI = nullptr;
152 LLVMBeforeIt = LLVMBB->end();
153 }
154
155 Ctx.getTracker().emplaceIfTracking<InsertIntoBB>(this);
156
157 // Insert the LLVM IR Instructions in program order.
158 for (llvm::Instruction *I : getLLVMInstrs())
159 I->insertInto(LLVMBB, LLVMBeforeIt);
160}

1155void PHINode::removeIncomingValueIf(function_ref<bool(unsigned)> Predicate) {
1156 // Avoid duplicate tracking by going through this->removeIncomingValue here at
1157 // the expense of some performance. Copy PHI::removeIncomingValueIf more
1158 // directly if performance becomes an issue.
1159
1160 // Removing the element at index X, moves the element previously at X + 1
1161 // to X. Working from the end avoids complications from that.
1162 unsigned Idx = getNumIncomingValues();
1163 while (Idx > 0) {
1164 if (Predicate(Idx - 1))
1165 removeIncomingValue(Idx - 1);
1166 --Idx;
1167 }
1168}
