50bool VPRecipeBase::mayWriteToMemory() const {
51 switch (getVPDefID()) {
52 case VPInstructionSC:
53 if (Instruction::isBinaryOp(cast<VPInstruction>(this)->getOpcode()))
54 return false;
55 switch (cast<VPInstruction>(this)->getOpcode()) {
56 case Instruction::Or:
57 case Instruction::ICmp:
58 case Instruction::Select:
59 case VPInstruction::Not:
60 case VPInstruction::CalculateTripCountMinusVF:
61 case VPInstruction::CanonicalIVIncrementForPart:
62 case VPInstruction::ExtractFromEnd:
63 case VPInstruction::FirstOrderRecurrenceSplice:
64 case VPInstruction::LogicalAnd:
65 case VPInstruction::PtrAdd:
66 return false;
67 default:
68 return true;
69 }
70 case VPInterleaveSC:
71 return cast<VPInterleaveRecipe>(this)->getNumStoreOperands() > 0;
72 case VPWidenStoreEVLSC:
73 case VPWidenStoreSC:
74 return true;
75 case VPReplicateSC:
76 return cast<Instruction>(getVPSingleValue()->getUnderlyingValue())
77 ->mayWriteToMemory();
78 case VPWidenCallSC:
79 return !cast<VPWidenCallRecipe>(this)
80 ->getCalledScalarFunction()
81 ->onlyReadsMemory();
82 case VPBranchOnMaskSC:
83 case VPScalarIVStepsSC:
84 case VPPredInstPHISC:
85 return false;
86 case VPBlendSC:
87 case VPReductionEVLSC:
88 case VPReductionSC:
89 case VPWidenCanonicalIVSC:
90 case VPWidenCastSC:
91 case VPWidenGEPSC:
92 case VPWidenIntOrFpInductionSC:
93 case VPWidenLoadEVLSC:
94 case VPWidenLoadSC:
95 case VPWidenPHISC:
96 case VPWidenSC:
97 case VPWidenEVLSC:
98 case VPWidenSelectSC: {
99 const Instruction *I =
100 dyn_cast_or_null<Instruction>(getVPSingleValue()->getUnderlyingValue());
101 (void)I;
102 assert((!I || !I->mayWriteToMemory()) &&
103 "underlying instruction may write to memory");
104 return false;
105 }
106 default:
107 return true;
108 }
109}

152bool VPRecipeBase::mayHaveSideEffects() const {
153 switch (getVPDefID()) {
154 case VPDerivedIVSC:
155 case VPPredInstPHISC:
156 case VPScalarCastSC:
157 return false;
158 case VPInstructionSC:
159 return mayWriteToMemory();
160 case VPWidenCallSC: {
161 Function *Fn = cast<VPWidenCallRecipe>(this)->getCalledScalarFunction();
162 return mayWriteToMemory() || !Fn->doesNotThrow() || !Fn->willReturn();
163 }
164 case VPBlendSC:
165 case VPReductionEVLSC:
166 case VPReductionSC:
167 case VPScalarIVStepsSC:
168 case VPWidenCanonicalIVSC:
169 case VPWidenCastSC:
170 case VPWidenGEPSC:
171 case VPWidenIntOrFpInductionSC:
172 case VPWidenPHISC:
173 case VPWidenPointerInductionSC:
174 case VPWidenSC:
175 case VPWidenEVLSC:
176 case VPWidenSelectSC: {
177 const Instruction *I =
178 dyn_cast_or_null<Instruction>(getVPSingleValue()->getUnderlyingValue());
179 (void)I;
180 assert((!I || !I->mayHaveSideEffects()) &&
181 "underlying instruction has side-effects");
182 return false;
183 }
184 case VPInterleaveSC:
185 return mayWriteToMemory();
186 case VPWidenLoadEVLSC:
187 case VPWidenLoadSC:
188 case VPWidenStoreEVLSC:
189 case VPWidenStoreSC:
190 assert(
191 cast<VPWidenMemoryRecipe>(this)->getIngredient().mayHaveSideEffects() ==
192 mayWriteToMemory() &&
193 "mayHaveSideffects result for ingredient differs from this "
194 "implementation");
195 return mayWriteToMemory();
196 case VPReplicateSC: {
197 auto *R = cast<VPReplicateRecipe>(this);
198 return R->getUnderlyingInstr()->mayHaveSideEffects();
199 }
200 default:
201 return true;
202 }

205void VPLiveOut::fixPhi(VPlan &Plan, VPTransformState &State) {
206 VPValue *ExitValue = getOperand(0);
207 VPBasicBlock *MiddleVPBB =
208 cast<VPBasicBlock>(Plan.getVectorLoopRegion()->getSingleSuccessor());
209 VPRecipeBase *ExitingRecipe = ExitValue->getDefiningRecipe();
210 auto *ExitingVPBB = ExitingRecipe ? ExitingRecipe->getParent() : nullptr;
211 // Values leaving the vector loop reach live out phi's in the exiting block
212 // via middle block.
213 auto *PredVPBB = !ExitingVPBB || ExitingVPBB->getEnclosingLoopRegion()
214 ? MiddleVPBB
215 : ExitingVPBB;
216 BasicBlock *PredBB = State.CFG.VPBB2IRBB[PredVPBB];
217 Value *V = State.get(ExitValue, VPLane(0));
218 if (Phi->getBasicBlockIndex(PredBB) != -1)
219 Phi->setIncomingValueForBlock(PredBB, V);
220 else
221 Phi->addIncoming(V, PredBB);
222}

277/// Return the underlying instruction to be used for computing \p R's cost via
278/// the legacy cost model. Return nullptr if there's no suitable instruction.
279static Instruction *getInstructionForCost(const VPRecipeBase *R) {
280 if (auto *S = dyn_cast<VPSingleDefRecipe>(R))
281 return dyn_cast_or_null<Instruction>(S->getUnderlyingValue());
282 if (auto *IG = dyn_cast<VPInterleaveRecipe>(R))
283 return IG->getInsertPos();
284 // Currently the legacy cost model only calculates the instruction cost with
285 // underlying instruction. Removing the WidenMem here will prevent
286 // force-target-instruction-cost overwriting the cost of recipe with
287 // underlying instruction which is inconsistent with the legacy model.
288 // TODO: Remove WidenMem from this function when we don't need to compare to
289 // the legacy model.
290 if (auto *WidenMem = dyn_cast<VPWidenMemoryRecipe>(R))
291 return &WidenMem->getIngredient();
292 return nullptr;
293}

295InstructionCost VPRecipeBase::cost(ElementCount VF, VPCostContext &Ctx) {
296 auto *UI = getInstructionForCost(this);
297 if (UI && Ctx.skipCostComputation(UI, VF.isVector()))
298 return 0;
299
300 InstructionCost RecipeCost = computeCost(VF, Ctx);
301 if (UI && ForceTargetInstructionCost.getNumOccurrences() > 0 &&
302 RecipeCost.isValid())
303 RecipeCost = InstructionCost(ForceTargetInstructionCost);
304
305 LLVM_DEBUG({
306 dbgs() << "Cost of " << RecipeCost << " for VF " << VF << ": ";
307 dump();
308 });
309 return RecipeCost;
310}

312InstructionCost VPRecipeBase::computeCost(ElementCount VF,
313 VPCostContext &Ctx) const {
314 // Compute the cost for the recipe falling back to the legacy cost model using
315 // the underlying instruction. If there is no underlying instruction, returns
316 // 0.
317 Instruction *UI = getInstructionForCost(this);
318 if (UI && isa<VPReplicateRecipe>(this)) {
319 // VPReplicateRecipe may be cloned as part of an existing VPlan-to-VPlan
320 // transform, avoid computing their cost multiple times for now.
321 Ctx.SkipCostComputation.insert(UI);
322 }
323 return UI ? Ctx.getLegacyCost(UI, VF) : 0;
324}

407Value *VPInstruction::generate(VPTransformState &State) {
408 IRBuilderBase &Builder = State.Builder;
409
410 if (Instruction::isBinaryOp(getOpcode())) {
411 bool OnlyFirstLaneUsed = vputils::onlyFirstLaneUsed(this);
412 Value *A = State.get(getOperand(0), OnlyFirstLaneUsed);
413 Value *B = State.get(getOperand(1), OnlyFirstLaneUsed);
414 auto *Res =
415 Builder.CreateBinOp((Instruction::BinaryOps)getOpcode(), A, B, Name);
416 if (auto *I = dyn_cast<Instruction>(Res))
417 setFlags(I);
418 return Res;
419 }
420
421 switch (getOpcode()) {
422 case VPInstruction::Not: {
423 Value *A = State.get(getOperand(0));
424 return Builder.CreateNot(A, Name);
425 }
426 case Instruction::ICmp: {
427 bool OnlyFirstLaneUsed = vputils::onlyFirstLaneUsed(this);
428 Value *A = State.get(getOperand(0), OnlyFirstLaneUsed);
429 Value *B = State.get(getOperand(1), OnlyFirstLaneUsed);
430 return Builder.CreateCmp(getPredicate(), A, B, Name);
431 }
432 case Instruction::Select: {
433 Value *Cond = State.get(getOperand(0));
434 Value *Op1 = State.get(getOperand(1));
435 Value *Op2 = State.get(getOperand(2));
436 return Builder.CreateSelect(Cond, Op1, Op2, Name);
437 }
438 case VPInstruction::ActiveLaneMask: {
439 // Get first lane of vector induction variable.
440 Value *VIVElem0 = State.get(getOperand(0), VPLane(0));
441 // Get the original loop tripcount.
442 Value *ScalarTC = State.get(getOperand(1), VPLane(0));
443
444 // If this part of the active lane mask is scalar, generate the CMP directly
445 // to avoid unnecessary extracts.
446 if (State.VF.isScalar())
447 return Builder.CreateCmp(CmpInst::Predicate::ICMP_ULT, VIVElem0, ScalarTC,
448 Name);
449
450 auto *Int1Ty = Type::getInt1Ty(Builder.getContext());
451 auto *PredTy = VectorType::get(Int1Ty, State.VF);
452 return Builder.CreateIntrinsic(Intrinsic::get_active_lane_mask,
453 {PredTy, ScalarTC->getType()},
454 {VIVElem0, ScalarTC}, nullptr, Name);
455 }
456 case VPInstruction::FirstOrderRecurrenceSplice: {
457 // Generate code to combine the previous and current values in vector v3.
458 //
459 // vector.ph:
460 // v_init = vector(..., ..., ..., a[-1])
461 // br vector.body
462 //
463 // vector.body
464 // i = phi [0, vector.ph], [i+4, vector.body]
465 // v1 = phi [v_init, vector.ph], [v2, vector.body]
466 // v2 = a[i, i+1, i+2, i+3];
467 // v3 = vector(v1(3), v2(0, 1, 2))
468
469 auto *V1 = State.get(getOperand(0));
470 if (!V1->getType()->isVectorTy())
471 return V1;
472 Value *V2 = State.get(getOperand(1));
473 return Builder.CreateVectorSplice(V1, V2, -1, Name);
474 }
475 case VPInstruction::CalculateTripCountMinusVF: {
476 unsigned UF = getParent()->getPlan()->getUF();
477 Value *ScalarTC = State.get(getOperand(0), VPLane(0));
478 Value *Step = createStepForVF(Builder, ScalarTC->getType(), State.VF, UF);
479 Value *Sub = Builder.CreateSub(ScalarTC, Step);
480 Value *Cmp = Builder.CreateICmp(CmpInst::Predicate::ICMP_UGT, ScalarTC, Step);
481 Value *Zero = ConstantInt::get(ScalarTC->getType(), 0);
482 return Builder.CreateSelect(Cmp, Sub, Zero);
483 }
484 case VPInstruction::ExplicitVectorLength: {
485 // TODO: Restructure this code with an explicit remainder loop, vsetvli can
486 // be outside of the main loop.
487 Value *AVL = State.get(getOperand(0), /*IsScalar*/ true);
488 // Compute EVL
489 assert(AVL->getType()->isIntegerTy() &&
490 "Requested vector length should be an integer.");
491
492 assert(State.VF.isScalable() && "Expected scalable vector factor.");
493 Value *VFArg = State.Builder.getInt32(State.VF.getKnownMinValue());
494
495 Value *EVL = State.Builder.CreateIntrinsic(
496 State.Builder.getInt32Ty(), Intrinsic::experimental_get_vector_length,
497 {AVL, VFArg, State.Builder.getTrue()});
498 return EVL;
499 }
500 case VPInstruction::CanonicalIVIncrementForPart: {
501 unsigned Part = getUnrollPart(*this);
502 auto *IV = State.get(getOperand(0), VPLane(0));
503 assert(Part != 0 && "Must have a positive part");
504 // The canonical IV is incremented by the vectorization factor (num of
505 // SIMD elements) times the unroll part.
506 Value *Step = createStepForVF(Builder, IV->getType(), State.VF, Part);
507 return Builder.CreateAdd(IV, Step, Name, hasNoUnsignedWrap(),
508 hasNoSignedWrap());
509 }
510 case VPInstruction::BranchOnCond: {
511 Value *Cond = State.get(getOperand(0), VPLane(0));
512 // Replace the temporary unreachable terminator with a new conditional
513 // branch, hooking it up to backward destination for exiting blocks now and
514 // to forward destination(s) later when they are created.
515 BranchInst *CondBr =
516 Builder.CreateCondBr(Cond, Builder.GetInsertBlock(), nullptr);
517 CondBr->setSuccessor(0, nullptr);
518 Builder.GetInsertBlock()->getTerminator()->eraseFromParent();
519
520 if (!getParent()->isExiting())
521 return CondBr;
522
523 VPRegionBlock *ParentRegion = getParent()->getParent();
524 VPBasicBlock *Header = ParentRegion->getEntryBasicBlock();
525 CondBr->setSuccessor(1, State.CFG.VPBB2IRBB[Header]);
526 return CondBr;
527 }
528 case VPInstruction::BranchOnCount: {
529 // First create the compare.
530 Value *IV = State.get(getOperand(0), /*IsScalar*/ true);
531 Value *TC = State.get(getOperand(1), /*IsScalar*/ true);
532 Value *Cond = Builder.CreateICmpEQ(IV, TC);
533
534 // Now create the branch.
535 auto *Plan = getParent()->getPlan();
536 VPRegionBlock *TopRegion = Plan->getVectorLoopRegion();
537 VPBasicBlock *Header = TopRegion->getEntry()->getEntryBasicBlock();
538
539 // Replace the temporary unreachable terminator with a new conditional
540 // branch, hooking it up to backward destination (the header) now and to the
541 // forward destination (the exit/middle block) later when it is created.
542 // Note that CreateCondBr expects a valid BB as first argument, so we need
543 // to set it to nullptr later.
544 BranchInst *CondBr = Builder.CreateCondBr(Cond, Builder.GetInsertBlock(),
545 State.CFG.VPBB2IRBB[Header]);
546 CondBr->setSuccessor(0, nullptr);
547 Builder.GetInsertBlock()->getTerminator()->eraseFromParent();
548 return CondBr;
549 }
550 case VPInstruction::ComputeReductionResult: {
551 // FIXME: The cross-recipe dependency on VPReductionPHIRecipe is temporary
552 // and will be removed by breaking up the recipe further.
553 auto *PhiR = cast<VPReductionPHIRecipe>(getOperand(0));
554 auto *OrigPhi = cast<PHINode>(PhiR->getUnderlyingValue());
555 // Get its reduction variable descriptor.
556 const RecurrenceDescriptor &RdxDesc = PhiR->getRecurrenceDescriptor();
557
558 RecurKind RK = RdxDesc.getRecurrenceKind();
559
560 Type *PhiTy = OrigPhi->getType();
561 // The recipe's operands are the reduction phi, followed by one operand for
562 // each part of the reduction.
563 unsigned UF = getNumOperands() - 1;
564 VectorParts RdxParts(UF);
565 for (unsigned Part = 0; Part < UF; ++Part)
566 RdxParts[Part] = State.get(getOperand(1 + Part), PhiR->isInLoop());
567
568 // If the vector reduction can be performed in a smaller type, we truncate
569 // then extend the loop exit value to enable InstCombine to evaluate the
570 // entire expression in the smaller type.
571 // TODO: Handle this in truncateToMinBW.
572 if (State.VF.isVector() && PhiTy != RdxDesc.getRecurrenceType()) {
573 Type *RdxVecTy = VectorType::get(RdxDesc.getRecurrenceType(), State.VF);
574 for (unsigned Part = 0; Part < UF; ++Part)
575 RdxParts[Part] = Builder.CreateTrunc(RdxParts[Part], RdxVecTy);
576 }
577 // Reduce all of the unrolled parts into a single vector.
578 Value *ReducedPartRdx = RdxParts[0];
579 unsigned Op = RecurrenceDescriptor::getOpcode(RK);
580 if (RecurrenceDescriptor::isAnyOfRecurrenceKind(RK))
581 Op = Instruction::Or;
582
583 if (PhiR->isOrdered()) {
584 ReducedPartRdx = RdxParts[UF - 1];
585 } else {
586 // Floating-point operations should have some FMF to enable the reduction.
587 IRBuilderBase::FastMathFlagGuard FMFG(Builder);
588 Builder.setFastMathFlags(RdxDesc.getFastMathFlags());
589 for (unsigned Part = 1; Part < UF; ++Part) {
590 Value *RdxPart = RdxParts[Part];
591 if (Op != Instruction::ICmp && Op != Instruction::FCmp)
592 ReducedPartRdx = Builder.CreateBinOp(
593 (Instruction::BinaryOps)Op, RdxPart, ReducedPartRdx, "bin.rdx");
594 else
595 ReducedPartRdx = createMinMaxOp(Builder, RK, ReducedPartRdx, RdxPart);
596 }
597 }
598
599 // Create the reduction after the loop. Note that inloop reductions create
600 // the target reduction in the loop using a Reduction recipe.
601 if ((State.VF.isVector() ||
602 RecurrenceDescriptor::isAnyOfRecurrenceKind(RK)) &&
603 !PhiR->isInLoop()) {
604 ReducedPartRdx =
605 createReduction(Builder, RdxDesc, ReducedPartRdx, OrigPhi);
606 // If the reduction can be performed in a smaller type, we need to extend
607 // the reduction to the wider type before we branch to the original loop.
608 if (PhiTy != RdxDesc.getRecurrenceType())
609 ReducedPartRdx = RdxDesc.isSigned()
610 ? Builder.CreateSExt(ReducedPartRdx, PhiTy)
611 : Builder.CreateZExt(ReducedPartRdx, PhiTy);
612 }
613
614 return ReducedPartRdx;
615 }
616 case VPInstruction::ExtractFromEnd: {
617 auto *CI = cast<ConstantInt>(getOperand(1)->getLiveInIRValue());
618 unsigned Offset = CI->getZExtValue();
619 assert(Offset > 0 && "Offset from end must be positive");
620 Value *Res;
621 if (State.VF.isVector()) {
622 assert(Offset <= State.VF.getKnownMinValue() &&
623 "invalid offset to extract from");
624 // Extract lane VF - Offset from the operand.
625 Res = State.get(getOperand(0), VPLane::getLaneFromEnd(State.VF, Offset));
626 } else {
627 assert(Offset <= 1 && "invalid offset to extract from");
628 Res = State.get(getOperand(0));
629 }
630 if (isa<ExtractElementInst>(Res))
631 Res->setName(Name);
632 return Res;
633 }
634 case VPInstruction::LogicalAnd: {
635 Value *A = State.get(getOperand(0));
636 Value *B = State.get(getOperand(1));
637 return Builder.CreateLogicalAnd(A, B, Name);
638 }
639 case VPInstruction::PtrAdd: {
640 assert(vputils::onlyFirstLaneUsed(this) &&
641 "can only generate first lane for PtrAdd");
642 Value *Ptr = State.get(getOperand(0), /* IsScalar */ true);
643 Value *Addend = State.get(getOperand(1), /* IsScalar */ true);
644 return Builder.CreatePtrAdd(Ptr, Addend, Name);
645 }
646 case VPInstruction::ResumePhi: {
647 Value *IncomingFromVPlanPred =
648 State.get(getOperand(0), /* IsScalar */ true);
649 Value *IncomingFromOtherPreds =
650 State.get(getOperand(1), /* IsScalar */ true);
651 auto *NewPhi =
652 Builder.CreatePHI(IncomingFromOtherPreds->getType(), 2, Name);
653 BasicBlock *VPlanPred =
654 State.CFG
655 .VPBB2IRBB[cast<VPBasicBlock>(getParent()->getSinglePredecessor())];
656 NewPhi->addIncoming(IncomingFromVPlanPred, VPlanPred);
657 for (auto *OtherPred : predecessors(Builder.GetInsertBlock())) {
658 assert(OtherPred != VPlanPred &&
659 "VPlan predecessors should not be connected yet");
660 NewPhi->addIncoming(IncomingFromOtherPreds, OtherPred);
661 }
662 return NewPhi;
663 }
664
665 default:
666 llvm_unreachable("Unsupported opcode for instruction");
667 }
668}

880void VPWidenCallRecipe::execute(VPTransformState &State) {
881 assert(State.VF.isVector() && "not widening");
882 Function *CalledScalarFn = getCalledScalarFunction();
883 assert(!isDbgInfoIntrinsic(CalledScalarFn->getIntrinsicID()) &&
884 "DbgInfoIntrinsic should have been dropped during VPlan construction");
885 State.setDebugLocFrom(getDebugLoc());
886
887 bool UseIntrinsic = VectorIntrinsicID != Intrinsic::not_intrinsic;
888 FunctionType *VFTy = nullptr;
889 if (Variant)
890 VFTy = Variant->getFunctionType();
891 SmallVector<Type *, 2> TysForDecl;
892 // Add return type if intrinsic is overloaded on it.
893 if (UseIntrinsic &&
894 isVectorIntrinsicWithOverloadTypeAtArg(VectorIntrinsicID, -1))
895 TysForDecl.push_back(VectorType::get(
896 CalledScalarFn->getReturnType()->getScalarType(), State.VF));
897 SmallVector<Value *, 4> Args;
898 for (const auto &I : enumerate(arg_operands())) {
899 // Some intrinsics have a scalar argument - don't replace it with a
900 // vector.
901 Value *Arg;
902 if (UseIntrinsic &&
903 isVectorIntrinsicWithScalarOpAtArg(VectorIntrinsicID, I.index()))
904 Arg = State.get(I.value(), VPLane(0));
905 // Some vectorized function variants may also take a scalar argument,
906 // e.g. linear parameters for pointers. This needs to be the scalar value
907 // from the start of the respective part when interleaving.
908 else if (VFTy && !VFTy->getParamType(I.index())->isVectorTy())
909 Arg = State.get(I.value(), VPLane(0));
910 else
911 Arg = State.get(I.value());
912 if (UseIntrinsic &&
913 isVectorIntrinsicWithOverloadTypeAtArg(VectorIntrinsicID, I.index()))
914 TysForDecl.push_back(Arg->getType());
915 Args.push_back(Arg);
916 }
917
918 Function *VectorF;
919 if (UseIntrinsic) {
920 // Use vector version of the intrinsic.
921 Module *M = State.Builder.GetInsertBlock()->getModule();
922 VectorF = Intrinsic::getDeclaration(M, VectorIntrinsicID, TysForDecl);
923 assert(VectorF && "Can't retrieve vector intrinsic.");
924 } else {
925#ifndef NDEBUG
926 assert(Variant != nullptr && "Can't create vector function.");
927#endif
928 VectorF = Variant;
929 }
930
931 auto *CI = cast_or_null<CallInst>(getUnderlyingInstr());
932 SmallVector<OperandBundleDef, 1> OpBundles;
933 if (CI)
934 CI->getOperandBundlesAsDefs(OpBundles);
935
936 CallInst *V = State.Builder.CreateCall(VectorF, Args, OpBundles);
937
938 if (isa<FPMathOperator>(V))
939 V->copyFastMathFlags(CI);
940
941 if (!V->getType()->isVoidTy())
942 State.set(this, V);
943 State.addMetadata(V, CI);
944}

946InstructionCost VPWidenCallRecipe::computeCost(ElementCount VF,
947 VPCostContext &Ctx) const {
948 TTI::TargetCostKind CostKind = TTI::TCK_RecipThroughput;
949 if (Variant) {
950 return Ctx.TTI.getCallInstrCost(nullptr, Variant->getReturnType(),
951 Variant->getFunctionType()->params(),
952 CostKind);
953 }
954
955 FastMathFlags FMF;
956 // TODO: Manage flags via VPRecipeWithIRFlags.
957 if (auto *FPMO = dyn_cast_or_null<FPMathOperator>(getUnderlyingValue()))
958 FMF = FPMO->getFastMathFlags();
959
960 // Some backends analyze intrinsic arguments to determine cost. Use the
961 // underlying value for the operand if it has one. Otherwise try to use the
962 // operand of the underlying call instruction, if there is one. Otherwise
963 // clear Arguments.
964 // TODO: Rework TTI interface to be independent of concrete IR values.
965 SmallVector<const Value *> Arguments;
966 for (const auto &[Idx, Op] : enumerate(operands())) {
967 auto *V = Op->getUnderlyingValue();
968 if (!V) {
969 if (auto *UI = dyn_cast_or_null<CallBase>(getUnderlyingValue())) {
970 Arguments.push_back(UI->getArgOperand(Idx));
971 continue;
972 }
973 Arguments.clear();
974 break;
975 }
976 Arguments.push_back(V);
977 }
978
979 Type *RetTy =
980 ToVectorTy(Ctx.Types.inferScalarType(this->getVPSingleValue()), VF);
981 SmallVector<Type *> ParamTys;
982 for (unsigned I = 0; I != getNumOperands(); ++I)
983 ParamTys.push_back(
984 ToVectorTy(Ctx.Types.inferScalarType(getOperand(I)), VF));
985
986 // TODO: Rework TTI interface to avoid reliance on underlying IntrinsicInst.
987 IntrinsicCostAttributes CostAttrs(
988 VectorIntrinsicID, RetTy, Arguments, ParamTys, FMF,
989 dyn_cast_or_null<IntrinsicInst>(getUnderlyingValue()));
990 return Ctx.TTI.getIntrinsicInstrCost(CostAttrs, CostKind);
991}

1023void VPHistogramRecipe::execute(VPTransformState &State) {
1024 State.setDebugLocFrom(getDebugLoc());
1025 IRBuilderBase &Builder = State.Builder;
1026
1027 Value *Address = State.get(getOperand(0));
1028 Value *IncAmt = State.get(getOperand(1), /*IsScalar=*/true);
1029 VectorType *VTy = cast<VectorType>(Address->getType());
1030
1031 // The histogram intrinsic requires a mask even if the recipe doesn't;
1032 // if the mask operand was omitted then all lanes should be executed and
1033 // we just need to synthesize an all-true mask.
1034 Value *Mask = nullptr;
1035 if (VPValue *VPMask = getMask())
1036 Mask = State.get(VPMask);
1037 else
1038 Mask =
1039 Builder.CreateVectorSplat(VTy->getElementCount(), Builder.getInt1(1));
1040
1041 // If this is a subtract, we want to invert the increment amount. We may
1042 // add a separate intrinsic in future, but for now we'll try this.
1043 if (Opcode == Instruction::Sub)
1044 IncAmt = Builder.CreateNeg(IncAmt);
1045 else
1046 assert(Opcode == Instruction::Add && "only add or sub supported for now");
1047
1048 State.Builder.CreateIntrinsic(Intrinsic::experimental_vector_histogram_add,
1049 {VTy, IncAmt->getType()},
1050 {Address, IncAmt, Mask});
1051}

1053InstructionCost VPHistogramRecipe::computeCost(ElementCount VF,
1054 VPCostContext &Ctx) const {
1055 // FIXME: Take the gather and scatter into account as well. For now we're
1056 // generating the same cost as the fallback path, but we'll likely
1057 // need to create a new TTI method for determining the cost, including
1058 // whether we can use base + vec-of-smaller-indices or just
1059 // vec-of-pointers.
1060 assert(VF.isVector() && "Invalid VF for histogram cost");
1061 Type *AddressTy = Ctx.Types.inferScalarType(getOperand(0));
1062 VPValue *IncAmt = getOperand(1);
1063 Type *IncTy = Ctx.Types.inferScalarType(IncAmt);
1064 VectorType *VTy = VectorType::get(IncTy, VF);
1065
1066 // Assume that a non-constant update value (or a constant != 1) requires
1067 // a multiply, and add that into the cost.
1068 InstructionCost MulCost =
1069 Ctx.TTI.getArithmeticInstrCost(Instruction::Mul, VTy);
1070 if (IncAmt->isLiveIn()) {
1071 ConstantInt *CI = dyn_cast<ConstantInt>(IncAmt->getLiveInIRValue());
1072
1073 if (CI && CI->getZExtValue() == 1)
1074 MulCost = TTI::TCC_Free;
1075 }
1076
1077 // Find the cost of the histogram operation itself.
1078 Type *PtrTy = VectorType::get(AddressTy, VF);
1079 Type *MaskTy = VectorType::get(Type::getInt1Ty(Ctx.LLVMCtx), VF);
1080 IntrinsicCostAttributes ICA(Intrinsic::experimental_vector_histogram_add,
1081 Type::getVoidTy(Ctx.LLVMCtx),
1082 {PtrTy, IncTy, MaskTy});
1083
1084 // Add the costs together with the add/sub operation.
1085 return Ctx.TTI.getIntrinsicInstrCost(
1086 ICA, TargetTransformInfo::TCK_RecipThroughput) +
1087 MulCost + Ctx.TTI.getArithmeticInstrCost(Opcode, VTy);
1088}

1192void VPWidenRecipe::execute(VPTransformState &State) {
1193 State.setDebugLocFrom(getDebugLoc());
1194 auto &Builder = State.Builder;
1195 switch (Opcode) {
1196 case Instruction::Call:
1197 case Instruction::Br:
1198 case Instruction::PHI:
1199 case Instruction::GetElementPtr:
1200 case Instruction::Select:
1201 llvm_unreachable("This instruction is handled by a different recipe.");
1202 case Instruction::UDiv:
1203 case Instruction::SDiv:
1204 case Instruction::SRem:
1205 case Instruction::URem:
1206 case Instruction::Add:
1207 case Instruction::FAdd:
1208 case Instruction::Sub:
1209 case Instruction::FSub:
1210 case Instruction::FNeg:
1211 case Instruction::Mul:
1212 case Instruction::FMul:
1213 case Instruction::FDiv:
1214 case Instruction::FRem:
1215 case Instruction::Shl:
1216 case Instruction::LShr:
1217 case Instruction::AShr:
1218 case Instruction::And:
1219 case Instruction::Or:
1220 case Instruction::Xor: {
1221 // Just widen unops and binops.
1222 SmallVector<Value *, 2> Ops;
1223 for (VPValue *VPOp : operands())
1224 Ops.push_back(State.get(VPOp));
1225
1226 Value *V = Builder.CreateNAryOp(Opcode, Ops);
1227
1228 if (auto *VecOp = dyn_cast<Instruction>(V))
1229 setFlags(VecOp);
1230
1231 // Use this vector value for all users of the original instruction.
1232 State.set(this, V);
1233 State.addMetadata(V, dyn_cast_or_null<Instruction>(getUnderlyingValue()));
1234 break;
1235 }
1236 case Instruction::Freeze: {
1237 Value *Op = State.get(getOperand(0));
1238
1239 Value *Freeze = Builder.CreateFreeze(Op);
1240 State.set(this, Freeze);
1241 break;
1242 }
1243 case Instruction::ICmp:
1244 case Instruction::FCmp: {
1245 // Widen compares. Generate vector compares.
1246 bool FCmp = Opcode == Instruction::FCmp;
1247 Value *A = State.get(getOperand(0));
1248 Value *B = State.get(getOperand(1));
1249 Value *C = nullptr;
1250 if (FCmp) {
1251 // Propagate fast math flags.
1252 IRBuilder<>::FastMathFlagGuard FMFG(Builder);
1253 if (auto *I = dyn_cast_or_null<Instruction>(getUnderlyingValue()))
1254 Builder.setFastMathFlags(I->getFastMathFlags());
1255 C = Builder.CreateFCmp(getPredicate(), A, B);
1256 } else {
1257 C = Builder.CreateICmp(getPredicate(), A, B);
1258 }
1259 State.set(this, C);
1260 State.addMetadata(C, dyn_cast_or_null<Instruction>(getUnderlyingValue()));
1261 break;
1262 }
1263 default:
1264 // This instruction is not vectorized by simple widening.
1265 LLVM_DEBUG(dbgs() << "LV: Found an unhandled opcode : "
1266 << Instruction::getOpcodeName(Opcode));
1267 llvm_unreachable("Unhandled instruction!");
1268 } // end of switch.
1269
1270#if !defined(NDEBUG)
1271 // Verify that VPlan type inference results agree with the type of the
1272 // generated values.
1273 assert(VectorType::get(State.TypeAnalysis.inferScalarType(this), State.VF) ==
1274 State.get(this)->getType() &&
1275 "inferred type and type from generated instructions do not match");
1276#endif
1277}

1279InstructionCost VPWidenRecipe::computeCost(ElementCount VF,
1280 VPCostContext &Ctx) const {
1281 TTI::TargetCostKind CostKind = TTI::TCK_RecipThroughput;
1282 switch (Opcode) {
1283 case Instruction::FNeg: {
1284 Type *VectorTy =
1285 ToVectorTy(Ctx.Types.inferScalarType(this->getVPSingleValue()), VF);
1286 return Ctx.TTI.getArithmeticInstrCost(
1287 Opcode, VectorTy, CostKind,
1288 {TargetTransformInfo::OK_AnyValue, TargetTransformInfo::OP_None},
1289 {TargetTransformInfo::OK_AnyValue, TargetTransformInfo::OP_None});
1290 }
1291
1292 case Instruction::UDiv:
1293 case Instruction::SDiv:
1294 case Instruction::SRem:
1295 case Instruction::URem:
1296 // More complex computation, let the legacy cost-model handle this for now.
1297 return Ctx.getLegacyCost(cast<Instruction>(getUnderlyingValue()), VF);
1298 case Instruction::Add:
1299 case Instruction::FAdd:
1300 case Instruction::Sub:
1301 case Instruction::FSub:
1302 case Instruction::Mul:
1303 case Instruction::FMul:
1304 case Instruction::FDiv:
1305 case Instruction::FRem:
1306 case Instruction::Shl:
1307 case Instruction::LShr:
1308 case Instruction::AShr:
1309 case Instruction::And:
1310 case Instruction::Or:
1311 case Instruction::Xor: {
1312 VPValue *RHS = getOperand(1);
1313 // Certain instructions can be cheaper to vectorize if they have a constant
1314 // second vector operand. One example of this are shifts on x86.
1315 TargetTransformInfo::OperandValueInfo RHSInfo = {
1316 TargetTransformInfo::OK_AnyValue, TargetTransformInfo::OP_None};
1317 if (RHS->isLiveIn())
1318 RHSInfo = Ctx.TTI.getOperandInfo(RHS->getLiveInIRValue());
1319
1320 if (RHSInfo.Kind == TargetTransformInfo::OK_AnyValue &&
1321 getOperand(1)->isDefinedOutsideLoopRegions())
1322 RHSInfo.Kind = TargetTransformInfo::OK_UniformValue;
1323 Type *VectorTy =
1324 ToVectorTy(Ctx.Types.inferScalarType(this->getVPSingleValue()), VF);
1325 Instruction *CtxI = dyn_cast_or_null<Instruction>(getUnderlyingValue());
1326
1327 SmallVector<const Value *, 4> Operands;
1328 if (CtxI)
1329 Operands.append(CtxI->value_op_begin(), CtxI->value_op_end());
1330 return Ctx.TTI.getArithmeticInstrCost(
1331 Opcode, VectorTy, CostKind,
1332 {TargetTransformInfo::OK_AnyValue, TargetTransformInfo::OP_None},
1333 RHSInfo, Operands, CtxI, &Ctx.TLI);
1334 }
1335 case Instruction::Freeze: {
1336 // This opcode is unknown. Assume that it is the same as 'mul'.
1337 Type *VectorTy =
1338 ToVectorTy(Ctx.Types.inferScalarType(this->getVPSingleValue()), VF);
1339 return Ctx.TTI.getArithmeticInstrCost(Instruction::Mul, VectorTy, CostKind);
1340 }
1341 case Instruction::ICmp:
1342 case Instruction::FCmp: {
1343 Instruction *CtxI = dyn_cast_or_null<Instruction>(getUnderlyingValue());
1344 Type *VectorTy = ToVectorTy(Ctx.Types.inferScalarType(getOperand(0)), VF);
1345 return Ctx.TTI.getCmpSelInstrCost(Opcode, VectorTy, nullptr, getPredicate(),
1346 CostKind,
1347 {TTI::OK_AnyValue, TTI::OP_None},
1348 {TTI::OK_AnyValue, TTI::OP_None}, CtxI);
1349 }
1350 default:
1351 llvm_unreachable("Unsupported opcode for instruction");
1352 }
1353}

1436/// This function adds
1437/// (StartIdx * Step, (StartIdx + 1) * Step, (StartIdx + 2) * Step, ...)
1438/// to each vector element of Val. The sequence starts at StartIndex.
1439/// \p Opcode is relevant for FP induction variable.
1440static Value *getStepVector(Value *Val, Value *StartIdx, Value *Step,
1441 Instruction::BinaryOps BinOp, ElementCount VF,
1442 IRBuilderBase &Builder) {
1443 assert(VF.isVector() && "only vector VFs are supported");
1444
1445 // Create and check the types.
1446 auto *ValVTy = cast<VectorType>(Val->getType());
1447 ElementCount VLen = ValVTy->getElementCount();
1448
1449 Type *STy = Val->getType()->getScalarType();
1450 assert((STy->isIntegerTy() || STy->isFloatingPointTy()) &&
1451 "Induction Step must be an integer or FP");
1452 assert(Step->getType() == STy && "Step has wrong type");
1453
1454 SmallVector<Constant *, 8> Indices;
1455
1456 // Create a vector of consecutive numbers from zero to VF.
1457 VectorType *InitVecValVTy = ValVTy;
1458 if (STy->isFloatingPointTy()) {
1459 Type *InitVecValSTy =
1460 IntegerType::get(STy->getContext(), STy->getScalarSizeInBits());
1461 InitVecValVTy = VectorType::get(InitVecValSTy, VLen);
1462 }
1463 Value *InitVec = Builder.CreateStepVector(InitVecValVTy);
1464
1465 // Splat the StartIdx
1466 Value *StartIdxSplat = Builder.CreateVectorSplat(VLen, StartIdx);
1467
1468 if (STy->isIntegerTy()) {
1469 InitVec = Builder.CreateAdd(InitVec, StartIdxSplat);
1470 Step = Builder.CreateVectorSplat(VLen, Step);
1471 assert(Step->getType() == Val->getType() && "Invalid step vec");
1472 // FIXME: The newly created binary instructions should contain nsw/nuw
1473 // flags, which can be found from the original scalar operations.
1474 Step = Builder.CreateMul(InitVec, Step);
1475 return Builder.CreateAdd(Val, Step, "induction");
1476 }
1477
1478 // Floating point induction.
1479 assert((BinOp == Instruction::FAdd || BinOp == Instruction::FSub) &&
1480 "Binary Opcode should be specified for FP induction");
1481 InitVec = Builder.CreateUIToFP(InitVec, ValVTy);
1482 InitVec = Builder.CreateFAdd(InitVec, StartIdxSplat);
1483
1484 Step = Builder.CreateVectorSplat(VLen, Step);
1485 Value *MulOp = Builder.CreateFMul(InitVec, Step);
1486 return Builder.CreateBinOp(BinOp, Val, MulOp, "induction");
1487}

1489/// A helper function that returns an integer or floating-point constant with
1490/// value C.
1491static Constant *getSignedIntOrFpConstant(Type *Ty, int64_t C) {
1492 return Ty->isIntegerTy() ? ConstantInt::getSigned(Ty, C)
1493 : ConstantFP::get(Ty, C);

1496void VPWidenIntOrFpInductionRecipe::execute(VPTransformState &State) {
1497 assert(!State.Lane && "Int or FP induction being replicated.");
1498
1499 Value *Start = getStartValue()->getLiveInIRValue();
1500 const InductionDescriptor &ID = getInductionDescriptor();
1501 TruncInst *Trunc = getTruncInst();
1502 IRBuilderBase &Builder = State.Builder;
1503 assert(IV->getType() == ID.getStartValue()->getType() && "Types must match");
1504 assert(State.VF.isVector() && "must have vector VF");
1505
1506 // The value from the original loop to which we are mapping the new induction
1507 // variable.
1508 Instruction *EntryVal = Trunc ? cast<Instruction>(Trunc) : IV;
1509
1510 // Fast-math-flags propagate from the original induction instruction.
1511 IRBuilder<>::FastMathFlagGuard FMFG(Builder);
1512 if (ID.getInductionBinOp() && isa<FPMathOperator>(ID.getInductionBinOp()))
1513 Builder.setFastMathFlags(ID.getInductionBinOp()->getFastMathFlags());
1514
1515 // Now do the actual transformations, and start with fetching the step value.
1516 Value *Step = State.get(getStepValue(), VPLane(0));
1517
1518 assert((isa<PHINode>(EntryVal) || isa<TruncInst>(EntryVal)) &&
1519 "Expected either an induction phi-node or a truncate of it!");
1520
1521 // Construct the initial value of the vector IV in the vector loop preheader
1522 auto CurrIP = Builder.saveIP();
1523 BasicBlock *VectorPH = State.CFG.getPreheaderBBFor(this);
1524 Builder.SetInsertPoint(VectorPH->getTerminator());
1525 if (isa<TruncInst>(EntryVal)) {
1526 assert(Start->getType()->isIntegerTy() &&
1527 "Truncation requires an integer type");
1528 auto *TruncType = cast<IntegerType>(EntryVal->getType());
1529 Step = Builder.CreateTrunc(Step, TruncType);
1530 Start = Builder.CreateCast(Instruction::Trunc, Start, TruncType);
1531 }
1532
1533 Value *Zero = getSignedIntOrFpConstant(Start->getType(), 0);
1534 Value *SplatStart = Builder.CreateVectorSplat(State.VF, Start);
1535 Value *SteppedStart = getStepVector(
1536 SplatStart, Zero, Step, ID.getInductionOpcode(), State.VF, State.Builder);
1537
1538 // We create vector phi nodes for both integer and floating-point induction
1539 // variables. Here, we determine the kind of arithmetic we will perform.
1540 Instruction::BinaryOps AddOp;
1541 Instruction::BinaryOps MulOp;
1542 if (Step->getType()->isIntegerTy()) {
1543 AddOp = Instruction::Add;
1544 MulOp = Instruction::Mul;
1545 } else {
1546 AddOp = ID.getInductionOpcode();
1547 MulOp = Instruction::FMul;
1548 }
1549
1550 Value *SplatVF;
1551 if (VPValue *SplatVFOperand = getSplatVFValue()) {
1552 // The recipe has been unrolled. In that case, fetch the splat value for the
1553 // induction increment.
1554 SplatVF = State.get(SplatVFOperand);
1555 } else {
1556 // Multiply the vectorization factor by the step using integer or
1557 // floating-point arithmetic as appropriate.
1558 Type *StepType = Step->getType();
1559 Value *RuntimeVF = State.get(getVFValue(), VPLane(0));
1560 if (Step->getType()->isFloatingPointTy())
1561 RuntimeVF = Builder.CreateUIToFP(RuntimeVF, StepType);
1562 else
1563 RuntimeVF = Builder.CreateZExtOrTrunc(RuntimeVF, StepType);
1564 Value *Mul = Builder.CreateBinOp(MulOp, Step, RuntimeVF);
1565
1566 // Create a vector splat to use in the induction update.
1567 //
1568 // FIXME: If the step is non-constant, we create the vector splat with
1569 // IRBuilder. IRBuilder can constant-fold the multiply, but it
1570 // doesn't handle a constant vector splat.
1571 SplatVF = isa<Constant>(Mul)
1572 ? ConstantVector::getSplat(State.VF, cast<Constant>(Mul))
1573 : Builder.CreateVectorSplat(State.VF, Mul);
1574 }
1575
1576 Builder.restoreIP(CurrIP);
1577
1578 // We may need to add the step a number of times, depending on the unroll
1579 // factor. The last of those goes into the PHI.
1580 PHINode *VecInd = PHINode::Create(SteppedStart->getType(), 2, "vec.ind");
1581 VecInd->insertBefore(State.CFG.PrevBB->getFirstInsertionPt());
1582 VecInd->setDebugLoc(EntryVal->getDebugLoc());
1583 State.set(this, VecInd);
1584
1585 Instruction *LastInduction = cast<Instruction>(
1586 Builder.CreateBinOp(AddOp, VecInd, SplatVF, "vec.ind.next"));
1587 if (isa<TruncInst>(EntryVal))
1588 State.addMetadata(LastInduction, EntryVal);
1589 LastInduction->setDebugLoc(EntryVal->getDebugLoc());
1590
1591 VecInd->addIncoming(SteppedStart, VectorPH);
1592 // Add induction update using an incorrect block temporarily. The phi node
1593 // will be fixed after VPlan execution. Note that at this point the latch
1594 // block cannot be used, as it does not exist yet.
1595 // TODO: Model increment value in VPlan, by turning the recipe into a
1596 // multi-def and a subclass of VPHeaderPHIRecipe.
1597 VecInd->addIncoming(LastInduction, VectorPH);
1598}

1647void VPScalarIVStepsRecipe::execute(VPTransformState &State) {
1648 // Fast-math-flags propagate from the original induction instruction.
1649 IRBuilder<>::FastMathFlagGuard FMFG(State.Builder);
1650 if (hasFastMathFlags())
1651 State.Builder.setFastMathFlags(getFastMathFlags());
1652
1653 /// Compute scalar induction steps. \p ScalarIV is the scalar induction
1654 /// variable on which to base the steps, \p Step is the size of the step.
1655
1656 Value *BaseIV = State.get(getOperand(0), VPLane(0));
1657 Value *Step = State.get(getStepValue(), VPLane(0));
1658 IRBuilderBase &Builder = State.Builder;
1659
1660 // Ensure step has the same type as that of scalar IV.
1661 Type *BaseIVTy = BaseIV->getType()->getScalarType();
1662 assert(BaseIVTy == Step->getType() && "Types of BaseIV and Step must match!");
1663
1664 // We build scalar steps for both integer and floating-point induction
1665 // variables. Here, we determine the kind of arithmetic we will perform.
1666 Instruction::BinaryOps AddOp;
1667 Instruction::BinaryOps MulOp;
1668 if (BaseIVTy->isIntegerTy()) {
1669 AddOp = Instruction::Add;
1670 MulOp = Instruction::Mul;
1671 } else {
1672 AddOp = InductionOpcode;
1673 MulOp = Instruction::FMul;
1674 }
1675
1676 // Determine the number of scalars we need to generate for each unroll
1677 // iteration.
1678 bool FirstLaneOnly = vputils::onlyFirstLaneUsed(this);
1679 // Compute the scalar steps and save the results in State.
1680 Type *IntStepTy =
1681 IntegerType::get(BaseIVTy->getContext(), BaseIVTy->getScalarSizeInBits());
1682 Type *VecIVTy = nullptr;
1683 Value *UnitStepVec = nullptr, *SplatStep = nullptr, *SplatIV = nullptr;
1684 if (!FirstLaneOnly && State.VF.isScalable()) {
1685 VecIVTy = VectorType::get(BaseIVTy, State.VF);
1686 UnitStepVec =
1687 Builder.CreateStepVector(VectorType::get(IntStepTy, State.VF));
1688 SplatStep = Builder.CreateVectorSplat(State.VF, Step);
1689 SplatIV = Builder.CreateVectorSplat(State.VF, BaseIV);
1690 }
1691
1692 unsigned StartLane = 0;
1693 unsigned EndLane = FirstLaneOnly ? 1 : State.VF.getKnownMinValue();
1694 if (State.Lane) {
1695 StartLane = State.Lane->getKnownLane();
1696 EndLane = StartLane + 1;
1697 }
1698 Value *StartIdx0 =
1699 createStepForVF(Builder, IntStepTy, State.VF, getUnrollPart(*this));
1700
1701 if (!FirstLaneOnly && State.VF.isScalable()) {
1702 auto *SplatStartIdx = Builder.CreateVectorSplat(State.VF, StartIdx0);
1703 auto *InitVec = Builder.CreateAdd(SplatStartIdx, UnitStepVec);
1704 if (BaseIVTy->isFloatingPointTy())
1705 InitVec = Builder.CreateSIToFP(InitVec, VecIVTy);
1706 auto *Mul = Builder.CreateBinOp(MulOp, InitVec, SplatStep);
1707 auto *Add = Builder.CreateBinOp(AddOp, SplatIV, Mul);
1708 State.set(this, Add);
1709 // It's useful to record the lane values too for the known minimum number
1710 // of elements so we do those below. This improves the code quality when
1711 // trying to extract the first element, for example.
1712 }
1713
1714 if (BaseIVTy->isFloatingPointTy())
1715 StartIdx0 = Builder.CreateSIToFP(StartIdx0, BaseIVTy);
1716
1717 for (unsigned Lane = StartLane; Lane < EndLane; ++Lane) {
1718 Value *StartIdx = Builder.CreateBinOp(
1719 AddOp, StartIdx0, getSignedIntOrFpConstant(BaseIVTy, Lane));
1720 // The step returned by `createStepForVF` is a runtime-evaluated value
1721 // when VF is scalable. Otherwise, it should be folded into a Constant.
1722 assert((State.VF.isScalable() || isa<Constant>(StartIdx)) &&
1723 "Expected StartIdx to be folded to a constant when VF is not "
1724 "scalable");
1725 auto *Mul = Builder.CreateBinOp(MulOp, StartIdx, Step);
1726 auto *Add = Builder.CreateBinOp(AddOp, BaseIV, Mul);
1727 State.set(this, Add, VPLane(Lane));
1728 }
1729}

1741void VPWidenGEPRecipe::execute(VPTransformState &State) {
1742 assert(State.VF.isVector() && "not widening");
1743 auto *GEP = cast<GetElementPtrInst>(getUnderlyingInstr());
1744 // Construct a vector GEP by widening the operands of the scalar GEP as
1745 // necessary. We mark the vector GEP 'inbounds' if appropriate. A GEP
1746 // results in a vector of pointers when at least one operand of the GEP
1747 // is vector-typed. Thus, to keep the representation compact, we only use
1748 // vector-typed operands for loop-varying values.
1749
1750 if (areAllOperandsInvariant()) {
1751 // If we are vectorizing, but the GEP has only loop-invariant operands,
1752 // the GEP we build (by only using vector-typed operands for
1753 // loop-varying values) would be a scalar pointer. Thus, to ensure we
1754 // produce a vector of pointers, we need to either arbitrarily pick an
1755 // operand to broadcast, or broadcast a clone of the original GEP.
1756 // Here, we broadcast a clone of the original.
1757 //
1758 // TODO: If at some point we decide to scalarize instructions having
1759 // loop-invariant operands, this special case will no longer be
1760 // required. We would add the scalarization decision to
1761 // collectLoopScalars() and teach getVectorValue() to broadcast
1762 // the lane-zero scalar value.
1763 SmallVector<Value *> Ops;
1764 for (unsigned I = 0, E = getNumOperands(); I != E; I++)
1765 Ops.push_back(State.get(getOperand(I), VPLane(0)));
1766
1767 auto *NewGEP =
1768 State.Builder.CreateGEP(GEP->getSourceElementType(), Ops[0],
1769 ArrayRef(Ops).drop_front(), "", isInBounds());
1770 Value *Splat = State.Builder.CreateVectorSplat(State.VF, NewGEP);
1771 State.set(this, Splat);
1772 State.addMetadata(Splat, GEP);
1773 } else {
1774 // If the GEP has at least one loop-varying operand, we are sure to
1775 // produce a vector of pointers unless VF is scalar.
1776 // The pointer operand of the new GEP. If it's loop-invariant, we
1777 // won't broadcast it.
1778 auto *Ptr = isPointerLoopInvariant() ? State.get(getOperand(0), VPLane(0))
1779 : State.get(getOperand(0));
1780
1781 // Collect all the indices for the new GEP. If any index is
1782 // loop-invariant, we won't broadcast it.
1783 SmallVector<Value *, 4> Indices;
1784 for (unsigned I = 1, E = getNumOperands(); I < E; I++) {
1785 VPValue *Operand = getOperand(I);
1786 if (isIndexLoopInvariant(I - 1))
1787 Indices.push_back(State.get(Operand, VPLane(0)));
1788 else
1789 Indices.push_back(State.get(Operand));
1790 }
1791
1792 // Create the new GEP. Note that this GEP may be a scalar if VF == 1,
1793 // but it should be a vector, otherwise.
1794 auto *NewGEP = State.Builder.CreateGEP(GEP->getSourceElementType(), Ptr,
1795 Indices, "", isInBounds());
1796 assert((State.VF.isScalar() || NewGEP->getType()->isVectorTy()) &&
1797 "NewGEP is not a pointer vector");
1798 State.set(this, NewGEP);
1799 State.addMetadata(NewGEP, GEP);
1800 }

1868void VPBlendRecipe::execute(VPTransformState &State) {
1869 assert(isNormalized() && "Expected blend to be normalized!");
1870 State.setDebugLocFrom(getDebugLoc());
1871 // We know that all PHIs in non-header blocks are converted into
1872 // selects, so we don't have to worry about the insertion order and we
1873 // can just use the builder.
1874 // At this point we generate the predication tree. There may be
1875 // duplications since this is a simple recursive scan, but future
1876 // optimizations will clean it up.
1877
1878 unsigned NumIncoming = getNumIncomingValues();
1879
1880 // Generate a sequence of selects of the form:
1881 // SELECT(Mask3, In3,
1882 // SELECT(Mask2, In2,
1883 // SELECT(Mask1, In1,
1884 // In0)))
1885 // Note that Mask0 is never used: lanes for which no path reaches this phi and
1886 // are essentially undef are taken from In0.
1887 bool OnlyFirstLaneUsed = vputils::onlyFirstLaneUsed(this);
1888 Value *Result = nullptr;
1889 for (unsigned In = 0; In < NumIncoming; ++In) {
1890 // We might have single edge PHIs (blocks) - use an identity
1891 // 'select' for the first PHI operand.
1892 Value *In0 = State.get(getIncomingValue(In), OnlyFirstLaneUsed);
1893 if (In == 0)
1894 Result = In0; // Initialize with the first incoming value.
1895 else {
1896 // Select between the current value and the previous incoming edge
1897 // based on the incoming mask.
1898 Value *Cond = State.get(getMask(In), OnlyFirstLaneUsed);
1899 Result = State.Builder.CreateSelect(Cond, In0, Result, "predphi");
1900 }
1901 }
1902 State.set(this, Result, OnlyFirstLaneUsed);
1903}

2509static Value *createBitOrPointerCast(IRBuilderBase &Builder, Value *V,
2510 VectorType *DstVTy, const DataLayout &DL) {
2511 // Verify that V is a vector type with same number of elements as DstVTy.
2512 auto VF = DstVTy->getElementCount();
2513 auto *SrcVecTy = cast<VectorType>(V->getType());
2514 assert(VF == SrcVecTy->getElementCount() && "Vector dimensions do not match");
2515 Type *SrcElemTy = SrcVecTy->getElementType();
2516 Type *DstElemTy = DstVTy->getElementType();
2517 assert((DL.getTypeSizeInBits(SrcElemTy) == DL.getTypeSizeInBits(DstElemTy)) &&
2518 "Vector elements must have same size");
2519
2520 // Do a direct cast if element types are castable.
2521 if (CastInst::isBitOrNoopPointerCastable(SrcElemTy, DstElemTy, DL)) {
2522 return Builder.CreateBitOrPointerCast(V, DstVTy);
2523 }
2524 // V cannot be directly casted to desired vector type.
2525 // May happen when V is a floating point vector but DstVTy is a vector of
2526 // pointers or vice-versa. Handle this using a two-step bitcast using an
2527 // intermediate Integer type for the bitcast i.e. Ptr <-> Int <-> Float.
2528 assert((DstElemTy->isPointerTy() != SrcElemTy->isPointerTy()) &&
2529 "Only one type should be a pointer type");
2530 assert((DstElemTy->isFloatingPointTy() != SrcElemTy->isFloatingPointTy()) &&
2531 "Only one type should be a floating point type");
2532 Type *IntTy =
2533 IntegerType::getIntNTy(V->getContext(), DL.getTypeSizeInBits(SrcElemTy));
2534 auto *VecIntTy = VectorType::get(IntTy, VF);
2535 Value *CastVal = Builder.CreateBitOrPointerCast(V, VecIntTy);
2536 return Builder.CreateBitOrPointerCast(CastVal, DstVTy);
2537}

2539/// Return a vector containing interleaved elements from multiple
2540/// smaller input vectors.
2541static Value *interleaveVectors(IRBuilderBase &Builder, ArrayRef<Value *> Vals,
2542 const Twine &Name) {
2543 unsigned Factor = Vals.size();
2544 assert(Factor > 1 && "Tried to interleave invalid number of vectors");
2545
2546 VectorType *VecTy = cast<VectorType>(Vals[0]->getType());
2547#ifndef NDEBUG
2548 for (Value *Val : Vals)
2549 assert(Val->getType() == VecTy && "Tried to interleave mismatched types");
2550#endif
2551
2552 // Scalable vectors cannot use arbitrary shufflevectors (only splats), so
2553 // must use intrinsics to interleave.
2554 if (VecTy->isScalableTy()) {
2555 VectorType *WideVecTy = VectorType::getDoubleElementsVectorType(VecTy);
2556 return Builder.CreateIntrinsic(WideVecTy, Intrinsic::vector_interleave2,
2557 Vals,
2558 /*FMFSource=*/nullptr, Name);
2559 }
2560
2561 // Fixed length. Start by concatenating all vectors into a wide vector.
2562 Value *WideVec = concatenateVectors(Builder, Vals);
2563
2564 // Interleave the elements into the wide vector.
2565 const unsigned NumElts = VecTy->getElementCount().getFixedValue();
2566 return Builder.CreateShuffleVector(
2567 WideVec, createInterleaveMask(NumElts, Factor), Name);
2568}
2569
2570// Try to vectorize the interleave group that \p Instr belongs to.
2571//
2572// E.g. Translate following interleaved load group (factor = 3):
2573// for (i = 0; i < N; i+=3) {
2574// R = Pic[i]; // Member of index 0
2575// G = Pic[i+1]; // Member of index 1
2576// B = Pic[i+2]; // Member of index 2
2577// ... // do something to R, G, B
2578// }
2579// To:
2580// %wide.vec = load <12 x i32> ; Read 4 tuples of R,G,B
2581// %R.vec = shuffle %wide.vec, poison, <0, 3, 6, 9> ; R elements
2582// %G.vec = shuffle %wide.vec, poison, <1, 4, 7, 10> ; G elements
2583// %B.vec = shuffle %wide.vec, poison, <2, 5, 8, 11> ; B elements
2584//
2585// Or translate following interleaved store group (factor = 3):
2586// for (i = 0; i < N; i+=3) {
2587// ... do something to R, G, B
2588// Pic[i] = R; // Member of index 0
2589// Pic[i+1] = G; // Member of index 1
2590// Pic[i+2] = B; // Member of index 2
2591// }
2592// To:
2593// %R_G.vec = shuffle %R.vec, %G.vec, <0, 1, 2, ..., 7>
2594// %B_U.vec = shuffle %B.vec, poison, <0, 1, 2, 3, u, u, u, u>
2595// %interleaved.vec = shuffle %R_G.vec, %B_U.vec,
2596// <0, 4, 8, 1, 5, 9, 2, 6, 10, 3, 7, 11> ; Interleave R,G,B elements
2597// store <12 x i32> %interleaved.vec ; Write 4 tuples of R,G,B
2598void VPInterleaveRecipe::execute(VPTransformState &State) {
2599 assert(!State.Lane && "Interleave group being replicated.");
2600 const InterleaveGroup<Instruction> *Group = IG;
2601 Instruction *Instr = Group->getInsertPos();
2602
2603 // Prepare for the vector type of the interleaved load/store.
2604 Type *ScalarTy = getLoadStoreType(Instr);
2605 unsigned InterleaveFactor = Group->getFactor();
2606 auto *VecTy = VectorType::get(ScalarTy, State.VF * InterleaveFactor);
2607
2608 // Prepare for the new pointers.
2609 unsigned Index = Group->getIndex(Instr);
2610
2611 // TODO: extend the masked interleaved-group support to reversed access.
2612 VPValue *BlockInMask = getMask();
2613 assert((!BlockInMask || !Group->isReverse()) &&
2614 "Reversed masked interleave-group not supported.");
2615
2616 Value *Idx;
2617 // If the group is reverse, adjust the index to refer to the last vector lane
2618 // instead of the first. We adjust the index from the first vector lane,
2619 // rather than directly getting the pointer for lane VF - 1, because the
2620 // pointer operand of the interleaved access is supposed to be uniform.
2621 if (Group->isReverse()) {
2622 Value *RuntimeVF =
2623 getRuntimeVF(State.Builder, State.Builder.getInt32Ty(), State.VF);
2624 Idx = State.Builder.CreateSub(RuntimeVF, State.Builder.getInt32(1));
2625 Idx = State.Builder.CreateMul(Idx,
2626 State.Builder.getInt32(Group->getFactor()));
2627 Idx = State.Builder.CreateAdd(Idx, State.Builder.getInt32(Index));
2628 Idx = State.Builder.CreateNeg(Idx);
2629 } else
2630 Idx = State.Builder.getInt32(-Index);
2631
2632 VPValue *Addr = getAddr();
2633 Value *ResAddr = State.get(Addr, VPLane(0));
2634 if (auto *I = dyn_cast<Instruction>(ResAddr))
2635 State.setDebugLocFrom(I->getDebugLoc());
2636
2637 // Notice current instruction could be any index. Need to adjust the address
2638 // to the member of index 0.
2639 //
2640 // E.g. a = A[i+1]; // Member of index 1 (Current instruction)
2641 // b = A[i]; // Member of index 0
2642 // Current pointer is pointed to A[i+1], adjust it to A[i].
2643 //
2644 // E.g. A[i+1] = a; // Member of index 1
2645 // A[i] = b; // Member of index 0
2646 // A[i+2] = c; // Member of index 2 (Current instruction)
2647 // Current pointer is pointed to A[i+2], adjust it to A[i].
2648
2649 bool InBounds = false;
2650 if (auto *gep = dyn_cast<GetElementPtrInst>(ResAddr->stripPointerCasts()))
2651 InBounds = gep->isInBounds();
2652 ResAddr = State.Builder.CreateGEP(ScalarTy, ResAddr, Idx, "", InBounds);
2653
2654 State.setDebugLocFrom(Instr->getDebugLoc());
2655 Value *PoisonVec = PoisonValue::get(VecTy);
2656
2657 auto CreateGroupMask = [&BlockInMask, &State,
2658 &InterleaveFactor](Value *MaskForGaps) -> Value * {
2659 if (State.VF.isScalable()) {
2660 assert(!MaskForGaps && "Interleaved groups with gaps are not supported.");
2661 assert(InterleaveFactor == 2 &&
2662 "Unsupported deinterleave factor for scalable vectors");
2663 auto *ResBlockInMask = State.get(BlockInMask);
2664 SmallVector<Value *, 2> Ops = {ResBlockInMask, ResBlockInMask};
2665 auto *MaskTy = VectorType::get(State.Builder.getInt1Ty(),
2666 State.VF.getKnownMinValue() * 2, true);
2667 return State.Builder.CreateIntrinsic(
2668 MaskTy, Intrinsic::vector_interleave2, Ops,
2669 /*FMFSource=*/nullptr, "interleaved.mask");
2670 }
2671
2672 if (!BlockInMask)
2673 return MaskForGaps;
2674
2675 Value *ResBlockInMask = State.get(BlockInMask);
2676 Value *ShuffledMask = State.Builder.CreateShuffleVector(
2677 ResBlockInMask,
2678 createReplicatedMask(InterleaveFactor, State.VF.getKnownMinValue()),
2679 "interleaved.mask");
2680 return MaskForGaps ? State.Builder.CreateBinOp(Instruction::And,
2681 ShuffledMask, MaskForGaps)
2682 : ShuffledMask;
2683 };
2684
2685 const DataLayout &DL = Instr->getDataLayout();
2686 // Vectorize the interleaved load group.
2687 if (isa<LoadInst>(Instr)) {
2688 Value *MaskForGaps = nullptr;
2689 if (NeedsMaskForGaps) {
2690 MaskForGaps = createBitMaskForGaps(State.Builder,
2691 State.VF.getKnownMinValue(), *Group);
2692 assert(MaskForGaps && "Mask for Gaps is required but it is null");
2693 }
2694
2695 Instruction *NewLoad;
2696 if (BlockInMask || MaskForGaps) {
2697 Value *GroupMask = CreateGroupMask(MaskForGaps);
2698 NewLoad = State.Builder.CreateMaskedLoad(VecTy, ResAddr,
2699 Group->getAlign(), GroupMask,
2700 PoisonVec, "wide.masked.vec");
2701 } else
2702 NewLoad = State.Builder.CreateAlignedLoad(VecTy, ResAddr,
2703 Group->getAlign(), "wide.vec");
2704 Group->addMetadata(NewLoad);
2705
2706 ArrayRef<VPValue *> VPDefs = definedValues();
2707 const DataLayout &DL = State.CFG.PrevBB->getDataLayout();
2708 if (VecTy->isScalableTy()) {
2709 assert(InterleaveFactor == 2 &&
2710 "Unsupported deinterleave factor for scalable vectors");
2711
2712 // Scalable vectors cannot use arbitrary shufflevectors (only splats),
2713 // so must use intrinsics to deinterleave.
2714 Value *DI = State.Builder.CreateIntrinsic(
2715 Intrinsic::vector_deinterleave2, VecTy, NewLoad,
2716 /*FMFSource=*/nullptr, "strided.vec");
2717 unsigned J = 0;
2718 for (unsigned I = 0; I < InterleaveFactor; ++I) {
2719 Instruction *Member = Group->getMember(I);
2720
2721 if (!Member)
2722 continue;
2723
2724 Value *StridedVec = State.Builder.CreateExtractValue(DI, I);
2725 // If this member has different type, cast the result type.
2726 if (Member->getType() != ScalarTy) {
2727 VectorType *OtherVTy = VectorType::get(Member->getType(), State.VF);
2728 StridedVec =
2729 createBitOrPointerCast(State.Builder, StridedVec, OtherVTy, DL);
2730 }
2731
2732 if (Group->isReverse())
2733 StridedVec = State.Builder.CreateVectorReverse(StridedVec, "reverse");
2734
2735 State.set(VPDefs[J], StridedVec);
2736 ++J;
2737 }
2738
2739 return;
2740 }
2741
2742 // For each member in the group, shuffle out the appropriate data from the
2743 // wide loads.
2744 unsigned J = 0;
2745 for (unsigned I = 0; I < InterleaveFactor; ++I) {
2746 Instruction *Member = Group->getMember(I);
2747
2748 // Skip the gaps in the group.
2749 if (!Member)
2750 continue;
2751
2752 auto StrideMask =
2753 createStrideMask(I, InterleaveFactor, State.VF.getKnownMinValue());
2754 Value *StridedVec =
2755 State.Builder.CreateShuffleVector(NewLoad, StrideMask, "strided.vec");
2756
2757 // If this member has different type, cast the result type.
2758 if (Member->getType() != ScalarTy) {
2759 assert(!State.VF.isScalable() && "VF is assumed to be non scalable.");
2760 VectorType *OtherVTy = VectorType::get(Member->getType(), State.VF);
2761 StridedVec =
2762 createBitOrPointerCast(State.Builder, StridedVec, OtherVTy, DL);
2763 }
2764
2765 if (Group->isReverse())
2766 StridedVec = State.Builder.CreateVectorReverse(StridedVec, "reverse");
2767
2768 State.set(VPDefs[J], StridedVec);
2769 ++J;
2770 }
2771 return;
2772 }
2773
2774 // The sub vector type for current instruction.
2775 auto *SubVT = VectorType::get(ScalarTy, State.VF);
2776
2777 // Vectorize the interleaved store group.
2778 Value *MaskForGaps =
2779 createBitMaskForGaps(State.Builder, State.VF.getKnownMinValue(), *Group);
2780 assert((!MaskForGaps || !State.VF.isScalable()) &&
2781 "masking gaps for scalable vectors is not yet supported.");
2782 ArrayRef<VPValue *> StoredValues = getStoredValues();
2783 // Collect the stored vector from each member.
2784 SmallVector<Value *, 4> StoredVecs;
2785 unsigned StoredIdx = 0;
2786 for (unsigned i = 0; i < InterleaveFactor; i++) {
2787 assert((Group->getMember(i) || MaskForGaps) &&
2788 "Fail to get a member from an interleaved store group");
2789 Instruction *Member = Group->getMember(i);
2790
2791 // Skip the gaps in the group.
2792 if (!Member) {
2793 Value *Undef = PoisonValue::get(SubVT);
2794 StoredVecs.push_back(Undef);
2795 continue;
2796 }
2797
2798 Value *StoredVec = State.get(StoredValues[StoredIdx]);
2799 ++StoredIdx;
2800
2801 if (Group->isReverse())
2802 StoredVec = State.Builder.CreateVectorReverse(StoredVec, "reverse");
2803
2804 // If this member has different type, cast it to a unified type.
2805
2806 if (StoredVec->getType() != SubVT)
2807 StoredVec = createBitOrPointerCast(State.Builder, StoredVec, SubVT, DL);
2808
2809 StoredVecs.push_back(StoredVec);
2810 }
2811
2812 // Interleave all the smaller vectors into one wider vector.
2813 Value *IVec = interleaveVectors(State.Builder, StoredVecs, "interleaved.vec");
2814 Instruction *NewStoreInstr;
2815 if (BlockInMask || MaskForGaps) {
2816 Value *GroupMask = CreateGroupMask(MaskForGaps);
2817 NewStoreInstr = State.Builder.CreateMaskedStore(
2818 IVec, ResAddr, Group->getAlign(), GroupMask);
2819 } else
2820 NewStoreInstr =
2821 State.Builder.CreateAlignedStore(IVec, ResAddr, Group->getAlign());
2822
2823 Group->addMetadata(NewStoreInstr);
2824}
