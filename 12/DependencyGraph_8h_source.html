9// This file declares the dependency graph used by the vectorizer's instruction
10// scheduler.
11//
12// The nodes of the graph are objects of the `DGNode` class. Each `DGNode`
13// object points to an instruction.
14// The edges between `DGNode`s are implicitly defined by an ordered set of
15// predecessor nodes, to save memory.
16// Finally the whole dependency graph is an object of the `DependencyGraph`
17// class, which also provides the API for creating/extending the graph from
18// input Sandbox IR.
19//
20//===----------------------------------------------------------------------===//

32/// A DependencyGraph Node that points to an Instruction and contains memory
33/// dependency edges.
34class DGNode {
35 Instruction *I;
36 /// Memory predecessors.
37 DenseSet<DGNode *> MemPreds;
38 /// This is true if this may read/write memory, or if it has some ordering
39 /// constraints, like with stacksave/stackrestore and alloca/inalloca.
40 bool IsMem;
41
42public:
43 DGNode(Instruction *I) : I(I) {
44 IsMem = I->isMemDepCandidate() ||
45 (isa<AllocaInst>(I) && cast<AllocaInst>(I)->isUsedWithInAlloca()) ||
46 I->isStackSaveOrRestoreIntrinsic();
47 }
48 Instruction *getInstruction() const { return I; }
49 void addMemPred(DGNode *PredN) { MemPreds.insert(PredN); }
50 /// \Returns all memory dependency predecessors.
51 iterator_range<DenseSet<DGNode *>::const_iterator> memPreds() const {
52 return make_range(MemPreds.begin(), MemPreds.end());
53 }
54 /// \Returns true if there is a memory dependency N->this.
55 bool hasMemPred(DGNode *N) const { return MemPreds.count(N); }
56 /// \Returns true if this may read/write memory, or if it has some ordering
57 /// constraints, like with stacksave/stackrestore and alloca/inalloca.
58 bool isMem() const { return IsMem; }
59#ifndef NDEBUG
60 void print(raw_ostream &OS, bool PrintDeps = true) const;
61 friend raw_ostream &operator<<(DGNode &N, raw_ostream &OS) {
62 N.print(OS);
63 return OS;
64 }
65 LLVM_DUMP_METHOD void dump() const;
66#endif // NDEBUG
67};
68
69class DependencyGraph {
70private:
71 DenseMap<Instruction *, std::unique_ptr<DGNode>> InstrToNodeMap;
72
73public:
74 DependencyGraph() {}
75
76 DGNode *getNode(Instruction *I) const {
77 auto It = InstrToNodeMap.find(I);
78 return It != InstrToNodeMap.end() ? It->second.get() : nullptr;
79 }
80 DGNode *getOrCreateNode(Instruction *I) {
81 auto [It, NotInMap] = InstrToNodeMap.try_emplace(I);
82 if (NotInMap)
83 It->second = std::make_unique<DGNode>(I);
84 return It->second.get();
85 }
86 /// Build/extend the dependency graph such that it includes \p Instrs. Returns
87 /// the interval spanning \p Instrs.
88 Interval<Instruction> extend(ArrayRef<Instruction *> Instrs);
89#ifndef NDEBUG
90 void print(raw_ostream &OS) const;
91 LLVM_DUMP_METHOD void dump() const;
92#endif // NDEBUG
93};
