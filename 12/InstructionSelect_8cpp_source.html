70/// This class observes instruction insertions/removals.
71/// InstructionSelect stores an iterator of the instruction prior to the one
72/// that is currently being selected to determine which instruction to select
73/// next. Previously this meant that selecting multiple instructions at once was
74/// illegal behavior due to potential invalidation of this iterator. This is
75/// a non-obvious limitation for selector implementers. Therefore, to allow
76/// deletion of arbitrary instructions, we detect this case and continue
77/// selection with the predecessor of the deleted instruction.
78class InstructionSelect::MIIteratorMaintainer : public GISelChangeObserver {
79#ifndef NDEBUG
80 SmallSetVector<const MachineInstr *, 32> CreatedInstrs;
81#endif
82public:
83 MachineBasicBlock::reverse_iterator MII;
84
85 void changingInstr(MachineInstr &MI) override {
86 llvm_unreachable("InstructionSelect does not track changed instructions!");
87 }
88 void changedInstr(MachineInstr &MI) override {
89 llvm_unreachable("InstructionSelect does not track changed instructions!");
90 }
91
92 void createdInstr(MachineInstr &MI) override {
93 LLVM_DEBUG(dbgs() << "Creating: " << MI; CreatedInstrs.insert(&MI));
94 }
95
96 void erasingInstr(MachineInstr &MI) override {
97 LLVM_DEBUG(dbgs() << "Erasing: " << MI; CreatedInstrs.remove(&MI));
98 if (MII.getInstrIterator().getNodePtr() == &MI) {
99 // If the iterator points to the MI that will be erased (i.e. the MI prior
100 // to the MI that is currently being selected), the iterator would be
101 // invalidated. Continue selection with its predecessor.
102 ++MII;
103 LLVM_DEBUG(dbgs() << "Instruction removal updated iterator.\n");
104 }
105 }
106
107 void reportFullyCreatedInstrs() {
108 LLVM_DEBUG({
109 if (CreatedInstrs.empty()) {
110 dbgs() << "Created no instructions.\n";
111 } else {
112 dbgs() << "Created:\n";
113 for (const auto *MI : CreatedInstrs) {
114 dbgs() << " " << *MI;
115 }
116 CreatedInstrs.clear();
117 }
118 });
119 }
120};

135bool InstructionSelect::runOnMachineFunction(MachineFunction &MF) {
136 // If the ISel pipeline failed, do not bother running that pass.
137 if (MF.getProperties().hasProperty(
138 MachineFunctionProperties::Property::FailedISel))
139 return false;
140
141 ISel = MF.getSubtarget().getInstructionSelector();
142 ISel->TPC = &getAnalysis<TargetPassConfig>();
143
144 // FIXME: Properly override OptLevel in TargetMachine. See OptLevelChanger
145 CodeGenOptLevel OldOptLevel = OptLevel;
146 auto RestoreOptLevel = make_scope_exit([=]() { OptLevel = OldOptLevel; });
147 OptLevel = MF.getFunction().hasOptNone() ? CodeGenOptLevel::None
148 : MF.getTarget().getOptLevel();
149
150 KB = &getAnalysis<GISelKnownBitsAnalysis>().get(MF);
151 if (OptLevel != CodeGenOptLevel::None) {
152 PSI = &getAnalysis<ProfileSummaryInfoWrapperPass>().getPSI();
153 if (PSI && PSI->hasProfileSummary())
154 BFI = &getAnalysis<LazyBlockFrequencyInfoPass>().getBFI();
155 }
156
157 return selectMachineFunction(MF);
158}

160bool InstructionSelect::selectMachineFunction(MachineFunction &MF) {
161 LLVM_DEBUG(dbgs() << "Selecting function: " << MF.getName() << '\n');
162 assert(ISel && "Cannot work without InstructionSelector");
163
164 const TargetPassConfig &TPC = *ISel->TPC;
165 CodeGenCoverage CoverageInfo;
166 ISel->setupMF(MF, KB, &CoverageInfo, PSI, BFI);
167
168 // An optimization remark emitter. Used to report failures.
169 MachineOptimizationRemarkEmitter MORE(MF, /*MBFI=*/nullptr);
170 ISel->MORE = &MORE;
171
172 // FIXME: There are many other MF/MFI fields we need to initialize.
173
174 MachineRegisterInfo &MRI = MF.getRegInfo();
175#ifndef NDEBUG
176 // Check that our input is fully legal: we require the function to have the
177 // Legalized property, so it should be.
178 // FIXME: This should be in the MachineVerifier, as the RegBankSelected
179 // property check already is.
180 if (!DisableGISelLegalityCheck)
181 if (const MachineInstr *MI = machineFunctionIsIllegal(MF)) {
182 reportGISelFailure(MF, TPC, MORE, "gisel-select",
183 "instruction is not legal", *MI);
184 return false;
185 }
186 // FIXME: We could introduce new blocks and will need to fix the outer loop.
187 // Until then, keep track of the number of blocks to assert that we don't.
188 const size_t NumBlocks = MF.size();
189#endif
190 // Keep track of selected blocks, so we can delete unreachable ones later.
191 DenseSet<MachineBasicBlock *> SelectedBlocks;
192
193 {
194 // Observe IR insertions and removals during selection.
195 // We only install a MachineFunction::Delegate instead of a
196 // GISelChangeObserver, because we do not want notifications about changed
197 // instructions. This prevents significant compile-time regressions from
198 // e.g. constrainOperandRegClass().
199 GISelObserverWrapper AllObservers;
200 MIIteratorMaintainer MIIMaintainer;
201 AllObservers.addObserver(&MIIMaintainer);
202 RAIIDelegateInstaller DelInstaller(MF, &AllObservers);
203 ISel->AllObservers = &AllObservers;
204
205 for (MachineBasicBlock *MBB : post_order(&MF)) {
206 ISel->CurMBB = MBB;
207 SelectedBlocks.insert(MBB);
208
209 // Select instructions in reverse block order.
210 MIIMaintainer.MII = MBB->rbegin();
211 for (auto End = MBB->rend(); MIIMaintainer.MII != End;) {
212 MachineInstr &MI = *MIIMaintainer.MII;
213 // Increment early to skip instructions inserted by select().
214 ++MIIMaintainer.MII;
215
216 LLVM_DEBUG(dbgs() << "\nSelect: " << MI);
217 if (!selectInstr(MI)) {
218 LLVM_DEBUG(dbgs() << "Selection failed!\n";
219 MIIMaintainer.reportFullyCreatedInstrs());
220 reportGISelFailure(MF, TPC, MORE, "gisel-select", "cannot select",
221 MI);
222 return false;
223 }
224 LLVM_DEBUG(MIIMaintainer.reportFullyCreatedInstrs());
225 }
226 }
227 }
228
229 for (MachineBasicBlock &MBB : MF) {
230 if (MBB.empty())
231 continue;
232
233 if (!SelectedBlocks.contains(&MBB)) {
234 // This is an unreachable block and therefore hasn't been selected, since
235 // the main selection loop above uses a postorder block traversal.
236 // We delete all the instructions in this block since it's unreachable.
237 MBB.clear();
238 // Don't delete the block in case the block has it's address taken or is
239 // still being referenced by a phi somewhere.
240 continue;
241 }
242 // Try to find redundant copies b/w vregs of the same register class.
243 for (auto MII = MBB.rbegin(), End = MBB.rend(); MII != End;) {
244 MachineInstr &MI = *MII;
245 ++MII;
246
247 if (MI.getOpcode() != TargetOpcode::COPY)
248 continue;
249 Register SrcReg = MI.getOperand(1).getReg();
250 Register DstReg = MI.getOperand(0).getReg();
251 if (SrcReg.isVirtual() && DstReg.isVirtual()) {
252 auto SrcRC = MRI.getRegClass(SrcReg);
253 auto DstRC = MRI.getRegClass(DstReg);
254 if (SrcRC == DstRC) {
255 MRI.replaceRegWith(DstReg, SrcReg);
256 MI.eraseFromParent();
257 }
258 }
259 }
260 }
261
262#ifndef NDEBUG
263 const TargetRegisterInfo &TRI = *MF.getSubtarget().getRegisterInfo();
264 // Now that selection is complete, there are no more generic vregs. Verify
265 // that the size of the now-constrained vreg is unchanged and that it has a
266 // register class.
267 for (unsigned I = 0, E = MRI.getNumVirtRegs(); I != E; ++I) {
268 Register VReg = Register::index2VirtReg(I);
269
270 MachineInstr *MI = nullptr;
271 if (!MRI.def_empty(VReg))
272 MI = &*MRI.def_instr_begin(VReg);
273 else if (!MRI.use_empty(VReg)) {
274 MI = &*MRI.use_instr_begin(VReg);
275 // Debug value instruction is permitted to use undefined vregs.
276 if (MI->isDebugValue())
277 continue;
278 }
279 if (!MI)
280 continue;
281
282 const TargetRegisterClass *RC = MRI.getRegClassOrNull(VReg);
283 if (!RC) {
284 reportGISelFailure(MF, TPC, MORE, "gisel-select",
285 "VReg has no regclass after selection", *MI);
286 return false;
287 }
288
289 const LLT Ty = MRI.getType(VReg);
290 if (Ty.isValid() &&
291 TypeSize::isKnownGT(Ty.getSizeInBits(), TRI.getRegSizeInBits(*RC))) {
292 reportGISelFailure(
293 MF, TPC, MORE, "gisel-select",
294 "VReg's low-level type and register class have different sizes", *MI);
295 return false;
296 }
297 }
298
299 if (MF.size() != NumBlocks) {
300 MachineOptimizationRemarkMissed R("gisel-select", "GISelFailure",
301 MF.getFunction().getSubprogram(),
302 /*MBB=*/nullptr);
303 R << "inserting blocks is not supported yet";
304 reportGISelFailure(MF, TPC, MORE, R);
305 return false;
306 }
307#endif
308
309 if (!DebugCounter::shouldExecute(GlobalISelCounter)) {
310 dbgs() << "Falling back for function " << MF.getName() << "\n";
311 MF.getProperties().set(MachineFunctionProperties::Property::FailedISel);
312 return false;
313 }
314
315 // Determine if there are any calls in this machine function. Ported from
316 // SelectionDAG.
317 MachineFrameInfo &MFI = MF.getFrameInfo();
318 for (const auto &MBB : MF) {
319 if (MFI.hasCalls() && MF.hasInlineAsm())
320 break;
321
322 for (const auto &MI : MBB) {
323 if ((MI.isCall() && !MI.isReturn()) || MI.isStackAligningInlineAsm())
324 MFI.setHasCalls(true);
325 if (MI.isInlineAsm())
326 MF.setHasInlineAsm(true);
327 }
328 }
329
330 // FIXME: FinalizeISel pass calls finalizeLowering, so it's called twice.
331 auto &TLI = *MF.getSubtarget().getTargetLowering();
332 TLI.finalizeLowering(MF);
333
334 LLVM_DEBUG({
335 dbgs() << "Rules covered by selecting function: " << MF.getName() << ":";
336 for (auto RuleID : CoverageInfo.covered())
337 dbgs() << " id" << RuleID;
338 dbgs() << "\n\n";
339 });
340 CoverageInfo.emit(CoveragePrefix,
341 TLI.getTargetMachine().getTarget().getBackendName());
342
343 // If we successfully selected the function nothing is going to use the vreg
344 // types after us (otherwise MIRPrinter would need them). Make sure the types
345 // disappear.
346 MRI.clearVirtRegTypes();
347
348 // FIXME: Should we accurately track changes?
349 return true;
350}

352bool InstructionSelect::selectInstr(MachineInstr &MI) {
353 MachineRegisterInfo &MRI = ISel->MF->getRegInfo();
354
355 // We could have folded this instruction away already, making it dead.
356 // If so, erase it.
357 if (isTriviallyDead(MI, MRI)) {
358 LLVM_DEBUG(dbgs() << "Is dead.\n");
359 salvageDebugInfo(MRI, MI);
360 MI.eraseFromParent();
361 return true;
362 }
363
364 // Eliminate hints or G_CONSTANT_FOLD_BARRIER.
365 if (isPreISelGenericOptimizationHint(MI.getOpcode()) ||
366 MI.getOpcode() == TargetOpcode::G_CONSTANT_FOLD_BARRIER) {
367 auto [DstReg, SrcReg] = MI.getFirst2Regs();
368
369 // At this point, the destination register class of the op may have
370 // been decided.
371 //
372 // Propagate that through to the source register.
373 const TargetRegisterClass *DstRC = MRI.getRegClassOrNull(DstReg);
374 if (DstRC)
375 MRI.setRegClass(SrcReg, DstRC);
376 assert(canReplaceReg(DstReg, SrcReg, MRI) &&
377 "Must be able to replace dst with src!");
378 MI.eraseFromParent();
379 MRI.replaceRegWith(DstReg, SrcReg);
380 return true;
381 }
382
383 if (MI.getOpcode() == TargetOpcode::G_INVOKE_REGION_START) {
384 MI.eraseFromParent();
385 return true;
386 }
387
388 return ISel->select(MI);
389}
